46/* Glob-style pattern matching. */
47int stringmatchlen(const char *pattern, int patternLen,
48 const char *string, int stringLen, int nocase)
49{
50 while(patternLen) {
51 switch(pattern[0]) {
52 case '*':
53 while (pattern[1] == '*') {
54 pattern++;
55 patternLen--;
56 }
57 if (patternLen == 1)
58 return 1; /* match */
59 while(stringLen) {
60 if (stringmatchlen(pattern+1, patternLen-1,
61 string, stringLen, nocase))
62 return 1; /* match */
63 string++;
64 stringLen--;
65 }
66 return 0; /* no match */
67 break;
68 case '?':
69 if (stringLen == 0)
70 return 0; /* no match */
71 string++;
72 stringLen--;
73 break;
74 case '[':
75 {
76 int not, match;
77
78 pattern++;
79 patternLen--;
80 not = pattern[0] == '^';
81 if (not) {
82 pattern++;
83 patternLen--;
84 }
85 match = 0;
86 while(1) {
87 if (pattern[0] == '\\') {
88 pattern++;
89 patternLen--;
90 if (pattern[0] == string[0])
91 match = 1;
92 } else if (pattern[0] == ']') {
93 break;
94 } else if (patternLen == 0) {
95 pattern--;
96 patternLen++;
97 break;
98 } else if (pattern[1] == '-' && patternLen >= 3) {
99 int start = pattern[0];
100 int end = pattern[2];
101 int c = string[0];
102 if (start > end) {
103 int t = start;
104 start = end;
105 end = t;
106 }
107 if (nocase) {
108 start = tolower(start);
109 end = tolower(end);
110 c = tolower(c);
111 }
112 pattern += 2;
113 patternLen -= 2;
114 if (c >= start && c <= end)
115 match = 1;
116 } else {
117 if (!nocase) {
118 if (pattern[0] == string[0])
119 match = 1;
120 } else {
121 if (tolower((int)pattern[0]) == tolower((int)string[0]))
122 match = 1;
123 }
124 }
125 pattern++;
126 patternLen--;
127 }
128 if (not)
129 match = !match;
130 if (!match)
131 return 0; /* no match */
132 string++;
133 stringLen--;
134 break;
135 }
136 case '\\':
137 if (patternLen >= 2) {
138 pattern++;
139 patternLen--;
140 }
141 /* fall through */
142 default:
143 if (!nocase) {
144 if (pattern[0] != string[0])
145 return 0; /* no match */
146 } else {
147 if (tolower((int)pattern[0]) != tolower((int)string[0]))
148 return 0; /* no match */
149 }
150 string++;
151 stringLen--;
152 break;
153 }
154 pattern++;
155 patternLen--;
156 if (stringLen == 0) {
157 while(*pattern == '*') {
158 pattern++;
159 patternLen--;
160 }
161 break;
162 }
163 }
164 if (patternLen == 0 && stringLen == 0)
165 return 1;
166 return 0;
167}

173/* Convert a string representing an amount of memory into the number of
174 * bytes, so for instance memtoll("1Gb") will return 1073741824 that is
175 * (1024*1024*1024).
176 *
177 * On parsing error, if *err is not NULL, it's set to 1, otherwise it's
178 * set to 0. On error the function return value is 0, regardless of the
179 * fact 'err' is NULL or not. */
180long long memtoll(const char *p, int *err) {
181 const char *u;
182 char buf[128];
183 long mul; /* unit multiplier */
184 long long val;
185 unsigned int digits;
186
187 if (err) *err = 0;
188
189 /* Search the first non digit character. */
190 u = p;
191 if (*u == '-') u++;
192 while(*u && isdigit(*u)) u++;
193 if (*u == '\0' || !strcasecmp(u,"b")) {
194 mul = 1;
195 } else if (!strcasecmp(u,"k")) {
196 mul = 1000;
197 } else if (!strcasecmp(u,"kb")) {
198 mul = 1024;
199 } else if (!strcasecmp(u,"m")) {
200 mul = 1000*1000;
201 } else if (!strcasecmp(u,"mb")) {
202 mul = 1024*1024;
203 } else if (!strcasecmp(u,"g")) {
204 mul = 1000L*1000*1000;
205 } else if (!strcasecmp(u,"gb")) {
206 mul = 1024L*1024*1024;
207 } else {
208 if (err) *err = 1;
209 return 0;
210 }
211
212 /* Copy the digits into a buffer, we'll use strtoll() to convert
213 * the digit (without the unit) into a number. */
214 digits = u-p;
215 if (digits >= sizeof(buf)) {
216 if (err) *err = 1;
217 return 0;
218 }
219 memcpy(buf,p,digits);
220 buf[digits] = '\0';
221
222 char *endptr;
223 errno = 0;
224 val = strtoll(buf,&endptr,10);
225 if ((val == 0 && errno == EINVAL) || *endptr != '\0') {
226 if (err) *err = 1;
227 return 0;
228 }
229 return val*mul;
230}

232/* Return the number of digits of 'v' when converted to string in radix 10.
233 * See ll2string() for more information. */
234uint32_t digits10(uint64_t v) {
235 if (v < 10) return 1;
236 if (v < 100) return 2;
237 if (v < 1000) return 3;
238 if (v < 1000000000000UL) {
239 if (v < 100000000UL) {
240 if (v < 1000000) {
241 if (v < 10000) return 4;
242 return 5 + (v >= 100000);
243 }
244 return 7 + (v >= 10000000UL);
245 }
246 if (v < 10000000000UL) {
247 return 9 + (v >= 1000000000UL);
248 }
249 return 11 + (v >= 100000000000UL);
250 }
251 return 12 + digits10(v / 1000000000000UL);
252}

254/* Like digits10() but for signed values. */
255uint32_t sdigits10(int64_t v) {
256 if (v < 0) {
257 /* Abs value of LLONG_MIN requires special handling. */
258 uint64_t uv = (v != LLONG_MIN) ?
259 (uint64_t)-v : ((uint64_t) LLONG_MAX)+1;
260 return digits10(uv)+1; /* +1 for the minus. */
261 } else {
262 return digits10(v);
263 }
264}

266/* Convert a long long into a string. Returns the number of
267 * characters needed to represent the number.
268 * If the buffer is not big enough to store the string, 0 is returned.
269 *
270 * Based on the following article (that apparently does not provide a
271 * novel approach but only publicizes an already used technique):
272 *
273 * https://www.facebook.com/notes/facebook-engineering/three-optimization-tips-for-c/10151361643253920
274 *
275 * Modified in order to handle signed integers since the original code was
276 * designed for unsigned integers. */
277int ll2string(char *dst, size_t dstlen, long long svalue) {
278 static const char digits[201] =
279 "0001020304050607080910111213141516171819"
280 "2021222324252627282930313233343536373839"
281 "4041424344454647484950515253545556575859"
282 "6061626364656667686970717273747576777879"
283 "8081828384858687888990919293949596979899";
284 int negative;
285 unsigned long long value;
286
287 /* The main loop works with 64bit unsigned integers for simplicity, so
288 * we convert the number here and remember if it is negative. */
289 if (svalue < 0) {
290 if (svalue != LLONG_MIN) {
291 value = -svalue;
292 } else {
293 value = ((unsigned long long) LLONG_MAX)+1;
294 }
295 negative = 1;
296 } else {
297 value = svalue;
298 negative = 0;
299 }
300
301 /* Check length. */
302 uint32_t const length = digits10(value)+negative;
303 if (length >= dstlen) return 0;
304
305 /* Null term. */
306 uint32_t next = length;
307 dst[next] = '\0';
308 next--;
309 while (value >= 100) {
310 int const i = (value % 100) * 2;
311 value /= 100;
312 dst[next] = digits[i + 1];
313 dst[next - 1] = digits[i];
314 next -= 2;
315 }
316
317 /* Handle last 1-2 digits. */
318 if (value < 10) {
319 dst[next] = '0' + (uint32_t) value;
320 } else {
321 int i = (uint32_t) value * 2;
322 dst[next] = digits[i + 1];
323 dst[next - 1] = digits[i];
324 }
325
326 /* Add sign. */
327 if (negative) dst[0] = '-';
328 return length;
329}

331/* Convert a string into a long long. Returns 1 if the string could be parsed
332 * into a (non-overflowing) long long, 0 otherwise. The value will be set to
333 * the parsed value when appropriate.
334 *
335 * Note that this function demands that the string strictly represents
336 * a long long: no spaces or other characters before or after the string
337 * representing the number are accepted, nor zeroes at the start if not
338 * for the string "0" representing the zero number.
339 *
340 * Because of its strictness, it is safe to use this function to check if
341 * you can convert a string into a long long, and obtain back the string
342 * from the number without any loss in the string representation. */
343int string2ll(const char *s, size_t slen, long long *value) {
344 const char *p = s;
345 size_t plen = 0;
346 int negative = 0;
347 unsigned long long v;
348
349 if (plen == slen)
350 return 0;
351
352 /* Special case: first and only digit is 0. */
353 if (slen == 1 && p[0] == '0') {
354 if (value != NULL) *value = 0;
355 return 1;
356 }
357
358 if (p[0] == '-') {
359 negative = 1;
360 p++; plen++;
361
362 /* Abort on only a negative sign. */
363 if (plen == slen)
364 return 0;
365 }
366
367 /* First digit should be 1-9, otherwise the string should just be 0. */
368 if (p[0] >= '1' && p[0] <= '9') {
369 v = p[0]-'0';
370 p++; plen++;
371 } else if (p[0] == '0' && slen == 1) {
372 *value = 0;
373 return 1;
374 } else {
375 return 0;
376 }
377
378 while (plen < slen && p[0] >= '0' && p[0] <= '9') {
379 if (v > (ULLONG_MAX / 10)) /* Overflow. */
380 return 0;
381 v *= 10;
382
383 if (v > (ULLONG_MAX - (p[0]-'0'))) /* Overflow. */
384 return 0;
385 v += p[0]-'0';
386
387 p++; plen++;
388 }
389
390 /* Return if not all bytes were used. */
391 if (plen < slen)
392 return 0;
393
394 if (negative) {
395 if (v > ((unsigned long long)(-(LLONG_MIN+1))+1)) /* Overflow. */
396 return 0;
397 if (value != NULL) *value = -v;
398 } else {
399 if (v > LLONG_MAX) /* Overflow. */
400 return 0;
401 if (value != NULL) *value = v;
402 }
403 return 1;
404}

406/* Convert a string into a long. Returns 1 if the string could be parsed into a
407 * (non-overflowing) long, 0 otherwise. The value will be set to the parsed
408 * value when appropriate. */
409int string2l(const char *s, size_t slen, long *lval) {
410 long long llval;
411
412 if (!string2ll(s,slen,&llval))
413 return 0;
414
415 if (llval < LONG_MIN || llval > LONG_MAX)
416 return 0;
417
418 *lval = (long)llval;
419 return 1;
420}

422/* Convert a string into a double. Returns 1 if the string could be parsed
423 * into a (non-overflowing) double, 0 otherwise. The value will be set to
424 * the parsed value when appropriate.
425 *
426 * Note that this function demands that the string strictly represents
427 * a double: no spaces or other characters before or after the string
428 * representing the number are accepted. */
429int string2ld(const char *s, size_t slen, long double *dp) {
430 char buf[256];
431 long double value;
432 char *eptr;
433
434 if (slen >= sizeof(buf)) return 0;
435 memcpy(buf,s,slen);
436 buf[slen] = '\0';
437
438 errno = 0;
439 value = strtold(buf, &eptr);
440 if (isspace(buf[0]) || eptr[0] != '\0' ||
441 (errno == ERANGE &&
442 (value == HUGE_VAL || value == -HUGE_VAL || value == 0)) ||
443 errno == EINVAL ||
444 isnan(value))
445 return 0;
446
447 if (dp) *dp = value;
448 return 1;
449}

451/* Convert a double to a string representation. Returns the number of bytes
452 * required. The representation should always be parsable by strtod(3).
453 * This function does not support human-friendly formatting like ld2string
454 * does. It is intented mainly to be used inside t_zset.c when writing scores
455 * into a ziplist representing a sorted set. */
456int d2string(char *buf, size_t len, double value) {
457 if (isnan(value)) {
458 len = snprintf(buf,len,"nan");
459 } else if (isinf(value)) {
460 if (value < 0)
461 len = snprintf(buf,len,"-inf");
462 else
463 len = snprintf(buf,len,"inf");
464 } else if (value == 0) {
465 /* See: http://en.wikipedia.org/wiki/Signed_zero, "Comparisons". */
466 if (1.0/value < 0)
467 len = snprintf(buf,len,"-0");
468 else
469 len = snprintf(buf,len,"0");
470 } else {
471#if (DBL_MANT_DIG >= 52) && (LLONG_MAX == 0x7fffffffffffffffLL)
472 /* Check if the float is in a safe range to be casted into a
473 * long long. We are assuming that long long is 64 bit here.
474 * Also we are assuming that there are no implementations around where
475 * double has precision < 52 bit.
476 *
477 * Under this assumptions we test if a double is inside an interval
478 * where casting to long long is safe. Then using two castings we
479 * make sure the decimal part is zero. If all this is true we use
480 * integer printing function that is much faster. */
481 double min = -4503599627370495; /* (2^52)-1 */
482 double max = 4503599627370496; /* -(2^52) */
483 if (value > min && value < max && value == ((double)((long long)value)))
484 len = ll2string(buf,len,(long long)value);
485 else
486#endif
487 len = snprintf(buf,len,"%.17g",value);
488 }
489
490 return len;
491}

493/* Convert a long double into a string. If humanfriendly is non-zero
494 * it does not use exponential format and trims trailing zeroes at the end,
495 * however this results in loss of precision. Otherwise exp format is used
496 * and the output of snprintf() is not modified.
497 *
498 * The function returns the length of the string or zero if there was not
499 * enough buffer room to store it. */
500int ld2string(char *buf, size_t len, long double value, int humanfriendly) {
501 size_t l;
502
503 if (isinf(value)) {
504 /* Libc in odd systems (Hi Solaris!) will format infinite in a
505 * different way, so better to handle it in an explicit way. */
506 if (len < 5) return 0; /* No room. 5 is "-inf\0" */
507 if (value > 0) {
508 memcpy(buf,"inf",3);
509 l = 3;
510 } else {
511 memcpy(buf,"-inf",4);
512 l = 4;
513 }
514 } else if (humanfriendly) {
515 /* We use 17 digits precision since with 128 bit floats that precision
516 * after rounding is able to represent most small decimal numbers in a
517 * way that is "non surprising" for the user (that is, most small
518 * decimal numbers will be represented in a way that when converted
519 * back into a string are exactly the same as what the user typed.) */
520 l = snprintf(buf,len,"%.17Lf", value);
521 if (l+1 > len) return 0; /* No room. */
522 /* Now remove trailing zeroes after the '.' */
523 if (strchr(buf,'.') != NULL) {
524 char *p = buf+l-1;
525 while(*p == '0') {
526 p--;
527 l--;
528 }
529 if (*p == '.') l--;
530 }
531 } else {
532 l = snprintf(buf,len,"%.17Lg", value);
533 if (l+1 > len) return 0; /* No room. */
534 }
535 buf[l] = '\0';
536 return l;
537}

539/* Generate the Redis "Run ID", a SHA1-sized random number that identifies a
540 * given execution of Redis, so that if you are talking with an instance
541 * having run_id == A, and you reconnect and it has run_id == B, you can be
542 * sure that it is either a different instance or it was restarted. */
543void getRandomHexChars(char *p, unsigned int len) {
544 char *charset = "0123456789abcdef";
545 unsigned int j;
546
547 /* Global state. */
548 static int seed_initialized = 0;
549 static unsigned char seed[20]; /* The SHA1 seed, from /dev/urandom. */
550 static uint64_t counter = 0; /* The counter we hash with the seed. */
551
552 if (!seed_initialized) {
553 /* Initialize a seed and use SHA1 in counter mode, where we hash
554 * the same seed with a progressive counter. For the goals of this
555 * function we just need non-colliding strings, there are no
556 * cryptographic security needs. */
557 FILE *fp = fopen("/dev/urandom","r");
558 if (fp && fread(seed,sizeof(seed),1,fp) == 1)
559 seed_initialized = 1;
560 if (fp) fclose(fp);
561 }
562
563 if (seed_initialized) {
564 while(len) {
565 unsigned char digest[20];
566 SHA1_CTX ctx;
567 unsigned int copylen = len > 20 ? 20 : len;
568
569 SHA1Init(&ctx);
570 SHA1Update(&ctx, seed, sizeof(seed));
571 SHA1Update(&ctx, (unsigned char*)&counter,sizeof(counter));
572 SHA1Final(digest, &ctx);
573 counter++;
574
575 memcpy(p,digest,copylen);
576 /* Convert to hex digits. */
577 for (j = 0; j < copylen; j++) p[j] = charset[p[j] & 0x0F];
578 len -= copylen;
579 p += copylen;
580 }
581 } else {
582 /* If we can't read from /dev/urandom, do some reasonable effort
583 * in order to create some entropy, since this function is used to
584 * generate run_id and cluster instance IDs */
585 char *x = p;
586 unsigned int l = len;
587 struct timeval tv;
588 pid_t pid = getpid();
589
590 /* Use time and PID to fill the initial array. */
591 gettimeofday(&tv,NULL);
592 if (l >= sizeof(tv.tv_usec)) {
593 memcpy(x,&tv.tv_usec,sizeof(tv.tv_usec));
594 l -= sizeof(tv.tv_usec);
595 x += sizeof(tv.tv_usec);
596 }
597 if (l >= sizeof(tv.tv_sec)) {
598 memcpy(x,&tv.tv_sec,sizeof(tv.tv_sec));
599 l -= sizeof(tv.tv_sec);
600 x += sizeof(tv.tv_sec);
601 }
602 if (l >= sizeof(pid)) {
603 memcpy(x,&pid,sizeof(pid));
604 l -= sizeof(pid);
605 x += sizeof(pid);
606 }
607 /* Finally xor it with rand() output, that was already seeded with
608 * time() at startup, and convert to hex digits. */
609 for (j = 0; j < len; j++) {
610 p[j] ^= rand();
611 p[j] = charset[p[j] & 0x0F];
612 }
613 }
614}

616/* Given the filename, return the absolute path as an SDS string, or NULL
617 * if it fails for some reason. Note that "filename" may be an absolute path
618 * already, this will be detected and handled correctly.
619 *
620 * The function does not try to normalize everything, but only the obvious
621 * case of one or more "../" appearning at the start of "filename"
622 * relative path. */
623sds getAbsolutePath(char *filename) {
624 char cwd[1024];
625 sds abspath;
626 sds relpath = sdsnew(filename);
627
628 relpath = sdstrim(relpath," \r\n\t");
629 if (relpath[0] == '/') return relpath; /* Path is already absolute. */
630
631 /* If path is relative, join cwd and relative path. */
632 if (getcwd(cwd,sizeof(cwd)) == NULL) {
633 sdsfree(relpath);
634 return NULL;
635 }
636 abspath = sdsnew(cwd);
637 if (sdslen(abspath) && abspath[sdslen(abspath)-1] != '/')
638 abspath = sdscat(abspath,"/");
639
640 /* At this point we have the current path always ending with "/", and
641 * the trimmed relative path. Try to normalize the obvious case of
642 * trailing ../ elements at the start of the path.
643 *
644 * For every "../" we find in the filename, we remove it and also remove
645 * the last element of the cwd, unless the current cwd is "/". */
646 while (sdslen(relpath) >= 3 &&
647 relpath[0] == '.' && relpath[1] == '.' && relpath[2] == '/')
648 {
649 sdsrange(relpath,3,-1);
650 if (sdslen(abspath) > 1) {
651 char *p = abspath + sdslen(abspath)-2;
652 int trimlen = 1;
653
654 while(*p != '/') {
655 p--;
656 trimlen++;
657 }
658 sdsrange(abspath,0,-(trimlen+1));
659 }
660 }
661
662 /* Finally glue the two parts together. */
663 abspath = sdscatsds(abspath,relpath);
664 sdsfree(relpath);
665 return abspath;
666}

668/* Return true if the specified path is just a file basename without any
669 * relative or absolute path. This function just checks that no / or \
670 * character exists inside the specified path, that's enough in the
671 * environments where Redis runs. */
672int pathIsBaseName(char *path) {
673 return strchr(path,'/') == NULL && strchr(path,'\\') == NULL;
674}

679static void test_string2ll(void) {

681 long long v;
682
683 /* May not start with +. */
684 strcpy(buf,"+1");

694

696 strcpy(buf,"01");
697 assert(string2ll(buf,strlen(buf),&v) == 0);
698
699 strcpy(buf,"-1");
700 assert(string2ll(buf,strlen(buf),&v) == 1);
701 assert(v == -1);
702
703 strcpy(buf,"0");
704 assert(string2ll(buf,strlen(buf),&v) == 1);
705 assert(v == 0);
706
707 strcpy(buf,"1");
708 assert(string2ll(buf,strlen(buf),&v) == 1);
709 assert(v == 1);
710
711 strcpy(buf,"99");
712 assert(string2ll(buf,strlen(buf),&v) == 1);
713 assert(v == 99);
714
715 strcpy(buf,"-99");
716 assert(string2ll(buf,strlen(buf),&v) == 1);
717 assert(v == -99);
718
719 strcpy(buf,"-9223372036854775808");
720 assert(string2ll(buf,strlen(buf),&v) == 1);
721 assert(v == LLONG_MIN);
722

724 assert(string2ll(buf,strlen(buf),&v) == 0);
725

727 assert(string2ll(buf,strlen(buf),&v) == 1);
728 assert(v == LLONG_MAX);
729
730 strcpy(buf,"9223372036854775808"); /* overflow */
731 assert(string2ll(buf,strlen(buf),&v) == 0);
