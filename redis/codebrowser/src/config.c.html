105/* Get enum value from name. If there is no match INT_MIN is returned. */
106int configEnumGetValue(configEnum *ce, char *name) {
107 while(ce->name != NULL) {
108 if (!strcasecmp(ce->name,name)) return ce->val;
109 ce++;
110 }
111 return INT_MIN;
112}
113
114/* Get enum name from value. If no match is found NULL is returned. */
115const char *configEnumGetName(configEnum *ce, int val) {
116 while(ce->name != NULL) {
117 if (ce->val == val) return ce->name;
118 ce++;
119 }
120 return NULL;
121}
122
123/* Wrapper for configEnumGetName() returning "unknown" insetad of NULL if
124 * there is no match. */
125const char *configEnumGetNameOrUnknown(configEnum *ce, int val) {
126 const char *name = configEnumGetName(ce,val);
127 return name ? name : "unknown";
128}

178 lines = sdssplitlen(config,strlen(config),"\n",1,&totlines);
179
180 for (i = 0; i < totlines; i++) {
181 sds *argv;
182 int argc;
183
184 linenum = i+1;

186
187 /* Skip comments and blank lines */
188 if (lines[i][0] == '#' || lines[i][0] == '\0') continue;
189
190 /* Split into arguments */
191 argv = sdssplitargs(lines[i],&argc);
192 if (argv == NULL) {
193 err = "Unbalanced quotes in configuration line";

195 }
196
197 /* Skip this line if the resulting command vector is empty. */
198 if (argc == 0) {

200 continue;
201 }
202 sdstolower(argv[0]);
203
204 /* Execute config directives */
205 if (!strcasecmp(argv[0],"timeout") && argc == 2) {
206 server.maxidletime = atoi(argv[1]);
207 if (server.maxidletime < 0) {
208 err = "Invalid timeout value"; goto loaderr;
209 }
210 } else if (!strcasecmp(argv[0],"tcp-keepalive") && argc == 2) {
211 server.tcpkeepalive = atoi(argv[1]);
212 if (server.tcpkeepalive < 0) {
213 err = "Invalid tcp-keepalive value"; goto loaderr;
214 }
215 } else if (!strcasecmp(argv[0],"protected-mode") && argc == 2) {
216 if ((server.protected_mode = yesnotoi(argv[1])) == -1) {
217 err = "argument must be 'yes' or 'no'"; goto loaderr;
218 }
219 } else if (!strcasecmp(argv[0],"port") && argc == 2) {

754/* Load the server configuration from the specified filename.
755 * The function appends the additional configuration directives stored
756 * in the 'options' string to the config file before loading.
757 *
758 * Both filename and options can be NULL, in such a case are considered
759 * empty. This way loadServerConfig can be used to just load a file or
760 * just load a string. */
761void loadServerConfig(char *filename, char *options) {
762 sds config = sdsempty();
763 char buf[CONFIG_MAX_LINE+1];
764
765 /* Load the file content */
766 if (filename) {
767 FILE *fp;
768
769 if (filename[0] == '-' && filename[1] == '\0') {
770 fp = stdin;
771 } else {
772 if ((fp = fopen(filename,"r")) == NULL) {
773 serverLog(LL_WARNING,
774 "Fatal error, can't open config file '%s'", filename);
775 exit(1);
776 }
777 }
778 while(fgets(buf,CONFIG_MAX_LINE+1,fp) != NULL)
779 config = sdscat(config,buf);
780 if (fp != stdin) fclose(fp);
781 }
782 /* Append the additional options */
783 if (options) {
784 config = sdscat(config,"\n");
785 config = sdscat(config,options);
786 }
787 loadServerConfigFromString(config);
788 sdsfree(config);
789}

1490/* Read the old file, split it into lines to populate a newly created
1491 * config rewrite state, and return it to the caller.
1492 *
1493 * If it is impossible to read the old file, NULL is returned.
1494 * If the old file does not exist at all, an empty state is returned. */
1495struct rewriteConfigState *rewriteConfigReadOldFile(char *path) {
1496 FILE *fp = fopen(path,"r");
1497 struct rewriteConfigState *state = zmalloc(sizeof(*state));
1498 char buf[CONFIG_MAX_LINE+1];
1499 int linenum = -1;
1500
1501 if (fp == NULL && errno != ENOENT) return NULL;
1502
1503 state->option_to_line = dictCreate(&optionToLineDictType,NULL);
1504 state->rewritten = dictCreate(&optionSetDictType,NULL);
1505 state->numlines = 0;
1506 state->lines = NULL;
1507 state->has_tail = 0;
1508 if (fp == NULL) return state;
1509
1510 /* Read the old file line by line, populate the state. */
1511 while(fgets(buf,CONFIG_MAX_LINE+1,fp) != NULL) {
1512 int argc;
1513 sds *argv;
1514 sds line = sdstrim(sdsnew(buf),"\r\n\t ");
1515
1516 linenum++; /* Zero based, so we init at -1 */
1517
1518 /* Handle comments and empty lines. */
1519 if (line[0] == '#' || line[0] == '\0') {
1520 if (!state->has_tail && !strcmp(line,REDIS_CONFIG_REWRITE_SIGNATURE))
1521 state->has_tail = 1;
1522 rewriteConfigAppendLine(state,line);
1523 continue;
1524 }
1525
1526 /* Not a comment, split into arguments. */
1527 argv = sdssplitargs(line,&argc);
1528 if (argv == NULL) {
1529 /* Apparently the line is unparsable for some reason, for
1530 * instance it may have unbalanced quotes. Load it as a
1531 * comment. */
1532 sds aux = sdsnew("# ??? ");
1533 aux = sdscatsds(aux,line);
1534 sdsfree(line);
1535 rewriteConfigAppendLine(state,aux);
1536 continue;
1537 }
1538
1539 sdstolower(argv[0]); /* We only want lowercase config directives. */
1540
1541 /* Now we populate the state according to the content of this line.
1542 * Append the line and populate the option -> line numbers map. */
1543 rewriteConfigAppendLine(state,line);
1544 rewriteConfigAddLineNumberToOption(state,argv[0],linenum);
1545
1546 sdsfreesplitres(argv,argc);
1547 }
1548 fclose(fp);
1549 return state;
1550}

1552/* Rewrite the specified configuration option with the new "line".
1553 * It progressively uses lines of the file that were already used for the same
1554 * configuration option in the old version of the file, removing that line from
1555 * the map of options -> line numbers.
1556 *
1557 * If there are lines associated with a given configuration option and
1558 * "force" is non-zero, the line is appended to the configuration file.
1559 * Usually "force" is true when an option has not its default value, so it
1560 * must be rewritten even if not present previously.
1561 *
1562 * The first time a line is appended into a configuration file, a comment
1563 * is added to show that starting from that point the config file was generated
1564 * by CONFIG REWRITE.
1565 *
1566 * "line" is either used, or freed, so the caller does not need to free it
1567 * in any way. */
1568void rewriteConfigRewriteLine(struct rewriteConfigState *state, const char *option, sds line, int force) {
1569 sds o = sdsnew(option);
1570 list *l = dictFetchValue(state->option_to_line,o);
1571
1572 rewriteConfigMarkAsProcessed(state,option);
1573
1574 if (!l && !force) {
1575 /* Option not used previously, and we are not forced to use it. */
1576 sdsfree(line);
1577 sdsfree(o);
1578 return;
1579 }
1580
1581 if (l) {
1582 listNode *ln = listFirst(l);
1583 int linenum = (long) ln->value;
1584
1585 /* There are still lines in the old configuration file we can reuse
1586 * for this option. Replace the line with the new one. */
1587 listDelNode(l,ln);
1588 if (listLength(l) == 0) dictDelete(state->option_to_line,o);
1589 sdsfree(state->lines[linenum]);
1590 state->lines[linenum] = line;
1591 } else {
1592 /* Append a new line. */
1593 if (!state->has_tail) {
1594 rewriteConfigAppendLine(state,
1595 sdsnew(REDIS_CONFIG_REWRITE_SIGNATURE));
1596 state->has_tail = 1;
1597 }
1598 rewriteConfigAppendLine(state,line);
1599 }
1600 sdsfree(o);
1601}

1811/* Glue together the configuration lines in the current configuration
1812 * rewrite state into a single string, stripping multiple empty lines. */
1813sds rewriteConfigGetContentFromState(struct rewriteConfigState *state) {
1814 sds content = sdsempty();
1815 int j, was_empty = 0;
1816
1817 for (j = 0; j < state->numlines; j++) {
1818 /* Every cluster of empty lines is turned into a single empty line. */
1819 if (sdslen(state->lines[j]) == 0) {
1820 if (was_empty) continue;
1821 was_empty = 1;
1822 } else {
1823 was_empty = 0;
1824 }
1825 content = sdscatsds(content,state->lines[j]);
1826 content = sdscatlen(content,"\n",1);
1827 }
1828 return content;
1829}

1839/* At the end of the rewrite process the state contains the remaining
1840 * map between "option name" => "lines in the original config file".
1841 * Lines used by the rewrite process were removed by the function
1842 * rewriteConfigRewriteLine(), all the other lines are "orphaned" and
1843 * should be replaced by empty lines.
1844 *
1845 * This function does just this, iterating all the option names and
1846 * blanking all the lines still associated. */
1847void rewriteConfigRemoveOrphaned(struct rewriteConfigState *state) {
1848 dictIterator *di = dictGetIterator(state->option_to_line);
1849 dictEntry *de;
1850
1851 while((de = dictNext(di)) != NULL) {
1852 list *l = dictGetVal(de);
1853 sds option = dictGetKey(de);
1854
1855 /* Don't blank lines about options the rewrite process
1856 * don't understand. */
1857 if (dictFind(state->rewritten,option) == NULL) {
1858 serverLog(LL_DEBUG,"Not rewritten option: %s", option);
1859 continue;
1860 }
1861
1862 while(listLength(l)) {
1863 listNode *ln = listFirst(l);
1864 int linenum = (long) ln->value;
1865
1866 sdsfree(state->lines[linenum]);
1867 state->lines[linenum] = sdsempty();
1868 listDelNode(l,ln);
1869 }
1870 }
1871 dictReleaseIterator(di);
1872}

1874/* This function overwrites the old configuration file with the new content.
1875 *
1876 * 1) The old file length is obtained.
1877 * 2) If the new content is smaller, padding is added.
1878 * 3) A single write(2) call is used to replace the content of the file.
1879 * 4) Later the file is truncated to the length of the new content.
1880 *
1881 * This way we are sure the file is left in a consistent state even if the
1882 * process is stopped between any of the four operations.
1883 *
1884 * The function returns 0 on success, otherwise -1 is returned and errno
1885 * set accordingly. */
1886int rewriteConfigOverwriteFile(char *configfile, sds content) {
1887 int retval = 0;
1888 int fd = open(configfile,O_RDWR|O_CREAT,0644);
1889 int content_size = sdslen(content), padding = 0;
1890 struct stat sb;
1891 sds content_padded;
1892
1893 /* 1) Open the old file (or create a new one if it does not
1894 * exist), get the size. */
1895 if (fd == -1) return -1; /* errno set by open(). */
1896 if (fstat(fd,&sb) == -1) {
1897 close(fd);
1898 return -1; /* errno set by fstat(). */
1899 }
1900
1901 /* 2) Pad the content at least match the old file size. */
1902 content_padded = sdsdup(content);
1903 if (content_size < sb.st_size) {
1904 /* If the old file was bigger, pad the content with
1905 * a newline plus as many "#" chars as required. */
1906 padding = sb.st_size - content_size;
1907 content_padded = sdsgrowzero(content_padded,sb.st_size);
1908 content_padded[content_size] = '\n';
1909 memset(content_padded+content_size+1,'#',padding-1);
1910 }
1911
1912 /* 3) Write the new content using a single write(2). */
1913 if (write(fd,content_padded,strlen(content_padded)) == -1) {
1914 retval = -1;
1915 goto cleanup;
1916 }
1917
1918 /* 4) Truncate the file to the right length if we used padding. */
1919 if (padding) {
1920 if (ftruncate(fd,content_size) == -1) {
1921 /* Non critical error... */
1922 }
1923 }
1924
1925cleanup:
1926 sdsfree(content_padded);
1927 close(fd);
1928 return retval;
1929}
