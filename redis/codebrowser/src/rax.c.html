69/* ------------------------- raxStack functions --------------------------
70 * The raxStack is a simple stack of pointers that is capable of switching
71 * from using a stack-allocated array to dynamic heap once a given number of
72 * items are reached. It is used in order to retain the list of parent nodes
73 * while walking the radix tree in order to implement certain operations that
74 * need to navigate the tree upward.
75 * ------------------------------------------------------------------------- */
76
77/* Initialize the stack. */
78static inline void raxStackInit(raxStack *ts) {
79 ts->stack = ts->static_items;
80 ts->items = 0;
81 ts->maxitems = RAX_STACK_STATIC_ITEMS;
82 ts->oom = 0;
83}
84
85/* Push an item into the stack, returns 1 on success, 0 on out of memory. */
86static inline int raxStackPush(raxStack *ts, void *ptr) {
87 if (ts->items == ts->maxitems) {
88 if (ts->stack == ts->static_items) {
89 ts->stack = rax_malloc(sizeof(void*)*ts->maxitems*2);
90 if (ts->stack == NULL) {
91 ts->stack = ts->static_items;
92 ts->oom = 1;
93 errno = ENOMEM;
94 return 0;
95 }
96 memcpy(ts->stack,ts->static_items,sizeof(void*)*ts->maxitems);
97 } else {
98 void **newalloc = rax_realloc(ts->stack,sizeof(void*)*ts->maxitems*2);
99 if (newalloc == NULL) {
100 ts->oom = 1;
101 errno = ENOMEM;
102 return 0;
103 }
104 ts->stack = newalloc;
105 }
106 ts->maxitems *= 2;
107 }
108 ts->stack[ts->items] = ptr;
109 ts->items++;
110 return 1;
111}
112
113/* Pop an item from the stack, the function returns NULL if there are no
114 * items to pop. */
115static inline void *raxStackPop(raxStack *ts) {
116 if (ts->items == 0) return NULL;
117 ts->items--;
118 return ts->stack[ts->items];
119}
120
121/* Return the stack item at the top of the stack without actually consuming
122 * it. */
123static inline void *raxStackPeek(raxStack *ts) {
124 if (ts->items == 0) return NULL;
125 return ts->stack[ts->items-1];
126}
127
128/* Free the stack in case we used heap allocation. */
129static inline void raxStackFree(raxStack *ts) {
130 if (ts->stack != ts->static_items) rax_free(ts->stack);
131}

137/* Allocate a new non compressed node with the specified number of children.
138 * If datafiled is true, the allocation is made large enough to hold the
139 * associated data pointer.
140 * Returns the new node pointer. On out of memory NULL is returned. */
141raxNode *raxNewNode(size_t children, int datafield) {
142 size_t nodesize = sizeof(raxNode)+children+
143 sizeof(raxNode*)*children;
144 if (datafield) nodesize += sizeof(void*);
145 raxNode *node = rax_malloc(nodesize);
146 if (node == NULL) return NULL;
147 node->iskey = 0;
148 node->isnull = 0;
149 node->iscompr = 0;
150 node->size = children;
151 return node;
152}

154/* Allocate a new rax and return its pointer. On out of memory the function
155 * returns NULL. */
156rax *raxNew(void) {
157 rax *rax = rax_malloc(sizeof(*rax));
158 if (rax == NULL) return NULL;
159 rax->numele = 0;
160 rax->numnodes = 1;
161 rax->head = raxNewNode(0,0);
162 if (rax->head == NULL) {
163 rax_free(rax);
164 return NULL;
165 } else {
166 return rax;
167 }
168}

177/* realloc the node to make room for auxiliary data in order
178 * to store an item in that node. On out of memory NULL is returned. */
179raxNode *raxReallocForData(raxNode *n, void *data) {
180 if (data == NULL) return n; /* No reallocation needed, setting isnull=1 */
181 size_t curlen = raxNodeCurrentLength(n);
182 return rax_realloc(n,curlen+sizeof(void*));

185/* Set the node auxiliary data to the specified pointer. */
186void raxSetData(raxNode *n, void *data) {
187 n->iskey = 1;
188 if (data != NULL) {
189 n->isnull = 0;
190 void **ndata = (void**)
191 ((char*)n+raxNodeCurrentLength(n)-sizeof(void*));
192 memcpy(ndata,&data,sizeof(data));
193 } else {
194 n->isnull = 1;
195 }
196}

207/* Add a new child to the node 'n' representing the character 'c' and return
208 * its new pointer, as well as the child pointer by reference. Additionally
209 * '***parentlink' is populated with the raxNode pointer-to-pointer of where
210 * the new child was stored, which is useful for the caller to replace the
211 * child pointer if it gets reallocated.
212 *
213 * On success the new parent node pointer is returned (it may change because
214 * of the realloc, so the caller should discard 'n' and use the new value).
215 * On out of memory NULL is returned, and the old node is still valid. */
216raxNode *raxAddChild(raxNode *n, unsigned char c, raxNode **childptr, raxNode ***parentlink) {
217 assert(n->iscompr == 0);
218
219 size_t curlen = sizeof(raxNode)+
220 n->size+
221 sizeof(raxNode*)*n->size;
222 size_t newlen;
223
224 /* Alloc the new child we will link to 'n'. */
225 raxNode *child = raxNewNode(0,0);
226 if (child == NULL) return NULL;
227
228 /* Make space in the original node. */
229 if (n->iskey) curlen += sizeof(void*);
230 newlen = curlen+sizeof(raxNode*)+1; /* Add 1 char and 1 pointer. */
231 raxNode *newn = rax_realloc(n,newlen);
232 if (newn == NULL) {
233 rax_free(child);
234 return NULL;
235 }
236 n = newn;
237
238 /* After the reallocation, we have 5/9 (depending on the system
239 * pointer size) bytes at the end, that is, the additional char
240 * in the 'data' section, plus one pointer to the new child:
241 *
242 * [numc][abx][ap][bp][xp]|auxp|.....
243 *
244 * Let's find where to insert the new child in order to make sure
245 * it is inserted in-place lexicographically. */
246 int pos;
247 for (pos = 0; pos < n->size; pos++) {
248 if (n->data[pos] > c) break;
249 }
250
251 /* Now, if present, move auxiliary data pointer at the end
252 * so that we can mess with the other data without overwriting it.
253 * We will obtain something like that:
254 *
255 * [numc][abx][ap][bp][xp].....|auxp| */
256 unsigned char *src;
257 if (n->iskey && !n->isnull) {
258 src = n->data+n->size+sizeof(raxNode*)*n->size;
259 memmove(src+1+sizeof(raxNode*),src,sizeof(void*));
260 }
261
262 /* Now imagine we are adding a node with edge 'c'. The insertion
263 * point is between 'b' and 'x', so the 'pos' variable value is
264 * To start, move all the child pointers after the insertion point
265 * of 1+sizeof(pointer) bytes on the right, to obtain:
266 *
267 * [numc][abx][ap][bp].....[xp]|auxp| */
268 src = n->data+n->size+sizeof(raxNode*)*pos;
269 memmove(src+1+sizeof(raxNode*),src,sizeof(raxNode*)*(n->size-pos));
270
271 /* Now make the space for the additional char in the data section,
272 * but also move the pointers before the insertion point in the right
273 * by 1 byte, in order to obtain the following:
274 *
275 * [numc][ab.x][ap][bp]....[xp]|auxp| */
276 src = n->data+pos;
277 memmove(src+1,src,n->size-pos+sizeof(raxNode*)*pos);
278
279 /* We can now set the character and its child node pointer to get:
280 *
281 * [numc][abcx][ap][bp][cp]....|auxp|
282 * [numc][abcx][ap][bp][cp][xp]|auxp| */
283 n->data[pos] = c;
284 n->size++;
285 raxNode **childfield = (raxNode**)(n->data+n->size+sizeof(raxNode*)*pos);
286 memcpy(childfield,&child,sizeof(child));
287 *childptr = child;
288 *parentlink = childfield;
289 return n;
290}

292/* Return the pointer to the last child pointer in a node. For the compressed
293 * nodes this is the only child pointer. */
294#define raxNodeLastChildPtr(n) ((raxNode**) ( \
295 ((char*)(n)) + \
296 raxNodeCurrentLength(n) - \
297 sizeof(raxNode*) - \
298 (((n)->iskey && !(n)->isnull) ? sizeof(void*) : 0) \
299))

301/* Return the pointer to the first child pointer. */
302#define raxNodeFirstChildPtr(n) ((raxNode**)((n)->data+(n)->size))

304/* Turn the node 'n', that must be a node without any children, into a
305 * compressed node representing a set of nodes linked one after the other
306 * and having exactly one child each. The node can be a key or not: this
307 * property and the associated value if any will be preserved.
308 *
309 * The function also returns a child node, since the last node of the
310 * compressed chain cannot be part of the chain: it has zero children while
311 * we can only compress inner nodes with exactly one child each. */
312raxNode *raxCompressNode(raxNode *n, unsigned char *s, size_t len, raxNode **child) {
313 assert(n->size == 0 && n->iscompr == 0);
314 void *data = NULL; /* Initialized only to avoid warnings. */
315 size_t newsize;
316
317 debugf("Compress node: %.*s\n", (int)len,s);
318
319 /* Allocate the child to link to this node. */
320 *child = raxNewNode(0,0);
321 if (*child == NULL) return NULL;
322
323 /* Make space in the parent node. */
324 newsize = sizeof(raxNode)+len+sizeof(raxNode*);
325 if (n->iskey) {
326 data = raxGetData(n); /* To restore it later. */
327 if (!n->isnull) newsize += sizeof(void*);
328 }
329 raxNode *newn = rax_realloc(n,newsize);
330 if (newn == NULL) {
331 rax_free(*child);
332 return NULL;
333 }
334 n = newn;
335
336 n->iscompr = 1;
337 n->size = len;
338 memcpy(n->data,s,len);
339 if (n->iskey) raxSetData(n,data);
340 raxNode **childfield = raxNodeLastChildPtr(n);
341 memcpy(childfield,child,sizeof(*child));
342 return n;
343}

345/* Low level function that walks the tree looking for the string
346 * 's' of 'len' bytes. The function returns the number of characters
347 * of the key that was possible to process: if the returned integer
348 * is the same as 'len', then it means that the node corresponding to the
349 * string was found (however it may not be a key in case the node->iskey is
350 * zero or if simply we stopped in the middle of a compressed node, so that
351 * 'splitpos' is non zero).
352 *
353 * Otherwise if the returned integer is not the same as 'len', there was an
354 * early stop during the tree walk because of a character mismatch.
355 *
356 * The node where the search ended (because the full string was processed
357 * or because there was an early stop) is returned by reference as
358 * '*stopnode' if the passed pointer is not NULL. This node link in the
359 * parent's node is returned as '*plink' if not NULL. Finally, if the
360 * search stopped in a compressed node, '*splitpos' returns the index
361 * inside the compressed node where the search ended. This is useful to
362 * know where to split the node for insertion. */
363static inline size_t raxLowWalk(rax *rax, unsigned char *s, size_t len, raxNode **stopnode, raxNode ***plink, int *splitpos, raxStack *ts) {
364 raxNode *h = rax->head;
365 raxNode **parentlink = &rax->head;
366
367 size_t i = 0; /* Position in the string. */
368 size_t j = 0; /* Position in the node children (or bytes if compressed).*/
369 while(h->size && i < len) {
370 debugnode("Lookup current node",h);
371 unsigned char *v = h->data;
372
373 if (h->iscompr) {
374 for (j = 0; j < h->size && i < len; j++, i++) {
375 if (v[j] != s[i]) break;
376 }
377 if (j != h->size) break;
378 } else {
379 /* Even when h->size is large, linear scan provides good
380 * performances compared to other approaches that are in theory
381 * more sounding, like performing a binary search. */
382 for (j = 0; j < h->size; j++) {
383 if (v[j] == s[i]) break;
384 }
385 if (j == h->size) break;
386 i++;
387 }
388
389 if (ts) raxStackPush(ts,h); /* Save stack of parent nodes. */
390 raxNode **children = raxNodeFirstChildPtr(h);
391 if (h->iscompr) j = 0; /* Compressed node only child is at index 0. */
392 memcpy(&h,children+j,sizeof(h));
393 parentlink = children+j;
394 j = 0; /* If the new node is compressed and we do not
395 iterate again (since i == l) set the split
396 position to 0 to signal this node represents
397 the searched key. */
398 }
399 debugnode("Lookup stop node is",h);
400 if (stopnode) *stopnode = h;
401 if (plink) *plink = parentlink;
402 if (splitpos && h->iscompr) *splitpos = j;
403 return i;
404}

406/* Insert the element 's' of size 'len', setting as auxiliary data
407 * the pointer 'data'. If the element is already present, the associated
408 * data is updated, and 0 is returned, otherwise the element is inserted
409 * and 1 is returned. On out of memory the function returns 0 as well but
410 * sets errno to ENOMEM, otherwise errno will be set to 0. */
411int raxInsert(rax *rax, unsigned char *s, size_t len, void *data, void **old) {
412 size_t i;
413 int j = 0; /* Split position. If raxLowWalk() stops in a compressed
414 node, the index 'j' represents the char we stopped within the
415 compressed node, that is, the position where to split the
416 node for insertion. */
417 raxNode *h, **parentlink;
418
419 debugf("### Insert %.*s with value %p\n", (int)len, s, data);
420 i = raxLowWalk(rax,s,len,&h,&parentlink,&j,NULL);
421
422 /* If i == len we walked following the whole string. If we are not
423 * in the middle of a compressed node, the string is either already
424 * inserted or this middle node is currently not a key, but can represent
425 * our key. We have just to reallocate the node and make space for the
426 * data pointer. */
427 if (i == len && (!h->iscompr || j == 0 /* not in the middle if j is 0 */)) {
428 debugf("### Insert: node representing key exists\n");
429 if (!h->iskey || h->isnull) {
430 h = raxReallocForData(h,data);
431 if (h) memcpy(parentlink,&h,sizeof(h));
432 }
433 if (h == NULL) {
434 errno = ENOMEM;
435 return 0;
436 }
437 if (h->iskey) {
438 if (old) *old = raxGetData(h);
439 raxSetData(h,data);
440 errno = 0;
441 return 0; /* Element already exists. */
442 }
443 raxSetData(h,data);
444 rax->numele++;
445 return 1; /* Element inserted. */
446 }
447
448 /* If the node we stopped at is a compressed node, we need to
449 * split it before to continue.
450 *
451 * Splitting a compressed node have a few possibile cases.
452 * Imagine that the node 'h' we are currently at is a compressed
453 * node contaning the string "ANNIBALE" (it means that it represents
454 * nodes A -> N -> N -> I -> B -> A -> L -> E with the only child
455 * pointer of this node pointing at the 'E' node, because remember that
456 * we have characters at the edges of the graph, not inside the nodes
457 * themselves.
458 *
459 * In order to show a real case imagine our node to also point to
460 * another compressed node, that finally points at the node without
461 * children, representing 'O':
462 *
463 * "ANNIBALE" -> "SCO" -> []
464 *
465 * When inserting we may face the following cases. Note that all the cases
466 * require the insertion of a non compressed node with exactly two
467 * children, except for the last case which just requires splitting a
468 * compressed node.
469 *
470 * 1) Inserting "ANNIENTARE"
471 *
472 * |B| -> "ALE" -> "SCO" -> []
473 * "ANNI" -> |-|
474 * |E| -> (... continue algo ...) "NTARE" -> []
475 *
476 * 2) Inserting "ANNIBALI"
477 *
478 * |E| -> "SCO" -> []
479 * "ANNIBAL" -> |-|
480 * |I| -> (... continue algo ...) []
481 *
482 * 3) Inserting "AGO" (Like case 1, but set iscompr = 0 into original node)
483 *
484 * |N| -> "NIBALE" -> "SCO" -> []
485 * |A| -> |-|
486 * |G| -> (... continue algo ...) |O| -> []
487 *
488 * 4) Inserting "CIAO"
489 *
490 * |A| -> "NNIBALE" -> "SCO" -> []
491 * |-|
492 * |C| -> (... continue algo ...) "IAO" -> []
493 *
494 * 5) Inserting "ANNI"
495 *
496 * "ANNI" -> "BALE" -> "SCO" -> []
497 *
498 * The final algorithm for insertion covering all the above cases is as
499 * follows.
500 *
501 * ============================= ALGO 1 =============================
502 *
503 * For the above cases 1 to 4, that is, all cases where we stopped in
504 * the middle of a compressed node for a character mismatch, do:
505 *
506 * Let $SPLITPOS be the zero-based index at which, in the
507 * compressed node array of characters, we found the mismatching
508 * character. For example if the node contains "ANNIBALE" and we add
509 * "ANNIENTARE" the $SPLITPOS is 4, that is, the index at which the
510 * mismatching character is found.
511 *
512 * 1. Save the current compressed node $NEXT pointer (the pointer to the
513 * child element, that is always present in compressed nodes).
514 *
515 * 2. Create "split node" having as child the non common letter
516 * at the compressed node. The other non common letter (at the key)
517 * will be added later as we continue the normal insertion algorithm
518 * at step "6".
519 *
520 * 3a. IF $SPLITPOS == 0:
521 * Replace the old node with the split node, by copying the auxiliary
522 * data if any. Fix parent's reference. Free old node eventually
523 * (we still need its data for the next steps of the algorithm).
524 *
525 * 3b. IF $SPLITPOS != 0:
526 * Trim the compressed node (reallocating it as well) in order to
527 * contain $splitpos characters. Change chilid pointer in order to link
528 * to the split node. If new compressed node len is just 1, set
529 * iscompr to 0 (layout is the same). Fix parent's reference.
530 *
531 * 4a. IF the postfix len (the length of the remaining string of the
532 * original compressed node after the split character) is non zero,
533 * create a "postfix node". If the postfix node has just one character
534 * set iscompr to 0, otherwise iscompr to 1. Set the postfix node
535 * child pointer to $NEXT.
536 *
537 * 4b. IF the postfix len is zero, just use $NEXT as postfix pointer.
538 *
539 * 5. Set child[0] of split node to postfix node.
540 *
541 * 6. Set the split node as the current node, set current index at child[1]
542 * and continue insertion algorithm as usually.
543 *
544 * ============================= ALGO 2 =============================
545 *
546 * For case 5, that is, if we stopped in the middle of a compressed
547 * node but no mismatch was found, do:
548 *
549 * Let $SPLITPOS be the zero-based index at which, in the
550 * compressed node array of characters, we stopped iterating because
551 * there were no more keys character to match. So in the example of
552 * the node "ANNIBALE", addig the string "ANNI", the $SPLITPOS is 4.
553 *
554 * 1. Save the current compressed node $NEXT pointer (the pointer to the
555 * child element, that is always present in compressed nodes).
556 *
557 * 2. Create a "postfix node" containing all the characters from $SPLITPOS
558 * to the end. Use $NEXT as the postfix node child pointer.
559 * If the postfix node length is 1, set iscompr to 0.
560 * Set the node as a key with the associated value of the new
561 * inserted key.
562 *
563 * 3. Trim the current node to contain the first $SPLITPOS characters.
564 * As usually if the new node length is just 1, set iscompr to 0.
565 * Take the iskey / associated value as it was in the orignal node.
566 * Fix the parent's reference.
567 *
568 * 4. Set the postfix node as the only child pointer of the trimmed
569 * node created at step 1.
570 */
571
572 /* ------------------------- ALGORITHM 1 --------------------------- */
573 if (h->iscompr && i != len) {
574 debugf("ALGO 1: Stopped at compressed node %.*s (%p)\n",
575 h->size, h->data, (void*)h);
576 debugf("Still to insert: %.*s\n", (int)(len-i), s+i);
577 debugf("Splitting at %d: '%c'\n", j, ((char*)h->data)[j]);
578 debugf("Other (key) letter is '%c'\n", s[i]);
579
580 /* 1: Save next pointer. */
581 raxNode **childfield = raxNodeLastChildPtr(h);
582 raxNode *next;
583 memcpy(&next,childfield,sizeof(next));
584 debugf("Next is %p\n", (void*)next);
585 debugf("iskey %d\n", h->iskey);
586 if (h->iskey) {
587 debugf("key value is %p\n", raxGetData(h));
588 }
589
590 /* Set the length of the additional nodes we will need. */
591 size_t trimmedlen = j;
592 size_t postfixlen = h->size - j - 1;
593 int split_node_is_key = !trimmedlen && h->iskey && !h->isnull;
594 size_t nodesize;
595
596 /* 2: Create the split node. Also allocate the other nodes we'll need
597 * ASAP, so that it will be simpler to handle OOM. */
598 raxNode *splitnode = raxNewNode(1, split_node_is_key);
599 raxNode *trimmed = NULL;
600 raxNode *postfix = NULL;
601
602 if (trimmedlen) {
603 nodesize = sizeof(raxNode)+trimmedlen+sizeof(raxNode*);
604 if (h->iskey && !h->isnull) nodesize += sizeof(void*);
605 trimmed = rax_malloc(nodesize);
606 }
607
608 if (postfixlen) {
609 nodesize = sizeof(raxNode)+postfixlen+
610 sizeof(raxNode*);
611 postfix = rax_malloc(nodesize);
612 }
613
614 /* OOM? Abort now that the tree is untouched. */
615 if (splitnode == NULL ||
616 (trimmedlen && trimmed == NULL) ||
617 (postfixlen && postfix == NULL))
618 {
619 rax_free(splitnode);
620 rax_free(trimmed);
621 rax_free(postfix);
622 errno = ENOMEM;
623 return 0;
624 }
625 splitnode->data[0] = h->data[j];
626
627 if (j == 0) {
628 /* 3a: Replace the old node with the split node. */
629 if (h->iskey) {
630 void *ndata = raxGetData(h);
631 raxSetData(splitnode,ndata);
632 }
633 memcpy(parentlink,&splitnode,sizeof(splitnode));
634 } else {
635 /* 3b: Trim the compressed node. */
636 trimmed->size = j;
637 memcpy(trimmed->data,h->data,j);
638 trimmed->iscompr = j > 1 ? 1 : 0;
639 trimmed->iskey = h->iskey;
640 trimmed->isnull = h->isnull;
641 if (h->iskey && !h->isnull) {
642 void *ndata = raxGetData(h);
643 raxSetData(trimmed,ndata);
644 }
645 raxNode **cp = raxNodeLastChildPtr(trimmed);
646 memcpy(cp,&splitnode,sizeof(splitnode));
647 memcpy(parentlink,&trimmed,sizeof(trimmed));
648 parentlink = cp; /* Set parentlink to splitnode parent. */
649 rax->numnodes++;
650 }
651
652 /* 4: Create the postfix node: what remains of the original
653 * compressed node after the split. */
654 if (postfixlen) {
655 /* 4a: create a postfix node. */
656 postfix->iskey = 0;
657 postfix->isnull = 0;
658 postfix->size = postfixlen;
659 postfix->iscompr = postfixlen > 1;
660 memcpy(postfix->data,h->data+j+1,postfixlen);
661 raxNode **cp = raxNodeLastChildPtr(postfix);
662 memcpy(cp,&next,sizeof(next));
663 rax->numnodes++;
664 } else {
665 /* 4b: just use next as postfix node. */
666 postfix = next;
667 }
668
669 /* 5: Set splitnode first child as the postfix node. */
670 raxNode **splitchild = raxNodeLastChildPtr(splitnode);
671 memcpy(splitchild,&postfix,sizeof(postfix));
672
673 /* 6. Continue insertion: this will cause the splitnode to
674 * get a new child (the non common character at the currently
675 * inserted key). */
676 rax_free(h);
677 h = splitnode;
678 } else if (h->iscompr && i == len) {
679 /* ------------------------- ALGORITHM 2 --------------------------- */
680 debugf("ALGO 2: Stopped at compressed node %.*s (%p) j = %d\n",
681 h->size, h->data, (void*)h, j);
682
683 /* Allocate postfix & trimmed nodes ASAP to fail for OOM gracefully. */
684 size_t postfixlen = h->size - j;
685 size_t nodesize = sizeof(raxNode)+postfixlen+sizeof(raxNode*);
686 if (data != NULL) nodesize += sizeof(void*);
687 raxNode *postfix = rax_malloc(nodesize);
688
689 nodesize = sizeof(raxNode)+j+sizeof(raxNode*);
690 if (h->iskey && !h->isnull) nodesize += sizeof(void*);
691 raxNode *trimmed = rax_malloc(nodesize);
692
693 if (postfix == NULL || trimmed == NULL) {
694 rax_free(postfix);
695 rax_free(trimmed);
696 errno = ENOMEM;
697 return 0;
698 }
699
700 /* 1: Save next pointer. */
701 raxNode **childfield = raxNodeLastChildPtr(h);
702 raxNode *next;
703 memcpy(&next,childfield,sizeof(next));
704
705 /* 2: Create the postfix node. */
706 postfix->size = postfixlen;
707 postfix->iscompr = postfixlen > 1;
708 postfix->iskey = 1;
709 postfix->isnull = 0;
710 memcpy(postfix->data,h->data+j,postfixlen);
711 raxSetData(postfix,data);
712 raxNode **cp = raxNodeLastChildPtr(postfix);
713 memcpy(cp,&next,sizeof(next));
714 rax->numnodes++;
715
716 /* 3: Trim the compressed node. */
717 trimmed->size = j;
718 trimmed->iscompr = j > 1;
719 trimmed->iskey = 0;
720 trimmed->isnull = 0;
721 memcpy(trimmed->data,h->data,j);
722 memcpy(parentlink,&trimmed,sizeof(trimmed));
723 if (h->iskey) {
724 void *aux = raxGetData(h);
725 raxSetData(trimmed,aux);
726 }
727
728 /* Fix the trimmed node child pointer to point to
729 * the postfix node. */
730 cp = raxNodeLastChildPtr(trimmed);
731 memcpy(cp,&postfix,sizeof(postfix));
732
733 /* Finish! We don't need to contine with the insertion
734 * algorithm for ALGO 2. The key is already inserted. */
735 rax->numele++;
736 rax_free(h);
737 return 1; /* Key inserted. */
738 }
739
740 /* We walked the radix tree as far as we could, but still there are left
741 * chars in our string. We need to insert the missing nodes. */
742 while(i < len) {
743 raxNode *child;
744
745 /* If this node is going to have a single child, and there
746 * are other characters, so that that would result in a chain
747 * of single-childed nodes, turn it into a compressed node. */
748 if (h->size == 0 && len-i > 1) {
749 debugf("Inserting compressed node\n");
750 size_t comprsize = len-i;
751 if (comprsize > RAX_NODE_MAX_SIZE)
752 comprsize = RAX_NODE_MAX_SIZE;
753 raxNode *newh = raxCompressNode(h,s+i,comprsize,&child);
754 if (newh == NULL) goto oom;
755 h = newh;
756 memcpy(parentlink,&h,sizeof(h));
757 parentlink = raxNodeLastChildPtr(h);
758 i += comprsize;
759 } else {
760 debugf("Inserting normal node\n");
761 raxNode **new_parentlink;
762 raxNode *newh = raxAddChild(h,s[i],&child,&new_parentlink);
763 if (newh == NULL) goto oom;
764 h = newh;
765 memcpy(parentlink,&h,sizeof(h));
766 parentlink = new_parentlink;
767 i++;
768 }
769 rax->numnodes++;
770 h = child;
771 }
772 raxNode *newh = raxReallocForData(h,data);
773 if (newh == NULL) goto oom;
774 h = newh;
775 if (!h->iskey) rax->numele++;
776 raxSetData(h,data);
777 memcpy(parentlink,&h,sizeof(h));
778 return 1; /* Element inserted. */
779
780oom:
781 /* This code path handles out of memory after part of the sub-tree was
782 * already modified. Set the node as a key, and then remove it. However we
783 * do that only if the node is a terminal node, otherwise if the OOM
784 * happened reallocating a node in the middle, we don't need to free
785 * anything. */
786 if (h->size == 0) {
787 h->isnull = 1;
788 h->iskey = 1;
789 rax->numele++; /* Compensate the next remove. */
790 assert(raxRemove(rax,s,i,NULL) != 0);
791 }
792 errno = ENOMEM;
793 return 0;
794}

796/* Find a key in the rax, returns raxNotFound special void pointer value
797 * if the item was not found, otherwise the value associated with the
798 * item is returned. */
799void *raxFind(rax *rax, unsigned char *s, size_t len) {
800 raxNode *h;
801
802 debugf("### Lookup: %.*s\n", (int)len, s);
803 int splitpos = 0;
804 size_t i = raxLowWalk(rax,s,len,&h,NULL,&splitpos,NULL);
805 if (i != len || (h->iscompr && splitpos != 0) || !h->iskey)
806 return raxNotFound;
807 return raxGetData(h);
808}
809
810/* Return the memory address where the 'parent' node stores the specified
811 * 'child' pointer, so that the caller can update the pointer with another
812 * one if needed. The function assumes it will find a match, otherwise the
813 * operation is an undefined behavior (it will continue scanning the
814 * memory without any bound checking). */
815raxNode **raxFindParentLink(raxNode *parent, raxNode *child) {
816 raxNode **cp = raxNodeFirstChildPtr(parent);
817 raxNode *c;
818 while(1) {
819 memcpy(&c,cp,sizeof(c));
820 if (c == child) break;
821 cp++;
822 }
823 return cp;
824}
825
826/* Low level child removal from node. The new node pointer (after the child
827 * removal) is returned. Note that this function does not fix the pointer
828 * of the parent node in its parent, so this task is up to the caller.
829 * The function never fails for out of memory. */
830raxNode *raxRemoveChild(raxNode *parent, raxNode *child) {
831 debugnode("raxRemoveChild before", parent);
832 /* If parent is a compressed node (having a single child, as for definition
833 * of the data structure), the removal of the child consists into turning
834 * it into a normal node without children. */
835 if (parent->iscompr) {
836 void *data = NULL;
837 if (parent->iskey) data = raxGetData(parent);
838 parent->isnull = 0;
839 parent->iscompr = 0;
840 parent->size = 0;
841 if (parent->iskey) raxSetData(parent,data);
842 debugnode("raxRemoveChild after", parent);
843 return parent;
844 }
845
846 /* Otherwise we need to scan for the children pointer and memmove()
847 * accordingly.
848 *
849 * 1. To start we seek the first element in both the children
850 * pointers and edge bytes in the node. */
851 raxNode **cp = raxNodeFirstChildPtr(parent);
852 raxNode **c = cp;
853 unsigned char *e = parent->data;
854
855 /* 2. Search the child pointer to remove inside the array of children
856 * pointers. */
857 while(1) {
858 raxNode *aux;
859 memcpy(&aux,c,sizeof(aux));
860 if (aux == child) break;
861 c++;
862 e++;
863 }
864
865 /* 3. Remove the edge and the pointer by memmoving the remaining children
866 * pointer and edge bytes one position before. */
867 int taillen = parent->size - (e - parent->data) - 1;
868 debugf("raxRemoveChild tail len: %d\n", taillen);
869 memmove(e,e+1,taillen);
870
871 /* Since we have one data byte less, also child pointers start one byte
872 * before now. */
873 memmove(((char*)cp)-1,cp,(parent->size-taillen-1)*sizeof(raxNode**));
874
875 /* Move the remaining "tail" pointer at the right position as well. */
876 memmove(((char*)c)-1,c+1,taillen*sizeof(raxNode**)+parent->iskey*sizeof(void*));
877
878 /* 4. Update size. */
879 parent->size--;
880
881 /* realloc the node according to the theoretical memory usage, to free
882 * data if we are over-allocating right now. */
883 raxNode *newnode = rax_realloc(parent,raxNodeCurrentLength(parent));
884 if (newnode) {
885 debugnode("raxRemoveChild after", newnode);
886 }
887 /* Note: if rax_realloc() fails we just return the old address, which
888 * is valid. */
889 return newnode ? newnode : parent;
890}

892/* Remove the specified item. Returns 1 if the item was found and
893 * deleted, 0 otherwise. */
894int raxRemove(rax *rax, unsigned char *s, size_t len, void **old) {
895 raxNode *h;
896 raxStack ts;
897
898 debugf("### Delete: %.*s\n", (int)len, s);
899 raxStackInit(&ts);
900 int splitpos = 0;
901 size_t i = raxLowWalk(rax,s,len,&h,NULL,&splitpos,&ts);
902 if (i != len || (h->iscompr && splitpos != 0) || !h->iskey) {
903 raxStackFree(&ts);
904 return 0;
905 }
906 if (old) *old = raxGetData(h);
907 h->iskey = 0;
908 rax->numele--;
909
910 /* If this node has no children, the deletion needs to reclaim the
911 * no longer used nodes. This is an iterative process that needs to
912 * walk the three upward, deleting all the nodes with just one child
913 * that are not keys, until the head of the rax is reached or the first
914 * node with more than one child is found. */
915
916 int trycompress = 0; /* Will be set to 1 if we should try to optimize the
917 tree resulting from the deletion. */
918
919 if (h->size == 0) {
920 debugf("Key deleted in node without children. Cleanup needed.\n");
921 raxNode *child = NULL;
922 while(h != rax->head) {
923 child = h;
924 debugf("Freeing child %p [%.*s] key:%d\n", (void*)child,
925 (int)child->size, (char*)child->data, child->iskey);
926 rax_free(child);
927 rax->numnodes--;
928 h = raxStackPop(&ts);
929 /* If this node has more then one child, or actually holds
930 * a key, stop here. */
931 if (h->iskey || (!h->iscompr && h->size != 1)) break;
932 }
933 if (child) {
934 debugf("Unlinking child %p from parent %p\n",
935 (void*)child, (void*)h);
936 raxNode *new = raxRemoveChild(h,child);
937 if (new != h) {
938 raxNode *parent = raxStackPeek(&ts);
939 raxNode **parentlink;
940 if (parent == NULL) {
941 parentlink = &rax->head;
942 } else {
943 parentlink = raxFindParentLink(parent,h);
944 }
945 memcpy(parentlink,&new,sizeof(new));
946 }
947
948 /* If after the removal the node has just a single child
949 * and is not a key, we need to try to compress it. */
950 if (new->size == 1 && new->iskey == 0) {
951 trycompress = 1;
952 h = new;
953 }
954 }
955 } else if (h->size == 1) {
956 /* If the node had just one child, after the removal of the key
957 * further compression with adjacent nodes is pontentially possible. */
958 trycompress = 1;
959 }
960
961 /* Don't try node compression if our nodes pointers stack is not
962 * complete because of OOM while executing raxLowWalk() */
963 if (trycompress && ts.oom) trycompress = 0;
964
965 /* Recompression: if trycompress is true, 'h' points to a radix tree node
966 * that changed in a way that could allow to compress nodes in this
967 * sub-branch. Compressed nodes represent chains of nodes that are not
968 * keys and have a single child, so there are two deletion events that
969 * may alter the tree so that further compression is needed:
970 *
971 * 1) A node with a single child was a key and now no longer is a key.
972 * 2) A node with two children now has just one child.
973 *
974 * We try to navigate upward till there are other nodes that can be
975 * compressed, when we reach the upper node which is not a key and has
976 * a single child, we scan the chain of children to collect the
977 * compressable part of the tree, and replace the current node with the
978 * new one, fixing the child pointer to reference the first non
979 * compressable node.
980 *
981 * Example of case "1". A tree stores the keys "FOO" = 1 and
982 * "FOOBAR" = 2:
983 *
984 *
985 * "FOO" -> "BAR" -> [] (2)
986 * (1)
987 *
988 * After the removal of "FOO" the tree can be compressed as:
989 *
990 * "FOOBAR" -> [] (2)
991 *
992 *
993 * Example of case "2". A tree stores the keys "FOOBAR" = 1 and
994 * "FOOTER" = 2:
995 *
996 * |B| -> "AR" -> [] (1)
997 * "FOO" -> |-|
998 * |T| -> "ER" -> [] (2)
999 *
1000 * After the removal of "FOOTER" the resulting tree is:
1001 *
1002 * "FOO" -> |B| -> "AR" -> [] (1)
1003 *
1004 * That can be compressed into:
1005 *
1006 * "FOOBAR" -> [] (1)
1007 */
1008 if (trycompress) {
1009 debugf("After removing %.*s:\n", (int)len, s);
1010 debugnode("Compression may be needed",h);
1011 debugf("Seek start node\n");
1012
1013 /* Try to reach the upper node that is compressible.
1014 * At the end of the loop 'h' will point to the first node we
1015 * can try to compress and 'parent' to its parent. */
1016 raxNode *parent;
1017 while(1) {
1018 parent = raxStackPop(&ts);
1019 if (!parent || parent->iskey ||
1020 (!parent->iscompr && parent->size != 1)) break;
1021 h = parent;
1022 debugnode("Going up to",h);
1023 }
1024 raxNode *start = h; /* Compression starting node. */
1025
1026 /* Scan chain of nodes we can compress. */
1027 size_t comprsize = h->size;
1028 int nodes = 1;
1029 while(h->size != 0) {
1030 raxNode **cp = raxNodeLastChildPtr(h);
1031 memcpy(&h,cp,sizeof(h));
1032 if (h->iskey || (!h->iscompr && h->size != 1)) break;
1033 /* Stop here if going to the next node would result into
1034 * a compressed node larger than h->size can hold. */
1035 if (comprsize + h->size > RAX_NODE_MAX_SIZE) break;
1036 nodes++;
1037 comprsize += h->size;
1038 }
1039 if (nodes > 1) {
1040 /* If we can compress, create the new node and populate it. */
1041 size_t nodesize =
1042 sizeof(raxNode)+comprsize+sizeof(raxNode*);
1043 raxNode *new = rax_malloc(nodesize);
1044 /* An out of memory here just means we cannot optimize this
1045 * node, but the tree is left in a consistent state. */
1046 if (new == NULL) {
1047 raxStackFree(&ts);
1048 return 1;
1049 }
1050 new->iskey = 0;
1051 new->isnull = 0;
1052 new->iscompr = 1;
1053 new->size = comprsize;
1054 rax->numnodes++;
1055
1056 /* Scan again, this time to populate the new node content and
1057 * to fix the new node child pointer. At the same time we free
1058 * all the nodes that we'll no longer use. */
1059 comprsize = 0;
1060 h = start;
1061 while(h->size != 0) {
1062 memcpy(new->data+comprsize,h->data,h->size);
1063 comprsize += h->size;
1064 raxNode **cp = raxNodeLastChildPtr(h);
1065 raxNode *tofree = h;
1066 memcpy(&h,cp,sizeof(h));
1067 rax_free(tofree); rax->numnodes--;
1068 if (h->iskey || (!h->iscompr && h->size != 1)) break;
1069 }
1070 debugnode("New node",new);
1071
1072 /* Now 'h' points to the first node that we still need to use,
1073 * so our new node child pointer will point to it. */
1074 raxNode **cp = raxNodeLastChildPtr(new);
1075 memcpy(cp,&h,sizeof(h));
1076
1077 /* Fix parent link. */
1078 if (parent) {
1079 raxNode **parentlink = raxFindParentLink(parent,start);
1080 memcpy(parentlink,&new,sizeof(new));
1081 } else {
1082 rax->head = new;
1083 }
1084
1085 debugf("Compressed %d nodes, %d total bytes\n",
1086 nodes, (int)comprsize);
1087 }
1088 }
1089 raxStackFree(&ts);
1090 return 1;
1091}

1093/* This is the core of raxFree(): performs a depth-first scan of the
1094 * tree and releases all the nodes found. */
1095void raxRecursiveFree(rax *rax, raxNode *n) {
1096 debugnode("free traversing",n);
1097 int numchildren = n->iscompr ? 1 : n->size;
1098 raxNode **cp = raxNodeLastChildPtr(n);
1099 while(numchildren--) {
1100 raxNode *child;
1101 memcpy(&child,cp,sizeof(child));
1102 raxRecursiveFree(rax,child);
1103 cp--;
1104 }
1105 debugnode("free depth-first",n);
1106 rax_free(n);
1107 rax->numnodes--;
1108}

1110/* Free a whole radix tree. */
1111void raxFree(rax *rax) {
1112 raxRecursiveFree(rax,rax->head);
1113 assert(rax->numnodes == 0);
1114 rax_free(rax);
1115}

1118
1119/* Initialize a Rax iterator. This call should be performed a single time
1120 * to initialize the iterator, and must be followed by a raxSeek() call,
1121 * otherwise the raxPrev()/raxNext() functions will just return EOF. */
1122void raxStart(raxIterator *it, rax *rt) {
1123 it->flags = RAX_ITER_EOF; /* No crash if the iterator is not seeked. */
1124 it->rt = rt;
1125 it->key_len = 0;
1126 it->key = it->key_static_string;
1127 it->key_max = RAX_ITER_STATIC_LEN;
1128 it->data = NULL;
1129 raxStackInit(&it->stack);
1130}
1131
1132/* Append characters at the current key string of the iterator 'it'. This
1133 * is a low level function used to implement the iterator, not callable by
1134 * the user. Returns 0 on out of memory, otherwise 1 is returned. */
1135int raxIteratorAddChars(raxIterator *it, unsigned char *s, size_t len) {
1136 if (it->key_max < it->key_len+len) {
1137 unsigned char *old = (it->key == it->key_static_string) ? NULL :
1138 it->key;
1139 size_t new_max = (it->key_len+len)*2;
1140 it->key = rax_realloc(old,new_max);
1141 if (it->key == NULL) {
1142 it->key = (!old) ? it->key_static_string : old;
1143 errno = ENOMEM;
1144 return 0;
1145 }
1146 if (old == NULL) memcpy(it->key,it->key_static_string,it->key_len);
1147 it->key_max = new_max;
1148 }
1149 /* Use memmove since there could be an overlap between 's' and
1150 * it->key when we use the current key in order to re-seek. */
1151 memmove(it->key+it->key_len,s,len);
1152 it->key_len += len;
1153 return 1;
1154}
1155
1156/* Remove the specified number of chars from the right of the current
1157 * iterator key. */
1158void raxIteratorDelChars(raxIterator *it, size_t count) {
1159 it->key_len -= count;
1160}
1161
1162/* Do an iteration step towards the next element. At the end of the step the
1163 * iterator key will represent the (new) current key. If it is not possible
1164 * to step in the specified direction since there are no longer elements, the
1165 * iterator is flagged with RAX_ITER_EOF.
1166 *
1167 * If 'noup' is true the function starts directly scanning for the next
1168 * lexicographically smaller children, and the current node is already assumed
1169 * to be the parent of the last key node, so the first operation to go back to
1170 * the parent will be skipped. This option is used by raxSeek() when
1171 * implementing seeking a non existing element with the ">" or "<" options:
1172 * the starting node is not a key in that particular case, so we start the scan
1173 * from a node that does not represent the key set.
1174 *
1175 * The function returns 1 on success or 0 on out of memory. */
1176int raxIteratorNextStep(raxIterator *it, int noup) {
1177 if (it->flags & RAX_ITER_EOF) {
1178 return 0;
1179 } else if (it->flags & RAX_ITER_JUST_SEEKED) {
1180 it->flags &= ~RAX_ITER_JUST_SEEKED;
1181 return 1;
1182 }
1183
1184 /* Save key len, stack items and the node where we are currently
1185 * so that on iterator EOF we can restore the current key and state. */
1186 size_t orig_key_len = it->key_len;
1187 size_t orig_stack_items = it->stack.items;
1188 raxNode *orig_node = it->node;
1189
1190 /* Clear the EOF flag: it will be set again if the EOF condition
1191 * is still valid. */
1192 it->flags &= ~RAX_ITER_EOF;
1193
1194 while(1) {
1195 int children = it->node->iscompr ? 1 : it->node->size;
1196 if (!noup && children) {
1197 debugf("GO DEEPER\n");
1198 /* Seek the lexicographically smaller key in this subtree, which
1199 * is the first one found always going torwards the first child
1200 * of every successive node. */
1201 if (!raxStackPush(&it->stack,it->node)) return 0;
1202 raxNode **cp = raxNodeFirstChildPtr(it->node);
1203 if (!raxIteratorAddChars(it,it->node->data,
1204 it->node->iscompr ? it->node->size : 1)) return 0;
1205 memcpy(&it->node,cp,sizeof(it->node));
1206 /* For "next" step, stop every time we find a key along the
1207 * way, since the key is lexicograhically smaller compared to
1208 * what follows in the sub-children. */
1209 if (it->node->iskey) {
1210 it->data = raxGetData(it->node);
1211 return 1;
1212 }
1213 } else {
1214 /* If we finished exporing the previous sub-tree, switch to the
1215 * new one: go upper until a node is found where there are
1216 * children representing keys lexicographically greater than the
1217 * current key. */
1218 while(1) {
1219 int old_noup = noup;
1220
1221 /* Already on head? Can't go up, iteration finished. */
1222 if (!noup && it->node == it->rt->head) {
1223 it->flags |= RAX_ITER_EOF;
1224 it->stack.items = orig_stack_items;
1225 it->key_len = orig_key_len;
1226 it->node = orig_node;
1227 return 1;
1228 }
1229 /* If there are no children at the current node, try parent's
1230 * next child. */
1231 unsigned char prevchild = it->key[it->key_len-1];
1232 if (!noup) {
1233 it->node = raxStackPop(&it->stack);
1234 } else {
1235 noup = 0;
1236 }
1237 /* Adjust the current key to represent the node we are
1238 * at. */
1239 int todel = it->node->iscompr ? it->node->size : 1;
1240 raxIteratorDelChars(it,todel);
1241
1242 /* Try visiting the next child if there was at least one
1243 * additional child. */
1244 if (!it->node->iscompr && it->node->size > (old_noup ? 0 : 1)) {
1245 raxNode **cp = raxNodeFirstChildPtr(it->node);
1246 int i = 0;
1247 while (i < it->node->size) {
1248 debugf("SCAN NEXT %c\n", it->node->data[i]);
1249 if (it->node->data[i] > prevchild) break;
1250 i++;
1251 cp++;
1252 }
1253 if (i != it->node->size) {
1254 debugf("SCAN found a new node\n");
1255 raxIteratorAddChars(it,it->node->data+i,1);
1256 if (!raxStackPush(&it->stack,it->node)) return 0;
1257 memcpy(&it->node,cp,sizeof(it->node));
1258 if (it->node->iskey) {
1259 it->data = raxGetData(it->node);
1260 return 1;
1261 }
1262 break;
1263 }
1264 }
1265 }
1266 }
1267 }

1270/* Seek the grestest key in the subtree at the current node. Return 0 on
1271 * out of memory, otherwise 1. This is an helper function for different
1272 * iteration functions below. */
1273int raxSeekGreatest(raxIterator *it) {
1274 while(it->node->size) {
1275 if (it->node->iscompr) {
1276 if (!raxIteratorAddChars(it,it->node->data,
1277 it->node->size)) return 0;
1278 } else {
1279 if (!raxIteratorAddChars(it,it->node->data+it->node->size-1,1))
1280 return 0;
1281 }
1282 raxNode **cp = raxNodeLastChildPtr(it->node);
1283 if (!raxStackPush(&it->stack,it->node)) return 0;
1284 memcpy(&it->node,cp,sizeof(it->node));
1285 }
1286 return 1;
1287}

1289/* Like raxIteratorNextStep() but implements an iteration step moving
1290 * to the lexicographically previous element. The 'noup' option has a similar
1291 * effect to the one of raxIteratorPrevSte(). */
1292int raxIteratorPrevStep(raxIterator *it, int noup) {
1293 if (it->flags & RAX_ITER_EOF) {
1294 return 0;
1295 } else if (it->flags & RAX_ITER_JUST_SEEKED) {
1296 it->flags &= ~RAX_ITER_JUST_SEEKED;
1297 return 1;
1298 }
1299
1300 /* Save key len, stack items and the node where we are currently
1301 * so that on iterator EOF we can restore the current key and state. */
1302 size_t orig_key_len = it->key_len;
1303 size_t orig_stack_items = it->stack.items;
1304 raxNode *orig_node = it->node;
1305
1306 while(1) {
1307 int old_noup = noup;
1308
1309 /* Already on head? Can't go up, iteration finished. */
1310 if (!noup && it->node == it->rt->head) {
1311 it->flags |= RAX_ITER_EOF;
1312 it->stack.items = orig_stack_items;
1313 it->key_len = orig_key_len;
1314 it->node = orig_node;
1315 return 1;
1316 }
1317
1318 unsigned char prevchild = it->key[it->key_len-1];
1319 if (!noup) {
1320 it->node = raxStackPop(&it->stack);
1321 } else {
1322 noup = 0;
1323 }
1324
1325 /* Adjust the current key to represent the node we are
1326 * at. */
1327 int todel = it->node->iscompr ? it->node->size : 1;
1328 raxIteratorDelChars(it,todel);
1329
1330 /* Try visiting the prev child if there is at least one
1331 * child. */
1332 if (!it->node->iscompr && it->node->size > (old_noup ? 0 : 1)) {
1333 raxNode **cp = raxNodeLastChildPtr(it->node);
1334 int i = it->node->size-1;
1335 while (i >= 0) {
1336 debugf("SCAN PREV %c\n", it->node->data[i]);
1337 if (it->node->data[i] < prevchild) break;
1338 i--;
1339 cp--;
1340 }
1341 /* If we found a new subtree to explore in this node,
1342 * go deeper following all the last children in order to
1343 * find the key lexicographically greater. */
1344 if (i != -1) {
1345 debugf("SCAN found a new node\n");
1346 /* Enter the node we just found. */
1347 if (!raxIteratorAddChars(it,it->node->data+i,1)) return 0;
1348 if (!raxStackPush(&it->stack,it->node)) return 0;
1349 memcpy(&it->node,cp,sizeof(it->node));
1350 /* Seek sub-tree max. */
1351 if (!raxSeekGreatest(it)) return 0;
1352 }
1353 }
1354
1355 /* Return the key: this could be the key we found scanning a new
1356 * subtree, or if we did not find a new subtree to explore here,
1357 * before giving up with this node, check if it's a key itself. */
1358 if (it->node->iskey) {
1359 it->data = raxGetData(it->node);
1360 return 1;
1361 }
1362 }
1363}

1365/* Seek an iterator at the specified element.
1366 * Return 0 if the seek failed for syntax error or out of memory. Otherwise
1367 * 1 is returned. When 0 is returned for out of memory, errno is set to
1368 * the ENOMEM value. */
1369int raxSeek(raxIterator *it, const char *op, unsigned char *ele, size_t len) {
1370 int eq = 0, lt = 0, gt = 0, first = 0, last = 0;
1371
1372 it->stack.items = 0; /* Just resetting. Intialized by raxStart(). */
1373 it->flags |= RAX_ITER_JUST_SEEKED;
1374 it->flags &= ~RAX_ITER_EOF;
1375 it->key_len = 0;
1376 it->node = NULL;
1377
1378 /* Set flags according to the operator used to perform the seek. */
1379 if (op[0] == '>') {
1380 gt = 1;
1381 if (op[1] == '=') eq = 1;
1382 } else if (op[0] == '<') {
1383 lt = 1;
1384 if (op[1] == '=') eq = 1;
1385 } else if (op[0] == '=') {
1386 eq = 1;
1387 } else if (op[0] == '^') {
1388 first = 1;
1389 } else if (op[0] == '$') {
1390 last = 1;
1391 } else {
1392 errno = 0;
1393 return 0; /* Error. */
1394 }
1395
1396 /* If there are no elements, set the EOF condition immediately and
1397 * return. */
1398 if (it->rt->numele == 0) {
1399 it->flags |= RAX_ITER_EOF;
1400 return 1;
1401 }
1402
1403 if (first) {
1404 /* Seeking the first key greater or equal to the empty string
1405 * is equivalent to seeking the smaller key available. */
1406 return raxSeek(it,">=",NULL,0);
1407 }
1408
1409 if (last) {
1410 /* Find the greatest key taking always the last child till a
1411 * final node is found. */
1412 it->node = it->rt->head;
1413 if (!raxSeekGreatest(it)) return 0;
1414 assert(it->node->iskey);
1415 return 1;
1416 }
1417
1418 /* We need to seek the specified key. What we do here is to actually
1419 * perform a lookup, and later invoke the prev/next key code that
1420 * we already use for iteration. */
1421 int splitpos = 0;
1422 size_t i = raxLowWalk(it->rt,ele,len,&it->node,NULL,&splitpos,&it->stack);
1423
1424 /* Return OOM on incomplete stack info. */
1425 if (it->stack.oom) return 0;
1426
1427 if (eq && i == len && (!it->node->iscompr || splitpos == 0) &&
1428 it->node->iskey)
1429 {
1430 /* We found our node, since the key matches and we have an
1431 * "equal" condition. */
1432 if (!raxIteratorAddChars(it,ele,len)) return 0; /* OOM. */
1433 } else if (lt || gt) {
1434 /* Exact key not found or eq flag not set. We have to set as current
1435 * key the one represented by the node we stopped at, and perform
1436 * a next/prev operation to seek. To reconstruct the key at this node
1437 * we start from the parent and go to the current node, accumulating
1438 * the characters found along the way. */
1439 if (!raxStackPush(&it->stack,it->node)) return 0;
1440 for (size_t j = 1; j < it->stack.items; j++) {
1441 raxNode *parent = it->stack.stack[j-1];
1442 raxNode *child = it->stack.stack[j];
1443 if (parent->iscompr) {
1444 if (!raxIteratorAddChars(it,parent->data,parent->size))
1445 return 0;
1446 } else {
1447 raxNode **cp = raxNodeFirstChildPtr(parent);
1448 unsigned char *p = parent->data;
1449 while(1) {
1450 raxNode *aux;
1451 memcpy(&aux,cp,sizeof(aux));
1452 if (aux == child) break;
1453 cp++;
1454 p++;
1455 }
1456 if (!raxIteratorAddChars(it,p,1)) return 0;
1457 }
1458 }
1459 raxStackPop(&it->stack);
1460
1461 /* We need to set the iterator in the correct state to call next/prev
1462 * step in order to seek the desired element. */
1463 debugf("After initial seek: i=%d len=%d key=%.*s\n",
1464 (int)i, (int)len, (int)it->key_len, it->key);
1465 if (i != len && !it->node->iscompr) {
1466 /* If we stopped in the middle of a normal node because of a
1467 * mismatch, add the mismatching character to the current key
1468 * and call the iterator with the 'noup' flag so that it will try
1469 * to seek the next/prev child in the current node directly based
1470 * on the mismatching character. */
1471 if (!raxIteratorAddChars(it,ele+i,1)) return 0;
1472 debugf("Seek normal node on mismatch: %.*s\n",
1473 (int)it->key_len, (char*)it->key);
1474
1475 it->flags &= ~RAX_ITER_JUST_SEEKED;
1476 if (lt && !raxIteratorPrevStep(it,1)) return 0;
1477 if (gt && !raxIteratorNextStep(it,1)) return 0;
1478 it->flags |= RAX_ITER_JUST_SEEKED; /* Ignore next call. */
1479 } else if (i != len && it->node->iscompr) {
1480 debugf("Compressed mismatch: %.*s\n",
1481 (int)it->key_len, (char*)it->key);
1482 /* In case of a mismatch within a compressed node. */
1483 int nodechar = it->node->data[splitpos];
1484 int keychar = ele[i];
1485 it->flags &= ~RAX_ITER_JUST_SEEKED;
1486 if (gt) {
1487 /* If the key the compressed node represents is greater
1488 * than our seek element, continue forward, otherwise set the
1489 * state in order to go back to the next sub-tree. */
1490 if (nodechar > keychar) {
1491 if (!raxIteratorNextStep(it,0)) return 0;
1492 } else {
1493 if (!raxIteratorAddChars(it,it->node->data,it->node->size))
1494 return 0;
1495 if (!raxIteratorNextStep(it,1)) return 0;
1496 }
1497 }
1498 if (lt) {
1499 /* If the key the compressed node represents is smaller
1500 * than our seek element, seek the greater key in this
1501 * subtree, otherwise set the state in order to go back to
1502 * the previous sub-tree. */
1503 if (nodechar < keychar) {
1504 if (!raxSeekGreatest(it)) return 0;
1505 } else {
1506 if (!raxIteratorAddChars(it,it->node->data,it->node->size))
1507 return 0;
1508 if (!raxIteratorPrevStep(it,1)) return 0;
1509 }
1510 }
1511 it->flags |= RAX_ITER_JUST_SEEKED; /* Ignore next call. */
1512 } else {
1513 debugf("No mismatch: %.*s\n",
1514 (int)it->key_len, (char*)it->key);
1515 /* If there was no mismatch we are into a node representing the
1516 * key, (but which is not a key or the seek operator does not
1517 * include 'eq'), or we stopped in the middle of a compressed node
1518 * after processing all the key. Cotinue iterating as this was
1519 * a legitimate key we stopped at. */
1520 it->flags &= ~RAX_ITER_JUST_SEEKED;
1521 if (gt && !raxIteratorNextStep(it,0)) return 0;
1522 if (lt && !raxIteratorPrevStep(it,0)) return 0;
1523 it->flags |= RAX_ITER_JUST_SEEKED; /* Ignore next call. */
1524 }
1525 } else {
1526 /* If we are here just eq was set but no match was found. */
1527 it->flags |= RAX_ITER_EOF;
1528 return 1;
1529 }
1530 return 1;
1531}
1532
1533/* Go to the next element in the scope of the iterator 'it'.
1534 * If EOF (or out of memory) is reached, 0 is returned, otherwise 1 is
1535 * returned. In case 0 is returned because of OOM, errno is set to ENOMEM. */
1536int raxNext(raxIterator *it) {
1537 if (!raxIteratorNextStep(it,0)) {
1538 errno = ENOMEM;
1539 return 0;
1540 }
1541 if (it->flags & RAX_ITER_EOF) {
1542 errno = 0;
1543 return 0;
1544 }
1545 return 1;
1546}
1547
1548/* Go to the previous element in the scope of the iterator 'it'.
1549 * If EOF (or out of memory) is reached, 0 is returned, otherwise 1 is
1550 * returned. In case 0 is returned because of OOM, errno is set to ENOMEM. */
1551int raxPrev(raxIterator *it) {
1552 if (!raxIteratorPrevStep(it,0)) {
1553 errno = ENOMEM;
1554 return 0;
1555 }
1556 if (it->flags & RAX_ITER_EOF) {
1557 errno = 0;
1558 return 0;
1559 }
1560 return 1;
1561}

1563/* Perform a random walk starting in the current position of the iterator.
1564 * Return 0 if the tree is empty or on out of memory. Otherwise 1 is returned
1565 * and the iterator is set to the node reached after doing a random walk
1566 * of 'steps' steps. If the 'steps' argument is 0, the random walk is performed
1567 * using a random number of steps between 1 and two times the logarithm of
1568 * the number of elements.
1569 *
1570 * NOTE: if you use this function to generate random elements from the radix
1571 * tree, expect a disappointing distribution. A random walk produces good
1572 * random elements if the tree is not sparse, however in the case of a radix
1573 * tree certain keys will be reported much more often than others. At least
1574 * this function should be able to expore every possible element eventually. */
1575int raxRandomWalk(raxIterator *it, size_t steps) {
1576 if (it->rt->numele == 0) {
1577 it->flags |= RAX_ITER_EOF;
1578 return 0;
1579 }
1580
1581 if (steps == 0) {
1582 size_t fle = floor(log(it->rt->numele));
1583 fle *= 2;
1584 steps = 1 + rand() % fle;
1585 }
1586
1587 raxNode *n = it->node;
1588 while(steps > 0 || !n->iskey) {
1589 int numchildren = n->iscompr ? 1 : n->size;
1590 int r = rand() % (numchildren+(n != it->rt->head));
1591
1592 if (r == numchildren) {
1593 /* Go up to parent. */
1594 n = raxStackPop(&it->stack);
1595 int todel = n->iscompr ? n->size : 1;
1596 raxIteratorDelChars(it,todel);
1597 } else {
1598 /* Select a random child. */
1599 if (n->iscompr) {
1600 if (!raxIteratorAddChars(it,n->data,n->size)) return 0;
1601 } else {
1602 if (!raxIteratorAddChars(it,n->data+r,1)) return 0;
1603 }
1604 raxNode **cp = raxNodeFirstChildPtr(n)+r;
1605 if (!raxStackPush(&it->stack,n)) return 0;
1606 memcpy(&n,cp,sizeof(n));
1607 }
1608 if (n->iskey) steps--;
1609 }
1610 it->node = n;
1611 return 1;
1612}

1614/* Compare the key currently pointed by the iterator to the specified
1615 * key according to the specified operator. Returns 1 if the comparison is
1616 * true, otherwise 0 is returned. */
1617int raxCompare(raxIterator *iter, const char *op, unsigned char *key, size_t key_len) {
1618 int eq = 0, lt = 0, gt = 0;
1619
1620 if (op[0] == '=' || op[1] == '=') eq = 1;
1621 if (op[1] == '>') gt = 1;
1622 else if (op[1] == '<') lt = 1;
1623 else if (op[1] != '=') return 0; /* Syntax error. */
1624
1625 size_t minlen = key_len < iter->key_len ? key_len : iter->key_len;
1626 int cmp = memcmp(iter->key,key,minlen);
1627
1628 /* Handle == */
1629 if (lt == 0 && gt == 0) return cmp == 0 && key_len == iter->key_len;
1630
1631 /* Handle >, >=, <, <= */
1632 if (cmp == 0) {
1633 /* Same prefix: longer wins. */
1634 if (eq && key_len == iter->key_len) return 1;
1635 else if (lt) return iter->key_len < key_len;
1636 else if (gt) return iter->key_len > key_len;
1637 } if (cmp > 0) {
1638 return gt ? 1 : 0;
1639 } else /* (cmp < 0) */ {
1640 return lt ? 1 : 0;
1641 }
1642}

1652/* This function is mostly used for debugging and learning purposes.
1653 * It shows an ASCII representation of a tree on standard output, outling
1654 * all the nodes and the contained keys.
1655 *
1656 * The representation is as follow:
1657 *
1658 * "foobar" (compressed node)
1659 * [abc] (normal node with three children)
1660 * [abc]=0x12345678 (node is a key, pointing to value 0x12345678)
1661 * [] (a normal empty node)
1662 *
1663 * Children are represented in new idented lines, each children prefixed by
1664 * the "`-(x)" string, where "x" is the edge byte.
1665 *
1666 * [abc]
1667 * `-(a) "ladin"
1668 * `-(b) [kj]
1669 * `-(c) []
1670 *
1671 * However when a node has a single child the following representation
1672 * is used instead:
1673 *
1674 * [abc] -> "ladin" -> []
1675 */
1676
1677/* The actual implementation of raxShow(). */
1678void raxRecursiveShow(int level, int lpad, raxNode *n) {
1679 char s = n->iscompr ? '"' : '[';
1680 char e = n->iscompr ? '"' : ']';
1681
1682 int numchars = printf("%c%.*s%c", s, n->size, n->data, e);
1683 if (n->iskey) {
1684 numchars += printf("=%p",raxGetData(n));
1685 }
1686
1687 int numchildren = n->iscompr ? 1 : n->size;
1688 /* Note that 7 and 4 magic constants are the string length
1689 * of " `-(x) " and " -> " respectively. */
1690 if (level) {
1691 lpad += (numchildren > 1) ? 7 : 4;
1692 if (numchildren == 1) lpad += numchars;
1693 }
1694 raxNode **cp = raxNodeFirstChildPtr(n);
1695 for (int i = 0; i < numchildren; i++) {
1696 char *branch = " `-(%c) ";
1697 if (numchildren > 1) {
1698 printf("\n");
1699 for (int j = 0; j < lpad; j++) putchar(' ');
1700 printf(branch,n->data[i]);
1701 } else {
1702 printf(" -> ");
1703 }
1704 raxNode *child;
1705 memcpy(&child,cp,sizeof(child));
1706 raxRecursiveShow(level+1,lpad,child);
1707 cp++;
1708 }
1709}
