35/*-----------------------------------------------------------------------------
36 * Incremental collection of expired keys.
37 *
38 * When keys are accessed they are expired on-access. However we need a
39 * mechanism in order to ensure keys are eventually removed when expired even
40 * if no access is performed on them.
41 *----------------------------------------------------------------------------*/
42
43/* Helper function for the activeExpireCycle() function.
44 * This function will try to expire the key that is stored in the hash table
45 * entry 'de' of the 'expires' hash table of a Redis database.
46 *
47 * If the key is found to be expired, it is removed from the database and
48 * 1 is returned. Otherwise no operation is performed and 0 is returned.
49 *
50 * When a key is expired, server.stat_expiredkeys is incremented.
51 *
52 * The parameter 'now' is the current time in milliseconds as is passed
53 * to the function to avoid too many gettimeofday() syscalls. */
54int activeExpireCycleTryExpire(redisDb *db, dictEntry *de, long long now) {
55 long long t = dictGetSignedIntegerVal(de);
56 if (now > t) {
57 sds key = dictGetKey(de);
58 robj *keyobj = createStringObject(key,sdslen(key));
59
60 propagateExpire(db,keyobj,server.lazyfree_lazy_expire);
61 if (server.lazyfree_lazy_expire)
62 dbAsyncDelete(db,keyobj);
63 else
64 dbSyncDelete(db,keyobj);
65 notifyKeyspaceEvent(NOTIFY_EXPIRED,
66 "expired",keyobj,db->id);
67 decrRefCount(keyobj);
68 server.stat_expiredkeys++;
69 return 1;
70 } else {
71 return 0;
72 }
73}

75/* Try to expire a few timed out keys. The algorithm used is adaptive and
76 * will use few CPU cycles if there are few expiring keys, otherwise
77 * it will get more aggressive to avoid that too much memory is used by
78 * keys that can be removed from the keyspace.
79 *
80 * No more than CRON_DBS_PER_CALL databases are tested at every
81 * iteration.
82 *
83 * This kind of call is used when Redis detects that timelimit_exit is
84 * true, so there is more work to do, and we do it more incrementally from
85 * the beforeSleep() function of the event loop.
86 *
87 * Expire cycle type:
88 *
89 * If type is ACTIVE_EXPIRE_CYCLE_FAST the function will try to run a
90 * "fast" expire cycle that takes no longer than EXPIRE_FAST_CYCLE_DURATION
91 * microseconds, and is not repeated again before the same amount of time.
92 *
93 * If type is ACTIVE_EXPIRE_CYCLE_SLOW, that normal expire cycle is
94 * executed, where the time limit is a percentage of the REDIS_HZ period
95 * as specified by the ACTIVE_EXPIRE_CYCLE_SLOW_TIME_PERC define. */
96
97void activeExpireCycle(int type) {
98 /* This function has some global state in order to continue the work
99 * incrementally across calls. */
100 static unsigned int current_db = 0; /* Last DB tested. */
101 static int timelimit_exit = 0; /* Time limit hit in previous call? */
102 static long long last_fast_cycle = 0; /* When last fast cycle ran. */
103
104 int j, iteration = 0;
105 int dbs_per_call = CRON_DBS_PER_CALL;
106 long long start = ustime(), timelimit;
107
108 /* When clients are paused the dataset should be static not just from the
109 * POV of clients not being able to write, but also from the POV of
110 * expires and evictions of keys not being performed. */
111 if (clientsArePaused()) return;
112
113 if (type == ACTIVE_EXPIRE_CYCLE_FAST) {
114 /* Don't start a fast cycle if the previous cycle did not exited
115 * for time limt. Also don't repeat a fast cycle for the same period
116 * as the fast cycle total duration itself. */
117 if (!timelimit_exit) return;
118 if (start < last_fast_cycle + ACTIVE_EXPIRE_CYCLE_FAST_DURATION*2) return;
119 last_fast_cycle = start;
120 }
121
122 /* We usually should test CRON_DBS_PER_CALL per iteration, with
123 * two exceptions:
124 *
125 * 1) Don't test more DBs than we have.
126 * 2) If last time we hit the time limit, we want to scan all DBs
127 * in this iteration, as there is work to do in some DB and we don't want
128 * expired keys to use memory for too much time. */
129 if (dbs_per_call > server.dbnum || timelimit_exit)
130 dbs_per_call = server.dbnum;
131
132 /* We can use at max ACTIVE_EXPIRE_CYCLE_SLOW_TIME_PERC percentage of CPU time
133 * per iteration. Since this function gets called with a frequency of
134 * server.hz times per second, the following is the max amount of
135 * microseconds we can spend in this function. */
136 timelimit = 1000000*ACTIVE_EXPIRE_CYCLE_SLOW_TIME_PERC/server.hz/100;
137 timelimit_exit = 0;
138 if (timelimit <= 0) timelimit = 1;
139
140 if (type == ACTIVE_EXPIRE_CYCLE_FAST)
141 timelimit = ACTIVE_EXPIRE_CYCLE_FAST_DURATION; /* in microseconds. */
142
143 for (j = 0; j < dbs_per_call; j++) {
144 int expired;
145 redisDb *db = server.db+(current_db % server.dbnum);
146
147 /* Increment the DB now so we are sure if we run out of time
148 * in the current DB we'll restart from the next. This allows to
149 * distribute the time evenly across DBs. */
150 current_db++;
151
152 /* Continue to expire if at the end of the cycle more than 25%
153 * of the keys were expired. */
154 do {
155 unsigned long num, slots;
156 long long now, ttl_sum;
157 int ttl_samples;
158
159 /* If there is nothing to expire try next DB ASAP. */
160 if ((num = dictSize(db->expires)) == 0) {
161 db->avg_ttl = 0;
162 break;
163 }
164 slots = dictSlots(db->expires);
165 now = mstime();
166
167 /* When there are less than 1% filled slots getting random
168 * keys is expensive, so stop here waiting for better times...
169 * The dictionary will be resized asap. */
170 if (num && slots > DICT_HT_INITIAL_SIZE &&
171 (num*100/slots < 1)) break;
172
173 /* The main collection cycle. Sample random keys among keys
174 * with an expire set, checking for expired ones. */
175 expired = 0;
176 ttl_sum = 0;
177 ttl_samples = 0;
178
179 if (num > ACTIVE_EXPIRE_CYCLE_LOOKUPS_PER_LOOP)
180 num = ACTIVE_EXPIRE_CYCLE_LOOKUPS_PER_LOOP;
181
182 while (num--) {
183 dictEntry *de;
184 long long ttl;
185
186 if ((de = dictGetRandomKey(db->expires)) == NULL) break;
187 ttl = dictGetSignedIntegerVal(de)-now;
188 if (activeExpireCycleTryExpire(db,de,now)) expired++;
189 if (ttl > 0) {
190 /* We want the average TTL of keys yet not expired. */
191 ttl_sum += ttl;
192 ttl_samples++;
193 }
194 }
195
196 /* Update the average TTL stats for this database. */
197 if (ttl_samples) {
198 long long avg_ttl = ttl_sum/ttl_samples;
199
200 /* Do a simple running average with a few samples.
201 * We just use the current estimate with a weight of 2%
202 * and the previous estimate with a weight of 98%. */
203 if (db->avg_ttl == 0) db->avg_ttl = avg_ttl;
204 db->avg_ttl = (db->avg_ttl/50)*49 + (avg_ttl/50);
205 }
206
207 /* We can't block forever here even if there are many keys to
208 * expire. So after a given amount of milliseconds return to the
209 * caller waiting for the other active expire cycle. */
210 iteration++;
211 if ((iteration & 0xf) == 0) { /* check once every 16 iterations. */
212 long long elapsed = ustime()-start;
213
214 latencyAddSampleIfNeeded("expire-cycle",elapsed/1000);
215 if (elapsed > timelimit) timelimit_exit = 1;
216 }
217 if (timelimit_exit) return;
218 /* We don't repeat the cycle if there are less than 25% of keys
219 * found expired in the current DB. */
220 } while (expired > ACTIVE_EXPIRE_CYCLE_LOOKUPS_PER_LOOP/4);
221 }
222}

224/*-----------------------------------------------------------------------------
225 * Expires of keys created in writable slaves
226 *
227 * Normally slaves do not process expires: they wait the masters to synthesize
228 * DEL operations in order to retain consistency. However writable slaves are
229 * an exception: if a key is created in the slave and an expire is assigned
230 * to it, we need a way to expire such a key, since the master does not know
231 * anything about such a key.
232 *
233 * In order to do so, we track keys created in the slave side with an expire
234 * set, and call the expireSlaveKeys() function from time to time in order to
235 * reclaim the keys if they already expired.
236 *
237 * Note that the use case we are trying to cover here, is a popular one where
238 * slaves are put in writable mode in order to compute slow operations in
239 * the slave side that are mostly useful to actually read data in a more
240 * processed way. Think at sets intersections in a tmp key, with an expire so
241 * that it is also used as a cache to avoid intersecting every time.
242 *
243 * This implementation is currently not perfect but a lot better than leaking
244 * the keys as implemented in 3.2.
245 *----------------------------------------------------------------------------*/
246
247/* The dictionary where we remember key names and database ID of keys we may
248 * want to expire from the slave. Since this function is not often used we
249 * don't even care to initialize the database at startup. We'll do it once
250 * the feature is used the first time, that is, when rememberSlaveKeyWithExpire()
251 * is called.
252 *
253 * The dictionary has an SDS string representing the key as the hash table
254 * key, while the value is a 64 bit unsigned integer with the bits corresponding
255 * to the DB where the keys may exist set to 1. Currently the keys created
256 * with a DB id > 63 are not expired, but a trivial fix is to set the bitmap
257 * to the max 64 bit unsigned value when we know there is a key with a DB
258 * ID greater than 63, and check all the configured DBs in such a case. */
259dict *slaveKeysWithExpire = NULL;
260
261/* Check the set of keys created by the master with an expire set in order to
262 * check if they should be evicted. */
263void expireSlaveKeys(void) {
264 if (slaveKeysWithExpire == NULL ||
265 dictSize(slaveKeysWithExpire) == 0) return;
266
267 int cycles = 0, noexpire = 0;
268 mstime_t start = mstime();
269 while(1) {
270 dictEntry *de = dictGetRandomKey(slaveKeysWithExpire);
271 sds keyname = dictGetKey(de);
272 uint64_t dbids = dictGetUnsignedIntegerVal(de);
273 uint64_t new_dbids = 0;
274
275 /* Check the key against every database corresponding to the
276 * bits set in the value bitmap. */
277 int dbid = 0;
278 while(dbids && dbid < server.dbnum) {
279 if ((dbids & 1) != 0) {
280 redisDb *db = server.db+dbid;
281 dictEntry *expire = dictFind(db->expires,keyname);
282 int expired = 0;
283
284 if (expire &&
285 activeExpireCycleTryExpire(server.db+dbid,expire,start))
286 {
287 expired = 1;
288 }
289
290 /* If the key was not expired in this DB, we need to set the
291 * corresponding bit in the new bitmap we set as value.
292 * At the end of the loop if the bitmap is zero, it means we
293 * no longer need to keep track of this key. */
294 if (expire && !expired) {
295 noexpire++;
296 new_dbids |= (uint64_t)1 << dbid;
297 }
298 }
299 dbid++;
300 dbids >>= 1;
301 }
302
303 /* Set the new bitmap as value of the key, in the dictionary
304 * of keys with an expire set directly in the writable slave. Otherwise
305 * if the bitmap is zero, we no longer need to keep track of it. */
306 if (new_dbids)
307 dictSetUnsignedIntegerVal(de,new_dbids);
308 else
309 dictDelete(slaveKeysWithExpire,keyname);
310
311 /* Stop conditions: found 3 keys we cna't expire in a row or
312 * time limit was reached. */
313 cycles++;
314 if (noexpire > 3) break;
315 if ((cycles % 64) == 0 && mstime()-start > 1) break;
316 if (dictSize(slaveKeysWithExpire) == 0) break;
317 }
318}

320/* Track keys that received an EXPIRE or similar command in the context
321 * of a writable slave. */
322void rememberSlaveKeyWithExpire(redisDb *db, robj *key) {
323 if (slaveKeysWithExpire == NULL) {
324 static dictType dt = {
325 dictSdsHash, /* hash function */
326 NULL, /* key dup */
327 NULL, /* val dup */
328 dictSdsKeyCompare, /* key compare */
329 dictSdsDestructor, /* key destructor */
330 NULL /* val destructor */
331 };
332 slaveKeysWithExpire = dictCreate(&dt,NULL);
333 }
334 if (db->id > 63) return;
335
336 dictEntry *de = dictAddOrFind(slaveKeysWithExpire,key->ptr);
337 /* If the entry was just created, set it to a copy of the SDS string
338 * representing the key: we don't want to need to take those keys
339 * in sync with the main DB. The keys will be removed by expireSlaveKeys()
340 * as it scans to find keys to remove. */
341 if (de->key == key->ptr) {
342 de->key = sdsdup(key->ptr);
343 dictSetUnsignedIntegerVal(de,0);
344 }
345
346 uint64_t dbids = dictGetUnsignedIntegerVal(de);
347 dbids |= (uint64_t)1 << db->id;
348 dictSetUnsignedIntegerVal(de,dbids);
349}
350
351/* Return the number of keys we are tracking. */
352size_t getSlaveKeyWithExpireCount(void) {
353 if (slaveKeysWithExpire == NULL) return 0;
354 return dictSize(slaveKeysWithExpire);
355}
356
357/* Remove the keys in the hash table. We need to do that when data is
358 * flushed from the server. We may receive new keys from the master with
359 * the same name/db and it is no longer a good idea to expire them.
360 *
361 * Note: technically we should handle the case of a single DB being flushed
362 * but it is not worth it since anyway race conditions using the same set
363 * of key names in a wriatable slave and in its master will lead to
364 * inconsistencies. This is just a best-effort thing we do. */
365void flushSlaveKeysWithExpireList(void) {
366 if (slaveKeysWithExpire) {
367 dictRelease(slaveKeysWithExpire);
368 slaveKeysWithExpire = NULL;
369 }
370}

376/* This is the generic command implementation for EXPIRE, PEXPIRE, EXPIREAT
377 * and PEXPIREAT. Because the commad second argument may be relative or absolute
378 * the "basetime" argument is used to signal what the base time is (either 0
379 * for *AT variants of the command, or the current time for relative expires).
380 *
381 * unit is either UNIT_SECONDS or UNIT_MILLISECONDS, and is only used for
382 * the argv[2] parameter. The basetime is always specified in milliseconds. */
383void expireGenericCommand(client *c, long long basetime, int unit) {
384 robj *key = c->argv[1], *param = c->argv[2];
385 long long when; /* unix time in milliseconds when the key will expire. */
386
387 if (getLongLongFromObjectOrReply(c, param, &when, NULL) != C_OK)
388 return;
389
390 if (unit == UNIT_SECONDS) when *= 1000;
391 when += basetime;
392
393 /* No key, return zero. */
394 if (lookupKeyWrite(c->db,key) == NULL) {
395 addReply(c,shared.czero);
396 return;
397 }
398
399 /* EXPIRE with negative TTL, or EXPIREAT with a timestamp into the past
400 * should never be executed as a DEL when load the AOF or in the context
401 * of a slave instance.
402 *
403 * Instead we take the other branch of the IF statement setting an expire
404 * (possibly in the past) and wait for an explicit DEL from the master. */
405 if (when <= mstime() && !server.loading && !server.masterhost) {
406 robj *aux;
407
408 int deleted = server.lazyfree_lazy_expire ? dbAsyncDelete(c->db,key) :
409 dbSyncDelete(c->db,key);
410 serverAssertWithInfo(c,key,deleted);
411 server.dirty++;
412
413 /* Replicate/AOF this as an explicit DEL or UNLINK. */
414 aux = server.lazyfree_lazy_expire ? shared.unlink : shared.del;
415 rewriteClientCommandVector(c,2,aux,key);
416 signalModifiedKey(c->db,key);
417 notifyKeyspaceEvent(NOTIFY_GENERIC,"del",key,c->db->id);
418 addReply(c, shared.cone);
419 return;
420 } else {
421 setExpire(c,c->db,key,when);
422 addReply(c,shared.cone);
423 signalModifiedKey(c->db,key);
424 notifyKeyspaceEvent(NOTIFY_GENERIC,"expire",key,c->db->id);
425 server.dirty++;
426 return;
427 }
428}

450/* Implements TTL and PTTL */
451void ttlGenericCommand(client *c, int output_ms) {
452 long long expire, ttl = -1;
453
454 /* If the key does not exist at all, return -2 */
455 if (lookupKeyReadWithFlags(c->db,c->argv[1],LOOKUP_NOTOUCH) == NULL) {
456 addReplyLongLong(c,-2);
457 return;
458 }
459 /* The key exists. Return -1 if it has no expire, or the actual
460 * TTL value otherwise. */
461 expire = getExpire(c->db,c->argv[1]);
462 if (expire != -1) {
463 ttl = expire-mstime();
464 if (ttl < 0) ttl = 0;
465 }
466 if (ttl == -1) {
467 addReplyLongLong(c,-1);
468 } else {
469 addReplyLongLong(c,output_ms ? ttl : ((ttl+500)/1000));
470 }
471}

473/* TTL key */
474void ttlCommand(client *c) {
475 ttlGenericCommand(c, 0);
476}

478/* PTTL key */
479void pttlCommand(client *c) {
480 ttlGenericCommand(c, 1);

483/* PERSIST key */
484void persistCommand(client *c) {
485 if (lookupKeyWrite(c->db,c->argv[1])) {
486 if (removeExpire(c->db,c->argv[1])) {
487 addReply(c,shared.cone);
488 server.dirty++;
489 } else {
490 addReply(c,shared.czero);
491 }
492 } else {
493 addReply(c,shared.czero);
494 }
495}

497/* TOUCH key1 [key2 key3 ... keyN] */
498void touchCommand(client *c) {
499 int touched = 0;
500 for (int j = 1; j < c->argc; j++)
501 if (lookupKeyRead(c->db,c->argv[j]) != NULL) touched++;
502 addReplyLongLong(c,touched);
503}
