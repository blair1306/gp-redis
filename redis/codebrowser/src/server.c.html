75/* Our command table.
76 *
77 * Every entry is composed of the following fields:
78 *
79 * name: a string representing the command name.
80 * function: pointer to the C function implementing the command.
81 * arity: number of arguments, it is possible to use -N to say >= N
82 * sflags: command flags as string. See below for a table of flags.
83 * flags: flags as bitmask. Computed by Redis using the 'sflags' field.
84 * get_keys_proc: an optional function to get key arguments from a command.
85 * This is only used when the following three fields are not
86 * enough to specify what arguments are keys.
87 * first_key_index: first argument that is a key
88 * last_key_index: last argument that is a key
89 * key_step: step to get all the keys from first to last argument. For instance
90 * in MSET the step is two since arguments are key,val,key,val,...
91 * microseconds: microseconds of total execution time for this command.
92 * calls: total number of calls of this command.
93 *
94 * The flags, microseconds and calls fields are computed by Redis and should
95 * always be set to zero.
96 *
97 * Command flags are expressed using strings where every character represents
98 * a flag. Later the populateCommandTable() function will take care of
99 * populating the real 'flags' field using this characters.
100 *
101 * This is the meaning of the flags:
102 *
103 * w: write command (may modify the key space).
104 * r: read command (will never modify the key space).
105 * m: may increase memory usage once called. Don't allow if out of memory.
106 * a: admin command, like SAVE or SHUTDOWN.
107 * p: Pub/Sub related command.
108 * f: force replication of this command, regardless of server.dirty.
109 * s: command not allowed in scripts.
110 * R: random command. Command is not deterministic, that is, the same command
111 * with the same arguments, with the same key space, may have different
112 * results. For instance SPOP and RANDOMKEY are two random commands.
113 * S: Sort command output array if called from script, so that the output
114 * is deterministic.
115 * l: Allow command while loading the database.
116 * t: Allow command while a slave has stale data but is not allowed to
117 * server this data. Normally no command is accepted in this condition
118 * but just a few.
119 * M: Do not automatically propagate the command on MONITOR.
120 * k: Perform an implicit ASKING for this command, so the command will be
121 * accepted in cluster mode if the slot is marked as 'importing'.
122 * F: Fast command: O(1) or O(log(N)) command that should never delay
123 * its execution as long as the kernel scheduler is giving us time.
124 * Note that commands that may trigger a DEL as a side effect (like SET)
125 * are not fast commands.
126 */

312/* Low level logging. To use only for very big messages, otherwise
313 * serverLog() is to prefer. */
314void serverLogRaw(int level, const char *msg) {
315 const int syslogLevelMap[] = { LOG_DEBUG, LOG_INFO, LOG_NOTICE, LOG_WARNING };
316 const char *c = ".-*#";
317 FILE *fp;
318 char buf[64];
319 int rawmode = (level & LL_RAW);
320 int log_to_stdout = server.logfile[0] == '\0';
321
322 level &= 0xff; /* clear flags */
323 if (level < server.verbosity) return;
324
325 fp = log_to_stdout ? stdout : fopen(server.logfile,"a");
326 if (!fp) return;
327
328 if (rawmode) {
329 fprintf(fp,"%s",msg);
330 } else {
331 int off;
332 struct timeval tv;
333 int role_char;
334 pid_t pid = getpid();
335
336 gettimeofday(&tv,NULL);
337 off = strftime(buf,sizeof(buf),"%d %b %H:%M:%S.",localtime(&tv.tv_sec));
338 snprintf(buf+off,sizeof(buf)-off,"%03d",(int)tv.tv_usec/1000);
339 if (server.sentinel_mode) {
340 role_char = 'X'; /* Sentinel. */
341 } else if (pid != server.pid) {
342 role_char = 'C'; /* RDB / AOF writing child. */
343 } else {
344 role_char = (server.masterhost ? 'S':'M'); /* Slave or Master. */
345 }
346 fprintf(fp,"%d:%c %s %c %s\n",
347 (int)getpid(),role_char, buf,c[level],msg);
348 }
349 fflush(fp);
350
351 if (!log_to_stdout) fclose(fp);
352 if (server.syslog_enabled) syslog(syslogLevelMap[level], "%s", msg);
353}

355/* Like serverLogRaw() but with printf-alike support. This is the function that
356 * is used across the code. The raw version is only used in order to dump
357 * the INFO output on crash. */
358void serverLog(int level, const char *fmt, ...) {
359 va_list ap;
360 char msg[LOG_MAX_LEN];
361
362 if ((level&0xff) < server.verbosity) return;
363
364 va_start(ap, fmt);
365 vsnprintf(msg, sizeof(msg), fmt, ap);
366 va_end(ap);
367
368 serverLogRaw(level,msg);
369}

371/* Log a fixed message without printf-alike capabilities, in a way that is
372 * safe to call from a signal handler.
373 *
374 * We actually use this only for signals that are not fatal from the point
375 * of view of Redis. Signals that are going to kill the server anyway and
376 * where we need printf-alike features are served by serverLog(). */
377void serverLogFromHandler(int level, const char *msg) {
378 int fd;
379 int log_to_stdout = server.logfile[0] == '\0';
380 char buf[64];
381
382 if ((level&0xff) < server.verbosity || (log_to_stdout && server.daemonize))
383 return;
384 fd = log_to_stdout ? STDOUT_FILENO :
385 open(server.logfile, O_APPEND|O_CREAT|O_WRONLY, 0644);
386 if (fd == -1) return;
387 ll2string(buf,sizeof(buf),getpid());
388 if (write(fd,buf,strlen(buf)) == -1) goto err;
389 if (write(fd,":signal-handler (",17) == -1) goto err;
390 ll2string(buf,sizeof(buf),time(NULL));

774/* Check for timeouts. Returns non-zero if the client was terminated.
775 * The function gets the current time in milliseconds as argument since
776 * it gets called multiple times in a loop, so calling gettimeofday() for
777 * each iteration would be costly without any actual gain. */
778int clientsCronHandleTimeout(client *c, mstime_t now_ms) {
779 time_t now = now_ms/1000;
780
781 if (server.maxidletime &&
782 !(c->flags & CLIENT_SLAVE) && /* no timeout for slaves */
783 !(c->flags & CLIENT_MASTER) && /* no timeout for masters */
784 !(c->flags & CLIENT_BLOCKED) && /* no timeout for BLPOP */
785 !(c->flags & CLIENT_PUBSUB) && /* no timeout for Pub/Sub clients */
786 (now - c->lastinteraction > server.maxidletime))
787 {
788 serverLog(LL_VERBOSE,"Closing idle client");
789 freeClient(c);
790 return 1;
791 } else if (c->flags & CLIENT_BLOCKED) {
792 /* Blocked OPS timeout is handled with milliseconds resolution.
793 * However note that the actual resolution is limited by
794 * server.hz. */
795
796 if (c->bpop.timeout != 0 && c->bpop.timeout < now_ms) {
797 /* Handle blocking operation specific timeout. */
798 replyToBlockedClientTimedOut(c);
799 unblockClient(c);
800 } else if (server.cluster_enabled) {
801 /* Cluster: handle unblock & redirect of clients blocked
802 * into keys no longer served by this server. */
803 if (clusterRedirectBlockedClientIfNeeded(c))
804 unblockClient(c);
805 }
806 }
807 return 0;
808}

810/* The client query buffer is an sds.c string that can end with a lot of
811 * free space not used, this function reclaims space if needed.
812 *
813 * The function always returns 0 as it never terminates the client. */
814int clientsCronResizeQueryBuffer(client *c) {
815 size_t querybuf_size = sdsAllocSize(c->querybuf);
816 time_t idletime = server.unixtime - c->lastinteraction;
817
818 /* There are two conditions to resize the query buffer:
819 * 1) Query buffer is > BIG_ARG and too big for latest peak.
820 * 2) Client is inactive and the buffer is bigger than 1k. */
821 if (((querybuf_size > PROTO_MBULK_BIG_ARG) &&
822 (querybuf_size/(c->querybuf_peak+1)) > 2) ||
823 (querybuf_size > 1024 && idletime > 2))
824 {
825 /* Only resize the query buffer if it is actually wasting space. */
826 if (sdsavail(c->querybuf) > 1024) {
827 c->querybuf = sdsRemoveFreeSpace(c->querybuf);
828 }
829 }
830 /* Reset the peak again to capture the peak memory usage in the next
831 * cycle. */
832 c->querybuf_peak = 0;
833 return 0;
834}

836#define CLIENTS_CRON_MIN_ITERATIONS 5
837void clientsCron(void) {
838 /* Make sure to process at least numclients/server.hz of clients
839 * per call. Since this function is called server.hz times per second
840 * we are sure that in the worst case we process all the clients in 1
841 * second. */
842 int numclients = listLength(server.clients);
843 int iterations = numclients/server.hz;
844 mstime_t now = mstime();
845
846 /* Process at least a few clients while we are at it, even if we need
847 * to process less than CLIENTS_CRON_MIN_ITERATIONS to meet our contract
848 * of processing each client once per second. */
849 if (iterations < CLIENTS_CRON_MIN_ITERATIONS)
850 iterations = (numclients < CLIENTS_CRON_MIN_ITERATIONS) ?
851 numclients : CLIENTS_CRON_MIN_ITERATIONS;
852
853 while(listLength(server.clients) && iterations--) {
854 client *c;
855 listNode *head;
856
857 /* Rotate the list, take the current head, process.
858 * This way if the client must be removed from the list it's the
859 * first element and we don't incur into O(N) computation. */
860 listRotate(server.clients);
861 head = listFirst(server.clients);
862 c = listNodeValue(head);
863 /* The following functions do different service checks on the client.
864 * The protocol is that they return non-zero if the client was
865 * terminated. */
866 if (clientsCronHandleTimeout(c,now)) continue;
867 if (clientsCronResizeQueryBuffer(c)) continue;
868 }
869}

871/* This function handles 'background' operations we are required to do
872 * incrementally in Redis databases, such as active key expiring, resizing,
873 * rehashing. */
874void databasesCron(void) {
875 /* Expire keys by random sampling. Not required for slaves
876 * as master will synthesize DELs for us. */
877 if (server.active_expire_enabled && server.masterhost == NULL) {
878 activeExpireCycle(ACTIVE_EXPIRE_CYCLE_SLOW);
879 } else if (server.masterhost != NULL) {
880 expireSlaveKeys();
881 }
882
883 /* Defrag keys gradually. */
884 if (server.active_defrag_enabled)
885 activeDefragCycle();
886
887 /* Perform hash tables rehashing if needed, but only if there are no
888 * other processes saving the DB on disk. Otherwise rehashing is bad
889 * as will cause a lot of copy-on-write of memory pages. */
890 if (server.rdb_child_pid == -1 && server.aof_child_pid == -1) {
891 /* We use global counters so if we stop the computation at a given
892 * DB we'll be able to start from the successive in the next
893 * cron loop iteration. */
894 static unsigned int resize_db = 0;
895 static unsigned int rehash_db = 0;
896 int dbs_per_call = CRON_DBS_PER_CALL;
897 int j;
898
899 /* Don't test more DBs than we have. */
900 if (dbs_per_call > server.dbnum) dbs_per_call = server.dbnum;
901
902 /* Resize */
903 for (j = 0; j < dbs_per_call; j++) {
904 tryResizeHashTables(resize_db % server.dbnum);
905 resize_db++;
906 }
907
908 /* Rehash */
909 if (server.activerehashing) {
910 for (j = 0; j < dbs_per_call; j++) {
911 int work_done = incrementallyRehash(rehash_db % server.dbnum);
912 rehash_db++;
913 if (work_done) {
914 /* If the function did some work, stop here, we'll do
915 * more at the next cron loop. */
916 break;
917 }
918 }
919 }
920 }
921}

934 * Here is where we do a number of things that need to be done asynchronously.
935 * For instance:
936 *
937 * - Active expired keys collection (it is also performed in a lazy way on
938 * lookup).
939 * - Software watchdog.
940 * - Update some statistic.
941 * - Incremental rehashing of the DBs hash tables.
942 * - Triggering BGSAVE / AOF rewrite, and handling of terminated children.
943 * - Clients timeout of different kinds.
944 * - Replication reconnection.
945 * - Many more...
946 *
947 * Everything directly called here will be called server.hz times per second,
948 * so in order to throttle execution of things we want to do less frequently
949 * a macro is used: run_with_period(milliseconds) { .... }

952int serverCron(struct aeEventLoop *eventLoop, long long id, void *clientData) {
953 int j;
954 UNUSED(eventLoop);
955 UNUSED(id);
956 UNUSED(clientData);
957
958 /* Software watchdog: deliver the SIGALRM that will reach the signal
959 * handler if we don't return here fast enough. */
960 if (server.watchdog_period) watchdogScheduleSignal(server.watchdog_period);
961
962 /* Update the time cache. */
963 updateCachedTime();

965 run_with_period(100) {
966 trackInstantaneousMetric(STATS_METRIC_COMMAND,server.stat_numcommands);
967 trackInstantaneousMetric(STATS_METRIC_NET_INPUT,
968 server.stat_net_input_bytes);
969 trackInstantaneousMetric(STATS_METRIC_NET_OUTPUT,
970 server.stat_net_output_bytes);

974 * So we use an (eventually wrapping) LRU clock.
975 *
976 * Note that even if the counter wraps it's not a big problem,
977 * everything will still work but some object will appear younger
978 * to Redis. However for this to happen a given object should never be
979 * touched for all the time needed to the counter to wrap, which is
980 * not likely.
981 *
982 * Note that you can change the resolution altering the
983 * LRU_CLOCK_RESOLUTION define. */
984 unsigned long lruclock = getLRUClock();
985 atomicSet(server.lruclock,lruclock);
986
987 /* Record the max memory used since the server was started. */
988 if (zmalloc_used_memory() > server.stat_peak_memory)
989 server.stat_peak_memory = zmalloc_used_memory();
990
991 /* Sample the RSS here since this is a relatively slow call. */
992 server.resident_set_size = zmalloc_get_rss();
993
994 /* We received a SIGTERM, shutting down here in a safe way, as it is
995 * not ok doing so inside the signal handler. */
996 if (server.shutdown_asap) {
997 if (prepareForShutdown(SHUTDOWN_NOFLAGS) == C_OK) exit(0);
998 serverLog(LL_WARNING,"SIGTERM received but errors trying to shut down the server, check the logs for more information");
999 server.shutdown_asap = 0;
1000 }

1002 /* Show some info about non-empty databases */
1003 run_with_period(5000) {
1004 for (j = 0; j < server.dbnum; j++) {
1005 long long size, used, vkeys;
1006
1007 size = dictSlots(server.db[j].dict);
1008 used = dictSize(server.db[j].dict);
1009 vkeys = dictSize(server.db[j].expires);
1010 if (used || vkeys) {
1011 serverLog(LL_VERBOSE,"DB %d: %lld keys (%lld volatile) in %lld slots HT.",j,used,vkeys,size);
1012 /* dictPrintStats(server.dict); */
1013 }
1014 }
1015 }
1016
1017 /* Show information about connected clients */
1018 if (!server.sentinel_mode) {
1019 run_with_period(5000) {
1020 serverLog(LL_VERBOSE,
1021 "%lu clients connected (%lu slaves), %zu bytes in use",
1022 listLength(server.clients)-listLength(server.slaves),
1023 listLength(server.slaves),
1024 zmalloc_used_memory());
1025 }
1026 }

1028 /* We need to do a few operations on clients asynchronously. */
1029 clientsCron();
1030
1031 /* Handle background operations on Redis databases. */
1032 databasesCron();
1033
1034 /* Start a scheduled AOF rewrite if this was requested by the user while
1035 * a BGSAVE was in progress. */
1036 if (server.rdb_child_pid == -1 && server.aof_child_pid == -1 &&
1037 server.aof_rewrite_scheduled)
1038 {
1039 rewriteAppendOnlyFileBackground();
1040 }
1041
1042 /* Check if a background saving or AOF rewrite in progress terminated. */
1043 if (server.rdb_child_pid != -1 || server.aof_child_pid != -1 ||
1044 ldbPendingChildren())
1045 {
1046 int statloc;
1047 pid_t pid;
1048
1049 if ((pid = wait3(&statloc,WNOHANG,NULL)) != 0) {
1050 int exitcode = WEXITSTATUS(statloc);
1051 int bysignal = 0;
1052
1053 if (WIFSIGNALED(statloc)) bysignal = WTERMSIG(statloc);
1054
1055 if (pid == -1) {
1056 serverLog(LL_WARNING,"wait3() returned an error: %s. "
1057 "rdb_child_pid = %d, aof_child_pid = %d",
1058 strerror(errno),
1059 (int) server.rdb_child_pid,
1060 (int) server.aof_child_pid);
1061 } else if (pid == server.rdb_child_pid) {
1062 backgroundSaveDoneHandler(exitcode,bysignal);
1063 if (!bysignal && exitcode == 0) receiveChildInfo();
1064 } else if (pid == server.aof_child_pid) {
1065 backgroundRewriteDoneHandler(exitcode,bysignal);
1066 if (!bysignal && exitcode == 0) receiveChildInfo();
1067 } else {
1068 if (!ldbRemoveChild(pid)) {
1069 serverLog(LL_WARNING,
1070 "Warning, detected child with unmatched pid: %ld",
1071 (long)pid);
1072 }
1073 }
1074 updateDictResizePolicy();
1075 closeChildInfoPipe();
1076 }
1077 } else {
1078 /* If there is not a background saving/rewrite in progress check if
1079 * we have to save/rewrite now */
1080 for (j = 0; j < server.saveparamslen; j++) {
1081 struct saveparam *sp = server.saveparams+j;
1082
1083 /* Save if we reached the given amount of changes,
1084 * the given amount of seconds, and if the latest bgsave was
1085 * successful or if, in case of an error, at least
1086 * CONFIG_BGSAVE_RETRY_DELAY seconds already elapsed. */
1087 if (server.dirty >= sp->changes &&
1088 server.unixtime-server.lastsave > sp->seconds &&
1089 (server.unixtime-server.lastbgsave_try >
1090 CONFIG_BGSAVE_RETRY_DELAY ||
1091 server.lastbgsave_status == C_OK))
1092 {
1093 serverLog(LL_NOTICE,"%d changes in %d seconds. Saving...",
1094 sp->changes, (int)sp->seconds);
1095 rdbSaveInfo rsi, *rsiptr;
1096 rsiptr = rdbPopulateSaveInfo(&rsi);
1097 rdbSaveBackground(server.rdb_filename,rsiptr);
1098 break;
1099 }
1100 }
1101
1102 /* Trigger an AOF rewrite if needed */
1103 if (server.rdb_child_pid == -1 &&
1104 server.aof_child_pid == -1 &&
1105 server.aof_rewrite_perc &&
1106 server.aof_current_size > server.aof_rewrite_min_size)
1107 {
1108 long long base = server.aof_rewrite_base_size ?
1109 server.aof_rewrite_base_size : 1;
1110 long long growth = (server.aof_current_size*100/base) - 100;
1111 if (growth >= server.aof_rewrite_perc) {
1112 serverLog(LL_NOTICE,"Starting automatic rewriting of AOF on %lld%% growth",growth);
1113 rewriteAppendOnlyFileBackground();
1114 }
1115 }
1116 }
1117
1118
1119 /* AOF postponed flush: Try at every cron cycle if the slow fsync
1120 * completed. */
1121 if (server.aof_flush_postponed_start) flushAppendOnlyFile(0);
1122
1123 /* AOF write errors: in this case we have a buffer to flush as well and
1124 * clear the AOF error in case of success to make the DB writable again,
1125 * however to try every second is enough in case of 'hz' is set to
1126 * an higher frequency. */
1127 run_with_period(1000) {
1128 if (server.aof_last_write_status == C_ERR)
1129 flushAppendOnlyFile(0);
1130 }
1131
1132 /* Close clients that need to be closed asynchronous */
1133 freeClientsInAsyncFreeQueue();
1134
1135 /* Clear the paused clients flag if needed. */
1136 clientsArePaused(); /* Don't check return value, just use the side effect.*/
1137
1138 /* Replication cron function -- used to reconnect to master,
1139 * detect transfer failures, start background RDB transfers and so forth. */
1140 run_with_period(1000) replicationCron();
1141
1142 /* Run the Redis Cluster cron. */
1143 run_with_period(100) {
1144 if (server.cluster_enabled) clusterCron();
1145 }
1146
1147 /* Run the Sentinel timer if we are in sentinel mode. */
1148 run_with_period(100) {
1149 if (server.sentinel_mode) sentinelTimer();
1150 }
1151
1152 /* Cleanup expired MIGRATE cached sockets. */
1153 run_with_period(1000) {
1154 migrateCloseTimedoutSockets();
1155 }
1156
1157 /* Start a scheduled BGSAVE if the corresponding flag is set. This is
1158 * useful when we are forced to postpone a BGSAVE because an AOF
1159 * rewrite is in progress.
1160 *
1161 * Note: this code must be after the replicationCron() call above so
1162 * make sure when refactoring this file to keep this order. This is useful
1163 * because we want to give priority to RDB savings for replication. */
1164 if (server.rdb_child_pid == -1 && server.aof_child_pid == -1 &&
1165 server.rdb_bgsave_scheduled &&
1166 (server.unixtime-server.lastbgsave_try > CONFIG_BGSAVE_RETRY_DELAY ||
1167 server.lastbgsave_status == C_OK))
1168 {
1169 rdbSaveInfo rsi, *rsiptr;
1170 rsiptr = rdbPopulateSaveInfo(&rsi);
1171 if (rdbSaveBackground(server.rdb_filename,rsiptr) == C_OK)
1172 server.rdb_bgsave_scheduled = 0;
1173 }
1174
1175 server.cronloops++;
1176 return 1000/server.hz;
1177}

1179/* This function gets called every time Redis is entering the
1180 * main loop of the event driven library, that is, before to sleep
1181 * for ready file descriptors. */
1182void beforeSleep(struct aeEventLoop *eventLoop) {
1183 UNUSED(eventLoop);
1184
1185 /* Call the Redis Cluster before sleep function. Note that this function
1186 * may change the state of Redis Cluster (from ok to fail or vice versa),
1187 * so it's a good idea to call it before serving the unblocked clients
1188 * later in this function. */
1189 if (server.cluster_enabled) clusterBeforeSleep();
1190
1191 /* Run a fast expire cycle (the called function will return
1192 * ASAP if a fast cycle is not needed). */
1193 if (server.active_expire_enabled && server.masterhost == NULL)
1194 activeExpireCycle(ACTIVE_EXPIRE_CYCLE_FAST);
1195
1196 /* Send all the slaves an ACK request if at least one client blocked
1197 * during the previous event loop iteration. */
1198 if (server.get_ack_from_slaves) {
1199 robj *argv[3];
1200
1201 argv[0] = createStringObject("REPLCONF",8);
1202 argv[1] = createStringObject("GETACK",6);
1203 argv[2] = createStringObject("*",1); /* Not used argument. */
1204 replicationFeedSlaves(server.slaves, server.slaveseldb, argv, 3);
1205 decrRefCount(argv[0]);
1206 decrRefCount(argv[1]);
1207 decrRefCount(argv[2]);
1208 server.get_ack_from_slaves = 0;
1209 }
1210
1211 /* Unblock all the clients blocked for synchronous replication
1212 * in WAIT. */
1213 if (listLength(server.clients_waiting_acks))
1214 processClientsWaitingReplicas();
1215
1216 /* Check if there are clients unblocked by modules that implement
1217 * blocking commands. */
1218 moduleHandleBlockedClients();
1219
1220 /* Try to process pending commands for clients that were just unblocked. */
1221 if (listLength(server.unblocked_clients))
1222 processUnblockedClients();
1223
1224 /* Write the AOF buffer on disk */
1225 flushAppendOnlyFile(0);
1226
1227 /* Handle writes with pending output buffers. */
1228 handleClientsWithPendingWrites();
1229
1230 /* Before we are going to sleep, let the threads access the dataset by
1231 * releasing the GIL. Redis main thread will not touch anything at this
1232 * time. */
1233 if (moduleCount()) moduleReleaseGIL();
1234}

1531 *
1532 * The function is designed to directly call execve() so that the new
1533 * server instance will retain the PID of the previous one.
1534 *
1535 * The list of flags, that may be bitwise ORed together, alter the
1536 * behavior of this function:
1537 *
1538 * RESTART_SERVER_NONE No flags.
1539 * RESTART_SERVER_GRACEFULLY Do a proper shutdown before restarting.
1540 * RESTART_SERVER_CONFIG_REWRITE Rewrite the config file before restarting.
1541 *
1542 * On success the function does not return, because the process turns into
1543 * a different process. On error C_ERR is returned. */
1544int restartServer(int flags, mstime_t delay) {
1545 int j;
1546
1547 /* Check if we still have accesses to the executable that started this
1548 * server instance. */
1549 if (access(server.executable,X_OK) == -1) return C_ERR;
1550
1551 /* Config rewriting. */
1552 if (flags & RESTART_SERVER_CONFIG_REWRITE &&
1553 server.configfile &&
1554 rewriteConfig(server.configfile) == -1) return C_ERR;
1555
1556 /* Perform a proper shutdown. */
1557 if (flags & RESTART_SERVER_GRACEFULLY &&
1558 prepareForShutdown(SHUTDOWN_NOFLAGS) != C_OK) return C_ERR;
1559
1560 /* Close all file descriptors, with the exception of stdin, stdout, strerr
1561 * which are useful if we restart a Redis server which is not daemonized. */
1562 for (j = 3; j < (int)server.maxclients + 1024; j++) {
1563 /* Test the descriptor validity before closing it, otherwise
1564 * Valgrind issues a warning on close(). */
1565 if (fcntl(j,F_GETFD) != -1) close(j);
1566 }
1567
1568 /* Execute the server with the original command line. */
1569 if (delay) usleep(delay*1000);
1570 execve(server.executable,server.exec_argv,environ);
1571
1572 /* If an error occurred here, there is nothing we can do, but exit. */
1573 _exit(1);
1574
1575 return C_ERR; /* Never reached. */
1576}

1578/* This function will try to raise the max number of open files accordingly to
1579 * the configured max number of clients. It also reserves a number of file
1580 * descriptors (CONFIG_MIN_RESERVED_FDS) for extra operations of
1581 * persistence, listening sockets, log files and so forth.
1582 *
1583 * If it will not be possible to set the limit accordingly to the configured
1584 * max number of clients, the function will do the reverse setting
1585 * server.maxclients to the value that we can actually handle. */
1586void adjustOpenFilesLimit(void) {
1587 rlim_t maxfiles = server.maxclients+CONFIG_MIN_RESERVED_FDS;
1588 struct rlimit limit;
1589
1590 if (getrlimit(RLIMIT_NOFILE,&limit) == -1) {
1591 serverLog(LL_WARNING,"Unable to obtain the current NOFILE limit (%s), assuming 1024 and setting the max clients configuration accordingly.",
1592 strerror(errno));
1593 server.maxclients = 1024-CONFIG_MIN_RESERVED_FDS;
1594 } else {
1595 rlim_t oldlimit = limit.rlim_cur;
1596
1597 /* Set the max number of files if the current limit is not enough
1598 * for our needs. */
1599 if (oldlimit < maxfiles) {
1600 rlim_t bestlimit;
1601 int setrlimit_error = 0;
1602
1603 /* Try to set the file limit to match 'maxfiles' or at least
1604 * to the higher value supported less than maxfiles. */
1605 bestlimit = maxfiles;
1606 while(bestlimit > oldlimit) {
1607 rlim_t decr_step = 16;
1608
1609 limit.rlim_cur = bestlimit;
1610 limit.rlim_max = bestlimit;
1611 if (setrlimit(RLIMIT_NOFILE,&limit) != -1) break;
1612 setrlimit_error = errno;
1613
1614 /* We failed to set file limit to 'bestlimit'. Try with a
1615 * smaller limit decrementing by a few FDs per iteration. */
1616 if (bestlimit < decr_step) break;
1617 bestlimit -= decr_step;
1618 }
1619
1620 /* Assume that the limit we get initially is still valid if
1621 * our last try was even lower. */
1622 if (bestlimit < oldlimit) bestlimit = oldlimit;
1623
1624 if (bestlimit < maxfiles) {
1625 unsigned int old_maxclients = server.maxclients;
1626 server.maxclients = bestlimit-CONFIG_MIN_RESERVED_FDS;
1627 /* maxclients is unsigned so may overflow: in order
1628 * to check if maxclients is now logically less than 1
1629 * we test indirectly via bestlimit. */
1630 if (bestlimit <= CONFIG_MIN_RESERVED_FDS) {
1631 serverLog(LL_WARNING,"Your current 'ulimit -n' "
1632 "of %llu is not enough for the server to start. "
1633 "Please increase your open file limit to at least "
1634 "%llu. Exiting.",
1635 (unsigned long long) oldlimit,
1636 (unsigned long long) maxfiles);
1637 exit(1);
1638 }
1639 serverLog(LL_WARNING,"You requested maxclients of %d "
1640 "requiring at least %llu max file descriptors.",
1641 old_maxclients,
1642 (unsigned long long) maxfiles);
1643 serverLog(LL_WARNING,"Server can't set maximum open files "
1644 "to %llu because of OS error: %s.",
1645 (unsigned long long) maxfiles, strerror(setrlimit_error));
1646 serverLog(LL_WARNING,"Current maximum open files is %llu. "
1647 "maxclients has been reduced to %d to compensate for "
1648 "low ulimit. "
1649 "If you need higher maxclients increase 'ulimit -n'.",
1650 (unsigned long long) bestlimit, server.maxclients);
1651 } else {
1652 serverLog(LL_NOTICE,"Increased maximum number of open files "
1653 "to %llu (it was originally set to %llu).",
1654 (unsigned long long) maxfiles,
1655 (unsigned long long) oldlimit);
1656 }
1657 }
1658 }

1678/* Initialize a set of file descriptors to listen to the specified 'port'
1679 * binding the addresses specified in the Redis server configuration.
1680 *
1681 * The listening file descriptors are stored in the integer array 'fds'
1682 * and their number is set in '*count'.
1683 *
1684 * The addresses to bind are specified in the global server.bindaddr array
1685 * and their number is server.bindaddr_count. If the server configuration
1686 * contains no specific addresses to bind, this function will try to
1687 * bind * (all addresses) for both the IPv4 and IPv6 protocols.
1688 *
1689 * On success the function returns C_OK.
1690 *
1691 * On error the function returns C_ERR. For the function to be on
1692 * error, at least one of the server.bindaddr addresses was
1693 * impossible to bind, or no bind addresses were specified in the server
1694 * configuration but the function is not able to bind * for at least
1695 * one of the IPv4 or IPv6 protocols. */
1696int listenToPort(int port, int *fds, int *count) {
1697 int j;
1698
1699 /* Force binding of 0.0.0.0 if no bind address is specified, always
1700 * entering the loop if j == 0. */
1701 if (server.bindaddr_count == 0) server.bindaddr[0] = NULL;
1702 for (j = 0; j < server.bindaddr_count || j == 0; j++) {
1703 if (server.bindaddr[j] == NULL) {
1704 int unsupported = 0;
1705 /* Bind * for both IPv6 and IPv4, we enter here only if
1706 * server.bindaddr_count == 0. */
1707 fds[*count] = anetTcp6Server(server.neterr,port,NULL,
1708 server.tcp_backlog);
1709 if (fds[*count] != ANET_ERR) {
1710 anetNonBlock(NULL,fds[*count]);
1711 (*count)++;
1712 } else if (errno == EAFNOSUPPORT) {
1713 unsupported++;
1714 serverLog(LL_WARNING,"Not listening to IPv6: unsupproted");
1715 }
1716
1717 if (*count == 1 || unsupported) {
1718 /* Bind the IPv4 address as well. */
1719 fds[*count] = anetTcpServer(server.neterr,port,NULL,
1720 server.tcp_backlog);
1721 if (fds[*count] != ANET_ERR) {
1722 anetNonBlock(NULL,fds[*count]);
1723 (*count)++;
1724 } else if (errno == EAFNOSUPPORT) {
1725 unsupported++;
1726 serverLog(LL_WARNING,"Not listening to IPv4: unsupproted");
1727 }
1728 }
1729 /* Exit the loop if we were able to bind * on IPv4 and IPv6,
1730 * otherwise fds[*count] will be ANET_ERR and we'll print an
1731 * error and return to the caller with an error. */
1732 if (*count + unsupported == 2) break;
1733 } else if (strchr(server.bindaddr[j],':')) {
1734 /* Bind IPv6 address. */
1735 fds[*count] = anetTcp6Server(server.neterr,port,server.bindaddr[j],
1736 server.tcp_backlog);
1737 } else {
1738 /* Bind IPv4 address. */
1739 fds[*count] = anetTcpServer(server.neterr,port,server.bindaddr[j],
1740 server.tcp_backlog);
1741 }
1742 if (fds[*count] == ANET_ERR) {
1743 serverLog(LL_WARNING,
1744 "Creating Server TCP listening socket %s:%d: %s",
1745 server.bindaddr[j] ? server.bindaddr[j] : "*",
1746 port, server.neterr);
1747 return C_ERR;
1748 }
1749 anetNonBlock(NULL,fds[*count]);
1750 (*count)++;
1751 }
1752 return C_OK;
1753}

2095 * after the current command is propagated to AOF / Replication.
2096 *
2097 * 'cmd' must be a pointer to the Redis command to replicate, dbid is the
2098 * database ID the command should be propagated into.
2099 * Arguments of the command to propagte are passed as an array of redis
2100 * objects pointers of len 'argc', using the 'argv' vector.
2101 *
2102 * The function does not take a reference to the passed 'argv' vector,
2103 * so it is up to the caller to release the passed argv (but it is usually
2104 * stack allocated). The function autoamtically increments ref count of
2105 * passed objects, so the caller does not need to. */
2106void alsoPropagate(struct redisCommand *cmd, int dbid, robj **argv, int argc,
2107 int target)
2108{
2109 robj **argvcopy;
2110 int j;
2111
2112 if (server.loading) return; /* No propagation during loading. */
2113
2114 argvcopy = zmalloc(sizeof(robj*)*argc);
2115 for (j = 0; j < argc; j++) {
2116 argvcopy[j] = argv[j];

2147/* Call() is the core of Redis execution of a command.
2148 *
2149 * The following flags can be passed:
2150 * CMD_CALL_NONE No flags.
2151 * CMD_CALL_SLOWLOG Check command speed and log in the slow log if needed.
2152 * CMD_CALL_STATS Populate command stats.
2153 * CMD_CALL_PROPAGATE_AOF Append command to AOF if it modified the dataset
2154 * or if the client flags are forcing propagation.
2155 * CMD_CALL_PROPAGATE_REPL Send command to salves if it modified the dataset
2156 * or if the client flags are forcing propagation.
2157 * CMD_CALL_PROPAGATE Alias for PROPAGATE_AOF|PROPAGATE_REPL.
2158 * CMD_CALL_FULL Alias for SLOWLOG|STATS|PROPAGATE.
2159 *
2160 * The exact propagation behavior depends on the client flags.
2161 * Specifically:
2162 *
2163 * 1. If the client flags CLIENT_FORCE_AOF or CLIENT_FORCE_REPL are set
2164 * and assuming the corresponding CMD_CALL_PROPAGATE_AOF/REPL is set
2165 * in the call flags, then the command is propagated even if the
2166 * dataset was not affected by the command.
2167 * 2. If the client flags CLIENT_PREVENT_REPL_PROP or CLIENT_PREVENT_AOF_PROP
2168 * are set, the propagation into AOF or to slaves is not performed even
2169 * if the command modified the dataset.
2170 *
2171 * Note that regardless of the client flags, if CMD_CALL_PROPAGATE_AOF
2172 * or CMD_CALL_PROPAGATE_REPL are not set, then respectively AOF or
2173 * slaves propagation will never occur.
2174 *
2175 * Client flags are modified by the implementation of a given command
2176 * using the following API:
2177 *
2178 * forceCommandPropagation(client *c, int flags);
2179 * preventCommandPropagation(client *c);
2180 * preventCommandAOF(client *c);
2181 * preventCommandReplication(client *c);
2182 *
2183 */
2184void call(client *c, int flags) {
2185 long long dirty, start, duration;
2186 int client_old_flags = c->flags;
2187
2188 /* Sent the command to clients in MONITOR mode, only if the commands are
2189 * not generated from reading an AOF. */
2190 if (listLength(server.monitors) &&
2191 !server.loading &&
2192 !(c->cmd->flags & (CMD_SKIP_MONITOR|CMD_ADMIN)))
2193 {
2194 replicationFeedMonitors(c,server.monitors,c->db->id,c->argv,c->argc);
2195 }
2196
2197 /* Initialization: clear the flags that must be set by the command on
2198 * demand, and initialize the array for additional commands propagation. */
2199 c->flags &= ~(CLIENT_FORCE_AOF|CLIENT_FORCE_REPL|CLIENT_PREVENT_PROP);
2200 redisOpArray prev_also_propagate = server.also_propagate;
2201 redisOpArrayInit(&server.also_propagate);
2202
2203 /* Call the command. */
2204 dirty = server.dirty;
2205 start = ustime();
2206 c->cmd->proc(c);
2207 duration = ustime()-start;
2208 dirty = server.dirty-dirty;
2209 if (dirty < 0) dirty = 0;
2210
2211 /* When EVAL is called loading the AOF we don't want commands called
2212 * from Lua to go into the slowlog or to populate statistics. */
2213 if (server.loading && c->flags & CLIENT_LUA)
2214 flags &= ~(CMD_CALL_SLOWLOG | CMD_CALL_STATS);
2215
2216 /* If the caller is Lua, we want to force the EVAL caller to propagate
2217 * the script if the command flag or client flag are forcing the
2218 * propagation. */
2219 if (c->flags & CLIENT_LUA && server.lua_caller) {
2220 if (c->flags & CLIENT_FORCE_REPL)
2221 server.lua_caller->flags |= CLIENT_FORCE_REPL;
2222 if (c->flags & CLIENT_FORCE_AOF)
2223 server.lua_caller->flags |= CLIENT_FORCE_AOF;
2224 }
2225
2226 /* Log the command into the Slow log if needed, and populate the
2227 * per-command statistics that we show in INFO commandstats. */
2228 if (flags & CMD_CALL_SLOWLOG && c->cmd->proc != execCommand) {
2229 char *latency_event = (c->cmd->flags & CMD_FAST) ?
2230 "fast-command" : "command";
2231 latencyAddSampleIfNeeded(latency_event,duration/1000);
2232 slowlogPushEntryIfNeeded(c,c->argv,c->argc,duration);
2233 }
2234 if (flags & CMD_CALL_STATS) {
2235 c->lastcmd->microseconds += duration;
2236 c->lastcmd->calls++;
2237 }
2238
2239 /* Propagate the command into the AOF and replication link */
2240 if (flags & CMD_CALL_PROPAGATE &&
2241 (c->flags & CLIENT_PREVENT_PROP) != CLIENT_PREVENT_PROP)
2242 {
2243 int propagate_flags = PROPAGATE_NONE;
2244
2245 /* Check if the command operated changes in the data set. If so
2246 * set for replication / AOF propagation. */
2247 if (dirty) propagate_flags |= (PROPAGATE_AOF|PROPAGATE_REPL);
2248
2249 /* If the client forced AOF / replication of the command, set
2250 * the flags regardless of the command effects on the data set. */
2251 if (c->flags & CLIENT_FORCE_REPL) propagate_flags |= PROPAGATE_REPL;
2252 if (c->flags & CLIENT_FORCE_AOF) propagate_flags |= PROPAGATE_AOF;
2253
2254 /* However prevent AOF / replication propagation if the command
2255 * implementatino called preventCommandPropagation() or similar,
2256 * or if we don't have the call() flags to do so. */
2257 if (c->flags & CLIENT_PREVENT_REPL_PROP ||
2258 !(flags & CMD_CALL_PROPAGATE_REPL))
2259 propagate_flags &= ~PROPAGATE_REPL;
2260 if (c->flags & CLIENT_PREVENT_AOF_PROP ||
2261 !(flags & CMD_CALL_PROPAGATE_AOF))
2262 propagate_flags &= ~PROPAGATE_AOF;
2263
2264 /* Call propagate() only if at least one of AOF / replication
2265 * propagation is needed. */
2266 if (propagate_flags != PROPAGATE_NONE)
2267 propagate(c->cmd,c->db->id,c->argv,c->argc,propagate_flags);
2268 }
2269
2270 /* Restore the old replication flags, since call() can be executed
2271 * recursively. */
2272 c->flags &= ~(CLIENT_FORCE_AOF|CLIENT_FORCE_REPL|CLIENT_PREVENT_PROP);
2273 c->flags |= client_old_flags &
2274 (CLIENT_FORCE_AOF|CLIENT_FORCE_REPL|CLIENT_PREVENT_PROP);
2275
2276 /* Handle the alsoPropagate() API to handle commands that want to propagate
2277 * multiple separated commands. Note that alsoPropagate() is not affected
2278 * by CLIENT_PREVENT_PROP flag. */
2279 if (server.also_propagate.numops) {
2280 int j;
2281 redisOp *rop;
2282
2283 if (flags & CMD_CALL_PROPAGATE) {
2284 for (j = 0; j < server.also_propagate.numops; j++) {
2285 rop = &server.also_propagate.ops[j];
2286 int target = rop->target;
2287 /* Whatever the command wish is, we honor the call() flags. */
2288 if (!(flags&CMD_CALL_PROPAGATE_AOF)) target &= ~PROPAGATE_AOF;
2289 if (!(flags&CMD_CALL_PROPAGATE_REPL)) target &= ~PROPAGATE_REPL;
2290 if (target)
2291 propagate(rop->cmd,rop->dbid,rop->argv,rop->argc,target);
2292 }
2293 }
2294 redisOpArrayFree(&server.also_propagate);
2295 }
2296 server.also_propagate = prev_also_propagate;
2297 server.stat_numcommands++;
2298}

2300/* If this function gets called we already read a whole
2301 * command, arguments are in the client argv/argc fields.
2302 * processCommand() execute the command or prepare the
2303 * server for a bulk read from the client.
2304 *
2305 * If C_OK is returned the client is still alive and valid and
2306 * other operations can be performed by the caller. Otherwise
2307 * if C_ERR is returned the client was destroyed (i.e. after QUIT). */
2308int processCommand(client *c) {
2309 /* The QUIT command is handled separately. Normal command procs will
2310 * go through checking for replication and QUIT will cause trouble
2311 * when FORCE_REPLICATION is enabled and would be implemented in
2312 * a regular command proc. */
2313 if (!strcasecmp(c->argv[0]->ptr,"quit")) {
2314 addReply(c,shared.ok);
2315 c->flags |= CLIENT_CLOSE_AFTER_REPLY;
2316 return C_ERR;
2317 }
2318
2319 /* Now lookup the command and check ASAP about trivial error conditions
2320 * such as wrong arity, bad command name and so forth. */
2321 c->cmd = c->lastcmd = lookupCommand(c->argv[0]->ptr);
2322 if (!c->cmd) {
2323 flagTransaction(c);
2324 addReplyErrorFormat(c,"unknown command '%s'",
2325 (char*)c->argv[0]->ptr);
2326 return C_OK;
2327 } else if ((c->cmd->arity > 0 && c->cmd->arity != c->argc) ||
2328 (c->argc < -c->cmd->arity)) {
2329 flagTransaction(c);
2330 addReplyErrorFormat(c,"wrong number of arguments for '%s' command",
2331 c->cmd->name);
2332 return C_OK;
2333 }
2334
2335 /* Check if the user is authenticated */
2336 if (server.requirepass && !c->authenticated && c->cmd->proc != authCommand)
2337 {
2338 flagTransaction(c);
2339 addReply(c,shared.noautherr);
2340 return C_OK;
2341 }
2342
2343 /* If cluster is enabled perform the cluster redirection here.
2344 * However we don't perform the redirection if:
2345 * 1) The sender of this command is our master.
2346 * 2) The command has no key arguments. */
2347 if (server.cluster_enabled &&
2348 !(c->flags & CLIENT_MASTER) &&
2349 !(c->flags & CLIENT_LUA &&
2350 server.lua_caller->flags & CLIENT_MASTER) &&
2351 !(c->cmd->getkeys_proc == NULL && c->cmd->firstkey == 0 &&
2352 c->cmd->proc != execCommand))
2353 {
2354 int hashslot;
2355 int error_code;
2356 clusterNode *n = getNodeByQuery(c,c->cmd,c->argv,c->argc,
2357 &hashslot,&error_code);
2358 if (n == NULL || n != server.cluster->myself) {
2359 if (c->cmd->proc == execCommand) {
2360 discardTransaction(c);
2361 } else {
2362 flagTransaction(c);
2363 }
2364 clusterRedirectClient(c,n,hashslot,error_code);
2365 return C_OK;
2366 }
2367 }

2369 /* Handle the maxmemory directive.
2370 *
2371 * First we try to free some memory if possible (if there are volatile
2372 * keys in the dataset). If there are not the only thing we can do
2373 * is returning an error. */
2374 if (server.maxmemory) {
2375 int retval = freeMemoryIfNeeded();
2376 /* freeMemoryIfNeeded may flush slave output buffers. This may result
2377 * into a slave, that may be the active client, to be freed. */
2378 if (server.current_client == NULL) return C_ERR;
2379
2380 /* It was impossible to free enough memory, and the command the client
2381 * is trying to execute is denied during OOM conditions? Error. */
2382 if ((c->cmd->flags & CMD_DENYOOM) && retval == C_ERR) {
2383 flagTransaction(c);
2384 addReply(c, shared.oomerr);
2385 return C_OK;
2386 }
2387 }
2388
2389 /* Don't accept write commands if there are problems persisting on disk
2390 * and if this is a master instance. */
2391 if (((server.stop_writes_on_bgsave_err &&
2392 server.saveparamslen > 0 &&
2393 server.lastbgsave_status == C_ERR) ||
2394 server.aof_last_write_status == C_ERR) &&
2395 server.masterhost == NULL &&
2396 (c->cmd->flags & CMD_WRITE ||
2397 c->cmd->proc == pingCommand))
2398 {
2399 flagTransaction(c);
2400 if (server.aof_last_write_status == C_OK)
2401 addReply(c, shared.bgsaveerr);
2402 else
2403 addReplySds(c,
2404 sdscatprintf(sdsempty(),

2406 strerror(server.aof_last_write_errno)));
2407 return C_OK;
2408 }
2409
2410 /* Don't accept write commands if there are not enough good slaves and
2411 * user configured the min-slaves-to-write option. */
2412 if (server.masterhost == NULL &&
2413 server.repl_min_slaves_to_write &&
2414 server.repl_min_slaves_max_lag &&
2415 c->cmd->flags & CMD_WRITE &&
2416 server.repl_good_slaves_count < server.repl_min_slaves_to_write)
2417 {
2418 flagTransaction(c);
2419 addReply(c, shared.noreplicaserr);
2420 return C_OK;
2421 }
2422
2423 /* Don't accept write commands if this is a read only slave. But
2424 * accept write commands if this is our master. */
2425 if (server.masterhost && server.repl_slave_ro &&
2426 !(c->flags & CLIENT_MASTER) &&
2427 c->cmd->flags & CMD_WRITE)
2428 {
2429 addReply(c, shared.roslaveerr);
2430 return C_OK;
2431 }
2432
2433 /* Only allow SUBSCRIBE and UNSUBSCRIBE in the context of Pub/Sub */
2434 if (c->flags & CLIENT_PUBSUB &&
2435 c->cmd->proc != pingCommand &&
2436 c->cmd->proc != subscribeCommand &&
2437 c->cmd->proc != unsubscribeCommand &&
2438 c->cmd->proc != psubscribeCommand &&
2439 c->cmd->proc != punsubscribeCommand) {
2440 addReplyError(c,"only (P)SUBSCRIBE / (P)UNSUBSCRIBE / PING / QUIT allowed in this context");
2441 return C_OK;
2442 }
2443
2444 /* Only allow INFO and SLAVEOF when slave-serve-stale-data is no and
2445 * we are a slave with a broken link with master. */
2446 if (server.masterhost && server.repl_state != REPL_STATE_CONNECTED &&
2447 server.repl_serve_stale_data == 0 &&
2448 !(c->cmd->flags & CMD_STALE))
2449 {
2450 flagTransaction(c);
2451 addReply(c, shared.masterdownerr);
2452 return C_OK;
2453 }
2454
2455 /* Loading DB? Return an error if the command has not the
2456 * CMD_LOADING flag. */
2457 if (server.loading && !(c->cmd->flags & CMD_LOADING)) {
2458 addReply(c, shared.loadingerr);
2459 return C_OK;
2460 }
2461
2462 /* Lua script too slow? Only allow a limited number of commands. */
2463 if (server.lua_timedout &&
2464 c->cmd->proc != authCommand &&
2465 c->cmd->proc != replconfCommand &&
2466 !(c->cmd->proc == shutdownCommand &&
2467 c->argc == 2 &&
2468 tolower(((char*)c->argv[1]->ptr)[0]) == 'n') &&
2469 !(c->cmd->proc == scriptCommand &&
2470 c->argc == 2 &&
2471 tolower(((char*)c->argv[1]->ptr)[0]) == 'k'))
2472 {
2473 flagTransaction(c);
2474 addReply(c, shared.slowscripterr);
2475 return C_OK;
2476 }
2477
2478 /* Exec the command */
2479 if (c->flags & CLIENT_MULTI &&
2480 c->cmd->proc != execCommand && c->cmd->proc != discardCommand &&
2481 c->cmd->proc != multiCommand && c->cmd->proc != watchCommand)
2482 {
2483 queueMultiCommand(c);
2484 addReply(c,shared.queued);
2485 } else {
2486 call(c,CMD_CALL_FULL);
2487 c->woff = server.master_repl_offset;
2488 if (listLength(server.ready_keys))
2489 handleClientsBlockedOnLists();
2490 }
2491 return C_OK;
2492}

2511int prepareForShutdown(int flags) {
2512 int save = flags & SHUTDOWN_SAVE;
2513 int nosave = flags & SHUTDOWN_NOSAVE;
2514
2515 serverLog(LL_WARNING,"User requested shutdown...");
2516
2517 /* Kill all the Lua debugger forked sessions. */
2518 ldbKillForkedSessions();
2519
2520 /* Kill the saving child if there is a background saving in progress.
2521 We want to avoid race conditions, for instance our saving child may
2522 overwrite the synchronous saving did by SHUTDOWN. */
2523 if (server.rdb_child_pid != -1) {
2524 serverLog(LL_WARNING,"There is a child saving an .rdb. Killing it!");
2525 kill(server.rdb_child_pid,SIGUSR1);
2526 rdbRemoveTempFile(server.rdb_child_pid);
2527 }
2528
2529 if (server.aof_state != AOF_OFF) {

2531 * but contains the full dataset anyway. */
2532 if (server.aof_child_pid != -1) {

2534 * shutdown or else the dataset will be lost. */

2536 serverLog(LL_WARNING, "Writing initial AOF, can't exit.");

2541 kill(server.aof_child_pid,SIGUSR1);
2542 }

2544 serverLog(LL_NOTICE,"Calling fsync() on the AOF file.");
2545 flushAppendOnlyFile(1);
2546 aof_fsync(server.aof_fd);
2547 }
2548
2549 /* Create a new RDB file before exiting. */
2550 if ((server.saveparamslen > 0 && !nosave) || save) {
2551 serverLog(LL_NOTICE,"Saving the final RDB snapshot before exiting.");
2552 /* Snapshotting. Perform a SYNC SAVE and exit */
2553 rdbSaveInfo rsi, *rsiptr;
2554 rsiptr = rdbPopulateSaveInfo(&rsi);
2555 if (rdbSave(server.rdb_filename,rsiptr) != C_OK) {
2556 /* Ooops.. error saving! The best we can do is to continue
2557 * operating. Note that if there was a background saving process,
2558 * in the next cron() Redis will be notified that the background
2559 * saving aborted, handling special stuff like slaves pending for
2560 * synchronization... */
2561 serverLog(LL_WARNING,"Error trying to save the DB, can't exit.");
2562 return C_ERR;
2563 }
2564 }
2565

2567 if (server.daemonize || server.pidfile) {
2568 serverLog(LL_NOTICE,"Removing the pid file.");
2569 unlink(server.pidfile);
2570 }
2571
2572 /* Best effort flush of slave output buffers, so that we hopefully
2573 * send them pending writes. */
2574 flushSlavesOutputBuffers();
2575
2576 /* Close the listening sockets. Apparently this allows faster restarts. */
2577 closeListeningSockets(1);
2578 serverLog(LL_WARNING,"%s is now ready to exit, bye bye...",
2579 server.sentinel_mode ? "Sentinel" : "Redis");
2580 return C_OK;
2581}

2585/* Return zero if strings are the same, non-zero if they are not.
2586 * The comparison is performed in a way that prevents an attacker to obtain
2587 * information about the nature of the strings just monitoring the execution
2588 * time of the function.
2589 *
2590 * Note that limiting the comparison length to strings up to 512 bytes we
2591 * can avoid leaking any information about the password length and any
2592 * possible branch misprediction related leak.
2593 */
2594int time_independent_strcmp(char *a, char *b) {
2595 char bufa[CONFIG_AUTHPASS_MAX_LEN], bufb[CONFIG_AUTHPASS_MAX_LEN];
2596 /* The above two strlen perform len(a) + len(b) operations where either
2597 * a or b are fixed (our password) length, and the difference is only
2598 * relative to the length of the user provided string, so no information
2599 * leak is possible in the following two lines of code. */
2600 unsigned int alen = strlen(a);
2601 unsigned int blen = strlen(b);
2602 unsigned int j;
2603 int diff = 0;
2604
2605 /* We can't compare strings longer than our static buffers.
2606 * Note that this will never pass the first test in practical circumstances
2607 * so there is no info leak. */
2608 if (alen > sizeof(bufa) || blen > sizeof(bufb)) return 1;
2609
2610 memset(bufa,0,sizeof(bufa)); /* Constant time. */
2611 memset(bufb,0,sizeof(bufb)); /* Constant time. */
2612 /* Again the time of the following two copies is proportional to
2613 * len(a) + len(b) so no info is leaked. */
2614 memcpy(bufa,a,alen);
2615 memcpy(bufb,b,blen);
2616
2617 /* Always compare all the chars in the two buffers without
2618 * conditional expressions. */
2619 for (j = 0; j < sizeof(bufa); j++) {
2620 diff |= (bufa[j] ^ bufb[j]);
2621 }
2622 /* Length must be equal as well. */
2623 diff |= alen ^ blen;
2624 return diff; /* If zero strings are the same. */
2625}

2771/* Convert an amount of bytes into a human readable string in the form
2772 * of 100B, 2G, 100M, 4K, and so forth. */
2773void bytesToHuman(char *s, unsigned long long n) {
2774 double d;
2775
2776 if (n < 1024) {
2777 /* Bytes */
2778 sprintf(s,"%lluB",n);
2779 return;
2780 } else if (n < (1024*1024)) {
2781 d = (double)n/(1024);
2782 sprintf(s,"%.2fK",d);
2783 } else if (n < (1024LL*1024*1024)) {
2784 d = (double)n/(1024*1024);
2785 sprintf(s,"%.2fM",d);
2786 } else if (n < (1024LL*1024*1024*1024)) {
2787 d = (double)n/(1024LL*1024*1024);
2788 sprintf(s,"%.2fG",d);
2789 } else if (n < (1024LL*1024*1024*1024*1024)) {
2790 d = (double)n/(1024LL*1024*1024*1024);
2791 sprintf(s,"%.2fT",d);
2792 } else if (n < (1024LL*1024*1024*1024*1024*1024)) {
2793 d = (double)n/(1024LL*1024*1024*1024*1024);
2794 sprintf(s,"%.2fP",d);
2795 } else {
2796 /* Let's hope we never need this */
2797 sprintf(s,"%lluB",n);
2798 }
2799}

3456static void sigShutdownHandler(int sig) {
3457 char *msg;
3458
3459 switch (sig) {
3460 case SIGINT:
3461 msg = "Received SIGINT scheduling shutdown...";
3462 break;
3463 case SIGTERM:
3464 msg = "Received SIGTERM scheduling shutdown...";
3465 break;
3466 default:
3467 msg = "Received shutdown signal, scheduling shutdown...";
3468 };
3469
3470 /* SIGINT is often delivered via Ctrl+C in an interactive session.
3471 * If we receive the signal the second time, we interpret this as
3472 * the user really wanting to quit ASAP without waiting to persist
3473 * on disk. */
3474 if (server.shutdown_asap && sig == SIGINT) {
3475 serverLogFromHandler(LL_WARNING, "You insist... exiting now.");
3476 rdbRemoveTempFile(getpid());
3477 exit(1); /* Exit with an error since this was not a clean shutdown. */
3478 } else if (server.loading) {
3479 exit(0);
3480 }
3481
3482 serverLogFromHandler(LL_WARNING, msg);
3483 server.shutdown_asap = 1;
3484}
