68/* The pool allocator block. Redis Modules can allocate memory via this special
69 * allocator that will automatically release it all once the callback returns.
70 * This means that it can only be used for ephemeral allocations. However
71 * there are two advantages for modules to use this API:
72 *
73 * 1) The memory is automatically released when the callback returns.
74 * 2) This allocator is faster for many small allocations since whole blocks
75 * are allocated, and small pieces returned to the caller just advancing
76 * the index of the allocation.
77 *
78 * Allocations are always rounded to the size of the void pointer in order
79 * to always return aligned memory chunks. */
80
81#define REDISMODULE_POOL_ALLOC_MIN_SIZE (1024*8)
82#define REDISMODULE_POOL_ALLOC_ALIGN (sizeof(void*))
83
84typedef struct RedisModulePoolAllocBlock {
85 uint32_t size;
86 uint32_t used;
87 struct RedisModulePoolAllocBlock *next;
88 char memory[];
89} RedisModulePoolAllocBlock;

272/* Release the chain of blocks used for pool allocations. */
273void poolAllocRelease(RedisModuleCtx *ctx) {
274 RedisModulePoolAllocBlock *head = ctx->pa_head, *next;
275
276 while(head != NULL) {
277 next = head->next;
278 zfree(head);
279 head = next;
280 }
281 ctx->pa_head = NULL;
282}

284/* Return heap allocated memory that will be freed automatically when the
285 * module callback function returns. Mostly suitable for small allocations
286 * that are short living and must be released when the callback returns
287 * anyway. The returned memory is aligned to the architecture word size
288 * if at least word size bytes are requested, otherwise it is just
289 * aligned to the next power of two, so for example a 3 bytes request is
290 * 4 bytes aligned while a 2 bytes request is 2 bytes aligned.
291 *
292 * There is no realloc style function since when this is needed to use the
293 * pool allocator is not a good idea.
294 *
295 * The function returns NULL if `bytes` is 0. */
296void *RM_PoolAlloc(RedisModuleCtx *ctx, size_t bytes) {
297 if (bytes == 0) return NULL;
298 RedisModulePoolAllocBlock *b = ctx->pa_head;
299 size_t left = b ? b->size - b->used : 0;
300
301 /* Fix alignment. */
302 if (left >= bytes) {
303 size_t alignment = REDISMODULE_POOL_ALLOC_ALIGN;
304 while (bytes < alignment && alignment/2 >= bytes) alignment /= 2;
305 if (b->used % alignment)
306 b->used += alignment - (b->used % alignment);
307 left = (b->used > b->size) ? 0 : b->size - b->used;
308 }
309
310 /* Create a new block if needed. */
311 if (left < bytes) {
312 size_t blocksize = REDISMODULE_POOL_ALLOC_MIN_SIZE;
313 if (blocksize < bytes) blocksize = bytes;
314 b = zmalloc(sizeof(*b) + blocksize);
315 b->size = blocksize;
316 b->used = 0;
317 b->next = ctx->pa_head;
318 ctx->pa_head = b;
319 }
320
321 char *retval = b->memory + b->used;
322 b->used += bytes;
323 return retval;
324}

330/* Create an empty key of the specified type. 'kp' must point to a key object
331 * opened for writing where the .value member is set to NULL because the
332 * key was found to be non existing.
333 *
334 * On success REDISMODULE_OK is returned and the key is populated with
335 * the value of the specified type. The function fails and returns
336 * REDISMODULE_ERR if:
337 *
338 * 1) The key is not open for writing.
339 * 2) The key is not empty.
340 * 3) The specified type is unknown.
341 */
342int moduleCreateEmptyKey(RedisModuleKey *key, int type) {
343 robj *obj;
344
345 /* The key must be open for writing and non existing to proceed. */
346 if (!(key->mode & REDISMODULE_WRITE) || key->value)
347 return REDISMODULE_ERR;
348
349 switch(type) {
350 case REDISMODULE_KEYTYPE_LIST:
351 obj = createQuicklistObject();
352 quicklistSetOptions(obj->ptr, server.list_max_ziplist_size,
353 server.list_compress_depth);
354 break;
355 case REDISMODULE_KEYTYPE_ZSET:
356 obj = createZsetZiplistObject();
357 break;
358 case REDISMODULE_KEYTYPE_HASH:
359 obj = createHashObject();
360 break;
361 default: return REDISMODULE_ERR;
362 }
363 dbAdd(key->db,key->key,obj);
364 key->value = obj;
365 return REDISMODULE_OK;
366}
367
368/* This function is called in low-level API implementation functions in order
369 * to check if the value associated with the key remained empty after an
370 * operation that removed elements from an aggregate data type.
371 *
372 * If this happens, the key is deleted from the DB and the key object state
373 * is set to the right one in order to be targeted again by write operations
374 * possibly recreating the key if needed.
375 *
376 * The function returns 1 if the key value object is found empty and is
377 * deleted, otherwise 0 is returned. */
378int moduleDelKeyIfEmpty(RedisModuleKey *key) {
379 if (!(key->mode & REDISMODULE_WRITE) || key->value == NULL) return 0;
380 int isempty;
381 robj *o = key->value;
382
383 switch(o->type) {
384 case OBJ_LIST: isempty = listTypeLength(o) == 0; break;
385 case OBJ_SET: isempty = setTypeSize(o) == 0; break;
386 case OBJ_ZSET: isempty = zsetLength(o) == 0; break;
387 case OBJ_HASH : isempty = hashTypeLength(o) == 0; break;
388 default: isempty = 0;
389 }
390
391 if (isempty) {
392 dbDelete(key->db,key->key);
393 key->value = NULL;
394 return 1;
395 } else {
396 return 0;
397 }
398}

423/* Free the context after the user function was called. */
424void moduleFreeContext(RedisModuleCtx *ctx) {
425 autoMemoryCollect(ctx);
426 poolAllocRelease(ctx);
427 if (ctx->postponed_arrays) {
428 zfree(ctx->postponed_arrays);
429 ctx->postponed_arrays_count = 0;
430 serverLog(LL_WARNING,
431 "API misuse detected in module %s: "
432 "RedisModule_ReplyWithArray(REDISMODULE_POSTPONED_ARRAY_LEN) "
433 "not matched by the same number of RedisModule_SetReplyArrayLen() "
434 "calls.",
435 ctx->module->name);
436 }
437 if (ctx->flags & REDISMODULE_CTX_THREAD_SAFE) freeClient(ctx->client);
438}

440/* Helper function for when a command callback is called, in order to handle
441 * details needed to correctly replicate commands. */
442void moduleHandlePropagationAfterCommandCallback(RedisModuleCtx *ctx) {
443 client *c = ctx->client;
444
445 /* We don't want any automatic propagation here since in modules we handle
446 * replication / AOF propagation in explicit ways. */
447 preventCommandPropagation(c);
448
449 /* Handle the replication of the final EXEC, since whatever a command
450 * emits is always wrappered around MULTI/EXEC. */
451 if (ctx->flags & REDISMODULE_CTX_MULTI_EMITTED) {
452 robj *propargv[1];
453 propargv[0] = createStringObject("EXEC",4);
454 alsoPropagate(server.execCommand,c->db->id,propargv,1,
455 PROPAGATE_AOF|PROPAGATE_REPL);
456 decrRefCount(propargv[0]);
457 }
458}

473/* This function returns the list of keys, with the same interface as the
474 * 'getkeys' function of the native commands, for module commands that exported
475 * the "getkeys-api" flag during the registration. This is done when the
476 * list of keys are not at fixed positions, so that first/last/step cannot
477 * be used.
478 *
479 * In order to accomplish its work, the module command is called, flagging
480 * the context in a way that the command can recognize this is a special
481 * "get keys" call by calling RedisModule_IsKeysPositionRequest(ctx). */
482int *moduleGetCommandKeysViaAPI(struct redisCommand *cmd, robj **argv, int argc, int *numkeys) {
483 RedisModuleCommandProxy *cp = (void*)(unsigned long)cmd->getkeys_proc;
484 RedisModuleCtx ctx = REDISMODULE_CTX_INIT;
485
486 ctx.module = cp->module;
487 ctx.client = NULL;
488 ctx.flags |= REDISMODULE_CTX_KEYS_POS_REQUEST;
489 cp->func(&ctx,(void**)argv,argc);
490 int *res = ctx.keys_pos;
491 if (numkeys) *numkeys = ctx.keys_count;
492 moduleFreeContext(&ctx);
493 return res;
494}

503/* When a module command is called in order to obtain the position of
504 * keys, since it was flagged as "getkeys-api" during the registration,
505 * the command implementation checks for this special call using the
506 * RedisModule_IsKeysPositionRequest() API and uses this function in
507 * order to report keys, like in the following example:
508 *
509 * if (RedisModule_IsKeysPositionRequest(ctx)) {
510 * RedisModule_KeyAtPos(ctx,1);
511 * RedisModule_KeyAtPos(ctx,2);
512 * }
513 *
514 * Note: in the example below the get keys API would not be needed since
515 * keys are at fixed positions. This interface is only used for commands
516 * with a more complex structure. */
517void RM_KeyAtPos(RedisModuleCtx *ctx, int pos) {
518 if (!(ctx->flags & REDISMODULE_CTX_KEYS_POS_REQUEST)) return;
519 if (pos <= 0) return;
520 ctx->keys_pos = zrealloc(ctx->keys_pos,sizeof(int)*(ctx->keys_count+1));
521 ctx->keys_pos[ctx->keys_count++] = pos;
522}

524/* Helper for RM_CreateCommand(). Truns a string representing command
525 * flags into the command flags used by the Redis core.
526 *
527 * It returns the set of flags, or -1 if unknown flags are found. */
528int commandFlagsFromString(char *s) {
529 int count, j;
530 int flags = 0;
531 sds *tokens = sdssplitlen(s,strlen(s)," ",1,&count);
532 for (j = 0; j < count; j++) {
533 char *t = tokens[j];
534 if (!strcasecmp(t,"write")) flags |= CMD_WRITE;
535 else if (!strcasecmp(t,"readonly")) flags |= CMD_READONLY;
536 else if (!strcasecmp(t,"admin")) flags |= CMD_ADMIN;
537 else if (!strcasecmp(t,"deny-oom")) flags |= CMD_DENYOOM;
538 else if (!strcasecmp(t,"deny-script")) flags |= CMD_NOSCRIPT;
539 else if (!strcasecmp(t,"allow-loading")) flags |= CMD_LOADING;
540 else if (!strcasecmp(t,"pubsub")) flags |= CMD_PUBSUB;
541 else if (!strcasecmp(t,"random")) flags |= CMD_RANDOM;
542 else if (!strcasecmp(t,"allow-stale")) flags |= CMD_STALE;
543 else if (!strcasecmp(t,"no-monitor")) flags |= CMD_SKIP_MONITOR;
544 else if (!strcasecmp(t,"fast")) flags |= CMD_FAST;
545 else if (!strcasecmp(t,"getkeys-api")) flags |= CMD_MODULE_GETKEYS;
546 else if (!strcasecmp(t,"no-cluster")) flags |= CMD_MODULE_NO_CLUSTER;
547 else break;
548 }
549 sdsfreesplitres(tokens,count);
550 if (j != count) return -1; /* Some token not processed correctly. */
551 return flags;
552}

554/* Register a new command in the Redis server, that will be handled by
555 * calling the function pointer 'func' using the RedisModule calling
556 * convention. The function returns REDISMODULE_ERR if the specified command
557 * name is already busy or a set of invalid flags were passed, otherwise
558 * REDISMODULE_OK is returned and the new command is registered.
559 *
560 * This function must be called during the initialization of the module
561 * inside the RedisModule_OnLoad() function. Calling this function outside
562 * of the initialization function is not defined.
563 *
564 * The command function type is the following:
565 *
566 * int MyCommand_RedisCommand(RedisModuleCtx *ctx, RedisModuleString **argv, int argc);
567 *
568 * And is supposed to always return REDISMODULE_OK.
569 *
570 * The set of flags 'strflags' specify the behavior of the command, and should
571 * be passed as a C string compoesd of space separated words, like for
572 * example "write deny-oom". The set of flags are:
573 *
574 * * **"write"**: The command may modify the data set (it may also read
575 * from it).
576 * * **"readonly"**: The command returns data from keys but never writes.
577 * * **"admin"**: The command is an administrative command (may change
578 * replication or perform similar tasks).
579 * * **"deny-oom"**: The command may use additional memory and should be
580 * denied during out of memory conditions.
581 * * **"deny-script"**: Don't allow this command in Lua scripts.
582 * * **"allow-loading"**: Allow this command while the server is loading data.
583 * Only commands not interacting with the data set
584 * should be allowed to run in this mode. If not sure
585 * don't use this flag.
586 * * **"pubsub"**: The command publishes things on Pub/Sub channels.
587 * * **"random"**: The command may have different outputs even starting
588 * from the same input arguments and key values.
589 * * **"allow-stale"**: The command is allowed to run on slaves that don't
590 * serve stale data. Don't use if you don't know what
591 * this means.
592 * * **"no-monitor"**: Don't propoagate the command on monitor. Use this if
593 * the command has sensible data among the arguments.
594 * * **"fast"**: The command time complexity is not greater
595 * than O(log(N)) where N is the size of the collection or
596 * anything else representing the normal scalability
597 * issue with the command.
598 * * **"getkeys-api"**: The command implements the interface to return
599 * the arguments that are keys. Used when start/stop/step
600 * is not enough because of the command syntax.
601 * * **"no-cluster"**: The command should not register in Redis Cluster
602 * since is not designed to work with it because, for
603 * example, is unable to report the position of the
604 * keys, programmatically creates key names, or any
605 * other reason.
606 */
607int RM_CreateCommand(RedisModuleCtx *ctx, const char *name, RedisModuleCmdFunc cmdfunc, const char *strflags, int firstkey, int lastkey, int keystep) {
608 int flags = strflags ? commandFlagsFromString((char*)strflags) : 0;
609 if (flags == -1) return REDISMODULE_ERR;
610 if ((flags & CMD_MODULE_NO_CLUSTER) && server.cluster_enabled)
611 return REDISMODULE_ERR;
612
613 struct redisCommand *rediscmd;
614 RedisModuleCommandProxy *cp;
615 sds cmdname = sdsnew(name);
616
617 /* Check if the command name is busy. */
618 if (lookupCommand((char*)name) != NULL) {
619 sdsfree(cmdname);
620 return REDISMODULE_ERR;
621 }
622
623 /* Create a command "proxy", which is a structure that is referenced
624 * in the command table, so that the generic command that works as
625 * binding between modules and Redis, can know what function to call
626 * and what the module is.
627 *
628 * Note that we use the Redis command table 'getkeys_proc' in order to
629 * pass a reference to the command proxy structure. */
630 cp = zmalloc(sizeof(*cp));
631 cp->module = ctx->module;
632 cp->func = cmdfunc;
633 cp->rediscmd = zmalloc(sizeof(*rediscmd));
634 cp->rediscmd->name = cmdname;
635 cp->rediscmd->proc = RedisModuleCommandDispatcher;
636 cp->rediscmd->arity = -1;
637 cp->rediscmd->flags = flags | CMD_MODULE;
638 cp->rediscmd->getkeys_proc = (redisGetKeysProc*)(unsigned long)cp;
639 cp->rediscmd->firstkey = firstkey;
640 cp->rediscmd->lastkey = lastkey;
641 cp->rediscmd->keystep = keystep;
642 cp->rediscmd->microseconds = 0;
643 cp->rediscmd->calls = 0;
644 dictAdd(server.commands,sdsdup(cmdname),cp->rediscmd);
645 dictAdd(server.orig_commands,sdsdup(cmdname),cp->rediscmd);
646 return REDISMODULE_OK;
647}

682/* Add a new object to release automatically when the callback returns. */
683void autoMemoryAdd(RedisModuleCtx *ctx, int type, void *ptr) {
684 if (!(ctx->flags & REDISMODULE_CTX_AUTO_MEMORY)) return;
685 if (ctx->amqueue_used == ctx->amqueue_len) {
686 ctx->amqueue_len *= 2;
687 if (ctx->amqueue_len < 16) ctx->amqueue_len = 16;
688 ctx->amqueue = zrealloc(ctx->amqueue,sizeof(struct AutoMemEntry)*ctx->amqueue_len);
689 }
690 ctx->amqueue[ctx->amqueue_used].type = type;
691 ctx->amqueue[ctx->amqueue_used].ptr = ptr;
692 ctx->amqueue_used++;
693}
694
695/* Mark an object as freed in the auto release queue, so that users can still
696 * free things manually if they want.
697 *
698 * The function returns 1 if the object was actually found in the auto memory
699 * pool, otherwise 0 is returned. */
700int autoMemoryFreed(RedisModuleCtx *ctx, int type, void *ptr) {
701 if (!(ctx->flags & REDISMODULE_CTX_AUTO_MEMORY)) return 0;
702
703 int count = (ctx->amqueue_used+1)/2;
704 for (int j = 0; j < count; j++) {
705 for (int side = 0; side < 2; side++) {
706 /* For side = 0 check right side of the array, for
707 * side = 1 check the left side instead (zig-zag scanning). */
708 int i = (side == 0) ? (ctx->amqueue_used - 1 - j) : j;
709 if (ctx->amqueue[i].type == type &&
710 ctx->amqueue[i].ptr == ptr)
711 {
712 ctx->amqueue[i].type = REDISMODULE_AM_FREED;
713
714 /* Switch the freed element and the last element, to avoid growing
715 * the queue unnecessarily if we allocate/free in a loop */
716 if (i != ctx->amqueue_used-1) {
717 ctx->amqueue[i] = ctx->amqueue[ctx->amqueue_used-1];
718 }
719
720 /* Reduce the size of the queue because we either moved the top
721 * element elsewhere or freed it */
722 ctx->amqueue_used--;
723 return 1;
724 }
725 }
726 }
727 return 0;
728}
729
730/* Release all the objects in queue. */
731void autoMemoryCollect(RedisModuleCtx *ctx) {
732 if (!(ctx->flags & REDISMODULE_CTX_AUTO_MEMORY)) return;
733 /* Clear the AUTO_MEMORY flag from the context, otherwise the functions
734 * we call to free the resources, will try to scan the auto release
735 * queue to mark the entries as freed. */
736 ctx->flags &= ~REDISMODULE_CTX_AUTO_MEMORY;
737 int j;
738 for (j = 0; j < ctx->amqueue_used; j++) {
739 void *ptr = ctx->amqueue[j].ptr;
740 switch(ctx->amqueue[j].type) {
741 case REDISMODULE_AM_STRING: decrRefCount(ptr); break;
742 case REDISMODULE_AM_REPLY: RM_FreeCallReply(ptr); break;
743 case REDISMODULE_AM_KEY: RM_CloseKey(ptr); break;
744 }
745 }
746 ctx->flags |= REDISMODULE_CTX_AUTO_MEMORY;
747 zfree(ctx->amqueue);
748 ctx->amqueue = NULL;
749 ctx->amqueue_len = 0;
750 ctx->amqueue_used = 0;
751}

822/* Every call to this function, will make the string 'str' requiring
823 * an additional call to RedisModule_FreeString() in order to really
824 * free the string. Note that the automatic freeing of the string obtained
825 * enabling modules automatic memory management counts for one
826 * RedisModule_FreeString() call (it is just executed automatically).
827 *
828 * Normally you want to call this function when, at the same time
829 * the following conditions are true:
830 *
831 * 1) You have automatic memory management enabled.
832 * 2) You want to create string objects.
833 * 3) Those string objects you create need to live *after* the callback
834 * function(for example a command implementation) creating them returns.
835 *
836 * Usually you want this in order to store the created string object
837 * into your own data structure, for example when implementing a new data
838 * type.
839 *
840 * Note that when memory management is turned off, you don't need
841 * any call to RetainString() since creating a string will always result
842 * into a string that lives after the callback function returns, if
843 * no FreeString() call is performed. */
844void RM_RetainString(RedisModuleCtx *ctx, RedisModuleString *str) {
845 if (!autoMemoryFreed(ctx,REDISMODULE_AM_STRING,str)) {
846 /* Increment the string reference counting only if we can't
847 * just remove the object from the list of objects that should
848 * be reclaimed. Why we do that, instead of just incrementing
849 * the refcount in any case, and let the automatic FreeString()
850 * call at the end to bring the refcount back at the desired
851 * value? Because this way we ensure that the object refcount
852 * value is 1 (instead of going to 2 to be dropped later to 1)
853 * after the call to this function. This is needed for functions
854 * like RedisModule_StringAppendBuffer() to work. */
855 incrRefCount(str);
856 }
857}

1029 * to `ReplyWith*` style functions must follow in order to emit the elements
1030 * of the array.
1031 *
1032 * When producing arrays with a number of element that is not known beforehand
1033 * the function can be called with the special count
1034 * REDISMODULE_POSTPONED_ARRAY_LEN, and the actual number of elements can be
1035 * later set with RedisModule_ReplySetArrayLength() (which will set the
1036 * latest "open" count if there are multiple ones).
1037 *
1038 * The function always returns REDISMODULE_OK. */
1039int RM_ReplyWithArray(RedisModuleCtx *ctx, long len) {
1040 client *c = moduleGetReplyClient(ctx);
1041 if (c == NULL) return REDISMODULE_OK;
1042 if (len == REDISMODULE_POSTPONED_ARRAY_LEN) {
1043 ctx->postponed_arrays = zrealloc(ctx->postponed_arrays,sizeof(void*)*
1044 (ctx->postponed_arrays_count+1));
1045 ctx->postponed_arrays[ctx->postponed_arrays_count] =
1046 addDeferredMultiBulkLength(c);
1047 ctx->postponed_arrays_count++;
1048 } else {
1049 addReplyMultiBulkLen(c,len);
1050 }
1051 return REDISMODULE_OK;
1052}

1054/* When RedisModule_ReplyWithArray() is used with the argument
1055 * REDISMODULE_POSTPONED_ARRAY_LEN, because we don't know beforehand the number
1056 * of items we are going to output as elements of the array, this function
1057 * will take care to set the array length.
1058 *
1059 * Since it is possible to have multiple array replies pending with unknown
1060 * length, this function guarantees to always set the latest array length
1061 * that was created in a postponed way.
1062 *
1063 * For example in order to output an array like [1,[10,20,30]] we
1064 * could write:
1065 *
1066 * RedisModule_ReplyWithArray(ctx,REDISMODULE_POSTPONED_ARRAY_LEN);
1067 * RedisModule_ReplyWithLongLong(ctx,1);
1068 * RedisModule_ReplyWithArray(ctx,REDISMODULE_POSTPONED_ARRAY_LEN);
1069 * RedisModule_ReplyWithLongLong(ctx,10);
1070 * RedisModule_ReplyWithLongLong(ctx,20);
1071 * RedisModule_ReplyWithLongLong(ctx,30);
1072 * RedisModule_ReplySetArrayLength(ctx,3); // Set len of 10,20,30 array.
1073 * RedisModule_ReplySetArrayLength(ctx,2); // Set len of top array
1074 *
1075 * Note that in the above example there is no reason to postpone the array
1076 * length, since we produce a fixed number of elements, but in the practice
1077 * the code may use an interator or other ways of creating the output so
1078 * that is not easy to calculate in advance the number of elements.
1079 */
1080void RM_ReplySetArrayLength(RedisModuleCtx *ctx, long len) {
1081 client *c = moduleGetReplyClient(ctx);
1082 if (c == NULL) return;
1083 if (ctx->postponed_arrays_count == 0) {
1084 serverLog(LL_WARNING,
1085 "API misuse detected in module %s: "
1086 "RedisModule_ReplySetArrayLength() called without previous "
1087 "RedisModule_ReplyWithArray(ctx,REDISMODULE_POSTPONED_ARRAY_LEN) "
1088 "call.", ctx->module->name);
1089 return;
1090 }
1091 ctx->postponed_arrays_count--;
1092 setDeferredMultiBulkLength(c,
1093 ctx->postponed_arrays[ctx->postponed_arrays_count],
1094 len);
1095 if (ctx->postponed_arrays_count == 0) {
1096 zfree(ctx->postponed_arrays);
1097 ctx->postponed_arrays = NULL;
1098 }
1099}

1177/* Replicate the specified command and arguments to slaves and AOF, as effect
1178 * of execution of the calling command implementation.
1179 *
1180 * The replicated commands are always wrapped into the MULTI/EXEC that
1181 * contains all the commands replicated in a given module command
1182 * execution. However the commands replicated with RedisModule_Call()
1183 * are the first items, the ones replicated with RedisModule_Replicate()
1184 * will all follow before the EXEC.
1185 *
1186 * Modules should try to use one interface or the other.
1187 *
1188 * This command follows exactly the same interface of RedisModule_Call(),
1189 * so a set of format specifiers must be passed, followed by arguments
1190 * matching the provided format specifiers.
1191 *
1192 * Please refer to RedisModule_Call() for more information.
1193 *
1194 * The command returns REDISMODULE_ERR if the format specifiers are invalid
1195 * or the command name does not belong to a known command. */
1196int RM_Replicate(RedisModuleCtx *ctx, const char *cmdname, const char *fmt, ...) {
1197 struct redisCommand *cmd;
1198 robj **argv = NULL;
1199 int argc = 0, flags = 0, j;
1200 va_list ap;
1201
1202 cmd = lookupCommandByCString((char*)cmdname);
1203 if (!cmd) return REDISMODULE_ERR;
1204
1205 /* Create the client and dispatch the command. */
1206 va_start(ap, fmt);
1207 argv = moduleCreateArgvFromUserFormat(cmdname,fmt,&argc,&flags,ap);
1208 va_end(ap);
1209 if (argv == NULL) return REDISMODULE_ERR;
1210
1211 /* Replicate! */

1494/* Prepare the key associated string value for DMA access, and returns
1495 * a pointer and size (by reference), that the user can use to read or
1496 * modify the string in-place accessing it directly via pointer.
1497 *
1498 * The 'mode' is composed by bitwise OR-ing the following flags:
1499 *
1500 * REDISMODULE_READ -- Read access
1501 * REDISMODULE_WRITE -- Write access
1502 *
1503 * If the DMA is not requested for writing, the pointer returned should
1504 * only be accessed in a read-only fashion.
1505 *
1506 * On error (wrong type) NULL is returned.
1507 *
1508 * DMA access rules:
1509 *
1510 * 1. No other key writing function should be called since the moment
1511 * the pointer is obtained, for all the time we want to use DMA access
1512 * to read or modify the string.
1513 *
1514 * 2. Each time RM_StringTruncate() is called, to continue with the DMA
1515 * access, RM_StringDMA() should be called again to re-obtain
1516 * a new pointer and length.
1517 *
1518 * 3. If the returned pointer is not NULL, but the length is zero, no
1519 * byte can be touched (the string is empty, or the key itself is empty)
1520 * so a RM_StringTruncate() call should be used if there is to enlarge
1521 * the string, and later call StringDMA() again to get the pointer.
1522 */
1523char *RM_StringDMA(RedisModuleKey *key, size_t *len, int mode) {
1524 /* We need to return *some* pointer for empty keys, we just return
1525 * a string literal pointer, that is the advantage to be mapped into
1526 * a read only memory page, so the module will segfault if a write
1527 * attempt is performed. */
1528 char *emptystring = "<dma-empty-string>";
1529 if (key->value == NULL) {
1530 *len = 0;
1531 return emptystring;
1532 }
1533
1534 if (key->value->type != OBJ_STRING) return NULL;
1535
1536 /* For write access, and even for read access if the object is encoded,
1537 * we unshare the string (that has the side effect of decoding it). */
1538 if ((mode & REDISMODULE_WRITE) || key->value->encoding != OBJ_ENCODING_RAW)
1539 key->value = dbUnshareStringValue(key->db, key->key, key->value);
1540
1541 *len = sdslen(key->value->ptr);
1542 return key->value->ptr;
1543}
1544
1545/* If the string is open for writing and is of string type, resize it, padding
1546 * with zero bytes if the new length is greater than the old one.
1547 *
1548 * After this call, RM_StringDMA() must be called again to continue
1549 * DMA access with the new pointer.
1550 *
1551 * The function returns REDISMODULE_OK on success, and REDISMODULE_ERR on
1552 * error, that is, the key is not open for writing, is not a string
1553 * or resizing for more than 512 MB is requested.
1554 *
1555 * If the key is empty, a string key is created with the new string value
1556 * unless the new length value requested is zero. */
1557int RM_StringTruncate(RedisModuleKey *key, size_t newlen) {
1558 if (!(key->mode & REDISMODULE_WRITE)) return REDISMODULE_ERR;
1559 if (key->value && key->value->type != OBJ_STRING) return REDISMODULE_ERR;
1560 if (newlen > 512*1024*1024) return REDISMODULE_ERR;
1561
1562 /* Empty key and new len set to 0. Just return REDISMODULE_OK without
1563 * doing anything. */
1564 if (key->value == NULL && newlen == 0) return REDISMODULE_OK;
1565
1566 if (key->value == NULL) {
1567 /* Empty key: create it with the new size. */
1568 robj *o = createObject(OBJ_STRING,sdsnewlen(NULL, newlen));
1569 setKey(key->db,key->key,o);
1570 key->value = o;
1571 decrRefCount(o);
1572 } else {
1573 /* Unshare and resize. */
1574 key->value = dbUnshareStringValue(key->db, key->key, key->value);
1575 size_t curlen = sdslen(key->value->ptr);
1576 if (newlen > curlen) {
1577 key->value->ptr = sdsgrowzero(key->value->ptr,newlen);
1578 } else if (newlen < curlen) {
1579 sdsrange(key->value->ptr,0,newlen-1);
1580 /* If the string is too wasteful, reallocate it. */
1581 if (sdslen(key->value->ptr) < sdsavail(key->value->ptr))
1582 key->value->ptr = sdsRemoveFreeSpace(key->value->ptr);
1583 }
1584 }
1585 return REDISMODULE_OK;
1586}

1651 * setup for writing.

1653 * Additional flags can be passed to the function via a pointer, the flags
1654 * are both used to receive input and to communicate state when the function
1655 * returns. 'flagsptr' can be NULL if no special flags are used.
1656 *
1657 * The input flags are:
1658 *
1659 * REDISMODULE_ZADD_XX: Element must already exist. Do nothing otherwise.
1660 * REDISMODULE_ZADD_NX: Element must not exist. Do nothing otherwise.
1661 *
1662 * The output flags are:
1663 *
1664 * REDISMODULE_ZADD_ADDED: The new element was added to the sorted set.
1665 * REDISMODULE_ZADD_UPDATED: The score of the element was updated.
1666 * REDISMODULE_ZADD_NOP: No operation was performed because XX or NX flags.
1667 *
1668 * On success the function returns REDISMODULE_OK. On the following errors
1669 * REDISMODULE_ERR is returned:
1670 *
1671 * * The key was not opened for writing.
1672 * * The key is of the wrong type.
1673 * * 'score' double value is not a number (NaN).
1674 */

1678 if (key->value && key->value->type != OBJ_ZSET) return REDISMODULE_ERR;
1679 if (key->value == NULL) moduleCreateEmptyKey(key,REDISMODULE_KEYTYPE_ZSET);
1680 if (flagsptr) flags = RM_ZsetAddFlagsToCoreFlags(*flagsptr);
1681 if (zsetAdd(key->value,score,ele->ptr,&flags,NULL) == 0) {
1682 if (flagsptr) *flagsptr = 0;
1683 return REDISMODULE_ERR;
1684 }
1685 if (flagsptr) *flagsptr = RM_ZsetAddFlagsFromCoreFlags(flags);
1686 return REDISMODULE_OK;
1687}

1690 * a new score, the score of the existing element is incremented, or if the
1691 * element does not already exist, it is added assuming the old score was
1692 * zero.
1693 *
1694 * The input and output flags, and the return value, have the same exact
1695 * meaning, with the only difference that this function will return
1696 * REDISMODULE_ERR even when 'score' is a valid double number, but adding it
1697 * to the existing score resuts into a NaN (not a number) condition.

1700 * with the new score of the element after the increment, if no error

1706 if (key->value == NULL) moduleCreateEmptyKey(key,REDISMODULE_KEYTYPE_ZSET);
1707 if (flagsptr) flags = RM_ZsetAddFlagsToCoreFlags(*flagsptr);

1716 return REDISMODULE_ERR;
1717 }

1719 return REDISMODULE_OK;

1942/* Go to the next element of the sorted set iterator. Returns 1 if there was
1943 * a next element, 0 if we are already at the latest element or the range
1944 * does not include any item at all. */
1945int RM_ZsetRangeNext(RedisModuleKey *key) {
1946 if (!key->ztype || !key->zcurrent) return 0; /* No active iterator. */
1947
1948 if (key->value->encoding == OBJ_ENCODING_ZIPLIST) {
1949 unsigned char *zl = key->value->ptr;
1950 unsigned char *eptr = key->zcurrent;
1951 unsigned char *next;
1952 next = ziplistNext(zl,eptr); /* Skip element. */
1953 if (next) next = ziplistNext(zl,next); /* Skip score. */
1954 if (next == NULL) {
1955 key->zer = 1;
1956 return 0;
1957 } else {
1958 /* Are we still within the range? */
1959 if (key->ztype == REDISMODULE_ZSET_RANGE_SCORE) {
1960 /* Fetch the next element score for the
1961 * range check. */
1962 unsigned char *saved_next = next;
1963 next = ziplistNext(zl,next); /* Skip next element. */
1964 double score = zzlGetScore(next); /* Obtain the next score. */
1965 if (!zslValueLteMax(score,&key->zrs)) {
1966 key->zer = 1;
1967 return 0;
1968 }
1969 next = saved_next;
1970 } else if (key->ztype == REDISMODULE_ZSET_RANGE_LEX) {
1971 if (!zzlLexValueLteMax(next,&key->zlrs)) {
1972 key->zer = 1;
1973 return 0;
1974 }
1975 }
1976 key->zcurrent = next;
1977 return 1;
1978 }
1979 } else if (key->value->encoding == OBJ_ENCODING_SKIPLIST) {
1980 zskiplistNode *ln = key->zcurrent, *next = ln->level[0].forward;
1981 if (next == NULL) {
1982 key->zer = 1;
1983 return 0;
1984 } else {
1985 /* Are we still within the range? */
1986 if (key->ztype == REDISMODULE_ZSET_RANGE_SCORE &&
1987 !zslValueLteMax(next->score,&key->zrs))
1988 {
1989 key->zer = 1;
1990 return 0;
1991 } else if (key->ztype == REDISMODULE_ZSET_RANGE_LEX) {
1992 if (!zslLexValueLteMax(next->ele,&key->zlrs)) {
1993 key->zer = 1;
1994 return 0;
1995 }
1996 }
1997 key->zcurrent = next;
1998 return 1;
1999 }
2000 } else {
2001 serverPanic("Unsupported zset encoding");
2002 }
2003}

2072/* Set the field of the specified hash field to the specified value.
2073 * If the key is an empty key open for writing, it is created with an empty
2074 * hash value, in order to set the specified field.
2075 *
2076 * The function is variadic and the user must specify pairs of field
2077 * names and values, both as RedisModuleString pointers (unless the
2078 * CFIELD option is set, see later).
2079 *
2080 * Example to set the hash argv[1] to the value argv[2]:
2081 *
2082 * RedisModule_HashSet(key,REDISMODULE_HASH_NONE,argv[1],argv[2],NULL);
2083 *
2084 * The function can also be used in order to delete fields (if they exist)
2085 * by setting them to the specified value of REDISMODULE_HASH_DELETE:
2086 *
2087 * RedisModule_HashSet(key,REDISMODULE_HASH_NONE,argv[1],
2088 * REDISMODULE_HASH_DELETE,NULL);
2089 *
2090 * The behavior of the command changes with the specified flags, that can be
2091 * set to REDISMODULE_HASH_NONE if no special behavior is needed.
2092 *
2093 * REDISMODULE_HASH_NX: The operation is performed only if the field was not
2094 * already existing in the hash.
2095 * REDISMODULE_HASH_XX: The operation is performed only if the field was
2096 * already existing, so that a new value could be
2097 * associated to an existing filed, but no new fields
2098 * are created.
2099 * REDISMODULE_HASH_CFIELDS: The field names passed are null terminated C
2100 * strings instead of RedisModuleString objects.
2101 *
2102 * Unless NX is specified, the command overwrites the old field value with
2103 * the new one.
2104 *
2105 * When using REDISMODULE_HASH_CFIELDS, field names are reported using
2106 * normal C strings, so for example to delete the field "foo" the following
2107 * code can be used:
2108 *
2109 * RedisModule_HashSet(key,REDISMODULE_HASH_CFIELDS,"foo",
2110 * REDISMODULE_HASH_DELETE,NULL);
2111 *
2112 * Return value:
2113 *
2114 * The number of fields updated (that may be less than the number of fields
2115 * specified because of the XX or NX options).
2116 *
2117 * In the following case the return value is always zero:
2118 *
2119 * * The key was not open for writing.
2120 * * The key was associated with a non Hash value.
2121 */
2122int RM_HashSet(RedisModuleKey *key, int flags, ...) {
2123 va_list ap;
2124 if (!(key->mode & REDISMODULE_WRITE)) return 0;
2125 if (key->value && key->value->type != OBJ_HASH) return 0;
2126 if (key->value == NULL) moduleCreateEmptyKey(key,REDISMODULE_KEYTYPE_HASH);
2127
2128 int updated = 0;
2129 va_start(ap, flags);
2130 while(1) {
2131 RedisModuleString *field, *value;
2132 /* Get the field and value objects. */
2133 if (flags & REDISMODULE_HASH_CFIELDS) {
2134 char *cfield = va_arg(ap,char*);
2135 if (cfield == NULL) break;
2136 field = createRawStringObject(cfield,strlen(cfield));
2137 } else {
2138 field = va_arg(ap,RedisModuleString*);
2139 if (field == NULL) break;
2140 }
2141 value = va_arg(ap,RedisModuleString*);
2142
2143 /* Handle XX and NX */
2144 if (flags & (REDISMODULE_HASH_XX|REDISMODULE_HASH_NX)) {
2145 int exists = hashTypeExists(key->value, field->ptr);
2146 if (((flags & REDISMODULE_HASH_XX) && !exists) ||
2147 ((flags & REDISMODULE_HASH_NX) && exists))
2148 {
2149 if (flags & REDISMODULE_HASH_CFIELDS) decrRefCount(field);
2150 continue;
2151 }
2152 }
2153
2154 /* Handle deletion if value is REDISMODULE_HASH_DELETE. */
2155 if (value == REDISMODULE_HASH_DELETE) {
2156 updated += hashTypeDelete(key->value, field->ptr);
2157 if (flags & REDISMODULE_HASH_CFIELDS) decrRefCount(field);
2158 continue;
2159 }
2160
2161 int low_flags = HASH_SET_COPY;
2162 /* If CFIELDS is active, we can pass the ownership of the
2163 * SDS object to the low level function that sets the field
2164 * to avoid a useless copy. */
2165 if (flags & REDISMODULE_HASH_CFIELDS)
2166 low_flags |= HASH_SET_TAKE_FIELD;
2167 updated += hashTypeSet(key->value, field->ptr, value->ptr, low_flags);
2168
2169 /* If CFIELDS is active, SDS string ownership is now of hashTypeSet(),
2170 * however we still have to release the 'field' object shell. */
2171 if (flags & REDISMODULE_HASH_CFIELDS) {
2172 field->ptr = NULL; /* Prevent the SDS string from being freed. */
2173 decrRefCount(field);
2174 }
2175 }
2176 va_end(ap);
2177 moduleDelKeyIfEmpty(key);
2178 return updated;
2179}

2363 if (arraylen == -1) {
2364 reply->protolen = p-proto;
2365 reply->type = REDISMODULE_REPLY_NULL;
2366 return;
2367 }
2368
2369 reply->val.array = zmalloc(sizeof(RedisModuleCallReply)*arraylen);
2370 reply->len = arraylen;
2371 for (j = 0; j < arraylen; j++) {
2372 RedisModuleCallReply *ele = reply->val.array+j;
2373 ele->flags = REDISMODULE_REPLYFLAG_NESTED |
2374 REDISMODULE_REPLYFLAG_TOPARSE;
2375 ele->proto = p;
2376 ele->ctx = reply->ctx;
2377 moduleParseCallReply(ele);
2378 p += ele->protolen;
2379 }
2380 reply->protolen = p-proto;
2381 reply->type = REDISMODULE_REPLY_ARRAY;
2382}

2384/* Free a Call reply and all the nested replies it contains if it's an
2385 * array. */
2386void RM_FreeCallReply_Rec(RedisModuleCallReply *reply, int freenested){
2387 /* Don't free nested replies by default: the user must always free the
2388 * toplevel reply. However be gentle and don't crash if the module
2389 * misuses the API. */
2390 if (!freenested && reply->flags & REDISMODULE_REPLYFLAG_NESTED) return;
2391
2392 if (!(reply->flags & REDISMODULE_REPLYFLAG_TOPARSE)) {
2393 if (reply->type == REDISMODULE_REPLY_ARRAY) {
2394 size_t j;
2395 for (j = 0; j < reply->len; j++)
2396 RM_FreeCallReply_Rec(reply->val.array+j,1);
2397 zfree(reply->val.array);
2398 }
2399 }

2402 * references the parent reply->proto buffer), nor the structure
2403 * itself which is allocated as an array of structures, and is freed
2404 * when the array value is released. */
2405 if (!(reply->flags & REDISMODULE_REPLYFLAG_NESTED)) {
2406 if (reply->proto) sdsfree(reply->proto);
2407 zfree(reply);
2408 }
2409}

2482/* Returns an array of robj pointers, and populates *argc with the number
2483 * of items, by parsing the format specifier "fmt" as described for
2484 * the RM_Call(), RM_Replicate() and other module APIs.
2485 *
2486 * The integer pointed by 'flags' is populated with flags according
2487 * to special modifiers in "fmt". For now only one exists:
2488 *
2489 * "!" -> REDISMODULE_ARGV_REPLICATE
2490 *
2491 * On error (format specifier error) NULL is returned and nothing is
2492 * allocated. On success the argument vector is returned. */
2493
2494#define REDISMODULE_ARGV_REPLICATE (1<<0)
2495
2496robj **moduleCreateArgvFromUserFormat(const char *cmdname, const char *fmt, int *argcp, int *flags, va_list ap) {
2497 int argc = 0, argv_size, j;
2498 robj **argv = NULL;
2499
2500 /* As a first guess to avoid useless reallocations, size argv to
2501 * hold one argument for each char specifier in 'fmt'. */
2502 argv_size = strlen(fmt)+1; /* +1 because of the command name. */
2503 argv = zrealloc(argv,sizeof(robj*)*argv_size);
2504
2505 /* Build the arguments vector based on the format specifier. */
2506 argv[0] = createStringObject(cmdname,strlen(cmdname));
2507 argc++;
2508
2509 /* Create the client and dispatch the command. */
2510 const char *p = fmt;
2511 while(*p) {
2512 if (*p == 'c') {
2513 char *cstr = va_arg(ap,char*);
2514 argv[argc++] = createStringObject(cstr,strlen(cstr));
2515 } else if (*p == 's') {
2516 robj *obj = va_arg(ap,void*);
2517 argv[argc++] = obj;
2518 incrRefCount(obj);
2519 } else if (*p == 'b') {
2520 char *buf = va_arg(ap,char*);
2521 size_t len = va_arg(ap,size_t);
2522 argv[argc++] = createStringObject(buf,len);
2523 } else if (*p == 'l') {
2524 long ll = va_arg(ap,long long);
2525 argv[argc++] = createObject(OBJ_STRING,sdsfromlonglong(ll));
2526 } else if (*p == 'v') {
2527 /* A vector of strings */
2528 robj **v = va_arg(ap, void*);
2529 size_t vlen = va_arg(ap, size_t);
2530
2531 /* We need to grow argv to hold the vector's elements.
2532 * We resize by vector_len-1 elements, because we held
2533 * one element in argv for the vector already */
2534 argv_size += vlen-1;
2535 argv = zrealloc(argv,sizeof(robj*)*argv_size);
2536
2537 size_t i = 0;
2538 for (i = 0; i < vlen; i++) {
2539 incrRefCount(v[i]);
2540 argv[argc++] = v[i];
2541 }
2542 } else if (*p == '!') {
2543 if (flags) (*flags) |= REDISMODULE_ARGV_REPLICATE;
2544 } else {
2545 goto fmterr;
2546 }
2547 p++;
2548 }
2549 *argcp = argc;
2550 return argv;
2551
2552fmterr:
2553 for (j = 0; j < argc; j++)
2554 decrRefCount(argv[j]);
2555 zfree(argv);
2556 return NULL;
2557}

2559/* Exported API to call any Redis command from modules.
2560 * On success a RedisModuleCallReply object is returned, otherwise
2561 * NULL is returned and errno is set to the following values:
2562 *
2563 * EINVAL: command non existing, wrong arity, wrong format specifier.
2564 * EPERM: operation in Cluster instance with key in non local slot. */
2565RedisModuleCallReply *RM_Call(RedisModuleCtx *ctx, const char *cmdname, const char *fmt, ...) {
2566 struct redisCommand *cmd;
2567 client *c = NULL;
2568 robj **argv = NULL;
2569 int argc = 0, flags = 0;
2570 va_list ap;
2571 RedisModuleCallReply *reply = NULL;
2572 int replicate = 0; /* Replicate this command? */
2573
2574 cmd = lookupCommandByCString((char*)cmdname);
2575 if (!cmd) {
2576 errno = EINVAL;
2577 return NULL;
2578 }
2579
2580 /* Create the client and dispatch the command. */
2581 va_start(ap, fmt);
2582 c = createClient(-1);
2583 argv = moduleCreateArgvFromUserFormat(cmdname,fmt,&argc,&flags,ap);
2584 replicate = flags & REDISMODULE_ARGV_REPLICATE;
2585 va_end(ap);
2586
2587 /* Setup our fake client for command execution. */
2588 c->flags |= CLIENT_MODULE;
2589 c->db = ctx->client->db;
2590 c->argv = argv;
2591 c->argc = argc;
2592 c->cmd = c->lastcmd = cmd;
2593 /* We handle the above format error only when the client is setup so that
2594 * we can free it normally. */
2595 if (argv == NULL) goto cleanup;
2596
2597 /* Basic arity checks. */
2598 if ((cmd->arity > 0 && cmd->arity != argc) || (argc < -cmd->arity)) {
2599 errno = EINVAL;
2600 goto cleanup;
2601 }
2602
2603 /* If this is a Redis Cluster node, we need to make sure the module is not
2604 * trying to access non-local keys, with the exception of commands
2605 * received from our master. */
2606 if (server.cluster_enabled && !(ctx->client->flags & CLIENT_MASTER)) {
2607 /* Duplicate relevant flags in the module client. */
2608 c->flags &= ~(CLIENT_READONLY|CLIENT_ASKING);
2609 c->flags |= ctx->client->flags & (CLIENT_READONLY|CLIENT_ASKING);
2610 if (getNodeByQuery(c,c->cmd,c->argv,c->argc,NULL,NULL) !=
2611 server.cluster->myself)
2612 {
2613 errno = EPERM;
2614 goto cleanup;
2615 }
2616 }
2617
2618 /* If we are using single commands replication, we need to wrap what
2619 * we propagate into a MULTI/EXEC block, so that it will be atomic like
2620 * a Lua script in the context of AOF and slaves. */
2621 if (replicate) moduleReplicateMultiIfNeeded(ctx);
2622
2623 /* Run the command */
2624 int call_flags = CMD_CALL_SLOWLOG | CMD_CALL_STATS;
2625 if (replicate) {
2626 call_flags |= CMD_CALL_PROPAGATE_AOF;
2627 call_flags |= CMD_CALL_PROPAGATE_REPL;
2628 }
2629 call(c,call_flags);
2630
2631 /* Convert the result of the Redis command into a suitable Lua type.
2632 * The first thing we need is to create a single string from the client
2633 * output buffers. */
2634 sds proto = sdsnewlen(c->buf,c->bufpos);
2635 c->bufpos = 0;
2636 while(listLength(c->reply)) {
2637 sds o = listNodeValue(listFirst(c->reply));
2638
2639 proto = sdscatsds(proto,o);
2640 listDelNode(c->reply,listFirst(c->reply));
2641 }
2642 reply = moduleCreateCallReplyFromProto(ctx,proto);
2643 autoMemoryAdd(ctx,REDISMODULE_AM_REPLY,reply);
2644
2645cleanup:
2646 freeClient(c);
2647 return reply;
2648}

2657/* --------------------------------------------------------------------------
2658 * Modules data types
2659 *
2660 * When String DMA or using existing data structures is not enough, it is
2661 * possible to create new data types from scratch and export them to
2662 * Redis. The module must provide a set of callbacks for handling the
2663 * new values exported (for example in order to provide RDB saving/loading,
2664 * AOF rewrite, and so forth). In this section we define this API.
2665 * -------------------------------------------------------------------------- */
2666
2667/* Turn a 9 chars name in the specified charset and a 10 bit encver into
2668 * a single 64 bit unsigned integer that represents this exact module name
2669 * and version. This final number is called a "type ID" and is used when
2670 * writing module exported values to RDB files, in order to re-associate the
2671 * value to the right module to load them during RDB loading.
2672 *
2673 * If the string is not of the right length or the charset is wrong, or
2674 * if encver is outside the unsigned 10 bit integer range, 0 is returned,
2675 * otherwise the function returns the right type ID.
2676 *
2677 * The resulting 64 bit integer is composed as follows:
2678 *
2679 * (high order bits) 6|6|6|6|6|6|6|6|6|10 (low order bits)
2680 *
2681 * The first 6 bits value is the first character, name[0], while the last
2682 * 6 bits value, immediately before the 10 bits integer, is name[8].
2683 * The last 10 bits are the encoding version.
2684 *
2685 * Note that a name and encver combo of "AAAAAAAAA" and 0, will produce
2686 * zero as return value, that is the same we use to signal errors, thus
2687 * this combination is invalid, and also useless since type names should
2688 * try to be vary to avoid collisions. */
2689
2690const char *ModuleTypeNameCharSet =
2691 "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
2692 "abcdefghijklmnopqrstuvwxyz"
2693 "0123456789-_";
2694
2695uint64_t moduleTypeEncodeId(const char *name, int encver) {
2696 /* We use 64 symbols so that we can map each character into 6 bits
2697 * of the final output. */
2698 const char *cset = ModuleTypeNameCharSet;
2699 if (strlen(name) != 9) return 0;
2700 if (encver < 0 || encver > 1023) return 0;
2701
2702 uint64_t id = 0;
2703 for (int j = 0; j < 9; j++) {
2704 char *p = strchr(cset,name[j]);
2705 if (!p) return 0;
2706 unsigned long pos = p-cset;
2707 id = (id << 6) | pos;
2708 }
2709 id = (id << 10) | encver;
2710 return id;
2711}

2713/* Search, in the list of exported data types of all the modules registered,
2714 * a type with the same name as the one given. Returns the moduleType
2715 * structure pointer if such a module is found, or NULL otherwise. */
2716moduleType *moduleTypeLookupModuleByName(const char *name) {
2717 dictIterator *di = dictGetIterator(modules);
2718 dictEntry *de;
2719
2720 while ((de = dictNext(di)) != NULL) {
2721 struct RedisModule *module = dictGetVal(de);
2722 listIter li;
2723 listNode *ln;
2724
2725 listRewind(module->types,&li);
2726 while((ln = listNext(&li))) {
2727 moduleType *mt = ln->value;
2728 if (memcmp(name,mt->name,sizeof(mt->name)) == 0) {
2729 dictReleaseIterator(di);
2730 return mt;
2731 }
2732 }
2733 }
2734 dictReleaseIterator(di);
2735 return NULL;
2736}

2738/* Lookup a module by ID, with caching. This function is used during RDB
2739 * loading. Modules exporting data types should never be able to unload, so
2740 * our cache does not need to expire. */
2741#define MODULE_LOOKUP_CACHE_SIZE 3
2742
2743moduleType *moduleTypeLookupModuleByID(uint64_t id) {
2744 static struct {
2745 uint64_t id;
2746 moduleType *mt;
2747 } cache[MODULE_LOOKUP_CACHE_SIZE];
2748
2749 /* Search in cache to start. */
2750 int j;
2751 for (j = 0; j < MODULE_LOOKUP_CACHE_SIZE && cache[j].mt != NULL; j++)
2752 if (cache[j].id == id) return cache[j].mt;
2753
2754 /* Slow module by module lookup. */
2755 moduleType *mt = NULL;
2756 dictIterator *di = dictGetIterator(modules);
2757 dictEntry *de;
2758
2759 while ((de = dictNext(di)) != NULL && mt == NULL) {
2760 struct RedisModule *module = dictGetVal(de);
2761 listIter li;
2762 listNode *ln;
2763
2764 listRewind(module->types,&li);
2765 while((ln = listNext(&li))) {
2766 moduleType *this_mt = ln->value;
2767 /* Compare only the 54 bit module identifier and not the
2768 * encoding version. */
2769 if (this_mt->id >> 10 == id >> 10) {
2770 mt = this_mt;
2771 break;
2772 }
2773 }
2774 }
2775 dictReleaseIterator(di);
2776
2777 /* Add to cache if possible. */
2778 if (mt && j < MODULE_LOOKUP_CACHE_SIZE) {
2779 cache[j].id = id;
2780 cache[j].mt = mt;
2781 }
2782 return mt;
2783}

2785/* Turn an (unresolved) module ID into a type name, to show the user an
2786 * error when RDB files contain module data we can't load.
2787 * The buffer pointed by 'name' must be 10 bytes at least. The function will
2788 * fill it with a null terminated module name. */
2789void moduleTypeNameByID(char *name, uint64_t moduleid) {
2790 const char *cset = ModuleTypeNameCharSet;
2791
2792 name[9] = '\0';
2793 char *p = name+8;
2794 moduleid >>= 10;
2795 for (int j = 0; j < 9; j++) {
2796 *p-- = cset[moduleid & 63];
2797 moduleid >>= 6;
2798 }
2799}

2801/* Register a new data type exported by the module. The parameters are the
2802 * following. Please for in depth documentation check the modules API
2803 * documentation, especially the TYPES.md file.
2804 *
2805 * * **name**: A 9 characters data type name that MUST be unique in the Redis
2806 * Modules ecosystem. Be creative... and there will be no collisions. Use
2807 * the charset A-Z a-z 9-0, plus the two "-_" characters. A good
2808 * idea is to use, for example `<typename>-<vendor>`. For example
2809 * "tree-AntZ" may mean "Tree data structure by @antirez". To use both
2810 * lower case and upper case letters helps in order to prevent collisions.
2811 * * **encver**: Encoding version, which is, the version of the serialization
2812 * that a module used in order to persist data. As long as the "name"
2813 * matches, the RDB loading will be dispatched to the type callbacks
2814 * whatever 'encver' is used, however the module can understand if
2815 * the encoding it must load are of an older version of the module.
2816 * For example the module "tree-AntZ" initially used encver=0. Later
2817 * after an upgrade, it started to serialize data in a different format
2818 * and to register the type with encver=1. However this module may
2819 * still load old data produced by an older version if the rdb_load
2820 * callback is able to check the encver value and act accordingly.
2821 * The encver must be a positive value between 0 and 1023.
2822 * * **typemethods_ptr** is a pointer to a RedisModuleTypeMethods structure
2823 * that should be populated with the methods callbacks and structure
2824 * version, like in the following example:
2825 *
2826 * RedisModuleTypeMethods tm = {
2827 * .version = REDISMODULE_TYPE_METHOD_VERSION,
2828 * .rdb_load = myType_RDBLoadCallBack,
2829 * .rdb_save = myType_RDBSaveCallBack,
2830 * .aof_rewrite = myType_AOFRewriteCallBack,
2831 * .free = myType_FreeCallBack,
2832 *
2833 * // Optional fields
2834 * .digest = myType_DigestCallBack,
2835 * .mem_usage = myType_MemUsageCallBack,
2836 * }
2837 *
2838 * * **rdb_load**: A callback function pointer that loads data from RDB files.
2839 * * **rdb_save**: A callback function pointer that saves data to RDB files.
2840 * * **aof_rewrite**: A callback function pointer that rewrites data as commands.
2841 * * **digest**: A callback function pointer that is used for `DEBUG DIGEST`.
2842 * * **free**: A callback function pointer that can free a type value.
2843 *
2844 * The **digest* and **mem_usage** methods should currently be omitted since
2845 * they are not yet implemented inside the Redis modules core.
2846 *
2847 * Note: the module name "AAAAAAAAA" is reserved and produces an error, it
2848 * happens to be pretty lame as well.
2849 *
2850 * If there is already a module registering a type with the same name,
2851 * and if the module name or encver is invalid, NULL is returned.
2852 * Otherwise the new type is registered into Redis, and a reference of
2853 * type RedisModuleType is returned: the caller of the function should store
2854 * this reference into a gobal variable to make future use of it in the
2855 * modules type API, since a single module may register multiple types.
2856 * Example code fragment:
2857 *
2858 * static RedisModuleType *BalancedTreeType;
2859 *
2860 * int RedisModule_OnLoad(RedisModuleCtx *ctx) {
2861 * // some code here ...
2862 * BalancedTreeType = RM_CreateDataType(...);
2863 * }
2864 */
2865moduleType *RM_CreateDataType(RedisModuleCtx *ctx, const char *name, int encver, void *typemethods_ptr) {
2866 uint64_t id = moduleTypeEncodeId(name,encver);
2867 if (id == 0) return NULL;
2868 if (moduleTypeLookupModuleByName(name) != NULL) return NULL;
2869
2870 long typemethods_version = ((long*)typemethods_ptr)[0];
2871 if (typemethods_version == 0) return NULL;
2872
2873 struct typemethods {
2874 uint64_t version;
2875 moduleTypeLoadFunc rdb_load;
2876 moduleTypeSaveFunc rdb_save;
2877 moduleTypeRewriteFunc aof_rewrite;
2878 moduleTypeMemUsageFunc mem_usage;
2879 moduleTypeDigestFunc digest;
2880 moduleTypeFreeFunc free;
2881 } *tms = (struct typemethods*) typemethods_ptr;
2882
2883 moduleType *mt = zcalloc(sizeof(*mt));
2884 mt->id = id;
2885 mt->module = ctx->module;
2886 mt->rdb_load = tms->rdb_load;
2887 mt->rdb_save = tms->rdb_save;
2888 mt->aof_rewrite = tms->aof_rewrite;
2889 mt->mem_usage = tms->mem_usage;
2890 mt->digest = tms->digest;
2891 mt->free = tms->free;
2892 memcpy(mt->name,name,sizeof(mt->name));
2893 listAddNodeTail(ctx->module->types,mt);
2894 return mt;
2895}

3162/* Add a new element to the digest. This function can be called multiple times
3163 * one element after the other, for all the elements that constitute a given
3164 * data structure. The function call must be followed by the call to
3165 * `RedisModule_DigestEndSequence` eventually, when all the elements that are
3166 * always in a given order are added. See the Redis Modules data types
3167 * documentation for more info. However this is a quick example that uses Redis
3168 * data types as an example.
3169 *
3170 * To add a sequence of unordered elements (for example in the case of a Redis
3171 * Set), the pattern to use is:
3172 *
3173 * foreach element {
3174 * AddElement(element);
3175 * EndSequence();
3176 * }
3177 *
3178 * Because Sets are not ordered, so every element added has a position that
3179 * does not depend from the other. However if instead our elements are
3180 * ordered in pairs, like field-value pairs of an Hash, then one should
3181 * use:
3182 *
3183 * foreach key,value {
3184 * AddElement(key);
3185 * AddElement(value);
3186 * EndSquence();
3187 * }
3188 *
3189 * Because the key and value will be always in the above order, while instead
3190 * the single key-value pairs, can appear in any position into a Redis hash.
3191 *
3192 * A list of ordered elements would be implemented with:
3193 *
3194 * foreach element {
3195 * AddElement(element);
3196 * }
3197 * EndSequence();
3198 *
3199 */
3200void RM_DigestAddStringBuffer(RedisModuleDigest *md, unsigned char *ele, size_t len) {
3201 mixDigest(md->o,ele,len);
3202}

3224 * by a module. The command works exactly like RedisModule_Call() in the way

3435/* Unblock a client blocked by `RedisModule_BlockedClient`. This will trigger
3436 * the reply callbacks to be called in order to reply to the client.
3437 * The 'privdata' argument will be accessible by the reply callback, so
3438 * the caller of this function can pass any value that is needed in order to
3439 * actually reply to the client.
3440 *
3441 * A common usage for 'privdata' is a thread that computes something that
3442 * needs to be passed to the client, included but not limited some slow
3443 * to compute reply or some reply obtained via networking.
3444 *
3445 * Note: this function can be called from threads spawned by the module. */

3448 bc->privdata = privdata;
3449 listAddNodeTail(moduleUnblockedClients,bc);
3450 if (write(server.module_blocked_pipe[1],"A",1) != 1) {
3451 /* Ignore the error, this is best-effort. */
3452 }
3453 pthread_mutex_unlock(&moduleUnblockedClientsMutex);

3464/* This function will check the moduleUnblockedClients queue in order to
3465 * call the reply callback and really unblock the client.
3466 *
3467 * Clients end into this list because of calls to RM_UnblockClient(),
3468 * however it is possible that while the module was doing work for the
3469 * blocked client, it was terminated by Redis (for timeout or other reasons).
3470 * When this happens the RedisModuleBlockedClient structure in the queue
3471 * will have the 'client' field set to NULL. */
3472void moduleHandleBlockedClients(void) {
3473 listNode *ln;
3474 RedisModuleBlockedClient *bc;
3475
3476 pthread_mutex_lock(&moduleUnblockedClientsMutex);
3477 /* Here we unblock all the pending clients blocked in modules operations
3478 * so we can read every pending "awake byte" in the pipe. */
3479 char buf[1];
3480 while (read(server.module_blocked_pipe[0],buf,1) == 1);
3481 while (listLength(moduleUnblockedClients)) {
3482 ln = listFirst(moduleUnblockedClients);
3483 bc = ln->value;
3484 client *c = bc->client;
3485 listDelNode(moduleUnblockedClients,ln);
3486 pthread_mutex_unlock(&moduleUnblockedClientsMutex);
3487
3488 /* Release the lock during the loop, as long as we don't
3489 * touch the shared list. */
3490
3491 /* Call the reply callback if the client is valid and we have
3492 * any callback. */
3493 if (c && bc->reply_callback) {
3494 RedisModuleCtx ctx = REDISMODULE_CTX_INIT;
3495 ctx.flags |= REDISMODULE_CTX_BLOCKED_REPLY;
3496 ctx.blocked_privdata = bc->privdata;
3497 ctx.module = bc->module;
3498 ctx.client = bc->client;
3499 bc->reply_callback(&ctx,(void**)c->argv,c->argc);
3500 moduleHandlePropagationAfterCommandCallback(&ctx);
3501 moduleFreeContext(&ctx);
3502 }
3503
3504 /* Free privdata if any. */
3505 if (bc->privdata && bc->free_privdata)
3506 bc->free_privdata(bc->privdata);
3507
3508 /* It is possible that this blocked client object accumulated
3509 * replies to send to the client in a thread safe context.
3510 * We need to glue such replies to the client output buffer and
3511 * free the temporary client we just used for the replies. */
3512 if (c) {
3513 if (bc->reply_client->bufpos)
3514 addReplyString(c,bc->reply_client->buf,
3515 bc->reply_client->bufpos);
3516 if (listLength(bc->reply_client->reply))
3517 listJoin(c->reply,bc->reply_client->reply);
3518 c->reply_bytes += bc->reply_client->reply_bytes;
3519 }
3520 freeClient(bc->reply_client);
3521
3522 if (c != NULL) {
3523 unblockClient(c);
3524 /* Put the client in the list of clients that need to write
3525 * if there are pending replies here. This is needed since
3526 * during a non blocking command the client may receive output. */
3527 if (clientHasPendingReplies(c) &&
3528 !(c->flags & CLIENT_PENDING_WRITE))
3529 {
3530 c->flags |= CLIENT_PENDING_WRITE;
3531 listAddNodeHead(server.clients_pending_write,c);
3532 }
3533 }
3534
3535 /* Free 'bc' only after unblocking the client, since it is
3536 * referenced in the client blocking context, and must be valid
3537 * when calling unblockClient(). */
3538 zfree(bc);
3539
3540 /* Lock again before to iterate the loop. */
3541 pthread_mutex_lock(&moduleUnblockedClientsMutex);
3542 }
3543 pthread_mutex_unlock(&moduleUnblockedClientsMutex);
3544}

3581/* Return a context which can be used inside threads to make Redis context
3582 * calls with certain modules APIs. If 'bc' is not NULL then the module will
3583 * be bound to a blocked client, and it will be possible to use the
3584 * `RedisModule_Reply*` family of functions to accumulate a reply for when the
3585 * client will be unblocked. Otherwise the thread safe context will be
3586 * detached by a specific client.
3587 *
3588 * To call non-reply APIs, the thread safe context must be prepared with:
3589 *
3590 * RedisModule_ThreadSafeCallStart(ctx);
3591 * ... make your call here ...
3592 * RedisModule_ThreadSafeCallStop(ctx);
3593 *
3594 * This is not needed when using `RedisModule_Reply*` functions, assuming
3595 * that a blocked client was used when the context was created, otherwise
3596 * no RedisModule_Reply* call should be made at all.
3597 *
3598 * TODO: thread safe contexts do not inherit the blocked client
3599 * selected database. */
3600RedisModuleCtx *RM_GetThreadSafeContext(RedisModuleBlockedClient *bc) {
3601 RedisModuleCtx *ctx = zmalloc(sizeof(*ctx));
3602 RedisModuleCtx empty = REDISMODULE_CTX_INIT;
3603 memcpy(ctx,&empty,sizeof(empty));
3604 if (bc) {
3605 ctx->blocked_client = bc;
3606 ctx->module = bc->module;
3607 }
3608 ctx->flags |= REDISMODULE_CTX_THREAD_SAFE;
3609 /* Even when the context is associated with a blocked client, we can't
3610 * access it safely from another thread, so we create a fake client here
3611 * in order to keep things like the currently selected database and similar
3612 * things. */
3613 ctx->client = createClient(-1);
3614 if (bc) selectDb(ctx->client,bc->dbid);
3615 return ctx;
3616}

3736/* Load a module and initialize it. On success C_OK is returned, otherwise
3737 * C_ERR is returned. */
3738int moduleLoad(const char *path, void **module_argv, int module_argc) {
3739 int (*onload)(void *, void **, int);
3740 void *handle;
3741 RedisModuleCtx ctx = REDISMODULE_CTX_INIT;
3742
3743 handle = dlopen(path,RTLD_NOW|RTLD_LOCAL);
3744 if (handle == NULL) {
3745 serverLog(LL_WARNING, "Module %s failed to load: %s", path, dlerror());
3746 return C_ERR;
3747 }
3748 onload = (int (*)(void *, void **, int))(unsigned long) dlsym(handle,"RedisModule_OnLoad");
3749 if (onload == NULL) {
3750 serverLog(LL_WARNING,
3751 "Module %s does not export RedisModule_OnLoad() "
3752 "symbol. Module not loaded.",path);
3753 return C_ERR;
3754 }
3755 if (onload((void*)&ctx,module_argv,module_argc) == REDISMODULE_ERR) {
3756 if (ctx.module) moduleFreeModuleStructure(ctx.module);
3757 dlclose(handle);
3758 serverLog(LL_WARNING,
3759 "Module %s initialization failed. Module not loaded",path);
3760 return C_ERR;
3761 }
3762
3763 /* Redis module loaded! Register it. */
3764 dictAdd(modules,ctx.module->name,ctx.module);
3765 ctx.module->handle = handle;
3766 serverLog(LL_NOTICE,"Module '%s' loaded from %s",ctx.module->name,path);
3767 moduleFreeContext(&ctx);
3768 return C_OK;
3769}

3771/* Unload the module registered with the specified name. On success
3772 * C_OK is returned, otherwise C_ERR is returned and errno is set
3773 * to the following values depending on the type of error:
3774 *
3775 * * ENONET: No such module having the specified name.
3776 * * EBUSY: The module exports a new data type and can only be reloaded. */
3777int moduleUnload(sds name) {
3778 struct RedisModule *module = dictFetchValue(modules,name);
3779
3780 if (module == NULL) {
3781 errno = ENOENT;
3782 return REDISMODULE_ERR;
3783 }
3784
3785 if (listLength(module->types)) {
3786 errno = EBUSY;
3787 return REDISMODULE_ERR;
3788 }
3789
3790 /* Unregister all the commands registered by this module. */
3791 dictIterator *di = dictGetSafeIterator(server.commands);
3792 dictEntry *de;
3793 while ((de = dictNext(di)) != NULL) {
3794 struct redisCommand *cmd = dictGetVal(de);
3795 if (cmd->proc == RedisModuleCommandDispatcher) {
3796 RedisModuleCommandProxy *cp =
3797 (void*)(unsigned long)cmd->getkeys_proc;
3798 sds cmdname = cp->rediscmd->name;
3799 if (cp->module == module) {
3800 dictDelete(server.commands,cmdname);
3801 dictDelete(server.orig_commands,cmdname);
3802 sdsfree(cmdname);
3803 zfree(cp->rediscmd);
3804 zfree(cp);
3805 }
3806 }
3807 }
3808 dictReleaseIterator(di);
3809
3810 /* Unregister all the hooks. TODO: Yet no hooks support here. */
3811
3812 /* Unload the dynamic library. */
3813 if (dlclose(module->handle) == -1) {
3814 char *error = dlerror();
3815 if (error == NULL) error = "Unknown error";
3816 serverLog(LL_WARNING,"Error when trying to close the %s module: %s",
3817 module->name, error);
3818 }
3819
3820 /* Remove from list of modules. */
3821 serverLog(LL_NOTICE,"Module %s unloaded",module->name);
3822 dictDelete(modules,module->name);
3823 module->name = NULL; /* The name was already freed by dictDelete(). */
3824 moduleFreeModuleStructure(module);
3825
3826 return REDISMODULE_OK;
3827}

3829/* Redis MODULE command.
3830 *
3831 * MODULE LOAD <path> [args...] */
3832void moduleCommand(client *c) {
3833 char *subcmd = c->argv[1]->ptr;
3834
3835 if (!strcasecmp(subcmd,"load") && c->argc >= 3) {
3836 robj **argv = NULL;
3837 int argc = 0;
3838
3839 if (c->argc > 3) {
3840 argc = c->argc - 3;
3841 argv = &c->argv[3];
3842 }
3843
3844 if (moduleLoad(c->argv[2]->ptr,(void **)argv,argc) == C_OK)
3845 addReply(c,shared.ok);
3846 else
3847 addReplyError(c,
3848 "Error loading the extension. Please check the server logs.");
3849 } else if (!strcasecmp(subcmd,"unload") && c->argc == 3) {
3850 if (moduleUnload(c->argv[2]->ptr) == C_OK)
3851 addReply(c,shared.ok);
3852 else {
3853 char *errmsg;
3854 switch(errno) {
3855 case ENOENT:
3856 errmsg = "no such module with that name";
3857 break;
3858 case EBUSY:
3859 errmsg = "the module exports one or more module-side data types, can't unload";
3860 break;
3861 default:
3862 errmsg = "operation not possible.";
3863 break;
3864 }
3865 addReplyErrorFormat(c,"Error unloading module: %s",errmsg);
3866 }
3867 } else if (!strcasecmp(subcmd,"list") && c->argc == 2) {
3868 dictIterator *di = dictGetIterator(modules);
3869 dictEntry *de;
3870
3871 addReplyMultiBulkLen(c,dictSize(modules));
3872 while ((de = dictNext(di)) != NULL) {
3873 sds name = dictGetKey(de);
3874 struct RedisModule *module = dictGetVal(de);
3875 addReplyMultiBulkLen(c,4);
3876 addReplyBulkCString(c,"name");
3877 addReplyBulkCBuffer(c,name,sdslen(name));
3878 addReplyBulkCString(c,"ver");
3879 addReplyLongLong(c,module->ver);
3880 }
3881 dictReleaseIterator(di);
3882 } else {
3883 addReply(c,shared.syntaxerr);
3884 }
3885}
