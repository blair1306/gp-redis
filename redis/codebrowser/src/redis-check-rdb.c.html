40struct {
41 rio *rio;
42 robj *key; /* Current key we are reading. */
43 int key_type; /* Current key type if != -1. */
44 unsigned long keys; /* Number of keys processed. */
45 unsigned long expires; /* Number of keys with an expire. */
46 unsigned long already_expired; /* Number of keys already expired. */
47 int doing; /* The state while reading the RDB. */
48 int error_set; /* True if error is populated. */
49 char error[1024];
50} rdbstate;

52/* At every loading step try to remember what we were about to do, so that
53 * we can log this information when an error is encountered. */
54#define RDB_CHECK_DOING_START 0
55#define RDB_CHECK_DOING_READ_TYPE 1
56#define RDB_CHECK_DOING_READ_EXPIRE 2
57#define RDB_CHECK_DOING_READ_KEY 3
58#define RDB_CHECK_DOING_READ_OBJECT_VALUE 4
59#define RDB_CHECK_DOING_CHECK_SUM 5
60#define RDB_CHECK_DOING_READ_LEN 6
61#define RDB_CHECK_DOING_READ_AUX 7

91/* Show a few stats collected into 'rdbstate' */
92void rdbShowGenericInfo(void) {
93 printf("[info] %lu keys read\n", rdbstate.keys);
94 printf("[info] %lu expires\n", rdbstate.expires);
95 printf("[info] %lu already expired\n", rdbstate.already_expired);
96}

98/* Called on RDB errors. Provides details about the RDB and the offset
99 * we were when the error was detected. */
100void rdbCheckError(const char *fmt, ...) {
101 char msg[1024];
102 va_list ap;
103
104 va_start(ap, fmt);
105 vsnprintf(msg, sizeof(msg), fmt, ap);
106 va_end(ap);
107
108 printf("--- RDB ERROR DETECTED ---\n");
109 printf("[offset %llu] %s\n",
110 (unsigned long long) (rdbstate.rio ?
111 rdbstate.rio->processed_bytes : 0), msg);
112 printf("[additional info] While doing: %s\n",
113 rdb_check_doing_string[rdbstate.doing]);
114 if (rdbstate.key)
115 printf("[additional info] Reading key '%s'\n",
116 (char*)rdbstate.key->ptr);
117 if (rdbstate.key_type != -1)
118 printf("[additional info] Reading type %d (%s)\n",
119 rdbstate.key_type,
120 ((unsigned)rdbstate.key_type <
121 sizeof(rdb_type_string)/sizeof(char*)) ?
122 rdb_type_string[rdbstate.key_type] : "unknown");
123 rdbShowGenericInfo();
124}

140/* Used inside rdb.c in order to log specific errors happening inside
141 * the RDB loading internals. */
142void rdbCheckSetError(const char *fmt, ...) {
143 va_list ap;
144
145 va_start(ap, fmt);
146 vsnprintf(rdbstate.error, sizeof(rdbstate.error), fmt, ap);
147 va_end(ap);
148 rdbstate.error_set = 1;
149}

151/* During RDB check we setup a special signal handler for memory violations
152 * and similar conditions, so that we can log the offending part of the RDB
153 * if the crash is due to broken content. */
154void rdbCheckHandleCrash(int sig, siginfo_t *info, void *secret) {
155 UNUSED(sig);
156 UNUSED(info);
157 UNUSED(secret);
158
159 rdbCheckError("Server crash checking the specified RDB file!");
160 exit(1);
161}
162
163void rdbCheckSetupSignals(void) {
164 struct sigaction act;
165
166 sigemptyset(&act.sa_mask);
167 act.sa_flags = SA_NODEFER | SA_RESETHAND | SA_SIGINFO;
168 act.sa_sigaction = rdbCheckHandleCrash;
169 sigaction(SIGSEGV, &act, NULL);
170 sigaction(SIGBUS, &act, NULL);
171 sigaction(SIGFPE, &act, NULL);
172 sigaction(SIGILL, &act, NULL);
173}

175/* Check the specified RDB file. Return 0 if the RDB looks sane, otherwise
176 * 1 is returned.
177 * The file is specified as a filename in 'rdbfilename' if 'fp' is not NULL,
178 * otherwise the already open file 'fp' is checked. */
179int redis_check_rdb(char *rdbfilename, FILE *fp) {
180 uint64_t dbid;
181 int type, rdbver;
182 char buf[1024];
183 long long expiretime, now = mstime();
184 static rio rdb; /* Pointed by global struct riostate. */
185
186 int closefile = (fp == NULL);
187 if (fp == NULL && (fp = fopen(rdbfilename,"r")) == NULL) return 1;
188
189 rioInitWithFile(&rdb,fp);
190 rdbstate.rio = &rdb;
191 rdb.update_cksum = rdbLoadProgressCallback;
192 if (rioRead(&rdb,buf,9) == 0) goto eoferr;
193 buf[9] = '\0';
194 if (memcmp(buf,"REDIS",5) != 0) {
195 rdbCheckError("Wrong signature trying to load DB from file");
196 return 1;
197 }
198 rdbver = atoi(buf+5);
199 if (rdbver < 1 || rdbver > RDB_VERSION) {
200 rdbCheckError("Can't handle RDB format version %d",rdbver);
201 return 1;
202 }
203
204 startLoading(fp);
205 while(1) {
206 robj *key, *val;
207 expiretime = -1;
208
209 /* Read type. */
210 rdbstate.doing = RDB_CHECK_DOING_READ_TYPE;
211 if ((type = rdbLoadType(&rdb)) == -1) goto eoferr;
212
213 /* Handle special types. */
214 if (type == RDB_OPCODE_EXPIRETIME) {
215 rdbstate.doing = RDB_CHECK_DOING_READ_EXPIRE;
216 /* EXPIRETIME: load an expire associated with the next key
217 * to load. Note that after loading an expire we need to
218 * load the actual type, and continue. */
219 if ((expiretime = rdbLoadTime(&rdb)) == -1) goto eoferr;
220 /* We read the time so we need to read the object type again. */
221 rdbstate.doing = RDB_CHECK_DOING_READ_TYPE;
222 if ((type = rdbLoadType(&rdb)) == -1) goto eoferr;
223 /* the EXPIRETIME opcode specifies time in seconds, so convert
224 * into milliseconds. */
225 expiretime *= 1000;
226 } else if (type == RDB_OPCODE_EXPIRETIME_MS) {
227 /* EXPIRETIME_MS: milliseconds precision expire times introduced
228 * with RDB v3. Like EXPIRETIME but no with more precision. */
229 rdbstate.doing = RDB_CHECK_DOING_READ_EXPIRE;
230 if ((expiretime = rdbLoadMillisecondTime(&rdb)) == -1) goto eoferr;
231 /* We read the time so we need to read the object type again. */
232 rdbstate.doing = RDB_CHECK_DOING_READ_TYPE;
233 if ((type = rdbLoadType(&rdb)) == -1) goto eoferr;
234 } else if (type == RDB_OPCODE_EOF) {
235 /* EOF: End of file, exit the main loop. */
236 break;
237 } else if (type == RDB_OPCODE_SELECTDB) {
238 /* SELECTDB: Select the specified database. */
239 rdbstate.doing = RDB_CHECK_DOING_READ_LEN;
240 if ((dbid = rdbLoadLen(&rdb,NULL)) == RDB_LENERR)
241 goto eoferr;
242 rdbCheckInfo("Selecting DB ID %d", dbid);
243 continue; /* Read type again. */
244 } else if (type == RDB_OPCODE_RESIZEDB) {
245 /* RESIZEDB: Hint about the size of the keys in the currently
246 * selected data base, in order to avoid useless rehashing. */
247 uint64_t db_size, expires_size;
248 rdbstate.doing = RDB_CHECK_DOING_READ_LEN;
249 if ((db_size = rdbLoadLen(&rdb,NULL)) == RDB_LENERR)
250 goto eoferr;
251 if ((expires_size = rdbLoadLen(&rdb,NULL)) == RDB_LENERR)
252 goto eoferr;
253 continue; /* Read type again. */
254 } else if (type == RDB_OPCODE_AUX) {
255 /* AUX: generic string-string fields. Use to add state to RDB
256 * which is backward compatible. Implementations of RDB loading
257 * are requierd to skip AUX fields they don't understand.
258 *
259 * An AUX field is composed of two strings: key and value. */
260 robj *auxkey, *auxval;
261 rdbstate.doing = RDB_CHECK_DOING_READ_AUX;
262 if ((auxkey = rdbLoadStringObject(&rdb)) == NULL) goto eoferr;
263 if ((auxval = rdbLoadStringObject(&rdb)) == NULL) goto eoferr;
264
265 rdbCheckInfo("AUX FIELD %s = '%s'",
266 (char*)auxkey->ptr, (char*)auxval->ptr);
267 decrRefCount(auxkey);
268 decrRefCount(auxval);
269 continue; /* Read type again. */
270 } else {
271 if (!rdbIsObjectType(type)) {
272 rdbCheckError("Invalid object type: %d", type);
273 return 1;
274 }
275 rdbstate.key_type = type;
276 }
277
278 /* Read key */
279 rdbstate.doing = RDB_CHECK_DOING_READ_KEY;
280 if ((key = rdbLoadStringObject(&rdb)) == NULL) goto eoferr;
281 rdbstate.key = key;
282 rdbstate.keys++;
283 /* Read value */
284 rdbstate.doing = RDB_CHECK_DOING_READ_OBJECT_VALUE;
285 if ((val = rdbLoadObject(type,&rdb)) == NULL) goto eoferr;
286 /* Check if the key already expired. This function is used when loading
287 * an RDB file from disk, either at startup, or when an RDB was
288 * received from the master. In the latter case, the master is
289 * responsible for key expiry. If we would expire keys here, the
290 * snapshot taken by the master may not be reflected on the slave. */
291 if (server.masterhost == NULL && expiretime != -1 && expiretime < now)
292 rdbstate.already_expired++;
293 if (expiretime != -1) rdbstate.expires++;
294 rdbstate.key = NULL;
295 decrRefCount(key);
296 decrRefCount(val);
297 rdbstate.key_type = -1;
298 }
299 /* Verify the checksum if RDB version is >= 5 */
300 if (rdbver >= 5 && server.rdb_checksum) {
301 uint64_t cksum, expected = rdb.cksum;
302
303 rdbstate.doing = RDB_CHECK_DOING_CHECK_SUM;
304 if (rioRead(&rdb,&cksum,8) == 0) goto eoferr;
305 memrev64ifbe(&cksum);
306 if (cksum == 0) {
307 rdbCheckInfo("RDB file was saved with checksum disabled: no check performed.");
308 } else if (cksum != expected) {
309 rdbCheckError("RDB CRC error");
310 } else {
311 rdbCheckInfo("Checksum OK");
312 }
313 }
314
315 if (closefile) fclose(fp);
316 return 0;
317
318eoferr: /* unexpected end of file is handled here with a fatal exit */
319 if (rdbstate.error_set) {
320 rdbCheckError(rdbstate.error);
321 } else {
322 rdbCheckError("Unexpected EOF reading RDB file");
323 }
324 return 1;
325}

327/* RDB check main: called form redis.c when Redis is executed with the
328 * redis-check-rdb alias, on during RDB loading errors.
329 *
330 * The function works in two ways: can be called with argc/argv as a
331 * standalone executable, or called with a non NULL 'fp' argument if we
332 * already have an open file to check. This happens when the function
333 * is used to check an RDB preamble inside an AOF file.
334 *
335 * When called with fp = NULL, the function never returns, but exits with the
336 * status code according to success (RDB is sane) or error (RDB is corrupted).
337 * Otherwise if called with a non NULL fp, the function returns C_OK or
338 * C_ERR depending on the success or failure. */
339int redis_check_rdb_main(int argc, char **argv, FILE *fp) {
340 if (argc != 2 && fp == NULL) {
341 fprintf(stderr, "Usage: %s <rdb-file-name>\n", argv[0]);
342 exit(1);
343 }
344 /* In order to call the loading functions we need to create the shared
345 * integer objects, however since this function may be called from
346 * an already initialized Redis instance, check if we really need to. */
347 if (shared.integers[0] == NULL)
348 createSharedObjects();
349 server.loading_process_events_interval_bytes = 0;
350 rdbCheckMode = 1;
351 rdbCheckInfo("Checking RDB file %s", argv[1]);
352 rdbCheckSetupSignals();
353 int retval = redis_check_rdb(argv[1],fp);
354 if (retval == 0) {
355 rdbCheckInfo("\\o/ RDB looks OK! \\o/");
356 rdbShowGenericInfo();
357 }
358 if (fp) return (retval == 0) ? C_OK : C_ERR;
359 exit(retval);
360}
