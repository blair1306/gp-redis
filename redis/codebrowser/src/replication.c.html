48/* Return the pointer to a string representing the slave ip:listening_port
49 * pair. Mostly useful for logging, since we want to log a slave using its
50 * IP address and its listening port which is more clear for the user, for
51 * example: "Closing connection with slave 10.1.2.3:6380". */
52char *replicationGetSlaveName(client *c) {
53 static char buf[NET_PEER_ID_LEN];
54 char ip[NET_IP_STR_LEN];
55
56 ip[0] = '\0';
57 buf[0] = '\0';
58 if (c->slave_ip[0] != '\0' ||
59 anetPeerToString(c->fd,ip,sizeof(ip),NULL) != -1)
60 {
61 /* Note that the 'ip' buffer is always larger than 'c->slave_ip' */
62 if (c->slave_ip[0] != '\0') memcpy(ip,c->slave_ip,sizeof(c->slave_ip));
63
64 if (c->slave_listening_port)
65 anetFormatAddr(buf,sizeof(buf),ip,c->slave_listening_port);
66 else
67 snprintf(buf,sizeof(buf),"%s:<unknown-slave-port>",ip);
68 } else {
69 snprintf(buf,sizeof(buf),"client id #%llu",
70 (unsigned long long) c->id);
71 }
72 return buf;
73}

77void createReplicationBacklog(void) {
78 serverAssert(server.repl_backlog == NULL);
79 server.repl_backlog = zmalloc(server.repl_backlog_size);
80 server.repl_backlog_histlen = 0;
81 server.repl_backlog_idx = 0;
82
83 /* We don't have any data inside our buffer, but virtually the first
84 * byte we have is the next byte that will be generated for the
85 * replication stream. */
86 server.repl_backlog_off = server.master_repl_offset+1;
87}
88
89/* This function is called when the user modifies the replication backlog
90 * size at runtime. It is up to the function to both update the
91 * server.repl_backlog_size and to resize the buffer and setup it so that
92 * it contains the same data as the previous one (possibly less data, but
93 * the most recent bytes, or the same data and more free space in case the
94 * buffer is enlarged). */
95void resizeReplicationBacklog(long long newsize) {
96 if (newsize < CONFIG_REPL_BACKLOG_MIN_SIZE)
97 newsize = CONFIG_REPL_BACKLOG_MIN_SIZE;
98 if (server.repl_backlog_size == newsize) return;
99
100 server.repl_backlog_size = newsize;
101 if (server.repl_backlog != NULL) {
102 /* What we actually do is to flush the old buffer and realloc a new
103 * empty one. It will refill with new data incrementally.
104 * The reason is that copying a few gigabytes adds latency and even
105 * worse often we need to alloc additional space before freeing the
106 * old buffer. */
107 zfree(server.repl_backlog);
108 server.repl_backlog = zmalloc(server.repl_backlog_size);
109 server.repl_backlog_histlen = 0;
110 server.repl_backlog_idx = 0;
111 /* Next byte we have is... the next since the buffer is empty. */
112 server.repl_backlog_off = server.master_repl_offset+1;
113 }
114}

116void freeReplicationBacklog(void) {
117 serverAssert(listLength(server.slaves) == 0);
118 zfree(server.repl_backlog);
119 server.repl_backlog = NULL;
120}

122/* Add data to the replication backlog.
123 * This function also increments the global replication offset stored at
124 * server.master_repl_offset, because there is no case where we want to feed
125 * the backlog without incrementing the offset. */
126void feedReplicationBacklog(void *ptr, size_t len) {
127 unsigned char *p = ptr;
128
129 server.master_repl_offset += len;
130
131 /* This is a circular buffer, so write as much data we can at every
132 * iteration and rewind the "idx" index if we reach the limit. */
133 while(len) {
134 size_t thislen = server.repl_backlog_size - server.repl_backlog_idx;
135 if (thislen > len) thislen = len;
136 memcpy(server.repl_backlog+server.repl_backlog_idx,p,thislen);
137 server.repl_backlog_idx += thislen;
138 if (server.repl_backlog_idx == server.repl_backlog_size)
139 server.repl_backlog_idx = 0;
140 len -= thislen;
141 p += thislen;
142 server.repl_backlog_histlen += thislen;
143 }
144 if (server.repl_backlog_histlen > server.repl_backlog_size)
145 server.repl_backlog_histlen = server.repl_backlog_size;
146 /* Set the offset of the first byte we have in the backlog. */
147 server.repl_backlog_off = server.master_repl_offset -
148 server.repl_backlog_histlen + 1;
149}

153void feedReplicationBacklogWithObject(robj *o) {
154 char llstr[LONG_STR_SIZE];
155 void *p;
156 size_t len;
157
158 if (o->encoding == OBJ_ENCODING_INT) {
159 len = ll2string(llstr,sizeof(llstr),(long)o->ptr);

168/* Propagate write commands to slaves, and populate the replication backlog
169 * as well. This function is used if the instance is a master: we use
170 * the commands received by our clients in order to create the replication
171 * stream. Instead if the instance is a slave and has sub-slaves attached,
172 * we use replicationFeedSlavesFromMaster() */
173void replicationFeedSlaves(list *slaves, int dictid, robj **argv, int argc) {
174 listNode *ln;
175 listIter li;
176 int j, len;
177 char llstr[LONG_STR_SIZE];
178
179 /* If the instance is not a top level master, return ASAP: we'll just proxy
180 * the stream of data we receive from our master instead, in order to
181 * propagate *identical* replication stream. In this way this slave can
182 * advertise the same replication ID as the master (since it shares the
183 * master replication history and has the same backlog and offsets). */
184 if (server.masterhost != NULL) return;
185
186 /* If there aren't slaves, and there is no backlog buffer to populate,
187 * we can return ASAP. */
188 if (server.repl_backlog == NULL && listLength(slaves) == 0) return;
189
190 /* We can't have slaves attached and no backlog. */
191 serverAssert(!(listLength(slaves) != 0 && server.repl_backlog == NULL));
192
193 /* Send SELECT command to every slave if needed. */
194 if (server.slaveseldb != dictid) {
195 robj *selectcmd;
196
197 /* For a few DBs we have pre-computed SELECT command. */
198 if (dictid >= 0 && dictid < PROTO_SHARED_SELECT_CMDS) {
199 selectcmd = shared.select[dictid];
200 } else {
201 int dictid_len;
202
203 dictid_len = ll2string(llstr,sizeof(llstr),dictid);
204 selectcmd = createObject(OBJ_STRING,
205 sdscatprintf(sdsempty(),
206 "*2\r\n$6\r\nSELECT\r\n$%d\r\n%s\r\n",
207 dictid_len, llstr));
208 }
209
210 /* Add the SELECT command into the backlog. */
211 if (server.repl_backlog) feedReplicationBacklogWithObject(selectcmd);
212
213 /* Send it to slaves. */
214 listRewind(slaves,&li);
215 while((ln = listNext(&li))) {
216 client *slave = ln->value;
217 if (slave->replstate == SLAVE_STATE_WAIT_BGSAVE_START) continue;
218 addReply(slave,selectcmd);
219 }
220
221 if (dictid < 0 || dictid >= PROTO_SHARED_SELECT_CMDS)
222 decrRefCount(selectcmd);
223 }
224 server.slaveseldb = dictid;
225
226 /* Write the command to the replication backlog if any. */
227 if (server.repl_backlog) {
228 char aux[LONG_STR_SIZE+3];
229
230 /* Add the multi bulk reply length. */
231 aux[0] = '*';
232 len = ll2string(aux+1,sizeof(aux)-1,argc);
233 aux[len+1] = '\r';
234 aux[len+2] = '\n';
235 feedReplicationBacklog(aux,len+3);
236
237 for (j = 0; j < argc; j++) {
238 long objlen = stringObjectLen(argv[j]);
239
240 /* We need to feed the buffer with the object as a bulk reply
241 * not just as a plain string, so create the $..CRLF payload len
242 * and add the final CRLF */
243 aux[0] = '$';
244 len = ll2string(aux+1,sizeof(aux)-1,objlen);
245 aux[len+1] = '\r';
246 aux[len+2] = '\n';
247 feedReplicationBacklog(aux,len+3);
248 feedReplicationBacklogWithObject(argv[j]);
249 feedReplicationBacklog(aux+len+1,2);
250 }
251 }
252
253 /* Write the command to every slave. */
254 listRewind(slaves,&li);
255 while((ln = listNext(&li))) {
256 client *slave = ln->value;
257
258 /* Don't feed slaves that are still waiting for BGSAVE to start */
259 if (slave->replstate == SLAVE_STATE_WAIT_BGSAVE_START) continue;
260
261 /* Feed slaves that are waiting for the initial SYNC (so these commands
262 * are queued in the output buffer until the initial SYNC completes),
263 * or are already in sync with the master. */
264
265 /* Add the multi bulk length. */
266 addReplyMultiBulkLen(slave,argc);
267
268 /* Finally any additional argument that was not stored inside the
269 * static buffer if any (from j to argc). */
270 for (j = 0; j < argc; j++)
271 addReplyBulk(slave,argv[j]);
272 }
273}

303void replicationFeedMonitors(client *c, list *monitors, int dictid, robj **argv, int argc) {
304 listNode *ln;
305 listIter li;
306 int j;
307 sds cmdrepr = sdsnew("+");
308 robj *cmdobj;
309 struct timeval tv;
310
311 gettimeofday(&tv,NULL);
312 cmdrepr = sdscatprintf(cmdrepr,"%ld.%06ld ",(long)tv.tv_sec,(long)tv.tv_usec);
313 if (c->flags & CLIENT_LUA) {
314 cmdrepr = sdscatprintf(cmdrepr,"[%d lua] ",dictid);
315 } else if (c->flags & CLIENT_UNIX_SOCKET) {
316 cmdrepr = sdscatprintf(cmdrepr,"[%d unix:%s] ",dictid,server.unixsocket);
317 } else {
318 cmdrepr = sdscatprintf(cmdrepr,"[%d %s] ",dictid,getClientPeerId(c));
319 }
320
321 for (j = 0; j < argc; j++) {
322 if (argv[j]->encoding == OBJ_ENCODING_INT) {
323 cmdrepr = sdscatprintf(cmdrepr, "\"%ld\"", (long)argv[j]->ptr);
324 } else {
325 cmdrepr = sdscatrepr(cmdrepr,(char*)argv[j]->ptr,
326 sdslen(argv[j]->ptr));
327 }
328 if (j != argc-1)
329 cmdrepr = sdscatlen(cmdrepr," ",1);
330 }
331 cmdrepr = sdscatlen(cmdrepr,"\r\n",2);
332 cmdobj = createObject(OBJ_STRING,cmdrepr);
333
334 listRewind(monitors,&li);
335 while((ln = listNext(&li))) {
336 client *monitor = ln->value;
337 addReply(monitor,cmdobj);
338 }
339 decrRefCount(cmdobj);
340}

342/* Feed the slave 'c' with the replication backlog starting from the
343 * specified 'offset' up to the end of the backlog. */
344long long addReplyReplicationBacklog(client *c, long long offset) {
345 long long j, skip, len;
346
347 serverLog(LL_DEBUG, "[PSYNC] Slave request offset: %lld", offset);
348
349 if (server.repl_backlog_histlen == 0) {
350 serverLog(LL_DEBUG, "[PSYNC] Backlog history len is zero");
351 return 0;
352 }
353
354 serverLog(LL_DEBUG, "[PSYNC] Backlog size: %lld",
355 server.repl_backlog_size);
356 serverLog(LL_DEBUG, "[PSYNC] First byte: %lld",
357 server.repl_backlog_off);
358 serverLog(LL_DEBUG, "[PSYNC] History len: %lld",
359 server.repl_backlog_histlen);
360 serverLog(LL_DEBUG, "[PSYNC] Current index: %lld",
361 server.repl_backlog_idx);
362
363 /* Compute the amount of bytes we need to discard. */
364 skip = offset - server.repl_backlog_off;
365 serverLog(LL_DEBUG, "[PSYNC] Skipping: %lld", skip);
366
367 /* Point j to the oldest byte, that is actually our
368 * server.repl_backlog_off byte. */
369 j = (server.repl_backlog_idx +
370 (server.repl_backlog_size-server.repl_backlog_histlen)) %
371 server.repl_backlog_size;
372 serverLog(LL_DEBUG, "[PSYNC] Index of first byte: %lld", j);
373
374 /* Discard the amount of data to seek to the specified 'offset'. */
375 j = (j + skip) % server.repl_backlog_size;
376
377 /* Feed slave with data. Since it is a circular buffer we have to
378 * split the reply in two parts if we are cross-boundary. */
379 len = server.repl_backlog_histlen - skip;
380 serverLog(LL_DEBUG, "[PSYNC] Reply total length: %lld", len);
381 while(len) {
382 long long thislen =
383 ((server.repl_backlog_size - j) < len) ?
384 (server.repl_backlog_size - j) : len;
385
386 serverLog(LL_DEBUG, "[PSYNC] addReply() length: %lld", thislen);
387 addReplySds(c,sdsnewlen(server.repl_backlog + j, thislen));
388 len -= thislen;
389 j = 0;
390 }
391 return server.repl_backlog_histlen - skip;
392}

402/* Send a FULLRESYNC reply in the specific case of a full resynchronization,
403 * as a side effect setup the slave for a full sync in different ways:
404 *
405 * 1) Remember, into the slave client structure, the replication offset
406 * we sent here, so that if new slaves will later attach to the same
407 * background RDB saving process (by duplicating this client output
408 * buffer), we can get the right offset from this slave.
409 * 2) Set the replication state of the slave to WAIT_BGSAVE_END so that
410 * we start accumulating differences from this point.
411 * 3) Force the replication stream to re-emit a SELECT statement so
412 * the new slave incremental differences will start selecting the
413 * right database number.
414 *
415 * Normally this function should be called immediately after a successful
416 * BGSAVE for replication was started, or when there is one already in
417 * progress that we attached our slave to. */
418int replicationSetupSlaveForFullResync(client *slave, long long offset) {
419 char buf[128];
420 int buflen;
421
422 slave->psync_initial_offset = offset;
423 slave->replstate = SLAVE_STATE_WAIT_BGSAVE_END;
424 /* We are going to accumulate the incremental changes for this
425 * slave as well. Set slaveseldb to -1 in order to force to re-emit
426 * a SELECT statement in the replication stream. */
427 server.slaveseldb = -1;
428
429 /* Don't send this reply to slaves that approached us with
430 * the old SYNC command. */
431 if (!(slave->flags & CLIENT_PRE_PSYNC)) {
432 buflen = snprintf(buf,sizeof(buf),"+FULLRESYNC %s %lld\r\n",
433 server.replid,offset);
434 if (write(slave->fd,buf,buflen) != buflen) {
435 freeClientAsync(slave);
436 return C_ERR;
437 }
438 }
439 return C_OK;
440}

442/* This function handles the PSYNC command from the point of view of a
443 * master receiving a request for partial resynchronization.
444 *
445 * On success return C_OK, otherwise C_ERR is returned and we proceed
446 * with the usual full resync. */
447int masterTryPartialResynchronization(client *c) {
448 long long psync_offset, psync_len;
449 char *master_replid = c->argv[1]->ptr;
450 char buf[128];
451 int buflen;
452
453 /* Parse the replication offset asked by the slave. Go to full sync
454 * on parse error: this should never happen but we try to handle
455 * it in a robust way compared to aborting. */
456 if (getLongLongFromObjectOrReply(c,c->argv[2],&psync_offset,NULL) !=
457 C_OK) goto need_full_resync;
458
459 /* Is the replication ID of this master the same advertised by the wannabe
460 * slave via PSYNC? If the replication ID changed this master has a
461 * different replication history, and there is no way to continue.
462 *
463 * Note that there are two potentially valid replication IDs: the ID1
464 * and the ID2. The ID2 however is only valid up to a specific offset. */
465 if (strcasecmp(master_replid, server.replid) &&
466 (strcasecmp(master_replid, server.replid2) ||
467 psync_offset > server.second_replid_offset))
468 {
469 /* Run id "?" is used by slaves that want to force a full resync. */
470 if (master_replid[0] != '?') {
471 if (strcasecmp(master_replid, server.replid) &&
472 strcasecmp(master_replid, server.replid2))
473 {
474 serverLog(LL_NOTICE,"Partial resynchronization not accepted: "
475 "Replication ID mismatch (Slave asked for '%s', my "
476 "replication IDs are '%s' and '%s')",
477 master_replid, server.replid, server.replid2);
478 } else {
479 serverLog(LL_NOTICE,"Partial resynchronization not accepted: "
480 "Requested offset for second ID was %lld, but I can reply "
481 "up to %lld", psync_offset, server.second_replid_offset);
482 }
483 } else {
484 serverLog(LL_NOTICE,"Full resync requested by slave %s",
485 replicationGetSlaveName(c));
486 }
487 goto need_full_resync;
488 }
489
490 /* We still have the data our slave is asking for? */
491 if (!server.repl_backlog ||
492 psync_offset < server.repl_backlog_off ||
493 psync_offset > (server.repl_backlog_off + server.repl_backlog_histlen))
494 {
495 serverLog(LL_NOTICE,
496 "Unable to partial resync with slave %s for lack of backlog (Slave request was: %lld).", replicationGetSlaveName(c), psync_offset);
497 if (psync_offset > server.master_repl_offset) {
498 serverLog(LL_WARNING,
499 "Warning: slave %s tried to PSYNC with an offset that is greater than the master replication offset.", replicationGetSlaveName(c));
500 }
501 goto need_full_resync;
502 }
503
504 /* If we reached this point, we are able to perform a partial resync:
505 * 1) Set client state to make it a slave.
506 * 2) Inform the client we can continue with +CONTINUE
507 * 3) Send the backlog data (from the offset to the end) to the slave. */
508 c->flags |= CLIENT_SLAVE;
509 c->replstate = SLAVE_STATE_ONLINE;
510 c->repl_ack_time = server.unixtime;
511 c->repl_put_online_on_ack = 0;
512 listAddNodeTail(server.slaves,c);
513 /* We can't use the connection buffers since they are used to accumulate
514 * new commands at this stage. But we are sure the socket send buffer is
515 * empty so this write will never fail actually. */
516 if (c->slave_capa & SLAVE_CAPA_PSYNC2) {
517 buflen = snprintf(buf,sizeof(buf),"+CONTINUE %s\r\n", server.replid);
518 } else {
519 buflen = snprintf(buf,sizeof(buf),"+CONTINUE\r\n");
520 }
521 if (write(c->fd,buf,buflen) != buflen) {
522 freeClientAsync(c);
523 return C_OK;
524 }
525 psync_len = addReplyReplicationBacklog(c,psync_offset);
526 serverLog(LL_NOTICE,
527 "Partial resynchronization request from %s accepted. Sending %lld bytes of backlog starting from offset %lld.",
528 replicationGetSlaveName(c),
529 psync_len, psync_offset);
530 /* Note that we don't need to set the selected DB at server.slaveseldb
531 * to -1 to force the master to emit SELECT, since the slave already
532 * has this state from the previous connection with the master. */
533
534 refreshGoodSlavesCount();
535 return C_OK; /* The caller can return, no full resync needed. */
536
537need_full_resync:
538 /* We need a full resync for some reason... Note that we can't
539 * reply to PSYNC right now if a full SYNC is needed. The reply
540 * must include the master offset at the time the RDB file we transfer
541 * is generated, so we need to delay the reply to that moment. */
542 return C_ERR;
543}

545/* Start a BGSAVE for replication goals, which is, selecting the disk or
546 * socket target depending on the configuration, and making sure that
547 * the script cache is flushed before to start.
548 *
549 * The mincapa argument is the bitwise AND among all the slaves capabilities
550 * of the slaves waiting for this BGSAVE, so represents the slave capabilities
551 * all the slaves support. Can be tested via SLAVE_CAPA_* macros.
552 *
553 * Side effects, other than starting a BGSAVE:
554 *
555 * 1) Handle the slaves in WAIT_START state, by preparing them for a full
556 * sync if the BGSAVE was succesfully started, or sending them an error
557 * and dropping them from the list of slaves.
558 *
559 * 2) Flush the Lua scripting script cache if the BGSAVE was actually
560 * started.
561 *
562 * Returns C_OK on success or C_ERR otherwise. */
563int startBgsaveForReplication(int mincapa) {
564 int retval;
565 int socket_target = server.repl_diskless_sync && (mincapa & SLAVE_CAPA_EOF);
566 listIter li;
567 listNode *ln;
568
569 serverLog(LL_NOTICE,"Starting BGSAVE for SYNC with target: %s",
570 socket_target ? "slaves sockets" : "disk");
571
572 rdbSaveInfo rsi, *rsiptr;
573 rsiptr = rdbPopulateSaveInfo(&rsi);
574 /* Only do rdbSave* when rsiptr is not NULL,
575 * otherwise slave will miss repl-stream-db. */
576 if (rsiptr) {
577 if (socket_target)
578 retval = rdbSaveToSlavesSockets(rsiptr);
579 else
580 retval = rdbSaveBackground(server.rdb_filename,rsiptr);
581 } else {
582 serverLog(LL_WARNING,"BGSAVE for replication: replication information not available, can't generate the RDB file right now. Try later.");
583 retval = C_ERR;
584 }
585
586 /* If we failed to BGSAVE, remove the slaves waiting for a full
587 * resynchorinization from the list of salves, inform them with
588 * an error about what happened, close the connection ASAP. */
589 if (retval == C_ERR) {
590 serverLog(LL_WARNING,"BGSAVE for replication failed");
591 listRewind(server.slaves,&li);
592 while((ln = listNext(&li))) {
593 client *slave = ln->value;
594
595 if (slave->replstate == SLAVE_STATE_WAIT_BGSAVE_START) {
596 slave->flags &= ~CLIENT_SLAVE;
597 listDelNode(server.slaves,ln);
598 addReplyError(slave,
599 "BGSAVE failed, replication can't continue");
600 slave->flags |= CLIENT_CLOSE_AFTER_REPLY;
601 }
602 }
603 return retval;
604 }
605
606 /* If the target is socket, rdbSaveToSlavesSockets() already setup
607 * the salves for a full resync. Otherwise for disk target do it now.*/
608 if (!socket_target) {
609 listRewind(server.slaves,&li);
610 while((ln = listNext(&li))) {
611 client *slave = ln->value;
612
613 if (slave->replstate == SLAVE_STATE_WAIT_BGSAVE_START) {
614 replicationSetupSlaveForFullResync(slave,
615 getPsyncInitialOffset());
616 }
617 }
618 }
619
620 /* Flush the script cache, since we need that slave differences are
621 * accumulated without requiring slaves to match our cached scripts. */
622 if (retval == C_OK) replicationScriptCacheFlush();
623 return retval;
624}

626/* SYNC and PSYNC command implemenation. */
627void syncCommand(client *c) {
628 /* ignore SYNC if already slave or in monitor mode */
629 if (c->flags & CLIENT_SLAVE) return;
630
631 /* Refuse SYNC requests if we are a slave but the link with our master
632 * is not ok... */
633 if (server.masterhost && server.repl_state != REPL_STATE_CONNECTED) {
634 addReplySds(c,sdsnew("-NOMASTERLINK Can't SYNC while not connected with my master\r\n"));
635 return;
636 }
637
638 /* SYNC can't be issued when the server has pending data to send to
639 * the client about already issued commands. We need a fresh reply
640 * buffer registering the differences between the BGSAVE and the current
641 * dataset, so that we can copy to other slaves if needed. */
642 if (clientHasPendingReplies(c)) {
643 addReplyError(c,"SYNC and PSYNC are invalid with pending output");
644 return;
645 }
646
647 serverLog(LL_NOTICE,"Slave %s asks for synchronization",
648 replicationGetSlaveName(c));
649
650 /* Try a partial resynchronization if this is a PSYNC command.
651 * If it fails, we continue with usual full resynchronization, however
652 * when this happens masterTryPartialResynchronization() already
653 * replied with:
654 *
655 * +FULLRESYNC <replid> <offset>
656 *
657 * So the slave knows the new replid and offset to try a PSYNC later
658 * if the connection with the master is lost. */
659 if (!strcasecmp(c->argv[0]->ptr,"psync")) {
660 if (masterTryPartialResynchronization(c) == C_OK) {
661 server.stat_sync_partial_ok++;
662 return; /* No full resync needed, return. */
663 } else {
664 char *master_replid = c->argv[1]->ptr;
665
666 /* Increment stats for failed PSYNCs, but only if the
667 * replid is not "?", as this is used by slaves to force a full
668 * resync on purpose when they are not albe to partially
669 * resync. */
670 if (master_replid[0] != '?') server.stat_sync_partial_err++;
671 }
672 } else {
673 /* If a slave uses SYNC, we are dealing with an old implementation
674 * of the replication protocol (like redis-cli --slave). Flag the client
675 * so that we don't expect to receive REPLCONF ACK feedbacks. */
676 c->flags |= CLIENT_PRE_PSYNC;
677 }
678
679 /* Full resynchronization. */
680 server.stat_sync_full++;
681
682 /* Setup the slave as one waiting for BGSAVE to start. The following code
683 * paths will change the state if we handle the slave differently. */
684 c->replstate = SLAVE_STATE_WAIT_BGSAVE_START;
685 if (server.repl_disable_tcp_nodelay)
686 anetDisableTcpNoDelay(NULL, c->fd); /* Non critical if it fails. */
687 c->repldbfd = -1;
688 c->flags |= CLIENT_SLAVE;
689 listAddNodeTail(server.slaves,c);
690
691 /* Create the replication backlog if needed. */
692 if (listLength(server.slaves) == 1 && server.repl_backlog == NULL) {
693 /* When we create the backlog from scratch, we always use a new
694 * replication ID and clear the ID2, since there is no valid
695 * past history. */
696 changeReplicationId();
697 clearReplicationId2();
698 createReplicationBacklog();
699 }
700
701 /* CASE 1: BGSAVE is in progress, with disk target. */
702 if (server.rdb_child_pid != -1 &&
703 server.rdb_child_type == RDB_CHILD_TYPE_DISK)
704 {
705 /* Ok a background save is in progress. Let's check if it is a good
706 * one for replication, i.e. if there is another slave that is
707 * registering differences since the server forked to save. */
708 client *slave;
709 listNode *ln;
710 listIter li;
711
712 listRewind(server.slaves,&li);
713 while((ln = listNext(&li))) {
714 slave = ln->value;
715 if (slave->replstate == SLAVE_STATE_WAIT_BGSAVE_END) break;
716 }
717 /* To attach this slave, we check that it has at least all the
718 * capabilities of the slave that triggered the current BGSAVE. */
719 if (ln && ((c->slave_capa & slave->slave_capa) == slave->slave_capa)) {
720 /* Perfect, the server is already registering differences for
721 * another slave. Set the right state, and copy the buffer. */
722 copyClientOutputBuffer(c,slave);
723 replicationSetupSlaveForFullResync(c,slave->psync_initial_offset);
724 serverLog(LL_NOTICE,"Waiting for end of BGSAVE for SYNC");
725 } else {
726 /* No way, we need to wait for the next BGSAVE in order to
727 * register differences. */
728 serverLog(LL_NOTICE,"Can't attach the slave to the current BGSAVE. Waiting for next BGSAVE for SYNC");
729 }
730
731 /* CASE 2: BGSAVE is in progress, with socket target. */
732 } else if (server.rdb_child_pid != -1 &&
733 server.rdb_child_type == RDB_CHILD_TYPE_SOCKET)
734 {
735 /* There is an RDB child process but it is writing directly to
736 * children sockets. We need to wait for the next BGSAVE
737 * in order to synchronize. */
738 serverLog(LL_NOTICE,"Current BGSAVE has socket target. Waiting for next BGSAVE for SYNC");
739
740 /* CASE 3: There is no BGSAVE is progress. */
741 } else {
742 if (server.repl_diskless_sync && (c->slave_capa & SLAVE_CAPA_EOF)) {
743 /* Diskless replication RDB child is created inside
744 * replicationCron() since we want to delay its start a
745 * few seconds to wait for more slaves to arrive. */
746 if (server.repl_diskless_sync_delay)
747 serverLog(LL_NOTICE,"Delay next BGSAVE for diskless SYNC");
748 } else {
749 /* Target is disk (or the slave is not capable of supporting
750 * diskless replication) and we don't have a BGSAVE in progress,
751 * let's start one. */
752 if (server.aof_child_pid == -1) {
753 startBgsaveForReplication(c->slave_capa);
754 } else {
755 serverLog(LL_NOTICE,
756 "No BGSAVE in progress, but an AOF rewrite is active. "
757 "BGSAVE for replication delayed");
758 }
759 }
760 }
761 return;
762}

764/* REPLCONF <option> <value> <option> <value> ...
765 * This command is used by a slave in order to configure the replication
766 * process before starting it with the SYNC command.
767 *
768 * Currently the only use of this command is to communicate to the master
769 * what is the listening port of the Slave redis instance, so that the
770 * master can accurately list slaves and their listening ports in
771 * the INFO output.
772 *
773 * In the future the same command can be used in order to configure
774 * the replication to initiate an incremental replication instead of a
775 * full resync. */
776void replconfCommand(client *c) {
777 int j;
778
779 if ((c->argc % 2) == 0) {
780 /* Number of arguments must be odd to make sure that every
781 * option has a corresponding value. */
782 addReply(c,shared.syntaxerr);
783 return;
784 }
785
786 /* Process every option-value pair. */
787 for (j = 1; j < c->argc; j+=2) {
788 if (!strcasecmp(c->argv[j]->ptr,"listening-port")) {
789 long port;
790
791 if ((getLongFromObjectOrReply(c,c->argv[j+1],
792 &port,NULL) != C_OK))
793 return;
794 c->slave_listening_port = port;
795 } else if (!strcasecmp(c->argv[j]->ptr,"ip-address")) {
796 sds ip = c->argv[j+1]->ptr;
797 if (sdslen(ip) < sizeof(c->slave_ip)) {
798 memcpy(c->slave_ip,ip,sdslen(ip)+1);
799 } else {
800 addReplyErrorFormat(c,"REPLCONF ip-address provided by "
801 "slave instance is too long: %zd bytes", sdslen(ip));
802 return;
803 }
804 } else if (!strcasecmp(c->argv[j]->ptr,"capa")) {
805 /* Ignore capabilities not understood by this master. */
806 if (!strcasecmp(c->argv[j+1]->ptr,"eof"))
807 c->slave_capa |= SLAVE_CAPA_EOF;
808 else if (!strcasecmp(c->argv[j+1]->ptr,"psync2"))
809 c->slave_capa |= SLAVE_CAPA_PSYNC2;
810 } else if (!strcasecmp(c->argv[j]->ptr,"ack")) {
811 /* REPLCONF ACK is used by slave to inform the master the amount
812 * of replication stream that it processed so far. It is an
813 * internal only command that normal clients should never use. */
814 long long offset;
815
816 if (!(c->flags & CLIENT_SLAVE)) return;
817 if ((getLongLongFromObject(c->argv[j+1], &offset) != C_OK))
818 return;
819 if (offset > c->repl_ack_off)
820 c->repl_ack_off = offset;
821 c->repl_ack_time = server.unixtime;
822 /* If this was a diskless replication, we need to really put
823 * the slave online when the first ACK is received (which
824 * confirms slave is online and ready to get more data). */
825 if (c->repl_put_online_on_ack && c->replstate == SLAVE_STATE_ONLINE)
826 putSlaveOnline(c);
827 /* Note: this command does not reply anything! */
828 return;
829 } else if (!strcasecmp(c->argv[j]->ptr,"getack")) {
830 /* REPLCONF GETACK is used in order to request an ACK ASAP
831 * to the slave. */
832 if (server.masterhost && server.master) replicationSendAck();
833 return;
834 } else {
835 addReplyErrorFormat(c,"Unrecognized REPLCONF option: %s",
836 (char*)c->argv[j]->ptr);
837 return;
838 }
839 }
840 addReply(c,shared.ok);
841}

843/* This function puts a slave in the online state, and should be called just
844 * after a slave received the RDB file for the initial synchronization, and
845 * we are finally ready to send the incremental stream of commands.
846 *
847 * It does a few things:
848 *
849 * 1) Put the slave in ONLINE state (useless when the function is called
850 * because state is already ONLINE but repl_put_online_on_ack is true).
851 * 2) Make sure the writable event is re-installed, since calling the SYNC
852 * command disables it, so that we can accumulate output buffer without
853 * sending it to the slave.
854 * 3) Update the count of good slaves. */
855void putSlaveOnline(client *slave) {
856 slave->replstate = SLAVE_STATE_ONLINE;
857 slave->repl_put_online_on_ack = 0;
858 slave->repl_ack_time = server.unixtime; /* Prevent false timeout. */
859 if (aeCreateFileEvent(server.el, slave->fd, AE_WRITABLE,
860 sendReplyToClient, slave) == AE_ERR) {
861 serverLog(LL_WARNING,"Unable to register writable event for slave bulk transfer: %s", strerror(errno));
862 freeClient(slave);
863 return;
864 }
865 refreshGoodSlavesCount();
866 serverLog(LL_NOTICE,"Synchronization with slave %s succeeded",
867 replicationGetSlaveName(slave));
868}

870void sendBulkToSlave(aeEventLoop *el, int fd, void *privdata, int mask) {
871 client *slave = privdata;
872 UNUSED(el);
873 UNUSED(mask);
874 char buf[PROTO_IOBUF_LEN];
875 ssize_t nwritten, buflen;
876
877 /* Before sending the RDB file, we send the preamble as configured by the
878 * replication process. Currently the preamble is just the bulk count of
879 * the file in the form "$<length>\r\n". */
880 if (slave->replpreamble) {
881 nwritten = write(fd,slave->replpreamble,sdslen(slave->replpreamble));
882 if (nwritten == -1) {
883 serverLog(LL_VERBOSE,"Write error sending RDB preamble to slave: %s",
884 strerror(errno));
885 freeClient(slave);
886 return;
887 }
888 server.stat_net_output_bytes += nwritten;
889 sdsrange(slave->replpreamble,nwritten,-1);
890 if (sdslen(slave->replpreamble) == 0) {
891 sdsfree(slave->replpreamble);
892 slave->replpreamble = NULL;
893 /* fall through sending data. */
894 } else {
895 return;
896 }
897 }
898
899 /* If the preamble was already transfered, send the RDB bulk data. */
900 lseek(slave->repldbfd,slave->repldboff,SEEK_SET);
901 buflen = read(slave->repldbfd,buf,PROTO_IOBUF_LEN);
902 if (buflen <= 0) {
903 serverLog(LL_WARNING,"Read error sending DB to slave: %s",
904 (buflen == 0) ? "premature EOF" : strerror(errno));
905 freeClient(slave);
906 return;
907 }
908 if ((nwritten = write(fd,buf,buflen)) == -1) {
909 if (errno != EAGAIN) {
910 serverLog(LL_WARNING,"Write error sending DB to slave: %s",
911 strerror(errno));
912 freeClient(slave);
913 }
914 return;
915 }
916 slave->repldboff += nwritten;
917 server.stat_net_output_bytes += nwritten;
918 if (slave->repldboff == slave->repldbsize) {
919 close(slave->repldbfd);
920 slave->repldbfd = -1;
921 aeDeleteFileEvent(server.el,slave->fd,AE_WRITABLE);
922 putSlaveOnline(slave);
923 }
924}

926/* This function is called at the end of every background saving,
927 * or when the replication RDB transfer strategy is modified from
928 * disk to socket or the other way around.
929 *
930 * The goal of this function is to handle slaves waiting for a successful
931 * background saving in order to perform non-blocking synchronization, and
932 * to schedule a new BGSAVE if there are slaves that attached while a
933 * BGSAVE was in progress, but it was not a good one for replication (no
934 * other slave was accumulating differences).
935 *
936 * The argument bgsaveerr is C_OK if the background saving succeeded
937 * otherwise C_ERR is passed to the function.
938 * The 'type' argument is the type of the child that terminated
939 * (if it had a disk or socket target). */
940void updateSlavesWaitingBgsave(int bgsaveerr, int type) {
941 listNode *ln;
942 int startbgsave = 0;
943 int mincapa = -1;
944 listIter li;
945
946 listRewind(server.slaves,&li);
947 while((ln = listNext(&li))) {
948 client *slave = ln->value;
949
950 if (slave->replstate == SLAVE_STATE_WAIT_BGSAVE_START) {
951 startbgsave = 1;
952 mincapa = (mincapa == -1) ? slave->slave_capa :
953 (mincapa & slave->slave_capa);
954 } else if (slave->replstate == SLAVE_STATE_WAIT_BGSAVE_END) {
955 struct redis_stat buf;
956
957 /* If this was an RDB on disk save, we have to prepare to send
958 * the RDB from disk to the slave socket. Otherwise if this was
959 * already an RDB -> Slaves socket transfer, used in the case of
960 * diskless replication, our work is trivial, we can just put
961 * the slave online. */
962 if (type == RDB_CHILD_TYPE_SOCKET) {
963 serverLog(LL_NOTICE,
964 "Streamed RDB transfer with slave %s succeeded (socket). Waiting for REPLCONF ACK from slave to enable streaming",
965 replicationGetSlaveName(slave));
966 /* Note: we wait for a REPLCONF ACK message from slave in
967 * order to really put it online (install the write handler
968 * so that the accumulated data can be transfered). However
969 * we change the replication state ASAP, since our slave
970 * is technically online now. */
971 slave->replstate = SLAVE_STATE_ONLINE;
972 slave->repl_put_online_on_ack = 1;
973 slave->repl_ack_time = server.unixtime; /* Timeout otherwise. */
974 } else {
975 if (bgsaveerr != C_OK) {
976 freeClient(slave);
977 serverLog(LL_WARNING,"SYNC failed. BGSAVE child returned an error");
978 continue;
979 }
980 if ((slave->repldbfd = open(server.rdb_filename,O_RDONLY)) == -1 ||
981 redis_fstat(slave->repldbfd,&buf) == -1) {
982 freeClient(slave);
983 serverLog(LL_WARNING,"SYNC failed. Can't open/stat DB after BGSAVE: %s", strerror(errno));
984 continue;
985 }
986 slave->repldboff = 0;
987 slave->repldbsize = buf.st_size;
988 slave->replstate = SLAVE_STATE_SEND_BULK;
989 slave->replpreamble = sdscatprintf(sdsempty(),"$%lld\r\n",
990 (unsigned long long) slave->repldbsize);
991
992 aeDeleteFileEvent(server.el,slave->fd,AE_WRITABLE);
993 if (aeCreateFileEvent(server.el, slave->fd, AE_WRITABLE, sendBulkToSlave, slave) == AE_ERR) {
994 freeClient(slave);
995 continue;
996 }
997 }
998 }
999 }
1000 if (startbgsave) startBgsaveForReplication(mincapa);
1001}

1021/* Use the current replication ID / offset as secondary replication
1022 * ID, and change the current one in order to start a new history.
1023 * This should be used when an instance is switched from slave to master
1024 * so that it can serve PSYNC requests performed using the master
1025 * replication ID. */
1026void shiftReplicationId(void) {
1027 memcpy(server.replid2,server.replid,sizeof(server.replid));
1028 /* We set the second replid offset to the master offset + 1, since
1029 * the slave will ask for the first byte it has not yet received, so
1030 * we need to add one to the offset: for example if, as a slave, we are
1031 * sure we have the same history as the master for 50 bytes, after we
1032 * are turned into a master, we can accept a PSYNC request with offset
1033 * 51, since the slave asking has the same history up to the 50th
1034 * byte, and is asking for the new bytes starting at offset 51. */
1035 server.second_replid_offset = server.master_repl_offset+1;
1036 changeReplicationId();
1037 serverLog(LL_WARNING,"Setting secondary replication ID to %s, valid up to offset: %lld. New replication ID is %s", server.replid2, server.second_replid_offset, server.replid);

1104/* Asynchronously read the SYNC payload we receive from a master */
1105#define REPL_MAX_WRITTEN_BEFORE_FSYNC (1024*1024*8) /* 8 MB */
1106void readSyncBulkPayload(aeEventLoop *el, int fd, void *privdata, int mask) {
1107 char buf[4096];
1108 ssize_t nread, readlen;
1109 off_t left;
1110 UNUSED(el);
1111 UNUSED(privdata);
1112 UNUSED(mask);
1113
1114 /* Static vars used to hold the EOF mark, and the last bytes received
1115 * form the server: when they match, we reached the end of the transfer. */
1116 static char eofmark[CONFIG_RUN_ID_SIZE];
1117 static char lastbytes[CONFIG_RUN_ID_SIZE];
1118 static int usemark = 0;
1119
1120 /* If repl_transfer_size == -1 we still have to read the bulk length
1121 * from the master reply. */
1122 if (server.repl_transfer_size == -1) {
1123 if (syncReadLine(fd,buf,1024,server.repl_syncio_timeout*1000) == -1) {
1124 serverLog(LL_WARNING,
1125 "I/O error reading bulk count from MASTER: %s",
1126 strerror(errno));
1127 goto error;
1128 }
1129
1130 if (buf[0] == '-') {
1131 serverLog(LL_WARNING,
1132 "MASTER aborted replication with an error: %s",
1133 buf+1);
1134 goto error;
1135 } else if (buf[0] == '\0') {
1136 /* At this stage just a newline works as a PING in order to take
1137 * the connection live. So we refresh our last interaction
1138 * timestamp. */
1139 server.repl_transfer_lastio = server.unixtime;
1140 return;
1141 } else if (buf[0] != '$') {
1142 serverLog(LL_WARNING,"Bad protocol from MASTER, the first byte is not '$' (we received '%s'), are you sure the host and port are right?", buf);
1143 goto error;
1144 }
1145
1146 /* There are two possible forms for the bulk payload. One is the
1147 * usual $<count> bulk format. The other is used for diskless transfers
1148 * when the master does not know beforehand the size of the file to
1149 * transfer. In the latter case, the following format is used:
1150 *
1151 * $EOF:<40 bytes delimiter>
1152 *
1153 * At the end of the file the announced delimiter is transmitted. The
1154 * delimiter is long and random enough that the probability of a
1155 * collision with the actual file content can be ignored. */
1156 if (strncmp(buf+1,"EOF:",4) == 0 && strlen(buf+5) >= CONFIG_RUN_ID_SIZE) {
1157 usemark = 1;
1158 memcpy(eofmark,buf+5,CONFIG_RUN_ID_SIZE);
1159 memset(lastbytes,0,CONFIG_RUN_ID_SIZE);
1160 /* Set any repl_transfer_size to avoid entering this code path
1161 * at the next call. */
1162 server.repl_transfer_size = 0;
1163 serverLog(LL_NOTICE,
1164 "MASTER <-> SLAVE sync: receiving streamed RDB from master");
1165 } else {
1166 usemark = 0;
1167 server.repl_transfer_size = strtol(buf+1,NULL,10);
1168 serverLog(LL_NOTICE,
1169 "MASTER <-> SLAVE sync: receiving %lld bytes from master",
1170 (long long) server.repl_transfer_size);
1171 }
1172 return;
1173 }
1174
1175 /* Read bulk data */
1176 if (usemark) {
1177 readlen = sizeof(buf);
1178 } else {
1179 left = server.repl_transfer_size - server.repl_transfer_read;
1180 readlen = (left < (signed)sizeof(buf)) ? left : (signed)sizeof(buf);
1181 }
1182
1183 nread = read(fd,buf,readlen);
1184 if (nread <= 0) {
1185 serverLog(LL_WARNING,"I/O error trying to sync with MASTER: %s",
1186 (nread == -1) ? strerror(errno) : "connection lost");
1187 cancelReplicationHandshake();
1188 return;
1189 }
1190 server.stat_net_input_bytes += nread;
1191
1192 /* When a mark is used, we want to detect EOF asap in order to avoid
1193 * writing the EOF mark into the file... */
1194 int eof_reached = 0;
1195
1196 if (usemark) {
1197 /* Update the last bytes array, and check if it matches our delimiter.*/
1198 if (nread >= CONFIG_RUN_ID_SIZE) {
1199 memcpy(lastbytes,buf+nread-CONFIG_RUN_ID_SIZE,CONFIG_RUN_ID_SIZE);
1200 } else {
1201 int rem = CONFIG_RUN_ID_SIZE-nread;
1202 memmove(lastbytes,lastbytes+nread,rem);
1203 memcpy(lastbytes+rem,buf,nread);
1204 }
1205 if (memcmp(lastbytes,eofmark,CONFIG_RUN_ID_SIZE) == 0) eof_reached = 1;
1206 }
1207
1208 server.repl_transfer_lastio = server.unixtime;
1209 if (write(server.repl_transfer_fd,buf,nread) != nread) {
1210 serverLog(LL_WARNING,"Write error or short write writing to the DB dump file needed for MASTER <-> SLAVE synchronization: %s", strerror(errno));
1211 goto error;
1212 }
1213 server.repl_transfer_read += nread;
1214
1215 /* Delete the last 40 bytes from the file if we reached EOF. */
1216 if (usemark && eof_reached) {
1217 if (ftruncate(server.repl_transfer_fd,
1218 server.repl_transfer_read - CONFIG_RUN_ID_SIZE) == -1)
1219 {
1220 serverLog(LL_WARNING,"Error truncating the RDB file received from the master for SYNC: %s", strerror(errno));
1221 goto error;
1222 }
1223 }
1224
1225 /* Sync data on disk from time to time, otherwise at the end of the transfer
1226 * we may suffer a big delay as the memory buffers are copied into the
1227 * actual disk. */
1228 if (server.repl_transfer_read >=
1229 server.repl_transfer_last_fsync_off + REPL_MAX_WRITTEN_BEFORE_FSYNC)
1230 {
1231 off_t sync_size = server.repl_transfer_read -
1232 server.repl_transfer_last_fsync_off;
1233 rdb_fsync_range(server.repl_transfer_fd,
1234 server.repl_transfer_last_fsync_off, sync_size);
1235 server.repl_transfer_last_fsync_off += sync_size;
1236 }
1237
1238 /* Check if the transfer is now complete */
1239 if (!usemark) {
1240 if (server.repl_transfer_read == server.repl_transfer_size)
1241 eof_reached = 1;
1242 }
1243
1244 if (eof_reached) {
1245 int aof_is_enabled = server.aof_state != AOF_OFF;
1246
1247 if (rename(server.repl_transfer_tmpfile,server.rdb_filename) == -1) {
1248 serverLog(LL_WARNING,"Failed trying to rename the temp DB into dump.rdb in MASTER <-> SLAVE synchronization: %s", strerror(errno));
1249 cancelReplicationHandshake();
1250 return;
1251 }
1252 serverLog(LL_NOTICE, "MASTER <-> SLAVE sync: Flushing old data");
1253 /* We need to stop any AOFRW fork before flusing and parsing
1254 * RDB, otherwise we'll create a copy-on-write disaster. */
1255 if(aof_is_enabled) stopAppendOnly();
1256 signalFlushedDb(-1);
1257 emptyDb(
1258 -1,
1259 server.repl_slave_lazy_flush ? EMPTYDB_ASYNC : EMPTYDB_NO_FLAGS,
1260 replicationEmptyDbCallback);
1261 /* Before loading the DB into memory we need to delete the readable
1262 * handler, otherwise it will get called recursively since
1263 * rdbLoad() will call the event loop to process events from time to
1264 * time for non blocking loading. */
1265 aeDeleteFileEvent(server.el,server.repl_transfer_s,AE_READABLE);
1266 serverLog(LL_NOTICE, "MASTER <-> SLAVE sync: Loading DB in memory");
1267 rdbSaveInfo rsi = RDB_SAVE_INFO_INIT;
1268 if (rdbLoad(server.rdb_filename,&rsi) != C_OK) {
1269 serverLog(LL_WARNING,"Failed trying to load the MASTER synchronization DB from disk");
1270 cancelReplicationHandshake();
1271 /* Re-enable the AOF if we disabled it earlier, in order to restore
1272 * the original configuration. */
1273 if (aof_is_enabled) restartAOF();
1274 return;
1275 }
1276 /* Final setup of the connected slave <- master link */
1277 zfree(server.repl_transfer_tmpfile);
1278 close(server.repl_transfer_fd);
1279 replicationCreateMasterClient(server.repl_transfer_s,rsi.repl_stream_db);
1280 server.repl_state = REPL_STATE_CONNECTED;
1281 /* After a full resynchroniziation we use the replication ID and
1282 * offset of the master. The secondary ID / offset are cleared since
1283 * we are starting a new history. */
1284 memcpy(server.replid,server.master->replid,sizeof(server.replid));
1285 server.master_repl_offset = server.master->reploff;
1286 clearReplicationId2();
1287 /* Let's create the replication backlog if needed. Slaves need to
1288 * accumulate the backlog regardless of the fact they have sub-slaves
1289 * or not, in order to behave correctly if they are promoted to
1290 * masters after a failover. */
1291 if (server.repl_backlog == NULL) createReplicationBacklog();
1292
1293 serverLog(LL_NOTICE, "MASTER <-> SLAVE sync: Finished with success");
1294 /* Restart the AOF subsystem now that we finished the sync. This
1295 * will trigger an AOF rewrite, and when done will start appending
1296 * to the new file. */
1297 if (aof_is_enabled) restartAOF();
1298 }
1299 return;
1300
1301error:
1302 cancelReplicationHandshake();
1303 return;
1304}

1306/* Send a synchronous command to the master. Used to send AUTH and
1307 * REPLCONF commands before starting the replication with SYNC.
1308 *
1309 * The command returns an sds string representing the result of the
1310 * operation. On error the first byte is a "-".
1311 */
1312#define SYNC_CMD_READ (1<<0)
1313#define SYNC_CMD_WRITE (1<<1)
1314#define SYNC_CMD_FULL (SYNC_CMD_READ|SYNC_CMD_WRITE)
1315char *sendSynchronousCommand(int flags, int fd, ...) {
1316
1317 /* Create the command to send to the master, we use simple inline
1318 * protocol for simplicity as currently we only send simple strings. */
1319 if (flags & SYNC_CMD_WRITE) {
1320 char *arg;
1321 va_list ap;
1322 sds cmd = sdsempty();
1323 va_start(ap,fd);
1324
1325 while(1) {
1326 arg = va_arg(ap, char*);
1327 if (arg == NULL) break;
1328
1329 if (sdslen(cmd) != 0) cmd = sdscatlen(cmd," ",1);
1330 cmd = sdscat(cmd,arg);
1331 }
1332 cmd = sdscatlen(cmd,"\r\n",2);
1333
1334 /* Transfer command to the server. */
1335 if (syncWrite(fd,cmd,sdslen(cmd),server.repl_syncio_timeout*1000)
1336 == -1)
1337 {
1338 sdsfree(cmd);
1339 return sdscatprintf(sdsempty(),"-Writing to master: %s",
1340 strerror(errno));
1341 }
1342 sdsfree(cmd);
1343 va_end(ap);
1344 }
1345
1346 /* Read the reply from the server. */
1347 if (flags & SYNC_CMD_READ) {
1348 char buf[256];
1349
1350 if (syncReadLine(fd,buf,sizeof(buf),server.repl_syncio_timeout*1000)
1351 == -1)
1352 {
1353 return sdscatprintf(sdsempty(),"-Reading from master: %s",
1354 strerror(errno));
1355 }
1356 server.repl_transfer_lastio = server.unixtime;
1357 return sdsnew(buf);
1358 }
1359 return NULL;
1360}

1362/* Try a partial resynchronization with the master if we are about to reconnect.
1363 * If there is no cached master structure, at least try to issue a
1364 * "PSYNC ? -1" command in order to trigger a full resync using the PSYNC
1365 * command in order to obtain the master run id and the master replication
1366 * global offset.
1367 *
1368 * This function is designed to be called from syncWithMaster(), so the
1369 * following assumptions are made:
1370 *
1371 * 1) We pass the function an already connected socket "fd".
1372 * 2) This function does not close the file descriptor "fd". However in case
1373 * of successful partial resynchronization, the function will reuse
1374 * 'fd' as file descriptor of the server.master client structure.
1375 *
1376 * The function is split in two halves: if read_reply is 0, the function
1377 * writes the PSYNC command on the socket, and a new function call is
1378 * needed, with read_reply set to 1, in order to read the reply of the
1379 * command. This is useful in order to support non blocking operations, so
1380 * that we write, return into the event loop, and read when there are data.
1381 *
1382 * When read_reply is 0 the function returns PSYNC_WRITE_ERR if there
1383 * was a write error, or PSYNC_WAIT_REPLY to signal we need another call
1384 * with read_reply set to 1. However even when read_reply is set to 1
1385 * the function may return PSYNC_WAIT_REPLY again to signal there were
1386 * insufficient data to read to complete its work. We should re-enter
1387 * into the event loop and wait in such a case.
1388 *
1389 * The function returns:
1390 *
1391 * PSYNC_CONTINUE: If the PSYNC command succeded and we can continue.
1392 * PSYNC_FULLRESYNC: If PSYNC is supported but a full resync is needed.
1393 * In this case the master run_id and global replication
1394 * offset is saved.
1395 * PSYNC_NOT_SUPPORTED: If the server does not understand PSYNC at all and
1396 * the caller should fall back to SYNC.
1397 * PSYNC_WRITE_ERROR: There was an error writing the command to the socket.
1398 * PSYNC_WAIT_REPLY: Call again the function with read_reply set to 1.
1399 * PSYNC_TRY_LATER: Master is currently in a transient error condition.
1400 *
1401 * Notable side effects:
1402 *
1403 * 1) As a side effect of the function call the function removes the readable
1404 * event handler from "fd", unless the return value is PSYNC_WAIT_REPLY.
1405 * 2) server.master_initial_offset is set to the right value according
1406 * to the master reply. This will be used to populate the 'server.master'
1407 * structure replication offset.
1408 */
1409
1410#define PSYNC_WRITE_ERROR 0
1411#define PSYNC_WAIT_REPLY 1
1412#define PSYNC_CONTINUE 2
1413#define PSYNC_FULLRESYNC 3
1414#define PSYNC_NOT_SUPPORTED 4
1415#define PSYNC_TRY_LATER 5
1416int slaveTryPartialResynchronization(int fd, int read_reply) {
1417 char *psync_replid;
1418 char psync_offset[32];
1419 sds reply;
1420
1421 /* Writing half */
1422 if (!read_reply) {
1423 /* Initially set master_initial_offset to -1 to mark the current
1424 * master run_id and offset as not valid. Later if we'll be able to do
1425 * a FULL resync using the PSYNC command we'll set the offset at the
1426 * right value, so that this information will be propagated to the
1427 * client structure representing the master into server.master. */
1428 server.master_initial_offset = -1;
1429
1430 if (server.cached_master) {
1431 psync_replid = server.cached_master->replid;
1432 snprintf(psync_offset,sizeof(psync_offset),"%lld", server.cached_master->reploff+1);
1433 serverLog(LL_NOTICE,"Trying a partial resynchronization (request %s:%s).", psync_replid, psync_offset);
1434 } else {
1435 serverLog(LL_NOTICE,"Partial resynchronization not possible (no cached master)");
1436 psync_replid = "?";
1437 memcpy(psync_offset,"-1",3);
1438 }
1439
1440 /* Issue the PSYNC command */
1441 reply = sendSynchronousCommand(SYNC_CMD_WRITE,fd,"PSYNC",psync_replid,psync_offset,NULL);
1442 if (reply != NULL) {
1443 serverLog(LL_WARNING,"Unable to send PSYNC to master: %s",reply);
1444 sdsfree(reply);
1445 aeDeleteFileEvent(server.el,fd,AE_READABLE);
1446 return PSYNC_WRITE_ERROR;
1447 }
1448 return PSYNC_WAIT_REPLY;
1449 }
1450
1451 /* Reading half */
1452 reply = sendSynchronousCommand(SYNC_CMD_READ,fd,NULL);
1453 if (sdslen(reply) == 0) {
1454 /* The master may send empty newlines after it receives PSYNC
1455 * and before to reply, just to keep the connection alive. */
1456 sdsfree(reply);
1457 return PSYNC_WAIT_REPLY;
1458 }
1459
1460 aeDeleteFileEvent(server.el,fd,AE_READABLE);
1461
1462 if (!strncmp(reply,"+FULLRESYNC",11)) {
1463 char *replid = NULL, *offset = NULL;
1464
1465 /* FULL RESYNC, parse the reply in order to extract the run id
1466 * and the replication offset. */
1467 replid = strchr(reply,' ');
1468 if (replid) {
1469 replid++;
1470 offset = strchr(replid,' ');
1471 if (offset) offset++;
1472 }
1473 if (!replid || !offset || (offset-replid-1) != CONFIG_RUN_ID_SIZE) {
1474 serverLog(LL_WARNING,
1475 "Master replied with wrong +FULLRESYNC syntax.");
1476 /* This is an unexpected condition, actually the +FULLRESYNC
1477 * reply means that the master supports PSYNC, but the reply
1478 * format seems wrong. To stay safe we blank the master
1479 * replid to make sure next PSYNCs will fail. */
1480 memset(server.master_replid,0,CONFIG_RUN_ID_SIZE+1);
1481 } else {
1482 memcpy(server.master_replid, replid, offset-replid-1);
1483 server.master_replid[CONFIG_RUN_ID_SIZE] = '\0';
1484 server.master_initial_offset = strtoll(offset,NULL,10);
1485 serverLog(LL_NOTICE,"Full resync from master: %s:%lld",
1486 server.master_replid,
1487 server.master_initial_offset);
1488 }
1489 /* We are going to full resync, discard the cached master structure. */
1490 replicationDiscardCachedMaster();
1491 sdsfree(reply);
1492 return PSYNC_FULLRESYNC;
1493 }
1494
1495 if (!strncmp(reply,"+CONTINUE",9)) {
1496 /* Partial resync was accepted. */
1497 serverLog(LL_NOTICE,
1498 "Successful partial resynchronization with master.");
1499
1500 /* Check the new replication ID advertised by the master. If it
1501 * changed, we need to set the new ID as primary ID, and set or
1502 * secondary ID as the old master ID up to the current offset, so
1503 * that our sub-slaves will be able to PSYNC with us after a
1504 * disconnection. */
1505 char *start = reply+10;
1506 char *end = reply+9;
1507 while(end[0] != '\r' && end[0] != '\n' && end[0] != '\0') end++;
1508 if (end-start == CONFIG_RUN_ID_SIZE) {
1509 char new[CONFIG_RUN_ID_SIZE+1];
1510 memcpy(new,start,CONFIG_RUN_ID_SIZE);
1511 new[CONFIG_RUN_ID_SIZE] = '\0';
1512
1513 if (strcmp(new,server.cached_master->replid)) {
1514 /* Master ID changed. */
1515 serverLog(LL_WARNING,"Master replication ID changed to %s",new);
1516
1517 /* Set the old ID as our ID2, up to the current offset+1. */
1518 memcpy(server.replid2,server.cached_master->replid,
1519 sizeof(server.replid2));
1520 server.second_replid_offset = server.master_repl_offset+1;
1521
1522 /* Update the cached master ID and our own primary ID to the
1523 * new one. */
1524 memcpy(server.replid,new,sizeof(server.replid));
1525 memcpy(server.cached_master->replid,new,sizeof(server.replid));
1526
1527 /* Disconnect all the sub-slaves: they need to be notified. */
1528 disconnectSlaves();
1529 }
1530 }
1531
1532 /* Setup the replication to continue. */
1533 sdsfree(reply);
1534 replicationResurrectCachedMaster(fd);
1535
1536 /* If this instance was restarted and we read the metadata to
1537 * PSYNC from the persistence file, our replication backlog could
1538 * be still not initialized. Create it. */
1539 if (server.repl_backlog == NULL) createReplicationBacklog();
1540 return PSYNC_CONTINUE;
1541 }
1542
1543 /* If we reach this point we received either an error (since the master does
1544 * not understand PSYNC or because it is in a special state and cannot
1545 * serve our request), or an unexpected reply from the master.
1546 *
1547 * Return PSYNC_NOT_SUPPORTED on errors we don't understand, otherwise
1548 * return PSYNC_TRY_LATER if we believe this is a transient error. */
1549
1550 if (!strncmp(reply,"-NOMASTERLINK",13) ||
1551 !strncmp(reply,"-LOADING",8))
1552 {
1553 serverLog(LL_NOTICE,
1554 "Master is currently unable to PSYNC "
1555 "but should be in the future: %s", reply);
1556 sdsfree(reply);
1557 return PSYNC_TRY_LATER;
1558 }
1559
1560 if (strncmp(reply,"-ERR",4)) {
1561 /* If it's not an error, log the unexpected event. */
1562 serverLog(LL_WARNING,
1563 "Unexpected reply to PSYNC from master: %s", reply);
1564 } else {
1565 serverLog(LL_NOTICE,
1566 "Master does not support PSYNC or is in "
1567 "error state (reply: %s)", reply);
1568 }
1569 sdsfree(reply);
1570 replicationDiscardCachedMaster();
1571 return PSYNC_NOT_SUPPORTED;
1572}

1574/* This handler fires when the non blocking connect was able to
1575 * establish a connection with the master. */
1576void syncWithMaster(aeEventLoop *el, int fd, void *privdata, int mask) {
1577 char tmpfile[256], *err = NULL;
1578 int dfd = -1, maxtries = 5;
1579 int sockerr = 0, psync_result;
1580 socklen_t errlen = sizeof(sockerr);
1581 UNUSED(el);
1582 UNUSED(privdata);
1583 UNUSED(mask);
1584
1585 /* If this event fired after the user turned the instance into a master
1586 * with SLAVEOF NO ONE we must just return ASAP. */
1587 if (server.repl_state == REPL_STATE_NONE) {
1588 close(fd);
1589 return;
1590 }
1591
1592 /* Check for errors in the socket: after a non blocking connect() we
1593 * may find that the socket is in error state. */
1594 if (getsockopt(fd, SOL_SOCKET, SO_ERROR, &sockerr, &errlen) == -1)
1595 sockerr = errno;
1596 if (sockerr) {
1597 serverLog(LL_WARNING,"Error condition on socket for SYNC: %s",
1598 strerror(sockerr));
1599 goto error;
1600 }
1601
1602 /* Send a PING to check the master is able to reply without errors. */
1603 if (server.repl_state == REPL_STATE_CONNECTING) {
1604 serverLog(LL_NOTICE,"Non blocking connect for SYNC fired the event.");
1605 /* Delete the writable event so that the readable event remains
1606 * registered and we can wait for the PONG reply. */
1607 aeDeleteFileEvent(server.el,fd,AE_WRITABLE);
1608 server.repl_state = REPL_STATE_RECEIVE_PONG;
1609 /* Send the PING, don't check for errors at all, we have the timeout
1610 * that will take care about this. */
1611 err = sendSynchronousCommand(SYNC_CMD_WRITE,fd,"PING",NULL);
1612 if (err) goto write_error;
1613 return;
1614 }
1615
1616 /* Receive the PONG command. */
1617 if (server.repl_state == REPL_STATE_RECEIVE_PONG) {
1618 err = sendSynchronousCommand(SYNC_CMD_READ,fd,NULL);
1619
1620 /* We accept only two replies as valid, a positive +PONG reply
1621 * (we just check for "+") or an authentication error.
1622 * Note that older versions of Redis replied with "operation not
1623 * permitted" instead of using a proper error code, so we test
1624 * both. */
1625 if (err[0] != '+' &&
1626 strncmp(err,"-NOAUTH",7) != 0 &&
1627 strncmp(err,"-ERR operation not permitted",28) != 0)
1628 {
1629 serverLog(LL_WARNING,"Error reply to PING from master: '%s'",err);
1630 sdsfree(err);
1631 goto error;
1632 } else {
1633 serverLog(LL_NOTICE,
1634 "Master replied to PING, replication can continue...");
1635 }
1636 sdsfree(err);
1637 server.repl_state = REPL_STATE_SEND_AUTH;
1638 }
1639
1640 /* AUTH with the master if required. */
1641 if (server.repl_state == REPL_STATE_SEND_AUTH) {
1642 if (server.masterauth) {
1643 err = sendSynchronousCommand(SYNC_CMD_WRITE,fd,"AUTH",server.masterauth,NULL);
1644 if (err) goto write_error;
1645 server.repl_state = REPL_STATE_RECEIVE_AUTH;
1646 return;
1647 } else {
1648 server.repl_state = REPL_STATE_SEND_PORT;
1649 }
1650 }
1651
1652 /* Receive AUTH reply. */
1653 if (server.repl_state == REPL_STATE_RECEIVE_AUTH) {
1654 err = sendSynchronousCommand(SYNC_CMD_READ,fd,NULL);
1655 if (err[0] == '-') {
1656 serverLog(LL_WARNING,"Unable to AUTH to MASTER: %s",err);
1657 sdsfree(err);
1658 goto error;
1659 }
1660 sdsfree(err);
1661 server.repl_state = REPL_STATE_SEND_PORT;
1662 }
1663
1664 /* Set the slave port, so that Master's INFO command can list the
1665 * slave listening port correctly. */
1666 if (server.repl_state == REPL_STATE_SEND_PORT) {
1667 sds port = sdsfromlonglong(server.slave_announce_port ?
1668 server.slave_announce_port : server.port);
1669 err = sendSynchronousCommand(SYNC_CMD_WRITE,fd,"REPLCONF",
1670 "listening-port",port, NULL);
1671 sdsfree(port);
1672 if (err) goto write_error;
1673 sdsfree(err);
1674 server.repl_state = REPL_STATE_RECEIVE_PORT;
1675 return;
1676 }
1677
1678 /* Receive REPLCONF listening-port reply. */
1679 if (server.repl_state == REPL_STATE_RECEIVE_PORT) {
1680 err = sendSynchronousCommand(SYNC_CMD_READ,fd,NULL);
1681 /* Ignore the error if any, not all the Redis versions support
1682 * REPLCONF listening-port. */
1683 if (err[0] == '-') {
1684 serverLog(LL_NOTICE,"(Non critical) Master does not understand "
1685 "REPLCONF listening-port: %s", err);
1686 }
1687 sdsfree(err);
1688 server.repl_state = REPL_STATE_SEND_IP;
1689 }
1690
1691 /* Skip REPLCONF ip-address if there is no slave-announce-ip option set. */
1692 if (server.repl_state == REPL_STATE_SEND_IP &&
1693 server.slave_announce_ip == NULL)
1694 {
1695 server.repl_state = REPL_STATE_SEND_CAPA;
1696 }
1697
1698 /* Set the slave ip, so that Master's INFO command can list the
1699 * slave IP address port correctly in case of port forwarding or NAT. */
1700 if (server.repl_state == REPL_STATE_SEND_IP) {
1701 err = sendSynchronousCommand(SYNC_CMD_WRITE,fd,"REPLCONF",
1702 "ip-address",server.slave_announce_ip, NULL);
1703 if (err) goto write_error;
1704 sdsfree(err);
1705 server.repl_state = REPL_STATE_RECEIVE_IP;
1706 return;
1707 }
1708
1709 /* Receive REPLCONF ip-address reply. */
1710 if (server.repl_state == REPL_STATE_RECEIVE_IP) {
1711 err = sendSynchronousCommand(SYNC_CMD_READ,fd,NULL);
1712 /* Ignore the error if any, not all the Redis versions support
1713 * REPLCONF listening-port. */
1714 if (err[0] == '-') {
1715 serverLog(LL_NOTICE,"(Non critical) Master does not understand "
1716 "REPLCONF ip-address: %s", err);
1717 }
1718 sdsfree(err);
1719 server.repl_state = REPL_STATE_SEND_CAPA;
1720 }
1721
1722 /* Inform the master of our (slave) capabilities.
1723 *
1724 * EOF: supports EOF-style RDB transfer for diskless replication.
1725 * PSYNC2: supports PSYNC v2, so understands +CONTINUE <new repl ID>.
1726 *
1727 * The master will ignore capabilities it does not understand. */
1728 if (server.repl_state == REPL_STATE_SEND_CAPA) {
1729 err = sendSynchronousCommand(SYNC_CMD_WRITE,fd,"REPLCONF",
1730 "capa","eof","capa","psync2",NULL);
1731 if (err) goto write_error;
1732 sdsfree(err);
1733 server.repl_state = REPL_STATE_RECEIVE_CAPA;
1734 return;
1735 }
1736
1737 /* Receive CAPA reply. */
1738 if (server.repl_state == REPL_STATE_RECEIVE_CAPA) {
1739 err = sendSynchronousCommand(SYNC_CMD_READ,fd,NULL);
1740 /* Ignore the error if any, not all the Redis versions support
1741 * REPLCONF capa. */
1742 if (err[0] == '-') {
1743 serverLog(LL_NOTICE,"(Non critical) Master does not understand "
1744 "REPLCONF capa: %s", err);
1745 }
1746 sdsfree(err);
1747 server.repl_state = REPL_STATE_SEND_PSYNC;
1748 }
1749
1750 /* Try a partial resynchonization. If we don't have a cached master
1751 * slaveTryPartialResynchronization() will at least try to use PSYNC
1752 * to start a full resynchronization so that we get the master run id
1753 * and the global offset, to try a partial resync at the next
1754 * reconnection attempt. */
1755 if (server.repl_state == REPL_STATE_SEND_PSYNC) {
1756 if (slaveTryPartialResynchronization(fd,0) == PSYNC_WRITE_ERROR) {
1757 err = sdsnew("Write error sending the PSYNC command.");
1758 goto write_error;
1759 }
1760 server.repl_state = REPL_STATE_RECEIVE_PSYNC;
1761 return;
1762 }
1763
1764 /* If reached this point, we should be in REPL_STATE_RECEIVE_PSYNC. */
1765 if (server.repl_state != REPL_STATE_RECEIVE_PSYNC) {
1766 serverLog(LL_WARNING,"syncWithMaster(): state machine error, "
1767 "state should be RECEIVE_PSYNC but is %d",
1768 server.repl_state);
1769 goto error;
1770 }
1771
1772 psync_result = slaveTryPartialResynchronization(fd,1);
1773 if (psync_result == PSYNC_WAIT_REPLY) return; /* Try again later... */
1774
1775 /* If the master is in an transient error, we should try to PSYNC
1776 * from scratch later, so go to the error path. This happens when
1777 * the server is loading the dataset or is not connected with its
1778 * master and so forth. */
1779 if (psync_result == PSYNC_TRY_LATER) goto error;
1780
1781 /* Note: if PSYNC does not return WAIT_REPLY, it will take care of
1782 * uninstalling the read handler from the file descriptor. */
1783
1784 if (psync_result == PSYNC_CONTINUE) {
1785 serverLog(LL_NOTICE, "MASTER <-> SLAVE sync: Master accepted a Partial Resynchronization.");
1786 return;
1787 }
1788
1789 /* PSYNC failed or is not supported: we want our slaves to resync with us
1790 * as well, if we have any sub-slaves. The master may transfer us an
1791 * entirely different data set and we have no way to incrementally feed
1792 * our slaves after that. */
1793 disconnectSlaves(); /* Force our slaves to resync with us as well. */
1794 freeReplicationBacklog(); /* Don't allow our chained slaves to PSYNC. */
1795
1796 /* Fall back to SYNC if needed. Otherwise psync_result == PSYNC_FULLRESYNC
1797 * and the server.master_replid and master_initial_offset are
1798 * already populated. */
1799 if (psync_result == PSYNC_NOT_SUPPORTED) {
1800 serverLog(LL_NOTICE,"Retrying with SYNC...");
1801 if (syncWrite(fd,"SYNC\r\n",6,server.repl_syncio_timeout*1000) == -1) {
1802 serverLog(LL_WARNING,"I/O error writing to MASTER: %s",
1803 strerror(errno));
1804 goto error;
1805 }
1806 }
1807
1808 /* Prepare a suitable temp file for bulk transfer */
1809 while(maxtries--) {
1810 snprintf(tmpfile,256,
1811 "temp-%d.%ld.rdb",(int)server.unixtime,(long int)getpid());
1812 dfd = open(tmpfile,O_CREAT|O_WRONLY|O_EXCL,0644);
1813 if (dfd != -1) break;
1814 sleep(1);
1815 }
1816 if (dfd == -1) {
1817 serverLog(LL_WARNING,"Opening the temp file needed for MASTER <-> SLAVE synchronization: %s",strerror(errno));
1818 goto error;
1819 }
1820
1821 /* Setup the non blocking download of the bulk file. */
1822 if (aeCreateFileEvent(server.el,fd, AE_READABLE,readSyncBulkPayload,NULL)
1823 == AE_ERR)
1824 {
1825 serverLog(LL_WARNING,
1826 "Can't create readable event for SYNC: %s (fd=%d)",
1827 strerror(errno),fd);
1828 goto error;
1829 }
1830
1831 server.repl_state = REPL_STATE_TRANSFER;
1832 server.repl_transfer_size = -1;
1833 server.repl_transfer_read = 0;
1834 server.repl_transfer_last_fsync_off = 0;
1835 server.repl_transfer_fd = dfd;
1836 server.repl_transfer_lastio = server.unixtime;
1837 server.repl_transfer_tmpfile = zstrdup(tmpfile);
1838 return;
1839
1840error:
1841 aeDeleteFileEvent(server.el,fd,AE_READABLE|AE_WRITABLE);
1842 if (dfd != -1) close(dfd);
1843 close(fd);
1844 server.repl_transfer_s = -1;
1845 server.repl_state = REPL_STATE_CONNECT;
1846 return;
1847
1848write_error: /* Handle sendSynchronousCommand(SYNC_CMD_WRITE) errors. */
1849 serverLog(LL_WARNING,"Sending command to master in replication handshake: %s", err);
1850 sdsfree(err);
1851 goto error;
1852}

1948/* Cancel replication, setting the instance as a master itself. */

1969 * with a SELECT statement. This is forced after a full resync, but
1970 * with PSYNC version 2, there is no need for full resync after a

2101/* ---------------------- MASTER CACHING FOR PSYNC -------------------------- */
2102
2103/* In order to implement partial synchronization we need to be able to cache
2104 * our master's client structure after a transient disconnection.
2105 * It is cached into server.cached_master and flushed away using the following
2106 * functions. */
2107
2108/* This function is called by freeClient() in order to cache the master
2109 * client structure instead of destryoing it. freeClient() will return
2110 * ASAP after this function returns, so every action needed to avoid problems
2111 * with a client that is really "suspended" has to be done by this function.
2112 *
2113 * The other functions that will deal with the cached master are:
2114 *
2115 * replicationDiscardCachedMaster() that will make sure to kill the client
2116 * as for some reason we don't want to use it in the future.
2117 *
2118 * replicationResurrectCachedMaster() that is used after a successful PSYNC
2119 * handshake in order to reactivate the cached master.
2120 */
2121void replicationCacheMaster(client *c) {
2122 serverAssert(server.master != NULL && server.cached_master == NULL);
2123 serverLog(LL_NOTICE,"Caching the disconnected master state.");
2124
2125 /* Unlink the client from the server structures. */
2126 unlinkClient(c);
2127
2128 /* Reset the master client so that's ready to accept new commands:
2129 * we want to discard te non processed query buffers and non processed
2130 * offsets, including pending transactions, already populated arguments,
2131 * pending outputs to the master. */
2132 sdsclear(server.master->querybuf);
2133 sdsclear(server.master->pending_querybuf);
2134 server.master->read_reploff = server.master->reploff;
2135 if (c->flags & CLIENT_MULTI) discardTransaction(c);
2136 listEmpty(c->reply);
2137 c->bufpos = 0;
2138 resetClient(c);
2139
2140 /* Save the master. Server.master will be set to null later by
2141 * replicationHandleMasterDisconnection(). */
2142 server.cached_master = server.master;
2143
2144 /* Invalidate the Peer ID cache. */
2145 if (c->peerid) {
2146 sdsfree(c->peerid);
2147 c->peerid = NULL;
2148 }
2149
2150 /* Caching the master happens instead of the actual freeClient() call,
2151 * so make sure to adjust the replication state. This function will
2152 * also set server.master to NULL. */
2153 replicationHandleMasterDisconnection();
2154}

2250/* ----------------------- REPLICATION SCRIPT CACHE --------------------------
2251 * The goal of this code is to keep track of scripts already sent to every
2252 * connected slave, in order to be able to replicate EVALSHA as it is without
2253 * translating it to EVAL every time it is possible.
2254 *
2255 * We use a capped collection implemented by a hash table for fast lookup
2256 * of scripts we can send as EVALSHA, plus a linked list that is used for
2257 * eviction of the oldest entry when the max number of items is reached.
2258 *
2259 * We don't care about taking a different cache for every different slave
2260 * since to fill the cache again is not very costly, the goal of this code
2261 * is to avoid that the same big script is trasmitted a big number of times
2262 * per second wasting bandwidth and processor speed, but it is not a problem
2263 * if we need to rebuild the cache from scratch from time to time, every used
2264 * script will need to be transmitted a single time to reappear in the cache.
2265 *
2266 * This is how the system works:
2267 *
2268 * 1) Every time a new slave connects, we flush the whole script cache.
2269 * 2) We only send as EVALSHA what was sent to the master as EVALSHA, without
2270 * trying to convert EVAL into EVALSHA specifically for slaves.
2271 * 3) Every time we trasmit a script as EVAL to the slaves, we also add the
2272 * corresponding SHA1 of the script into the cache as we are sure every
2273 * slave knows about the script starting from now.
2274 * 4) On SCRIPT FLUSH command, we replicate the command to all the slaves
2275 * and at the same time flush the script cache.
2276 * 5) When the last slave disconnects, flush the cache.
2277 * 6) We handle SCRIPT LOAD as well since that's how scripts are loaded
2278 * in the master sometimes.
2279 */
2280
2281/* Initialize the script cache, only called at startup. */
2282void replicationScriptCacheInit(void) {
2283 server.repl_scriptcache_size = 10000;
2284 server.repl_scriptcache_dict = dictCreate(&replScriptCacheDictType,NULL);
2285 server.repl_scriptcache_fifo = listCreate();
2286}
2287
2288/* Empty the script cache. Should be called every time we are no longer sure
2289 * that every slave knows about all the scripts in our set, or when the
2290 * current AOF "context" is no longer aware of the script. In general we
2291 * should flush the cache:
2292 *
2293 * 1) Every time a new slave reconnects to this master and performs a
2294 * full SYNC (PSYNC does not require flushing).
2295 * 2) Every time an AOF rewrite is performed.
2296 * 3) Every time we are left without slaves at all, and AOF is off, in order
2297 * to reclaim otherwise unused memory.
2298 */
2299void replicationScriptCacheFlush(void) {
2300 dictEmpty(server.repl_scriptcache_dict,NULL);
2301 listRelease(server.repl_scriptcache_fifo);
2302 server.repl_scriptcache_fifo = listCreate();
2303}

2306 * oldest is removed from the list. */
2307void replicationScriptCacheAdd(sds sha1) {

2310
2311 /* Evict oldest. */
2312 if (listLength(server.repl_scriptcache_fifo) == server.repl_scriptcache_size)

2318 serverAssert(retval == DICT_OK);

2321

2326}

2335 * Redis synchronous replication design can be summarized in points:
2336 *
2337 * - Redis masters have a global replication offset, used by PSYNC.
2338 * - Master increment the offset every time new commands are sent to slaves.
2339 * - Slaves ping back masters with the offset processed so far.
2340 *
2341 * So synchronous replication adds a new WAIT command in the form:
2342 *
2343 * WAIT <num_replicas> <milliseconds_timeout>
2344 *
2345 * That returns the number of replicas that processed the query when
2346 * we finally have at least num_replicas, or when the timeout was
2347 * reached.
2348 *
2349 * The command is implemented in this way:
2350 *
2351 * - Every time a client processes a command, we remember the replication
2352 * offset after sending that command to the slaves.
2353 * - When WAIT is called, we ask slaves to send an acknowledgement ASAP.
2354 * The client is blocked at the same time (see blocked.c).
2355 * - Once we receive enough ACKs for a given offset or when the timeout
2356 * is reached, the WAIT command is unblocked and the reply sent to the
2357 * client.
2358 */
2359
2360/* This just set a flag so that we broadcast a REPLCONF GETACK command
2361 * to all the slaves in the beforeSleep() function. Note that this way
2362 * we "group" all the clients that want to wait for synchronouns replication
2363 * in a given event loop iteration, and send a single GETACK for them all. */
2364void replicationRequestAckFromSlaves(void) {
2365 server.get_ack_from_slaves = 1;
2366}
2367
2368/* Return the number of slaves that already acknowledged the specified
2369 * replication offset. */
2370int replicationCountAcksByOffset(long long offset) {
2371 listIter li;
2372 listNode *ln;
2373 int count = 0;
2374
2375 listRewind(server.slaves,&li);
2376 while((ln = listNext(&li))) {
2377 client *slave = ln->value;
2378
2379 if (slave->replstate != SLAVE_STATE_ONLINE) continue;
2380 if (slave->repl_ack_off >= offset) count++;
2381 }
2382 return count;
2383}

2385/* WAIT for N replicas to acknowledge the processing of our latest
2386 * write command (and all the previous commands). */
2387void waitCommand(client *c) {
2388 mstime_t timeout;
2389 long numreplicas, ackreplicas;
2390 long long offset = c->woff;
2391
2392 if (server.masterhost) {
2393 addReplyError(c,"WAIT cannot be used with slave instances. Please also note that since Redis 4.0 if a slave is configured to be writable (which is not the default) writes to slaves are just local and are not propagated.");
2394 return;
2395 }
2396
2397 /* Argument parsing. */
2398 if (getLongFromObjectOrReply(c,c->argv[1],&numreplicas,NULL) != C_OK)
2399 return;
2400 if (getTimeoutFromObjectOrReply(c,c->argv[2],&timeout,UNIT_MILLISECONDS)
2401 != C_OK) return;
2402
2403 /* First try without blocking at all. */
2404 ackreplicas = replicationCountAcksByOffset(c->woff);
2405 if (ackreplicas >= numreplicas || c->flags & CLIENT_MULTI) {
2406 addReplyLongLong(c,ackreplicas);
2407 return;
2408 }
2409
2410 /* Otherwise block the client and put it into our list of clients
2411 * waiting for ack from slaves. */
2412 c->bpop.timeout = timeout;
2413 c->bpop.reploffset = offset;
2414 c->bpop.numreplicas = numreplicas;
2415 listAddNodeTail(server.clients_waiting_acks,c);
2416 blockClient(c,BLOCKED_WAIT);
2417
2418 /* Make sure that the server will send an ACK request to all the slaves
2419 * before returning to the event loop. */
2420 replicationRequestAckFromSlaves();
2421}
2422
2423/* This is called by unblockClient() to perform the blocking op type
2424 * specific cleanup. We just remove the client from the list of clients
2425 * waiting for replica acks. Never call it directly, call unblockClient()
2426 * instead. */
2427void unblockClientWaitingReplicas(client *c) {
2428 listNode *ln = listSearchKey(server.clients_waiting_acks,c);
2429 serverAssert(ln != NULL);
2430 listDelNode(server.clients_waiting_acks,ln);
2431}

2433/* Check if there are clients blocked in WAIT that can be unblocked since
2434 * we received enough ACKs from slaves. */
2435void processClientsWaitingReplicas(void) {
2436 long long last_offset = 0;
2437 int last_numreplicas = 0;
2438
2439 listIter li;
2440 listNode *ln;
2441
2442 listRewind(server.clients_waiting_acks,&li);
2443 while((ln = listNext(&li))) {
2444 client *c = ln->value;
2445
2446 /* Every time we find a client that is satisfied for a given
2447 * offset and number of replicas, we remember it so the next client
2448 * may be unblocked without calling replicationCountAcksByOffset()
2449 * if the requested offset / replicas were equal or less. */
2450 if (last_offset && last_offset > c->bpop.reploffset &&
2451 last_numreplicas > c->bpop.numreplicas)
2452 {
2453 unblockClient(c);
2454 addReplyLongLong(c,last_numreplicas);
2455 } else {
2456 int numreplicas = replicationCountAcksByOffset(c->bpop.reploffset);
2457
2458 if (numreplicas >= c->bpop.numreplicas) {
2459 last_offset = c->bpop.reploffset;
2460 last_numreplicas = numreplicas;
2461 unblockClient(c);
2462 addReplyLongLong(c,numreplicas);
2463 }
2464 }
2465 }
2466}

2468/* Return the slave replication offset for this instance, that is
2469 * the offset for which we already processed the master replication stream. */
2470long long replicationGetSlaveOffset(void) {
2471 long long offset = 0;
2472
2473 if (server.masterhost != NULL) {
2474 if (server.master) {
2475 offset = server.master->reploff;
2476 } else if (server.cached_master) {
2477 offset = server.cached_master->reploff;
2478 }
2479 }
2480 /* offset may be -1 when the master does not support it at all, however
2481 * this function is designed to return an offset that can express the
2482 * amount of data processed by the master, so we return a positive
2483 * integer. */
2484 if (offset < 0) offset = 0;
2485 return offset;
2486}

2489
2490/* Replication cron function, called 1 time per second. */
2491void replicationCron(void) {
2492 static long long replication_cron_loops = 0;
2493
2494 /* Non blocking connection timeout? */
2495 if (server.masterhost &&
2496 (server.repl_state == REPL_STATE_CONNECTING ||
2497 slaveIsInHandshakeState()) &&
2498 (time(NULL)-server.repl_transfer_lastio) > server.repl_timeout)
2499 {
2500 serverLog(LL_WARNING,"Timeout connecting to the MASTER...");
2501 cancelReplicationHandshake();
2502 }
2503
2504 /* Bulk transfer I/O timeout? */
2505 if (server.masterhost && server.repl_state == REPL_STATE_TRANSFER &&
2506 (time(NULL)-server.repl_transfer_lastio) > server.repl_timeout)
2507 {
2508 serverLog(LL_WARNING,"Timeout receiving bulk data from MASTER... If the problem persists try to set the 'repl-timeout' parameter in redis.conf to a larger value.");
2509 cancelReplicationHandshake();
2510 }
2511
2512 /* Timed out master when we are an already connected slave? */
2513 if (server.masterhost && server.repl_state == REPL_STATE_CONNECTED &&
2514 (time(NULL)-server.master->lastinteraction) > server.repl_timeout)
2515 {
2516 serverLog(LL_WARNING,"MASTER timeout: no data nor PING received...");
2517 freeClient(server.master);
2518 }
2519
2520 /* Check if we should connect to a MASTER */
2521 if (server.repl_state == REPL_STATE_CONNECT) {
2522 serverLog(LL_NOTICE,"Connecting to MASTER %s:%d",
2523 server.masterhost, server.masterport);
2524 if (connectWithMaster() == C_OK) {
2525 serverLog(LL_NOTICE,"MASTER <-> SLAVE sync started");
2526 }
2527 }
2528
2529 /* Send ACK to master from time to time.
2530 * Note that we do not send periodic acks to masters that don't
2531 * support PSYNC and replication offsets. */
2532 if (server.masterhost && server.master &&
2533 !(server.master->flags & CLIENT_PRE_PSYNC))
2534 replicationSendAck();
2535
2536 /* If we have attached slaves, PING them from time to time.
2537 * So slaves can implement an explicit timeout to masters, and will
2538 * be able to detect a link disconnection even if the TCP connection
2539 * will not actually go down. */
2540 listIter li;
2541 listNode *ln;
2542 robj *ping_argv[1];
2543
2544 /* First, send PING according to ping_slave_period. */
2545 if ((replication_cron_loops % server.repl_ping_slave_period) == 0 &&
2546 listLength(server.slaves))
2547 {
2548 ping_argv[0] = createStringObject("PING",4);
2549 replicationFeedSlaves(server.slaves, server.slaveseldb,
2550 ping_argv, 1);
2551 decrRefCount(ping_argv[0]);
2552 }
2553
2554 /* Second, send a newline to all the slaves in pre-synchronization
2555 * stage, that is, slaves waiting for the master to create the RDB file.
2556 *
2557 * Also send the a newline to all the chained slaves we have, if we lost
2558 * connection from our master, to keep the slaves aware that their
2559 * master is online. This is needed since sub-slaves only receive proxied
2560 * data from top-level masters, so there is no explicit pinging in order
2561 * to avoid altering the replication offsets. This special out of band
2562 * pings (newlines) can be sent, they will have no effect in the offset.
2563 *
2564 * The newline will be ignored by the slave but will refresh the
2565 * last interaction timer preventing a timeout. In this case we ignore the
2566 * ping period and refresh the connection once per second since certain
2567 * timeouts are set at a few seconds (example: PSYNC response). */
2568 listRewind(server.slaves,&li);
2569 while((ln = listNext(&li))) {
2570 client *slave = ln->value;
2571
2572 int is_presync =
2573 (slave->replstate == SLAVE_STATE_WAIT_BGSAVE_START ||
2574 (slave->replstate == SLAVE_STATE_WAIT_BGSAVE_END &&
2575 server.rdb_child_type != RDB_CHILD_TYPE_SOCKET));
2576
2577 if (is_presync) {
2578 if (write(slave->fd, "\n", 1) == -1) {
2579 /* Don't worry about socket errors, it's just a ping. */
2580 }
2581 }
2582 }
2583
2584 /* Disconnect timedout slaves. */
2585 if (listLength(server.slaves)) {
2586 listIter li;
2587 listNode *ln;
2588
2589 listRewind(server.slaves,&li);
2590 while((ln = listNext(&li))) {
2591 client *slave = ln->value;
2592
2593 if (slave->replstate != SLAVE_STATE_ONLINE) continue;
2594 if (slave->flags & CLIENT_PRE_PSYNC) continue;
2595 if ((server.unixtime - slave->repl_ack_time) > server.repl_timeout)
2596 {
2597 serverLog(LL_WARNING, "Disconnecting timedout slave: %s",
2598 replicationGetSlaveName(slave));
2599 freeClient(slave);
2600 }
2601 }
2602 }
2603
2604 /* If this is a master without attached slaves and there is a replication
2605 * backlog active, in order to reclaim memory we can free it after some
2606 * (configured) time. Note that this cannot be done for slaves: slaves
2607 * without sub-slaves attached should still accumulate data into the
2608 * backlog, in order to reply to PSYNC queries if they are turned into
2609 * masters after a failover. */
2610 if (listLength(server.slaves) == 0 && server.repl_backlog_time_limit &&
2611 server.repl_backlog && server.masterhost == NULL)
2612 {
2613 time_t idle = server.unixtime - server.repl_no_slaves_since;
2614
2615 if (idle > server.repl_backlog_time_limit) {
2616 freeReplicationBacklog();
2617 serverLog(LL_NOTICE,
2618 "Replication backlog freed after %d seconds "
2619 "without connected slaves.",
2620 (int) server.repl_backlog_time_limit);
2621 }
2622 }
2623
2624 /* If AOF is disabled and we no longer have attached slaves, we can
2625 * free our Replication Script Cache as there is no need to propagate
2626 * EVALSHA at all. */
2627 if (listLength(server.slaves) == 0 &&
2628 server.aof_state == AOF_OFF &&
2629 listLength(server.repl_scriptcache_fifo) != 0)
2630 {
2631 replicationScriptCacheFlush();
2632 }
2633
2634 /* Start a BGSAVE good for replication if we have slaves in
2635 * WAIT_BGSAVE_START state.
2636 *
2637 * In case of diskless replication, we make sure to wait the specified
2638 * number of seconds (according to configuration) so that other slaves
2639 * have the time to arrive before we start streaming. */
2640 if (server.rdb_child_pid == -1 && server.aof_child_pid == -1) {
2641 time_t idle, max_idle = 0;
2642 int slaves_waiting = 0;
2643 int mincapa = -1;
2644 listNode *ln;
2645 listIter li;
2646
2647 listRewind(server.slaves,&li);
2648 while((ln = listNext(&li))) {
2649 client *slave = ln->value;
2650 if (slave->replstate == SLAVE_STATE_WAIT_BGSAVE_START) {
2651 idle = server.unixtime - slave->lastinteraction;
2652 if (idle > max_idle) max_idle = idle;
2653 slaves_waiting++;
2654 mincapa = (mincapa == -1) ? slave->slave_capa :
2655 (mincapa & slave->slave_capa);
2656 }
2657 }
2658
2659 if (slaves_waiting &&
2660 (!server.repl_diskless_sync ||
2661 max_idle > server.repl_diskless_sync_delay))
2662 {
2663 /* Start the BGSAVE. The called function may start a
2664 * BGSAVE with socket target or disk target depending on the
2665 * configuration and slaves capabilities. */
2666 startBgsaveForReplication(mincapa);
2667 }
2668 }
2669
2670 /* Refresh the number of slaves with lag <= min-slaves-max-lag. */
2671 refreshGoodSlavesCount();
2672 replication_cron_loops++; /* Incremented with frequency 1 HZ. */
2673}
