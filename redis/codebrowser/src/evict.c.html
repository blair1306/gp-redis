41/* To improve the quality of the LRU approximation we take a set of keys
42 * that are good candidate for eviction across freeMemoryIfNeeded() calls.
43 *
44 * Entries inside the eviciton pool are taken ordered by idle time, putting
45 * greater idle times to the right (ascending order).
46 *
47 * When an LFU policy is used instead, a reverse frequency indication is used
48 * instead of the idle time, so that we still evict by larger value (larger
49 * inverse frequency means to evict keys with the least frequent accesses).
50 *
51 * Empty entries have the key pointer set to NULL. */
52#define EVPOOL_SIZE 16
53#define EVPOOL_CACHED_SDS_SIZE 255
54struct evictionPoolEntry {
55 unsigned long long idle; /* Object idle time (inverse frequency for LFU) */
56 sds key; /* Key name. */
57 sds cached; /* Cached SDS object for key name. */
58 int dbid; /* Key DB number. */
59};

69/* Return the LRU clock, based on the clock resolution. This is a time
70 * in a reduced-bits format that can be used to set and check the
71 * object->lru field of redisObject structures. */
72unsigned int getLRUClock(void) {
73 return (mstime()/LRU_CLOCK_RESOLUTION) & LRU_CLOCK_MAX;
74}
75
76/* This function is used to obtain the current LRU clock.
77 * If the current resolution is lower than the frequency we refresh the
78 * LRU clock (as it should be in production servers) we return the
79 * precomputed value, otherwise we need to resort to a system call. */
80unsigned int LRU_CLOCK(void) {
81 unsigned int lruclock;
82 if (1000/server.hz <= LRU_CLOCK_RESOLUTION) {
83 atomicGet(server.lruclock,lruclock);
84 } else {
85 lruclock = getLRUClock();
86 }
87 return lruclock;
88}
89
90/* Given an object returns the min number of milliseconds the object was never
91 * requested, using an approximated LRU algorithm. */
92unsigned long long estimateObjectIdleTime(robj *o) {
93 unsigned long long lruclock = LRU_CLOCK();
94 if (lruclock >= o->lru) {
95 return (lruclock - o->lru) * LRU_CLOCK_RESOLUTION;
96 } else {
97 return (lruclock + (LRU_CLOCK_MAX - o->lru)) *
98 LRU_CLOCK_RESOLUTION;
99 }
100}

102/* freeMemoryIfNeeded() gets called when 'maxmemory' is set on the config
103 * file to limit the max memory used by the server, before processing a
104 * command.
105 *
106 * The goal of the function is to free enough memory to keep Redis under the
107 * configured memory limit.
108 *
109 * The function starts calculating how many bytes should be freed to keep
110 * Redis under the limit, and enters a loop selecting the best keys to
111 * evict accordingly to the configured policy.
112 *
113 * If all the bytes needed to return back under the limit were freed the
114 * function returns C_OK, otherwise C_ERR is returned, and the caller
115 * should block the execution of commands that will result in more memory
116 * used by the server.
117 *
118 * ------------------------------------------------------------------------
119 *
120 * LRU approximation algorithm
121 *
122 * Redis uses an approximation of the LRU algorithm that runs in constant
123 * memory. Every time there is a key to expire, we sample N keys (with
124 * N very small, usually in around 5) to populate a pool of best keys to
125 * evict of M keys (the pool size is defined by EVPOOL_SIZE).
126 *
127 * The N keys sampled are added in the pool of good keys to expire (the one
128 * with an old access time) if they are better than one of the current keys
129 * in the pool.
130 *
131 * After the pool is populated, the best key we have in the pool is expired.
132 * However note that we don't remove keys from the pool when they are deleted
133 * so the pool may contain keys that no longer exist.
134 *
135 * When we try to evict a key, and all the entries in the pool don't exist
136 * we populate it again. This time we'll be sure that the pool has at least
137 * one key that can be evicted, if there is at least one key that can be
138 * evicted in the whole database. */
139
140/* Create a new eviction pool. */
141void evictionPoolAlloc(void) {
142 struct evictionPoolEntry *ep;
143 int j;
144
145 ep = zmalloc(sizeof(*ep)*EVPOOL_SIZE);
146 for (j = 0; j < EVPOOL_SIZE; j++) {
147 ep[j].idle = 0;
148 ep[j].key = NULL;
149 ep[j].cached = sdsnewlen(NULL,EVPOOL_CACHED_SDS_SIZE);
150 ep[j].dbid = 0;
151 }
152 EvictionPoolLRU = ep;
153}

155/* This is an helper function for freeMemoryIfNeeded(), it is used in order
156 * to populate the evictionPool with a few entries every time we want to
157 * expire a key. Keys with idle time smaller than one of the current
158 * keys are added. Keys are always added if there are free entries.
159 *
160 * We insert keys on place in ascending order, so keys with the smaller
161 * idle time are on the left, and keys with the higher idle time on the
162 * right. */
163
164void evictionPoolPopulate(int dbid, dict *sampledict, dict *keydict, struct evictionPoolEntry *pool) {
165 int j, k, count;
166 dictEntry *samples[server.maxmemory_samples];
167
168 count = dictGetSomeKeys(sampledict,samples,server.maxmemory_samples);
169 for (j = 0; j < count; j++) {
170 unsigned long long idle;
171 sds key;
172 robj *o;
173 dictEntry *de;
174
175 de = samples[j];
176 key = dictGetKey(de);
177
178 /* If the dictionary we are sampling from is not the main
179 * dictionary (but the expires one) we need to lookup the key
180 * again in the key dictionary to obtain the value object. */
181 if (server.maxmemory_policy != MAXMEMORY_VOLATILE_TTL) {
182 if (sampledict != keydict) de = dictFind(keydict, key);
183 o = dictGetVal(de);
184 }
185
186 /* Calculate the idle time according to the policy. This is called
187 * idle just because the code initially handled LRU, but is in fact
188 * just a score where an higher score means better candidate. */
189 if (server.maxmemory_policy & MAXMEMORY_FLAG_LRU) {
190 idle = estimateObjectIdleTime(o);
191 } else if (server.maxmemory_policy & MAXMEMORY_FLAG_LFU) {
192 /* When we use an LRU policy, we sort the keys by idle time
193 * so that we expire keys starting from greater idle time.
194 * However when the policy is an LFU one, we have a frequency
195 * estimation, and we want to evict keys with lower frequency
196 * first. So inside the pool we put objects using the inverted
197 * frequency subtracting the actual frequency to the maximum
198 * frequency of 255. */
199 idle = 255-LFUDecrAndReturn(o);
200 } else if (server.maxmemory_policy == MAXMEMORY_VOLATILE_TTL) {
201 /* In this case the sooner the expire the better. */
202 idle = ULLONG_MAX - (long)dictGetVal(de);
203 } else {
204 serverPanic("Unknown eviction policy in evictionPoolPopulate()");
205 }
206
207 /* Insert the element inside the pool.
208 * First, find the first empty bucket or the first populated
209 * bucket that has an idle time smaller than our idle time. */
210 k = 0;
211 while (k < EVPOOL_SIZE &&
212 pool[k].key &&
213 pool[k].idle < idle) k++;
214 if (k == 0 && pool[EVPOOL_SIZE-1].key != NULL) {
215 /* Can't insert if the element is < the worst element we have
216 * and there are no empty buckets. */
217 continue;
218 } else if (k < EVPOOL_SIZE && pool[k].key == NULL) {
219 /* Inserting into empty position. No setup needed before insert. */
220 } else {
221 /* Inserting in the middle. Now k points to the first element
222 * greater than the element to insert. */
223 if (pool[EVPOOL_SIZE-1].key == NULL) {
224 /* Free space on the right? Insert at k shifting
225 * all the elements from k to end to the right. */
226
227 /* Save SDS before overwriting. */
228 sds cached = pool[EVPOOL_SIZE-1].cached;
229 memmove(pool+k+1,pool+k,
230 sizeof(pool[0])*(EVPOOL_SIZE-k-1));
231 pool[k].cached = cached;
232 } else {
233 /* No free space on right? Insert at k-1 */
234 k--;
235 /* Shift all elements on the left of k (included) to the
236 * left, so we discard the element with smaller idle time. */
237 sds cached = pool[0].cached; /* Save SDS before overwriting. */
238 if (pool[0].key != pool[0].cached) sdsfree(pool[0].key);
239 memmove(pool,pool+1,sizeof(pool[0])*k);
240 pool[k].cached = cached;
241 }
242 }
243
244 /* Try to reuse the cached SDS string allocated in the pool entry,
245 * because allocating and deallocating this object is costly
246 * (according to the profiler, not my fantasy. Remember:
247 * premature optimizbla bla bla bla. */
248 int klen = sdslen(key);
249 if (klen > EVPOOL_CACHED_SDS_SIZE) {
250 pool[k].key = sdsdup(key);
251 } else {
252 memcpy(pool[k].cached,key,klen+1);
253 sdssetlen(pool[k].cached,klen);
254 pool[k].key = pool[k].cached;
255 }
256 pool[k].idle = idle;
257 pool[k].dbid = dbid;
258 }
259}

261/* ----------------------------------------------------------------------------
262 * LFU (Least Frequently Used) implementation.
263
264 * We have 24 total bits of space in each object in order to implement
265 * an LFU (Least Frequently Used) eviction policy, since we re-use the
266 * LRU field for this purpose.
267 *
268 * We split the 24 bits into two fields:
269 *
270 * 16 bits 8 bits
271 * +----------------+--------+
272 * + Last decr time | LOG_C |
273 * +----------------+--------+
274 *
275 * LOG_C is a logarithmic counter that provides an indication of the access
276 * frequency. However this field must also be decremented otherwise what used
277 * to be a frequently accessed key in the past, will remain ranked like that
278 * forever, while we want the algorithm to adapt to access pattern changes.
279 *
280 * So the remaining 16 bits are used in order to store the "decrement time",
281 * a reduced-precision Unix time (we take 16 bits of the time converted
282 * in minutes since we don't care about wrapping around) where the LOG_C
283 * counter is halved if it has an high value, or just decremented if it
284 * has a low value.
285 *
286 * New keys don't start at zero, in order to have the ability to collect
287 * some accesses before being trashed away, so they start at COUNTER_INIT_VAL.
288 * The logarithmic increment performed on LOG_C takes care of COUNTER_INIT_VAL
289 * when incrementing the key, so that keys starting at COUNTER_INIT_VAL
290 * (or having a smaller value) have a very high chance of being incremented
291 * on access.
292 *
293 * During decrement, the value of the logarithmic counter is halved if
294 * its current value is greater than two times the COUNTER_INIT_VAL, otherwise
295 * it is just decremented by one.
296 * --------------------------------------------------------------------------*/
297
298/* Return the current time in minutes, just taking the least significant
299 * 16 bits. The returned time is suitable to be stored as LDT (last decrement
300 * time) for the LFU implementation. */
301unsigned long LFUGetTimeInMinutes(void) {
302 return (server.unixtime/60) & 65535;
303}
304
305/* Given an object last decrement time, compute the minimum number of minutes
306 * that elapsed since the last decrement. Handle overflow (ldt greater than
307 * the current 16 bits minutes time) considering the time as wrapping
308 * exactly once. */
309unsigned long LFUTimeElapsed(unsigned long ldt) {
310 unsigned long now = LFUGetTimeInMinutes();
311 if (now >= ldt) return now-ldt;
312 return 65535-ldt+now;
313}
314
315/* Logarithmically increment a counter. The greater is the current counter value
316 * the less likely is that it gets really implemented. Saturate it at 255. */
317uint8_t LFULogIncr(uint8_t counter) {
318 if (counter == 255) return 255;
319 double r = (double)rand()/RAND_MAX;
320 double baseval = counter - LFU_INIT_VAL;
321 if (baseval < 0) baseval = 0;
322 double p = 1.0/(baseval*server.lfu_log_factor+1);
323 if (r < p) counter++;
324 return counter;
325}
326
327/* If the object decrement time is reached, decrement the LFU counter and
328 * update the decrement time field. Return the object frequency counter.
329 *
330 * This function is used in order to scan the dataset for the best object
331 * to fit: as we check for the candidate, we incrementally decrement the
332 * counter of the scanned objects if needed. */
333#define LFU_DECR_INTERVAL 1
334unsigned long LFUDecrAndReturn(robj *o) {
335 unsigned long ldt = o->lru >> 8;
336 unsigned long counter = o->lru & 255;
337 if (LFUTimeElapsed(ldt) >= server.lfu_decay_time && counter) {
338 if (counter > LFU_INIT_VAL*2) {
339 counter /= 2;
340 if (counter < LFU_INIT_VAL*2) counter = LFU_INIT_VAL*2;
341 } else {
342 counter--;
343 }
344 o->lru = (LFUGetTimeInMinutes()<<8) | counter;
345 }
346 return counter;
347}

349/* ----------------------------------------------------------------------------
350 * The external API for eviction: freeMemroyIfNeeded() is called by the
351 * server when there is data to add in order to make space if needed.
352 * --------------------------------------------------------------------------*/

354/* We don't want to count AOF buffers and slaves output buffers as
355 * used memory: the eviction should use mostly data size. This function
356 * returns the sum of AOF and slaves buffer. */
357size_t freeMemoryGetNotCountedMemory(void) {
358 size_t overhead = 0;
359 int slaves = listLength(server.slaves);
360
361 if (slaves) {
362 listIter li;
363 listNode *ln;
364
365 listRewind(server.slaves,&li);
366 while((ln = listNext(&li))) {
367 client *slave = listNodeValue(ln);
368 overhead += getClientOutputBufferMemoryUsage(slave);
369 }
370 }
371 if (server.aof_state != AOF_OFF) {
372 overhead += sdslen(server.aof_buf)+aofRewriteBufferSize();
373 }
374 return overhead;
375}

377int freeMemoryIfNeeded(void) {
378 size_t mem_reported, mem_used, mem_tofree, mem_freed;
379 mstime_t latency, eviction_latency;
380 long long delta;
381 int slaves = listLength(server.slaves);
382
383 /* When clients are paused the dataset should be static not just from the
384 * POV of clients not being able to write, but also from the POV of
385 * expires and evictions of keys not being performed. */
386 if (clientsArePaused()) return C_OK;
387
388 /* Check if we are over the memory usage limit. If we are not, no need
389 * to subtract the slaves output buffers. We can just return ASAP. */
390 mem_reported = zmalloc_used_memory();
391 if (mem_reported <= server.maxmemory) return C_OK;
392
393 /* Remove the size of slaves output buffers and AOF buffer from the
394 * count of used memory. */
395 mem_used = mem_reported;
396 size_t overhead = freeMemoryGetNotCountedMemory();
397 mem_used = (mem_used > overhead) ? mem_used-overhead : 0;
398
399 /* Check if we are still over the memory limit. */
400 if (mem_used <= server.maxmemory) return C_OK;
401
402 /* Compute how much memory we need to free. */
403 mem_tofree = mem_used - server.maxmemory;
404 mem_freed = 0;
405
406 if (server.maxmemory_policy == MAXMEMORY_NO_EVICTION)
407 goto cant_free; /* We need to free memory, but policy forbids. */
408
409 latencyStartMonitor(latency);
410 while (mem_freed < mem_tofree) {
411 int j, k, i, keys_freed = 0;
412 static int next_db = 0;
413 sds bestkey = NULL;
414 int bestdbid;
415 redisDb *db;
416 dict *dict;
417 dictEntry *de;
418
419 if (server.maxmemory_policy & (MAXMEMORY_FLAG_LRU|MAXMEMORY_FLAG_LFU) ||
420 server.maxmemory_policy == MAXMEMORY_VOLATILE_TTL)
421 {
422 struct evictionPoolEntry *pool = EvictionPoolLRU;
423
424 while(bestkey == NULL) {
425 unsigned long total_keys = 0, keys;
426
427 /* We don't want to make local-db choices when expiring keys,
428 * so to start populate the eviction pool sampling keys from
429 * every DB. */
430 for (i = 0; i < server.dbnum; i++) {
431 db = server.db+i;
432 dict = (server.maxmemory_policy & MAXMEMORY_FLAG_ALLKEYS) ?
433 db->dict : db->expires;
434 if ((keys = dictSize(dict)) != 0) {
435 evictionPoolPopulate(i, dict, db->dict, pool);
436 total_keys += keys;
437 }
438 }
439 if (!total_keys) break; /* No keys to evict. */
440
441 /* Go backward from best to worst element to evict. */
442 for (k = EVPOOL_SIZE-1; k >= 0; k--) {
443 if (pool[k].key == NULL) continue;
444 bestdbid = pool[k].dbid;
445
446 if (server.maxmemory_policy & MAXMEMORY_FLAG_ALLKEYS) {
447 de = dictFind(server.db[pool[k].dbid].dict,
448 pool[k].key);
449 } else {
450 de = dictFind(server.db[pool[k].dbid].expires,
451 pool[k].key);
452 }
453
454 /* Remove the entry from the pool. */
455 if (pool[k].key != pool[k].cached)
456 sdsfree(pool[k].key);
457 pool[k].key = NULL;
458 pool[k].idle = 0;
459
460 /* If the key exists, is our pick. Otherwise it is
461 * a ghost and we need to try the next element. */
462 if (de) {
463 bestkey = dictGetKey(de);
464 break;
465 } else {
466 /* Ghost... Iterate again. */
467 }
468 }
469 }
470 }
471
472 /* volatile-random and allkeys-random policy */
473 else if (server.maxmemory_policy == MAXMEMORY_ALLKEYS_RANDOM ||
474 server.maxmemory_policy == MAXMEMORY_VOLATILE_RANDOM)
475 {
476 /* When evicting a random key, we try to evict a key for
477 * each DB, so we use the static 'next_db' variable to
478 * incrementally visit all DBs. */
479 for (i = 0; i < server.dbnum; i++) {
480 j = (++next_db) % server.dbnum;
481 db = server.db+j;
482 dict = (server.maxmemory_policy == MAXMEMORY_ALLKEYS_RANDOM) ?
483 db->dict : db->expires;
484 if (dictSize(dict) != 0) {
485 de = dictGetRandomKey(dict);
486 bestkey = dictGetKey(de);
487 bestdbid = j;
488 break;
489 }
490 }
491 }
492
493 /* Finally remove the selected key. */
494 if (bestkey) {
495 db = server.db+bestdbid;
496 robj *keyobj = createStringObject(bestkey,sdslen(bestkey));
497 propagateExpire(db,keyobj,server.lazyfree_lazy_eviction);
498 /* We compute the amount of memory freed by db*Delete() alone.
499 * It is possible that actually the memory needed to propagate
500 * the DEL in AOF and replication link is greater than the one
501 * we are freeing removing the key, but we can't account for
502 * that otherwise we would never exit the loop.
503 *
504 * AOF and Output buffer memory will be freed eventually so
505 * we only care about memory used by the key space. */
506 delta = (long long) zmalloc_used_memory();
507 latencyStartMonitor(eviction_latency);
508 if (server.lazyfree_lazy_eviction)
509 dbAsyncDelete(db,keyobj);
510 else
511 dbSyncDelete(db,keyobj);
512 latencyEndMonitor(eviction_latency);
513 latencyAddSampleIfNeeded("eviction-del",eviction_latency);
514 latencyRemoveNestedEvent(latency,eviction_latency);
515 delta -= (long long) zmalloc_used_memory();
516 mem_freed += delta;
517 server.stat_evictedkeys++;
518 notifyKeyspaceEvent(NOTIFY_EVICTED, "evicted",
519 keyobj, db->id);
520 decrRefCount(keyobj);
521 keys_freed++;
522
523 /* When the memory to free starts to be big enough, we may
524 * start spending so much time here that is impossible to
525 * deliver data to the slaves fast enough, so we force the
526 * transmission here inside the loop. */
527 if (slaves) flushSlavesOutputBuffers();
528
529 /* Normally our stop condition is the ability to release
530 * a fixed, pre-computed amount of memory. However when we
531 * are deleting objects in another thread, it's better to
532 * check, from time to time, if we already reached our target
533 * memory, since the "mem_freed" amount is computed only
534 * across the dbAsyncDelete() call, while the thread can
535 * release the memory all the time. */
536 if (server.lazyfree_lazy_eviction && !(keys_freed % 16)) {
537 overhead = freeMemoryGetNotCountedMemory();
538 mem_used = zmalloc_used_memory();
539 mem_used = (mem_used > overhead) ? mem_used-overhead : 0;
540 if (mem_used <= server.maxmemory) {
541 mem_freed = mem_tofree;
542 }
543 }
544 }
545
546 if (!keys_freed) {
547 latencyEndMonitor(latency);
548 latencyAddSampleIfNeeded("eviction-cycle",latency);
549 goto cant_free; /* nothing to free... */
550 }
551 }
552 latencyEndMonitor(latency);
553 latencyAddSampleIfNeeded("eviction-cycle",latency);
554 return C_OK;
555
556cant_free:
557 /* We are here if we are not able to reclaim memory. There is only one
558 * last thing we can try: check if the lazyfree thread has jobs in queue
559 * and wait... */
560 while(bioPendingJobsOfType(BIO_LAZY_FREE)) {
561 if (((mem_reported - zmalloc_used_memory()) + mem_freed) >= mem_tofree)
562 break;
563 usleep(1000);
564 }
565 return C_ERR;
566}
