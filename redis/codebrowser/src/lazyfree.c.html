9/* Return the number of currently pending objects to free. */
10size_t lazyfreeGetPendingObjectsCount(void) {
11 size_t aux;
12 atomicGet(lazyfree_objects,aux);
13 return aux;
14}

16/* Return the amount of work needed in order to free an object.
17 * The return value is not always the actual number of allocations the
18 * object is compoesd of, but a number proportional to it.
19 *
20 * For strings the function always returns 1.
21 *
22 * For aggregated objects represented by hash tables or other data structures
23 * the function just returns the number of elements the object is composed of.
24 *
25 * Objects composed of single allocations are always reported as having a
26 * single item even if they are actaully logical composed of multiple
27 * elements.
28 *
29 * For lists the funciton returns the number of elements in the quicklist
30 * representing the list. */
31size_t lazyfreeGetFreeEffort(robj *obj) {
32 if (obj->type == OBJ_LIST) {
33 quicklist *ql = obj->ptr;
34 return ql->len;
35 } else if (obj->type == OBJ_SET && obj->encoding == OBJ_ENCODING_HT) {
36 dict *ht = obj->ptr;
37 return dictSize(ht);
38 } else if (obj->type == OBJ_ZSET && obj->encoding == OBJ_ENCODING_SKIPLIST){
39 zset *zs = obj->ptr;
40 return zs->zsl->length;
41 } else if (obj->type == OBJ_HASH && obj->encoding == OBJ_ENCODING_HT) {
42 dict *ht = obj->ptr;
43 return dictSize(ht);
44 } else {
45 return 1; /* Everything else is a single allocation. */
46 }
47}

49/* Delete a key, value, and associated expiration entry if any, from the DB.
50 * If there are enough allocations to free the value object may be put into
51 * a lazy free list instead of being freed synchronously. The lazy free list
52 * will be reclaimed in a different bio.c thread. */
53#define LAZYFREE_THRESHOLD 64
54int dbAsyncDelete(redisDb *db, robj *key) {
55 /* Deleting an entry from the expires dict will not free the sds of
56 * the key, because it is shared with the main dictionary. */
57 if (dictSize(db->expires) > 0) dictDelete(db->expires,key->ptr);
58
59 /* If the value is composed of a few allocations, to free in a lazy way
60 * is actually just slower... So under a certain limit we just free
61 * the object synchronously. */
62 dictEntry *de = dictUnlink(db->dict,key->ptr);
63 if (de) {
64 robj *val = dictGetVal(de);
65 size_t free_effort = lazyfreeGetFreeEffort(val);
66
67 /* If releasing the object is too much work, let's put it into the
68 * lazy free list. */
69 if (free_effort > LAZYFREE_THRESHOLD) {
70 atomicIncr(lazyfree_objects,1);
71 bioCreateBackgroundJob(BIO_LAZY_FREE,val,NULL,NULL);
72 dictSetVal(db->dict,de,NULL);
73 }
74 }
75
76 /* Release the key-val pair, or just the key if we set the val
77 * field to NULL in order to lazy free it later. */
78 if (de) {
79 dictFreeUnlinkedEntry(db->dict,de);
80 if (server.cluster_enabled) slotToKeyDel(key);
81 return 1;
82 } else {
83 return 0;
84 }
85}

87/* Empty a Redis DB asynchronously. What the function does actually is to
88 * create a new empty set of hash tables and scheduling the old ones for
89 * lazy freeing. */
90void emptyDbAsync(redisDb *db) {
91 dict *oldht1 = db->dict, *oldht2 = db->expires;
92 db->dict = dictCreate(&dbDictType,NULL);
93 db->expires = dictCreate(&keyptrDictType,NULL);
94 atomicIncr(lazyfree_objects,dictSize(oldht1));
95 bioCreateBackgroundJob(BIO_LAZY_FREE,NULL,oldht1,oldht2);
96}

98/* Empty the slots-keys map of Redis CLuster by creating a new empty one
99 * and scheduiling the old for lazy freeing. */
100void slotToKeyFlushAsync(void) {
101 rax *old = server.cluster->slots_to_keys;
102
103 server.cluster->slots_to_keys = raxNew();
104 memset(server.cluster->slots_keys_count,0,
105 sizeof(server.cluster->slots_keys_count));
106 atomicIncr(lazyfree_objects,old->numele);
107 bioCreateBackgroundJob(BIO_LAZY_FREE,NULL,NULL,old);
108}

110/* Release objects from the lazyfree thread. It's just decrRefCount()
111 * updating the count of objects to release. */
112void lazyfreeFreeObjectFromBioThread(robj *o) {
113 decrRefCount(o);
114 atomicDecr(lazyfree_objects,1);
115}

117/* Release a database from the lazyfree thread. The 'db' pointer is the
118 * database which was substitutied with a fresh one in the main thread
119 * when the database was logically deleted. 'sl' is a skiplist used by
120 * Redis Cluster in order to take the hash slots -> keys mapping. This
121 * may be NULL if Redis Cluster is disabled. */
122void lazyfreeFreeDatabaseFromBioThread(dict *ht1, dict *ht2) {
123 size_t numkeys = dictSize(ht1);
124 dictRelease(ht1);
125 dictRelease(ht2);
126 atomicDecr(lazyfree_objects,numkeys);
127}

129/* Release the skiplist mapping Redis Cluster keys to slots in the
130 * lazyfree thread. */
131void lazyfreeFreeSlotsMapFromBioThread(rax *rt) {
132 size_t len = rt->numele;
133 raxFree(rt);
134 atomicDecr(lazyfree_objects,len);
135}
