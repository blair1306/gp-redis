41/* Low level key lookup API, not actually called directly from commands
42 * implementations that should instead rely on lookupKeyRead(),
43 * lookupKeyWrite() and lookupKeyReadWithFlags(). */
44robj *lookupKey(redisDb *db, robj *key, int flags) {
45 dictEntry *de = dictFind(db->dict,key->ptr);
46 if (de) {
47 robj *val = dictGetVal(de);
48
49 /* Update the access time for the ageing algorithm.
50 * Don't do it if we have a saving child, as this will trigger
51 * a copy on write madness. */
52 if (server.rdb_child_pid == -1 &&
53 server.aof_child_pid == -1 &&
54 !(flags & LOOKUP_NOTOUCH))
55 {
56 if (server.maxmemory_policy & MAXMEMORY_FLAG_LFU) {
57 unsigned long ldt = val->lru >> 8;
58 unsigned long counter = LFULogIncr(val->lru & 255);
59 val->lru = (ldt << 8) | counter;
60 } else {
61 val->lru = LRU_CLOCK();
62 }
63 }
64 return val;
65 } else {
66 return NULL;
67 }
68}

70/* Lookup a key for read operations, or return NULL if the key is not found
71 * in the specified DB.
72 *
73 * As a side effect of calling this function:
74 * 1. A key gets expired if it reached it's TTL.
75 * 2. The key last access time is updated.
76 * 3. The global keys hits/misses stats are updated (reported in INFO).
77 *
78 * This API should not be used when we write to the key after obtaining
79 * the object linked to the key, but only for read only operations.
80 *
81 * Flags change the behavior of this command:
82 *
83 * LOOKUP_NONE (or zero): no special flags are passed.
84 * LOOKUP_NOTOUCH: don't alter the last access time of the key.
85 *
86 * Note: this function also returns NULL is the key is logically expired
87 * but still existing, in case this is a slave, since this API is called only
88 * for read operations. Even if the key expiry is master-driven, we can
89 * correctly report a key is expired on slaves even if the master is lagging
90 * expiring our key via DELs in the replication link. */
91robj *lookupKeyReadWithFlags(redisDb *db, robj *key, int flags) {
92 robj *val;
93
94 if (expireIfNeeded(db,key) == 1) {
95 /* Key expired. If we are in the context of a master, expireIfNeeded()
96 * returns 0 only when the key does not exist at all, so it's safe
97 * to return NULL ASAP. */
98 if (server.masterhost == NULL) return NULL;
99
100 /* However if we are in the context of a slave, expireIfNeeded() will
101 * not really try to expire the key, it only returns information
102 * about the "logical" status of the key: key expiring is up to the
103 * master in order to have a consistent view of master's data set.
104 *
105 * However, if the command caller is not the master, and as additional
106 * safety measure, the command invoked is a read-only command, we can
107 * safely return NULL here, and provide a more consistent behavior
108 * to clients accessign expired values in a read-only fashion, that
109 * will say the key as non exisitng.
110 *
111 * Notably this covers GETs when slaves are used to scale reads. */
112 if (server.current_client &&
113 server.current_client != server.master &&
114 server.current_client->cmd &&
115 server.current_client->cmd->flags & CMD_READONLY)
116 {
117 return NULL;
118 }
119 }
120 val = lookupKey(db,key,flags);
121 if (val == NULL)
122 server.stat_keyspace_misses++;
123 else
124 server.stat_keyspace_hits++;
125 return val;
126}

128/* Like lookupKeyReadWithFlags(), but does not use any flag, which is the
129 * common case. */
130robj *lookupKeyRead(redisDb *db, robj *key) {
131 return lookupKeyReadWithFlags(db,key,LOOKUP_NONE);
132}

134/* Lookup a key for write operations, and as a side effect, if needed, expires
135 * the key if its TTL is reached.
136 *
137 * Returns the linked value object if the key exists or NULL if the key
138 * does not exist in the specified DB. */
139robj *lookupKeyWrite(redisDb *db, robj *key) {
140 expireIfNeeded(db,key);
141 return lookupKey(db,key,LOOKUP_NONE);
142}

156/* Add the key to the DB. It's up to the caller to increment the reference
157 * counter of the value if needed.
158 *
159 * The program is aborted if the key already exists. */
160void dbAdd(redisDb *db, robj *key, robj *val) {
161 sds copy = sdsdup(key->ptr);
162 int retval = dictAdd(db->dict, copy, val);
163
164 serverAssertWithInfo(NULL,key,retval == DICT_OK);
165 if (val->type == OBJ_LIST) signalListAsReady(db, key);
166 if (server.cluster_enabled) slotToKeyAdd(key);
167 }
168
169/* Overwrite an existing key with a new value. Incrementing the reference
170 * count of the new value is up to the caller.
171 * This function does not modify the expire time of the existing key.
172 *
173 * The program is aborted if the key was not already present. */
174void dbOverwrite(redisDb *db, robj *key, robj *val) {
175 dictEntry *de = dictFind(db->dict,key->ptr);
176
177 serverAssertWithInfo(NULL,key,de != NULL);
178 if (server.maxmemory_policy & MAXMEMORY_FLAG_LFU) {
179 robj *old = dictGetVal(de);
180 int saved_lru = old->lru;
181 dictReplace(db->dict, key->ptr, val);
182 val->lru = saved_lru;
183 } else {
184 dictReplace(db->dict, key->ptr, val);
185 }
186}

188/* High level Set operation. This function can be used in order to set
189 * a key, whatever it was existing or not, to a new object.
190 *
191 * 1) The ref count of the value object is incremented.
192 * 2) clients WATCHing for the destination key notified.
193 * 3) The expire time of the key is reset (the key is made persistent).
194 *
195 * All the new keys in the database should be craeted via this interface. */
196void setKey(redisDb *db, robj *key, robj *val) {
197 if (lookupKeyWrite(db,key) == NULL) {
198 dbAdd(db,key,val);
199 } else {
200 dbOverwrite(db,key,val);
201 }
202 incrRefCount(val);
203 removeExpire(db,key);
204 signalModifiedKey(db,key);
205}

208 return dictFind(db->dict,key->ptr) != NULL;

211/* Return a random key, in form of a Redis object.
212 * If there are no keys, NULL is returned.
213 *
214 * The function makes sure to return keys not already expired. */
215robj *dbRandomKey(redisDb *db) {
216 dictEntry *de;
217
218 while(1) {
219 sds key;
220 robj *keyobj;
221
222 de = dictGetRandomKey(db->dict);
223 if (de == NULL) return NULL;
224
225 key = dictGetKey(de);
226 keyobj = createStringObject(key,sdslen(key));
227 if (dictFind(db->expires,key)) {
228 if (expireIfNeeded(db,keyobj)) {
229 decrRefCount(keyobj);
230 continue; /* search for another key. This expired. */
231 }
232 }
233 return keyobj;
234 }
235}

237/* Delete a key, value, and associated expiration entry if any, from the DB */
238int dbSyncDelete(redisDb *db, robj *key) {
239 /* Deleting an entry from the expires dict will not free the sds of
240 * the key, because it is shared with the main dictionary. */
241 if (dictSize(db->expires) > 0) dictDelete(db->expires,key->ptr);
242 if (dictDelete(db->dict,key->ptr) == DICT_OK) {
243 if (server.cluster_enabled) slotToKeyDel(key);
244 return 1;
245 } else {
246 return 0;
247 }
248}

257/* Prepare the string object stored at 'key' to be modified destructively
258 * to implement commands like SETBIT or APPEND.
259 *
260 * An object is usually ready to be modified unless one of the two conditions
261 * are true:
262 *
263 * 1) The object 'o' is shared (refcount > 1), we don't want to affect
264 * other users.
265 * 2) The object encoding is not "RAW".
266 *
267 * If the object is found in one of the above conditions (or both) by the
268 * function, an unshared / not-encoded copy of the string object is stored
269 * at 'key' in the specified 'db'. Otherwise the object 'o' itself is
270 * returned.
271 *
272 * USAGE:
273 *
274 * The object 'o' is what the caller already obtained by looking up 'key'
275 * in 'db', the usage pattern looks like this:
276 *
277 * o = lookupKeyWrite(db,key);
278 * if (checkType(c,o,OBJ_STRING)) return;
279 * o = dbUnshareStringValue(db,key,o);
280 *
281 * At this point the caller is ready to modify the object, for example
282 * using an sdscat() call to append some data, or anything else.
283 */
284robj *dbUnshareStringValue(redisDb *db, robj *key, robj *o) {
285 serverAssert(o->type == OBJ_STRING);
286 if (o->refcount != 1 || o->encoding != OBJ_ENCODING_RAW) {
287 robj *decoded = getDecodedObject(o);
288 o = createRawStringObject(decoded->ptr, sdslen(decoded->ptr));
289 decrRefCount(decoded);
290 dbOverwrite(db,key,o);
291 }
292 return o;
293}

295/* Remove all keys from all the databases in a Redis server.
296 * If callback is given the function is called from time to time to
297 * signal that work is in progress.
298 *
299 * The dbnum can be -1 if all teh DBs should be flushed, or the specified
300 * DB number if we want to flush only a single Redis database number.
301 *
302 * Flags are be EMPTYDB_NO_FLAGS if no special flags are specified or
303 * EMPTYDB_ASYNC if we want the memory to be freed in a different thread
304 * and the function to return ASAP.
305 *
306 * On success the fuction returns the number of keys removed from the
307 * database(s). Otherwise -1 is returned in the specific case the
308 * DB number is out of range, and errno is set to EINVAL. */
309long long emptyDb(int dbnum, int flags, void(callback)(void*)) {
310 int j, async = (flags & EMPTYDB_ASYNC);
311 long long removed = 0;
312
313 if (dbnum < -1 || dbnum >= server.dbnum) {
314 errno = EINVAL;
315 return -1;
316 }
317
318 for (j = 0; j < server.dbnum; j++) {
319 if (dbnum != -1 && dbnum != j) continue;
320 removed += dictSize(server.db[j].dict);
321 if (async) {
322 emptyDbAsync(&server.db[j]);
323 } else {
324 dictEmpty(server.db[j].dict,callback);
325 dictEmpty(server.db[j].expires,callback);
326 }
327 }
328 if (server.cluster_enabled) {
329 if (async) {
330 slotToKeyFlushAsync();
331 } else {
332 slotToKeyFlush();
333 }
334 }
335 if (dbnum == -1) flushSlaveKeysWithExpireList();
336 return removed;
337}

367/* Return the set of flags to use for the emptyDb() call for FLUSHALL
368 * and FLUSHDB commands.
369 *
370 * Currently the command just attempts to parse the "ASYNC" option. It
371 * also checks if the command arity is wrong.
372 *
373 * On success C_OK is returned and the flags are stored in *flags, otherwise
374 * C_ERR is returned and the function sends an error to the client. */
375int getFlushCommandFlags(client *c, int *flags) {
376 /* Parse the optional ASYNC option. */
377 if (c->argc > 1) {
378 if (c->argc > 2 || strcasecmp(c->argv[1]->ptr,"async")) {
379 addReply(c,shared.syntaxerr);
380 return C_ERR;
381 }
382 *flags = EMPTYDB_ASYNC;
383 } else {
384 *flags = EMPTYDB_NO_FLAGS;
385 }
386 return C_OK;
387}

389/* FLUSHDB [ASYNC]
390 *
391 * Flushes the currently SELECTed Redis DB. */
392void flushdbCommand(client *c) {
393 int flags;
394
395 if (getFlushCommandFlags(c,&flags) == C_ERR) return;
396 signalFlushedDb(c->db->id);
397 server.dirty += emptyDb(c->db->id,flags,NULL);
398 addReply(c,shared.ok);
399}
400
401/* FLUSHALL [ASYNC]
402 *
403 * Flushes the whole server data set. */
404void flushallCommand(client *c) {
405 int flags;
406
407 if (getFlushCommandFlags(c,&flags) == C_ERR) return;
408 signalFlushedDb(-1);
409 server.dirty += emptyDb(-1,flags,NULL);
410 addReply(c,shared.ok);
411 if (server.rdb_child_pid != -1) {
412 kill(server.rdb_child_pid,SIGUSR1);
413 rdbRemoveTempFile(server.rdb_child_pid);
414 }
415 if (server.saveparamslen > 0) {
416 /* Normally rdbSave() will reset dirty, but we don't want this here
417 * as otherwise FLUSHALL will not be replicated nor put into the AOF. */
418 int saved_dirty = server.dirty;
419 rdbSaveInfo rsi, *rsiptr;
420 rsiptr = rdbPopulateSaveInfo(&rsi);
421 rdbSave(server.rdb_filename,rsiptr);
422 server.dirty = saved_dirty;
423 }
424 server.dirty++;
425}

427/* This command implements DEL and LAZYDEL. */
428void delGenericCommand(client *c, int lazy) {
429 int numdel = 0, j;
430
431 for (j = 1; j < c->argc; j++) {
432 expireIfNeeded(c->db,c->argv[j]);
433 int deleted = lazy ? dbAsyncDelete(c->db,c->argv[j]) :
434 dbSyncDelete(c->db,c->argv[j]);
435 if (deleted) {
436 signalModifiedKey(c->db,c->argv[j]);
437 notifyKeyspaceEvent(NOTIFY_GENERIC,
438 "del",c->argv[j],c->db->id);
439 server.dirty++;
440 numdel++;
441 }
442 }
443 addReplyLongLong(c,numdel);
444}

497void keysCommand(client *c) {
498 dictIterator *di;
499 dictEntry *de;
500 sds pattern = c->argv[1]->ptr;
501 int plen = sdslen(pattern), allkeys;
502 unsigned long numkeys = 0;
503 void *replylen = addDeferredMultiBulkLength(c);
504
505 di = dictGetSafeIterator(c->db->dict);
506 allkeys = (pattern[0] == '*' && pattern[1] == '\0');
507 while((de = dictNext(di)) != NULL) {
508 sds key = dictGetKey(de);
509 robj *keyobj;
510
511 if (allkeys || stringmatchlen(pattern,plen,key,sdslen(key),0)) {
512 keyobj = createStringObject(key,sdslen(key));
513 if (expireIfNeeded(c->db,keyobj) == 0) {
514 addReplyBulk(c,keyobj);
515 numkeys++;
516 }
517 decrRefCount(keyobj);
518 }
519 }
520 dictReleaseIterator(di);
521 setDeferredMultiBulkLength(c,replylen,numkeys);
522}

524/* This callback is used by scanGenericCommand in order to collect elements
525 * returned by the dictionary iterator into a list. */
526void scanCallback(void *privdata, const dictEntry *de) {
527 void **pd = (void**) privdata;
528 list *keys = pd[0];
529 robj *o = pd[1];
530 robj *key, *val = NULL;
531
532 if (o == NULL) {
533 sds sdskey = dictGetKey(de);
534 key = createStringObject(sdskey, sdslen(sdskey));
535 } else if (o->type == OBJ_SET) {
536 sds keysds = dictGetKey(de);
537 key = createStringObject(keysds,sdslen(keysds));
538 } else if (o->type == OBJ_HASH) {
539 sds sdskey = dictGetKey(de);
540 sds sdsval = dictGetVal(de);
541 key = createStringObject(sdskey,sdslen(sdskey));
542 val = createStringObject(sdsval,sdslen(sdsval));
543 } else if (o->type == OBJ_ZSET) {
544 sds sdskey = dictGetKey(de);
545 key = createStringObject(sdskey,sdslen(sdskey));
546 val = createStringObjectFromLongDouble(*(double*)dictGetVal(de),0);
547 } else {
548 serverPanic("Type not handled in SCAN callback.");
549 }
550
551 listAddNodeTail(keys, key);
552 if (val) listAddNodeTail(keys, val);
553}

555/* Try to parse a SCAN cursor stored at object 'o':
556 * if the cursor is valid, store it as unsigned integer into *cursor and
557 * returns C_OK. Otherwise return C_ERR and send an error to the
558 * client. */
559int parseScanCursorOrReply(client *c, robj *o, unsigned long *cursor) {
560 char *eptr;
561
562 /* Use strtoul() because we need an *unsigned* long, so
563 * getLongLongFromObject() does not cover the whole cursor space. */
564 errno = 0;
565 *cursor = strtoul(o->ptr, &eptr, 10);
566 if (isspace(((char*)o->ptr)[0]) || eptr[0] != '\0' || errno == ERANGE)
567 {
568 addReplyError(c, "invalid cursor");
569 return C_ERR;
570 }
571 return C_OK;
572}

574/* This command implements SCAN, HSCAN and SSCAN commands.
575 * If object 'o' is passed, then it must be a Hash or Set object, otherwise
576 * if 'o' is NULL the command will operate on the dictionary associated with
577 * the current database.
578 *
579 * When 'o' is not NULL the function assumes that the first argument in
580 * the client arguments vector is a key so it skips it before iterating
581 * in order to parse options.
582 *
583 * In the case of a Hash object the function returns both the field and value
584 * of every element on the Hash. */
585void scanGenericCommand(client *c, robj *o, unsigned long cursor) {
586 int i, j;
587 list *keys = listCreate();
588 listNode *node, *nextnode;
589 long count = 10;
590 sds pat = NULL;
591 int patlen = 0, use_pattern = 0;
592 dict *ht;
593
594 /* Object must be NULL (to iterate keys names), or the type of the object
595 * must be Set, Sorted Set, or Hash. */
596 serverAssert(o == NULL || o->type == OBJ_SET || o->type == OBJ_HASH ||
597 o->type == OBJ_ZSET);
598
599 /* Set i to the first option argument. The previous one is the cursor. */
600 i = (o == NULL) ? 2 : 3; /* Skip the key argument if needed. */
601
602 /* Step 1: Parse options. */
603 while (i < c->argc) {
604 j = c->argc - i;
605 if (!strcasecmp(c->argv[i]->ptr, "count") && j >= 2) {
606 if (getLongFromObjectOrReply(c, c->argv[i+1], &count, NULL)
607 != C_OK)
608 {
609 goto cleanup;
610 }
611
612 if (count < 1) {
613 addReply(c,shared.syntaxerr);
614 goto cleanup;
615 }
616
617 i += 2;
618 } else if (!strcasecmp(c->argv[i]->ptr, "match") && j >= 2) {
619 pat = c->argv[i+1]->ptr;
620 patlen = sdslen(pat);
621
622 /* The pattern always matches if it is exactly "*", so it is
623 * equivalent to disabling it. */
624 use_pattern = !(pat[0] == '*' && patlen == 1);
625
626 i += 2;
627 } else {
628 addReply(c,shared.syntaxerr);
629 goto cleanup;
630 }
631 }
632
633 /* Step 2: Iterate the collection.
634 *
635 * Note that if the object is encoded with a ziplist, intset, or any other
636 * representation that is not a hash table, we are sure that it is also
637 * composed of a small number of elements. So to avoid taking state we
638 * just return everything inside the object in a single call, setting the
639 * cursor to zero to signal the end of the iteration. */
640
641 /* Handle the case of a hash table. */
642 ht = NULL;
643 if (o == NULL) {
644 ht = c->db->dict;
645 } else if (o->type == OBJ_SET && o->encoding == OBJ_ENCODING_HT) {
646 ht = o->ptr;
647 } else if (o->type == OBJ_HASH && o->encoding == OBJ_ENCODING_HT) {
648 ht = o->ptr;
649 count *= 2; /* We return key / value for this type. */
650 } else if (o->type == OBJ_ZSET && o->encoding == OBJ_ENCODING_SKIPLIST) {
651 zset *zs = o->ptr;
652 ht = zs->dict;
653 count *= 2; /* We return key / value for this type. */
654 }
655
656 if (ht) {
657 void *privdata[2];
658 /* We set the max number of iterations to ten times the specified
659 * COUNT, so if the hash table is in a pathological state (very
660 * sparsely populated) we avoid to block too much time at the cost
661 * of returning no or very few elements. */
662 long maxiterations = count*10;
663
664 /* We pass two pointers to the callback: the list to which it will
665 * add new elements, and the object containing the dictionary so that
666 * it is possible to fetch more data in a type-dependent way. */
667 privdata[0] = keys;
668 privdata[1] = o;
669 do {
670 cursor = dictScan(ht, cursor, scanCallback, NULL, privdata);
671 } while (cursor &&
672 maxiterations-- &&
673 listLength(keys) < (unsigned long)count);
674 } else if (o->type == OBJ_SET) {
675 int pos = 0;
676 int64_t ll;
677
678 while(intsetGet(o->ptr,pos++,&ll))
679 listAddNodeTail(keys,createStringObjectFromLongLong(ll));
680 cursor = 0;
681 } else if (o->type == OBJ_HASH || o->type == OBJ_ZSET) {
682 unsigned char *p = ziplistIndex(o->ptr,0);
683 unsigned char *vstr;
684 unsigned int vlen;
685 long long vll;
686
687 while(p) {
688 ziplistGet(p,&vstr,&vlen,&vll);
689 listAddNodeTail(keys,
690 (vstr != NULL) ? createStringObject((char*)vstr,vlen) :
691 createStringObjectFromLongLong(vll));
692 p = ziplistNext(o->ptr,p);
693 }
694 cursor = 0;
695 } else {
696 serverPanic("Not handled encoding in SCAN.");
697 }
698
699 /* Step 3: Filter elements. */
700 node = listFirst(keys);
701 while (node) {
702 robj *kobj = listNodeValue(node);
703 nextnode = listNextNode(node);
704 int filter = 0;
705
706 /* Filter element if it does not match the pattern. */
707 if (!filter && use_pattern) {
708 if (sdsEncodedObject(kobj)) {
709 if (!stringmatchlen(pat, patlen, kobj->ptr, sdslen(kobj->ptr), 0))
710 filter = 1;
711 } else {
712 char buf[LONG_STR_SIZE];
713 int len;
714
715 serverAssert(kobj->encoding == OBJ_ENCODING_INT);
716 len = ll2string(buf,sizeof(buf),(long)kobj->ptr);
717 if (!stringmatchlen(pat, patlen, buf, len, 0)) filter = 1;
718 }
719 }
720
721 /* Filter element if it is an expired key. */
722 if (!filter && o == NULL && expireIfNeeded(c->db, kobj)) filter = 1;
723
724 /* Remove the element and its associted value if needed. */
725 if (filter) {
726 decrRefCount(kobj);
727 listDelNode(keys, node);
728 }
729
730 /* If this is a hash or a sorted set, we have a flat list of
731 * key-value elements, so if this element was filtered, remove the
732 * value, or skip it if it was not filtered: we only match keys. */
733 if (o && (o->type == OBJ_ZSET || o->type == OBJ_HASH)) {
734 node = nextnode;
735 nextnode = listNextNode(node);
736 if (filter) {
737 kobj = listNodeValue(node);
738 decrRefCount(kobj);
739 listDelNode(keys, node);
740 }
741 }
742 node = nextnode;
743 }
744
745 /* Step 4: Reply to the client. */
746 addReplyMultiBulkLen(c, 2);
747 addReplyBulkLongLong(c,cursor);
748
749 addReplyMultiBulkLen(c, listLength(keys));
750 while ((node = listFirst(keys)) != NULL) {
751 robj *kobj = listNodeValue(node);
752 addReplyBulk(c, kobj);
753 decrRefCount(kobj);
754 listDelNode(keys, node);
755 }
756
757cleanup:
758 listSetFreeMethod(keys,decrRefCountVoid);
759 listRelease(keys);
760}

879void moveCommand(client *c) {
880 robj *o;
881 redisDb *src, *dst;
882 int srcid;
883 long long dbid, expire;
884
885 if (server.cluster_enabled) {
886 addReplyError(c,"MOVE is not allowed in cluster mode");
887 return;
888 }
889
890 /* Obtain source and target DB pointers */
891 src = c->db;
892 srcid = c->db->id;
893
894 if (getLongLongFromObject(c->argv[2],&dbid) == C_ERR ||
895 dbid < INT_MIN || dbid > INT_MAX ||
896 selectDb(c,dbid) == C_ERR)
897 {
898 addReply(c,shared.outofrangeerr);
899 return;
900 }
901 dst = c->db;
902 selectDb(c,srcid); /* Back to the source DB */
903
904 /* If the user is moving using as target the same
905 * DB as the source DB it is probably an error. */
906 if (src == dst) {
907 addReply(c,shared.sameobjecterr);
908 return;
909 }
910
911 /* Check if the element exists and get a reference */
912 o = lookupKeyWrite(c->db,c->argv[1]);
913 if (!o) {
914 addReply(c,shared.czero);
915 return;
916 }
917 expire = getExpire(c->db,c->argv[1]);
918
919 /* Return zero if the key already exists in the target DB */
920 if (lookupKeyWrite(dst,c->argv[1]) != NULL) {
921 addReply(c,shared.czero);
922 return;
923 }
924 dbAdd(dst,c->argv[1],o);
925 if (expire != -1) setExpire(c,dst,c->argv[1],expire);
926 incrRefCount(o);
927
928 /* OK! key moved, free the entry in the source DB */
929 dbDelete(src,c->argv[1]);
930 server.dirty++;
931 addReply(c,shared.cone);
932}

934/* Helper function for dbSwapDatabases(): scans the list of keys that have
935 * one or more blocked clients for B[LR]POP or other list blocking commands
936 * and signal the keys are ready if they are lists. See the comment where
937 * the function is used for more info. */
938void scanDatabaseForReadyLists(redisDb *db) {
939 dictEntry *de;
940 dictIterator *di = dictGetSafeIterator(db->blocking_keys);
941 while((de = dictNext(di)) != NULL) {
942 robj *key = dictGetKey(de);
943 robj *value = lookupKey(db,key,LOOKUP_NOTOUCH);
944 if (value && value->type == OBJ_LIST)
945 signalListAsReady(db, key);
946 }
947 dictReleaseIterator(di);
948}

950/* Swap two databases at runtime so that all clients will magically see
951 * the new database even if already connected. Note that the client
952 * structure c->db points to a given DB, so we need to be smarter and
953 * swap the underlying referenced structures, otherwise we would need
954 * to fix all the references to the Redis DB structure.
955 *
956 * Returns C_ERR if at least one of the DB ids are out of range, otherwise
957 * C_OK is returned. */
958int dbSwapDatabases(int id1, int id2) {
959 if (id1 < 0 || id1 >= server.dbnum ||
960 id2 < 0 || id2 >= server.dbnum) return C_ERR;
961 if (id1 == id2) return C_OK;
962 redisDb aux = server.db[id1];
963 redisDb *db1 = &server.db[id1], *db2 = &server.db[id2];
964
965 /* Swap hash tables. Note that we don't swap blocking_keys,
966 * ready_keys and watched_keys, since we want clients to
967 * remain in the same DB they were. */
968 db1->dict = db2->dict;
969 db1->expires = db2->expires;
970 db1->avg_ttl = db2->avg_ttl;
971
972 db2->dict = aux.dict;
973 db2->expires = aux.expires;
974 db2->avg_ttl = aux.avg_ttl;
975
976 /* Now we need to handle clients blocked on lists: as an effect
977 * of swapping the two DBs, a client that was waiting for list
978 * X in a given DB, may now actually be unblocked if X happens
979 * to exist in the new version of the DB, after the swap.
980 *
981 * However normally we only do this check for efficiency reasons
982 * in dbAdd() when a list is created. So here we need to rescan
983 * the list of clients blocked on lists and signal lists as ready
984 * if needed. */
985 scanDatabaseForReadyLists(db1);
986 scanDatabaseForReadyLists(db2);
987 return C_OK;
988}

1023int removeExpire(redisDb *db, robj *key) {
1024 /* An expire may only be removed if there is a corresponding entry in the
1025 * main dict. Otherwise, the key will never be freed. */
1026 serverAssertWithInfo(NULL,key,dictFind(db->dict,key->ptr) != NULL);
1027 return dictDelete(db->expires,key->ptr) == DICT_OK;
1028}

1030/* Set an expire to the specified key. If the expire is set in the context
1031 * of an user calling a command 'c' is the client, otherwise 'c' is set
1032 * to NULL. The 'when' parameter is the absolute unix time in milliseconds
1033 * after which the key will no longer be considered valid. */
1034void setExpire(client *c, redisDb *db, robj *key, long long when) {
1035 dictEntry *kde, *de;
1036
1037 /* Reuse the sds from the main dict in the expire dict */
1038 kde = dictFind(db->dict,key->ptr);
1039 serverAssertWithInfo(NULL,key,kde != NULL);
1040 de = dictAddOrFind(db->expires,dictGetKey(kde));
1041 dictSetSignedIntegerVal(de,when);
1042
1043 int writable_slave = server.masterhost && server.repl_slave_ro == 0;
1044 if (c && writable_slave && !(c->flags & CLIENT_MASTER))
1045 rememberSlaveKeyWithExpire(db,key);
1046}

1048/* Return the expire time of the specified key, or -1 if no expire
1049 * is associated with this key (i.e. the key is non volatile) */
1050long long getExpire(redisDb *db, robj *key) {
1051 dictEntry *de;
1052
1053 /* No expire? return ASAP */
1054 if (dictSize(db->expires) == 0 ||
1055 (de = dictFind(db->expires,key->ptr)) == NULL) return -1;
1056
1057 /* The entry was found in the expire dict, this means it should also
1058 * be present in the main dict (safety check). */
1059 serverAssertWithInfo(NULL,key,dictFind(db->dict,key->ptr) != NULL);
1060 return dictGetSignedIntegerVal(de);
1061}

1063/* Propagate expires into slaves and the AOF file.
1064 * When a key expires in the master, a DEL operation for this key is sent
1065 * to all the slaves and the AOF file if enabled.
1066 *
1067 * This way the key expiry is centralized in one place, and since both
1068 * AOF and the master->slave link guarantee operation ordering, everything
1069 * will be consistent even if we allow write operations against expiring
1070 * keys. */
1071void propagateExpire(redisDb *db, robj *key, int lazy) {
1072 robj *argv[2];
1073
1074 argv[0] = lazy ? shared.unlink : shared.del;
1075 argv[1] = key;
1076 incrRefCount(argv[0]);
1077 incrRefCount(argv[1]);
1078
1079 if (server.aof_state != AOF_OFF)
1080 feedAppendOnlyFile(server.delCommand,db->id,argv,2);
1081 replicationFeedSlaves(server.slaves,db->id,argv,2);
1082
1083 decrRefCount(argv[0]);
1084 decrRefCount(argv[1]);
1085}

1087int expireIfNeeded(redisDb *db, robj *key) {
1088 mstime_t when = getExpire(db,key);
1089 mstime_t now;
1090
1091 if (when < 0) return 0; /* No expire for this key */
1092
1093 /* Don't expire anything while loading. It will be done later. */
1094 if (server.loading) return 0;
1095
1096 /* If we are in the context of a Lua script, we claim that time is
1097 * blocked to when the Lua script started. This way a key can expire
1098 * only the first time it is accessed and not in the middle of the
1099 * script execution, making propagation to slaves / AOF consistent.
1100 * See issue #1525 on Github for more information. */
1101 now = server.lua_caller ? server.lua_time_start : mstime();
1102
1103 /* If we are running in the context of a slave, return ASAP:
1104 * the slave key expiration is controlled by the master that will
1105 * send us synthesized DEL operations for expired keys.
1106 *
1107 * Still we try to return the right information to the caller,
1108 * that is, 0 if we think the key should be still valid, 1 if
1109 * we think the key is expired at this time. */
1110 if (server.masterhost != NULL) return now > when;
1111
1112 /* Return when this key has not expired */
1113 if (now <= when) return 0;
1114
1115 /* Delete the key */
1116 server.stat_expiredkeys++;
1117 propagateExpire(db,key,server.lazyfree_lazy_expire);
1118 notifyKeyspaceEvent(NOTIFY_EXPIRED,
1119 "expired",key,db->id);
1120 return server.lazyfree_lazy_expire ? dbAsyncDelete(db,key) :
1121 dbSyncDelete(db,key);
1122}

1128/* The base case is to use the keys position as given in the command table
1129 * (firstkey, lastkey, step). */
1130int *getKeysUsingCommandTable(struct redisCommand *cmd,robj **argv, int argc, int *numkeys) {
1131 int j, i = 0, last, *keys;
1132 UNUSED(argv);
1133
1134 if (cmd->firstkey == 0) {
1135 *numkeys = 0;
1136 return NULL;
1137 }
1138
1139 last = cmd->lastkey;
1140 if (last < 0) last = argc+last;
1141 keys = zmalloc(sizeof(int)*((last - cmd->firstkey)+1));
1142 for (j = cmd->firstkey; j <= last; j += cmd->keystep) {
1143 if (j >= argc) {
1144 /* Modules command do not have dispatch time arity checks, so
1145 * we need to handle the case where the user passed an invalid
1146 * number of arguments here. In this case we return no keys
1147 * and expect the module command to report an arity error. */
1148 if (cmd->flags & CMD_MODULE) {
1149 zfree(keys);
1150 *numkeys = 0;
1151 return NULL;
1152 } else {
1153 serverPanic("Redis built-in command declared keys positions not matching the arity requirements.");
1154 }
1155 }
1156 keys[i++] = j;
1157 }
1158 *numkeys = i;
1159 return keys;
1160}

1162/* Return all the arguments that are keys in the command passed via argc / argv.
1163 *
1164 * The command returns the positions of all the key arguments inside the array,
1165 * so the actual return value is an heap allocated array of integers. The
1166 * length of the array is returned by reference into *numkeys.
1167 *
1168 * 'cmd' must be point to the corresponding entry into the redisCommand
1169 * table, according to the command name in argv[0].
1170 *
1171 * This function uses the command table if a command-specific helper function
1172 * is not required, otherwise it calls the command-specific function. */
1173int *getKeysFromCommand(struct redisCommand *cmd, robj **argv, int argc, int *numkeys) {
1174 if (cmd->flags & CMD_MODULE_GETKEYS) {
1175 return moduleGetCommandKeysViaAPI(cmd,argv,argc,numkeys);
1176 } else if (!(cmd->flags & CMD_MODULE) && cmd->getkeys_proc) {
1177 return cmd->getkeys_proc(cmd,argv,argc,numkeys);
1178 } else {
1179 return getKeysUsingCommandTable(cmd,argv,argc,numkeys);
1180 }
1181}

1188/* Helper function to extract keys from following commands:
1189 * ZUNIONSTORE <destkey> <num-keys> <key> <key> ... <key> <options>
1190 * ZINTERSTORE <destkey> <num-keys> <key> <key> ... <key> <options> */
1191int *zunionInterGetKeys(struct redisCommand *cmd, robj **argv, int argc, int *numkeys) {
1192 int i, num, *keys;
1193 UNUSED(cmd);
1194
1195 num = atoi(argv[2]->ptr);
1196 /* Sanity check. Don't return any key if the command is going to
1197 * reply with syntax error. */
1198 if (num > (argc-3)) {
1199 *numkeys = 0;
1200 return NULL;
1201 }
1202
1203 /* Keys in z{union,inter}store come from two places:
1204 * argv[1] = storage key,
1205 * argv[3...n] = keys to intersect */
1206 keys = zmalloc(sizeof(int)*(num+1));
1207
1208 /* Add all key positions for argv[3...n] to keys[] */
1209 for (i = 0; i < num; i++) keys[i] = 3+i;
1210
1211 /* Finally add the argv[1] key position (the storage key target). */
1212 keys[num] = 1;
1213 *numkeys = num+1; /* Total keys = {union,inter} keys + storage key */
1214 return keys;
1215}

1241/* Helper function to extract keys from the SORT command.
1242 *
1243 * SORT <sort-key> ... STORE <store-key> ...
1244 *
1245 * The first argument of SORT is always a key, however a list of options
1246 * follow in SQL-alike style. Here we parse just the minimum in order to
1247 * correctly identify keys in the "STORE" option. */
1248int *sortGetKeys(struct redisCommand *cmd, robj **argv, int argc, int *numkeys) {
1249 int i, j, num, *keys, found_store = 0;
1250 UNUSED(cmd);
1251
1252 num = 0;
1253 keys = zmalloc(sizeof(int)*2); /* Alloc 2 places for the worst case. */
1254
1255 keys[num++] = 1; /* <sort-key> is always present. */
1256
1257 /* Search for STORE option. By default we consider options to don't
1258 * have arguments, so if we find an unknown option name we scan the
1259 * next. However there are options with 1 or 2 arguments, so we
1260 * provide a list here in order to skip the right number of args. */
1261 struct {
1262 char *name;
1263 int skip;
1264 } skiplist[] = {
1265 {"limit", 2},
1266 {"get", 1},
1267 {"by", 1},
1268 {NULL, 0} /* End of elements. */
1269 };
1270
1271 for (i = 2; i < argc; i++) {
1272 for (j = 0; skiplist[j].name != NULL; j++) {
1273 if (!strcasecmp(argv[i]->ptr,skiplist[j].name)) {
1274 i += skiplist[j].skip;
1275 break;
1276 } else if (!strcasecmp(argv[i]->ptr,"store") && i+1 < argc) {
1277 /* Note: we don't increment "num" here and continue the loop
1278 * to be sure to process the *last* "STORE" option if multiple
1279 * ones are provided. This is same behavior as SORT. */
1280 found_store = 1;
1281 keys[num] = i+1; /* <store-key> */
1282 break;
1283 }
1284 }
1285 }
1286 *numkeys = num + found_store;
1287 return keys;
1288}

1323 UNUSED(cmd);
1324
1325 /* Check for the presence of the stored key in the command */
1326 int stored_key = -1;
1327 for (i = 5; i < argc; i++) {
1328 char *arg = argv[i]->ptr;
1329 /* For the case when user specifies both "store" and "storedist" options, the
1330 * second key specified would override the first key. This behavior is kept
1331 * the same as in georadiusCommand method.
1332 */
1333 if ((!strcasecmp(arg, "store") || !strcasecmp(arg, "storedist")) && ((i+1) < argc)) {
1334 stored_key = i+1;
1335 i++;
1336 }
1337 }
1338 num = 1 + (stored_key == -1 ? 0 : 1);
1339
1340 /* Keys in the command come from two places:
1341 * argv[1] = key,
1342 * argv[5...n] = stored key if present
1343 */
1344 keys = zmalloc(sizeof(int) * num);
1345
1346 /* Add all key positions to keys[] */
1347 keys[0] = 1;
1348 if(num > 1) {
1349 keys[1] = stored_key;
1350 }
1351 *numkeys = num;
1352 return keys;
1353}

1355/* Slot to Key API. This is used by Redis Cluster in order to obtain in
1356 * a fast way a key that belongs to a specified hash slot. This is useful
1357 * while rehashing the cluster and in other conditions when we need to
1358 * understand if we have keys for a given hash slot. */
1359void slotToKeyUpdateKey(robj *key, int add) {
1360 unsigned int hashslot = keyHashSlot(key->ptr,sdslen(key->ptr));
1361 unsigned char buf[64];
1362 unsigned char *indexed = buf;
1363 size_t keylen = sdslen(key->ptr);
1364
1365 server.cluster->slots_keys_count[hashslot] += add ? 1 : -1;
1366 if (keylen+2 > 64) indexed = zmalloc(keylen+2);
1367 indexed[0] = (hashslot >> 8) & 0xff;
1368 indexed[1] = hashslot & 0xff;
1369 memcpy(indexed+2,key->ptr,keylen);
1370 if (add) {
1371 raxInsert(server.cluster->slots_to_keys,indexed,keylen+2,NULL,NULL);
1372 } else {
1373 raxRemove(server.cluster->slots_to_keys,indexed,keylen+2,NULL);
1374 }
1375 if (indexed != buf) zfree(indexed);
1376}
1377
1378void slotToKeyAdd(robj *key) {
1379 slotToKeyUpdateKey(key,1);
1380}
1381
1382void slotToKeyDel(robj *key) {
1383 slotToKeyUpdateKey(key,0);
1384}

1386void slotToKeyFlush(void) {
1387 raxFree(server.cluster->slots_to_keys);
1388 server.cluster->slots_to_keys = raxNew();
1389 memset(server.cluster->slots_keys_count,0,
1390 sizeof(server.cluster->slots_keys_count));
1391}

1393/* Pupulate the specified array of objects with keys in the specified slot.
1394 * New objects are returned to represent keys, it's up to the caller to
1395 * decrement the reference count to release the keys names. */
1396unsigned int getKeysInSlot(unsigned int hashslot, robj **keys, unsigned int count) {
1397 raxIterator iter;
1398 int j = 0;
1399 unsigned char indexed[2];
1400
1401 indexed[0] = (hashslot >> 8) & 0xff;
1402 indexed[1] = hashslot & 0xff;
1403 raxStart(&iter,server.cluster->slots_to_keys);
1404 raxSeek(&iter,">=",indexed,2);
1405 while(count-- && raxNext(&iter)) {
1406 if (iter.key[0] != indexed[0] || iter.key[1] != indexed[1]) break;
1407 keys[j++] = createStringObject((char*)iter.key+2,iter.key_len-2);
1408 }
1409 raxStop(&iter);
1410 return j;
1411}

1413/* Remove all the keys in the specified hash slot.
1414 * The number of removed items is returned. */
1415unsigned int delKeysInSlot(unsigned int hashslot) {
1416 raxIterator iter;
1417 int j = 0;
1418 unsigned char indexed[2];
1419
1420 indexed[0] = (hashslot >> 8) & 0xff;
1421 indexed[1] = hashslot & 0xff;
1422 raxStart(&iter,server.cluster->slots_to_keys);
1423 while(server.cluster->slots_keys_count[hashslot]) {
1424 raxSeek(&iter,">=",indexed,2);
1425 raxNext(&iter);
1426
1427 robj *key = createStringObject((char*)iter.key+2,iter.key_len-2);
1428 dbDelete(&server.db[0],key);
1429 decrRefCount(key);
1430 j++;
1431 }
1432 raxStop(&iter);
1433 return j;
1434}
