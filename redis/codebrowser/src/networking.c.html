70client *createClient(int fd) {
71 client *c = zmalloc(sizeof(client));
72
73 /* passing -1 as fd it is possible to create a non connected client.
74 * This is useful since all the commands needs to be executed
75 * in the context of a client. When commands are executed in other
76 * contexts (for instance a Lua script) we need a non connected client. */
77 if (fd != -1) {
78 anetNonBlock(NULL,fd);
79 anetEnableTcpNoDelay(NULL,fd);
80 if (server.tcpkeepalive)
81 anetKeepAlive(NULL,fd,server.tcpkeepalive);
82 if (aeCreateFileEvent(server.el,fd,AE_READABLE,
83 readQueryFromClient, c) == AE_ERR)
84 {
85 close(fd);
86 zfree(c);
87 return NULL;
88 }
89 }
90
91 selectDb(c,0);
92 uint64_t client_id;
93 atomicGetIncr(server.next_client_id,client_id,1);
94 c->id = client_id;
95 c->fd = fd;
96 c->name = NULL;
97 c->bufpos = 0;
98 c->querybuf = sdsempty();
99 c->pending_querybuf = sdsempty();
100 c->querybuf_peak = 0;
101 c->reqtype = 0;
102 c->argc = 0;
103 c->argv = NULL;
104 c->cmd = c->lastcmd = NULL;
105 c->multibulklen = 0;
106 c->bulklen = -1;
107 c->sentlen = 0;
108 c->flags = 0;
109 c->ctime = c->lastinteraction = server.unixtime;
110 c->authenticated = 0;
111 c->replstate = REPL_STATE_NONE;
112 c->repl_put_online_on_ack = 0;
113 c->reploff = 0;
114 c->read_reploff = 0;
115 c->repl_ack_off = 0;
116 c->repl_ack_time = 0;
117 c->slave_listening_port = 0;
118 c->slave_ip[0] = '\0';
119 c->slave_capa = SLAVE_CAPA_NONE;
120 c->reply = listCreate();
121 c->reply_bytes = 0;
122 c->obuf_soft_limit_reached_time = 0;
123 listSetFreeMethod(c->reply,freeClientReplyValue);
124 listSetDupMethod(c->reply,dupClientReplyValue);
125 c->btype = BLOCKED_NONE;
126 c->bpop.timeout = 0;
127 c->bpop.keys = dictCreate(&objectKeyPointerValueDictType,NULL);
128 c->bpop.target = NULL;
129 c->bpop.numreplicas = 0;
130 c->bpop.reploffset = 0;
131 c->woff = 0;
132 c->watched_keys = listCreate();
133 c->pubsub_channels = dictCreate(&objectKeyPointerValueDictType,NULL);
134 c->pubsub_patterns = listCreate();
135 c->peerid = NULL;
136 listSetFreeMethod(c->pubsub_patterns,decrRefCountVoid);
137 listSetMatchMethod(c->pubsub_patterns,listMatchObjects);
138 if (fd != -1) listAddNodeTail(server.clients,c);
139 initClientMultiState(c);
140 return c;
141}

143/* This function is called every time we are going to transmit new data
144 * to the client. The behavior is the following:
145 *
146 * If the client should receive new data (normal clients will) the function
147 * returns C_OK, and make sure to install the write handler in our event
148 * loop so that when the socket is writable new data gets written.
149 *
150 * If the client should not receive new data, because it is a fake client
151 * (used to load AOF in memory), a master or because the setup of the write
152 * handler failed, the function returns C_ERR.
153 *
154 * The function may return C_OK without actually installing the write
155 * event handler in the following cases:
156 *
157 * 1) The event handler should already be installed since the output buffer
158 * already contains something.
159 * 2) The client is a slave but not yet online, so we want to just accumulate
160 * writes in the buffer but not actually sending them yet.
161 *
162 * Typically gets called every time a reply is built, before adding more
163 * data to the clients output buffers. If the function returns C_ERR no
164 * data should be appended to the output buffers. */
165int prepareClientToWrite(client *c) {
166 /* If it's the Lua client we always return ok without installing any
167 * handler since there is no socket at all. */
168 if (c->flags & (CLIENT_LUA|CLIENT_MODULE)) return C_OK;
169
170 /* CLIENT REPLY OFF / SKIP handling: don't send replies. */
171 if (c->flags & (CLIENT_REPLY_OFF|CLIENT_REPLY_SKIP)) return C_ERR;
172
173 /* Masters don't receive replies, unless CLIENT_MASTER_FORCE_REPLY flag
174 * is set. */
175 if ((c->flags & CLIENT_MASTER) &&
176 !(c->flags & CLIENT_MASTER_FORCE_REPLY)) return C_ERR;
177
178 if (c->fd <= 0) return C_ERR; /* Fake client for AOF loading. */
179
180 /* Schedule the client to write the output buffers to the socket only
181 * if not already done (there were no pending writes already and the client
182 * was yet not flagged), and, for slaves, if the slave can actually
183 * receive writes at this stage. */
184 if (!clientHasPendingReplies(c) &&
185 !(c->flags & CLIENT_PENDING_WRITE) &&
186 (c->replstate == REPL_STATE_NONE ||
187 (c->replstate == SLAVE_STATE_ONLINE && !c->repl_put_online_on_ack)))
188 {
189 /* Here instead of installing the write handler, we just flag the
190 * client and put it into a list of clients that have something
191 * to write to the socket. This way before re-entering the event
192 * loop, we can try to directly write to the client sockets avoiding
193 * a system call. We'll only really install the write handler if
194 * we'll not be able to write the whole reply at once. */
195 c->flags |= CLIENT_PENDING_WRITE;
196 listAddNodeHead(server.clients_pending_write,c);
197 }
198
199 /* Authorize the caller to queue in the output buffer of this client. */
200 return C_OK;
201}

207int _addReplyToBuffer(client *c, const char *s, size_t len) {
208 size_t available = sizeof(c->buf)-c->bufpos;
209
210 if (c->flags & CLIENT_CLOSE_AFTER_REPLY) return C_OK;
211
212 /* If there already are entries in the reply list, we cannot
213 * add anything more to the static buffer. */
214 if (listLength(c->reply) > 0) return C_ERR;
215
216 /* Check that the buffer has enough space available for this string. */
217 if (len > available) return C_ERR;
218
219 memcpy(c->buf+c->bufpos,s,len);
220 c->bufpos+=len;
221 return C_OK;
222}

224void _addReplyObjectToList(client *c, robj *o) {
225 if (c->flags & CLIENT_CLOSE_AFTER_REPLY) return;
226
227 if (listLength(c->reply) == 0) {
228 sds s = sdsdup(o->ptr);
229 listAddNodeTail(c->reply,s);
230 c->reply_bytes += sdslen(s);
231 } else {
232 listNode *ln = listLast(c->reply);
233 sds tail = listNodeValue(ln);
234
235 /* Append to this object when possible. If tail == NULL it was
236 * set via addDeferredMultiBulkLength(). */
237 if (tail && sdslen(tail)+sdslen(o->ptr) <= PROTO_REPLY_CHUNK_BYTES) {
238 tail = sdscatsds(tail,o->ptr);
239 listNodeValue(ln) = tail;
240 c->reply_bytes += sdslen(o->ptr);
241 } else {
242 sds s = sdsdup(o->ptr);
243 listAddNodeTail(c->reply,s);
244 c->reply_bytes += sdslen(s);
245 }
246 }
247 asyncCloseClientOnOutputBufferLimitReached(c);
248}

250/* This method takes responsibility over the sds. When it is no longer
251 * needed it will be free'd, otherwise it ends up in a robj. */
252void _addReplySdsToList(client *c, sds s) {
253 if (c->flags & CLIENT_CLOSE_AFTER_REPLY) {
254 sdsfree(s);
255 return;
256 }
257
258 if (listLength(c->reply) == 0) {
259 listAddNodeTail(c->reply,s);
260 c->reply_bytes += sdslen(s);
261 } else {
262 listNode *ln = listLast(c->reply);
263 sds tail = listNodeValue(ln);
264
265 /* Append to this object when possible. If tail == NULL it was
266 * set via addDeferredMultiBulkLength(). */
267 if (tail && sdslen(tail)+sdslen(s) <= PROTO_REPLY_CHUNK_BYTES) {
268 tail = sdscatsds(tail,s);
269 listNodeValue(ln) = tail;
270 c->reply_bytes += sdslen(s);
271 sdsfree(s);
272 } else {
273 listAddNodeTail(c->reply,s);
274 c->reply_bytes += sdslen(s);
275 }
276 }
277 asyncCloseClientOnOutputBufferLimitReached(c);
278}

280void _addReplyStringToList(client *c, const char *s, size_t len) {
281 if (c->flags & CLIENT_CLOSE_AFTER_REPLY) return;
282
283 if (listLength(c->reply) == 0) {
284 sds node = sdsnewlen(s,len);
285 listAddNodeTail(c->reply,node);
286 c->reply_bytes += len;
287 } else {
288 listNode *ln = listLast(c->reply);
289 sds tail = listNodeValue(ln);
290
291 /* Append to this object when possible. If tail == NULL it was
292 * set via addDeferredMultiBulkLength(). */
293 if (tail && sdslen(tail)+len <= PROTO_REPLY_CHUNK_BYTES) {
294 tail = sdscatlen(tail,s,len);
295 listNodeValue(ln) = tail;
296 c->reply_bytes += len;
297 } else {
298 sds node = sdsnewlen(s,len);
299 listAddNodeTail(c->reply,node);
300 c->reply_bytes += len;
301 }
302 }
303 asyncCloseClientOnOutputBufferLimitReached(c);
304}

311void addReply(client *c, robj *obj) {
312 if (prepareClientToWrite(c) != C_OK) return;
313
314 /* This is an important place where we can avoid copy-on-write
315 * when there is a saving child running, avoiding touching the
316 * refcount field of the object if it's not needed.
317 *
318 * If the encoding is RAW and there is room in the static buffer
319 * we'll be able to send the object to the client without
320 * messing with its page. */
321 if (sdsEncodedObject(obj)) {
322 if (_addReplyToBuffer(c,obj->ptr,sdslen(obj->ptr)) != C_OK)
323 _addReplyObjectToList(c,obj);
324 } else if (obj->encoding == OBJ_ENCODING_INT) {
325 /* Optimization: if there is room in the static buffer for 32 bytes
326 * (more than the max chars a 64 bit integer can take as string) we
327 * avoid decoding the object and go for the lower level approach. */
328 if (listLength(c->reply) == 0 && (sizeof(c->buf) - c->bufpos) >= 32) {
329 char buf[32];
330 int len;
331
332 len = ll2string(buf,sizeof(buf),(long)obj->ptr);
333 if (_addReplyToBuffer(c,buf,len) == C_OK)
334 return;
335 /* else... continue with the normal code path, but should never
336 * happen actually since we verified there is room. */
337 }
338 obj = getDecodedObject(obj);
339 if (_addReplyToBuffer(c,obj->ptr,sdslen(obj->ptr)) != C_OK)
340 _addReplyObjectToList(c,obj);
341 decrRefCount(obj);
342 } else {
343 serverPanic("Wrong obj->encoding in addReply()");
344 }
345}

347void addReplySds(client *c, sds s) {
348 if (prepareClientToWrite(c) != C_OK) {
349 /* The caller expects the sds to be free'd. */
350 sdsfree(s);
351 return;
352 }
353 if (_addReplyToBuffer(c,s,sdslen(s)) == C_OK) {
354 sdsfree(s);
355 } else {
356 /* This method free's the sds when it is no longer needed. */
357 _addReplySdsToList(c,s);
358 }
359}

361/* This low level function just adds whatever protocol you send it to the
362 * client buffer, trying the static buffer initially, and using the string
363 * of objects if not possible.
364 *
365 * It is efficient because does not create an SDS object nor an Redis object
366 * if not needed. The object will only be created by calling
367 * _addReplyStringToList() if we fail to extend the existing tail object
368 * in the list of objects. */
369void addReplyString(client *c, const char *s, size_t len) {
370 if (prepareClientToWrite(c) != C_OK) return;
371 if (_addReplyToBuffer(c,s,len) != C_OK)
372 _addReplyStringToList(c,s,len);
373}

386 size_t l, j;
387 va_list ap;

392 * is emitted. */
393 l = sdslen(s);

395 if (s[j] == '\r' || s[j] == '\n') s[j] = ' ';
396 }
397 addReplyErrorLength(c,s,sdslen(s));
398 sdsfree(s);

420/* Adds an empty object to the reply list that will contain the multi bulk
421 * length, which is not known when this function is called. */
422void *addDeferredMultiBulkLength(client *c) {
423 /* Note that we install the write event here even if the object is not
424 * ready to be sent, since we are sure that before returning to the
425 * event loop setDeferredMultiBulkLength() will be called. */
426 if (prepareClientToWrite(c) != C_OK) return NULL;
427 listAddNodeTail(c->reply,NULL); /* NULL is our placeholder. */
428 return listLast(c->reply);
429}

431/* Populate the length object and try gluing it to the next chunk. */
432void setDeferredMultiBulkLength(client *c, void *node, long length) {
433 listNode *ln = (listNode*)node;
434 sds len, next;
435
436 /* Abort when *node is NULL: when the client should not accept writes
437 * we return NULL in addDeferredMultiBulkLength() */
438 if (node == NULL) return;
439
440 len = sdscatprintf(sdsnewlen("*",1),"%ld\r\n",length);
441 listNodeValue(ln) = len;
442 c->reply_bytes += sdslen(len);
443 if (ln->next != NULL) {
444 next = listNodeValue(ln->next);
445
446 /* Only glue when the next node is non-NULL (an sds in this case) */
447 if (next != NULL) {
448 len = sdscatsds(len,next);
449 listDelNode(c->reply,ln->next);
450 listNodeValue(ln) = len;
451 /* No need to update c->reply_bytes: we are just moving the same
452 * amount of bytes from one node to another. */
453 }
454 }
455 asyncCloseClientOnOutputBufferLimitReached(c);
456}

458/* Add a double as a bulk reply */
459void addReplyDouble(client *c, double d) {
460 char dbuf[128], sbuf[128];
461 int dlen, slen;
462 if (isinf(d)) {
463 /* Libc in odd systems (Hi Solaris!) will format infinite in a
464 * different way, so better to handle it in an explicit way. */
465 addReplyBulkCString(c, d > 0 ? "inf" : "-inf");
466 } else {
467 dlen = snprintf(dbuf,sizeof(dbuf),"%.17g",d);
468 slen = snprintf(sbuf,sizeof(sbuf),"$%d\r\n%s\r\n",dlen,dbuf);
469 addReplyString(c,sbuf,slen);
470 }
471}

482/* Add a long long as integer reply or bulk len / multi bulk count.
483 * Basically this is used to output <prefix><long long><crlf>. */
484void addReplyLongLongWithPrefix(client *c, long long ll, char prefix) {
485 char buf[128];
486 int len;
487
488 /* Things like $3\r\n or *2\r\n are emitted very often by the protocol
489 * so we have a few shared objects to use if the integer is small
490 * like it is most of the times. */
491 if (prefix == '*' && ll < OBJ_SHARED_BULKHDR_LEN && ll >= 0) {
492 addReply(c,shared.mbulkhdr[ll]);
493 return;
494 } else if (prefix == '$' && ll < OBJ_SHARED_BULKHDR_LEN && ll >= 0) {
495 addReply(c,shared.bulkhdr[ll]);
496 return;
497 }
498
499 buf[0] = prefix;
500 len = ll2string(buf+1,sizeof(buf)-1,ll);
501 buf[len+1] = '\r';
502 buf[len+2] = '\n';
503 addReplyString(c,buf,len+3);
504}

522/* Create the length prefix of a bulk reply, example: $2234 */
523void addReplyBulkLen(client *c, robj *obj) {
524 size_t len;
525
526 if (sdsEncodedObject(obj)) {
527 len = sdslen(obj->ptr);
528 } else {
529 long n = (long)obj->ptr;
530
531 /* Compute how many bytes will take this integer as a radix 10 string */
532 len = 1;
533 if (n < 0) {
534 len++;
535 n = -n;
536 }
537 while((n = n/10) != 0) {
538 len++;
539 }
540 }
541
542 if (len < OBJ_SHARED_BULKHDR_LEN)
543 addReply(c,shared.bulkhdr[len]);
544 else
545 addReplyLongLongWithPrefix(c,len,'$');
546}

587/* Copy 'src' client output buffers into 'dst' client output buffers.
588 * The function takes care of freeing the old output buffers of the
589 * destination client. */
590void copyClientOutputBuffer(client *dst, client *src) {
591 listRelease(dst->reply);
592 dst->reply = listDup(src->reply);
593 memcpy(dst->buf,src->buf,src->bufpos);
594 dst->bufpos = src->bufpos;
595 dst->reply_bytes = src->reply_bytes;
596}

604#define MAX_ACCEPTS_PER_CALL 1000
605static void acceptCommonHandler(int fd, int flags, char *ip) {
606 client *c;
607 if ((c = createClient(fd)) == NULL) {
608 serverLog(LL_WARNING,
609 "Error registering fd event for the new client: %s (fd=%d)",
610 strerror(errno),fd);
611 close(fd); /* May be already closed, just ignore errors */
612 return;
613 }
614 /* If maxclient directive is set and this is one client more... close the
615 * connection. Note that we create the client instead to check before
616 * for this condition, since now the socket is already set in non-blocking
617 * mode and we can send an error for free using the Kernel I/O */
618 if (listLength(server.clients) > server.maxclients) {
619 char *err = "-ERR max number of clients reached\r\n";
620
621 /* That's a best effort error message, don't check write errors */
622 if (write(c->fd,err,strlen(err)) == -1) {
623 /* Nothing to do, Just to avoid the warning... */
624 }
625 server.stat_rejected_conn++;
626 freeClient(c);
627 return;
628 }

630 /* If the server is running in protected mode (the default) and there
631 * is no password set, nor a specific interface is bound, we don't accept
632 * requests from non loopback interfaces. Instead we try to explain the
633 * user what to do to fix it if needed. */
634 if (server.protected_mode &&
635 server.bindaddr_count == 0 &&
636 server.requirepass == NULL &&
637 !(flags & CLIENT_UNIX_SOCKET) &&
638 ip != NULL)
639 {
640 if (strcmp(ip,"127.0.0.1") && strcmp(ip,"::1")) {
641 char *err =
642 "-DENIED Redis is running in protected mode because protected "

733 * be referenced, not including the Pub/Sub channels.
734 * This is used by freeClient() and replicationCacheMaster(). */
735void unlinkClient(client *c) {

737
738 /* If this is marked as current client unset it. */
739 if (server.current_client == c) server.current_client = NULL;
740

742 * If the client was already unlinked or if it's a "fake client" the
743 * fd is already set to -1. */
744 if (c->fd != -1) {

760 serverAssert(ln != NULL);

762 c->flags &= ~CLIENT_PENDING_WRITE;
763 }
764
765 /* When client was just unblocked because of a blocking operation,
766 * remove it from the list of unblocked clients. */
767 if (c->flags & CLIENT_UNBLOCKED) {
768 ln = listSearchKey(server.unblocked_clients,c);
769 serverAssert(ln != NULL);
770 listDelNode(server.unblocked_clients,ln);
771 c->flags &= ~CLIENT_UNBLOCKED;
772 }

775void freeClient(client *c) {
776 listNode *ln;
777
778 /* If it is our master that's beging disconnected we should make sure
779 * to cache the state to try a partial resynchronization later.
780 *
781 * Note that before doing this we make sure that the client is not in
782 * some unexpected state, by checking its flags. */
783 if (server.master && c->flags & CLIENT_MASTER) {
784 serverLog(LL_WARNING,"Connection with master lost.");
785 if (!(c->flags & (CLIENT_CLOSE_AFTER_REPLY|
786 CLIENT_CLOSE_ASAP|
787 CLIENT_BLOCKED|
788 CLIENT_UNBLOCKED)))
789 {
790 replicationCacheMaster(c);
791 return;
792 }
793 }
794
795 /* Log link disconnection with slave */
796 if ((c->flags & CLIENT_SLAVE) && !(c->flags & CLIENT_MONITOR)) {
797 serverLog(LL_WARNING,"Connection with slave %s lost.",
798 replicationGetSlaveName(c));
799 }
800
801 /* Free the query buffer */
802 sdsfree(c->querybuf);
803 sdsfree(c->pending_querybuf);
804 c->querybuf = NULL;
805
806 /* Deallocate structures used to block on blocking ops. */
807 if (c->flags & CLIENT_BLOCKED) unblockClient(c);
808 dictRelease(c->bpop.keys);
809
810 /* UNWATCH all the keys */
811 unwatchAllKeys(c);
812 listRelease(c->watched_keys);
813
814 /* Unsubscribe from all the pubsub channels */
815 pubsubUnsubscribeAllChannels(c,0);
816 pubsubUnsubscribeAllPatterns(c,0);
817 dictRelease(c->pubsub_channels);
818 listRelease(c->pubsub_patterns);
819
820 /* Free data structures. */
821 listRelease(c->reply);
822 freeClientArgv(c);
823
824 /* Unlink the client: this will close the socket, remove the I/O
825 * handlers, and remove references of the client from different
826 * places where active clients may be referenced. */
827 unlinkClient(c);
828
829 /* Master/slave cleanup Case 1:
830 * we lost the connection with a slave. */
831 if (c->flags & CLIENT_SLAVE) {
832 if (c->replstate == SLAVE_STATE_SEND_BULK) {
833 if (c->repldbfd != -1) close(c->repldbfd);
834 if (c->replpreamble) sdsfree(c->replpreamble);
835 }
836 list *l = (c->flags & CLIENT_MONITOR) ? server.monitors : server.slaves;
837 ln = listSearchKey(l,c);
838 serverAssert(ln != NULL);
839 listDelNode(l,ln);
840 /* We need to remember the time when we started to have zero
841 * attached slaves, as after some time we'll free the replication
842 * backlog. */
843 if (c->flags & CLIENT_SLAVE && listLength(server.slaves) == 0)
844 server.repl_no_slaves_since = server.unixtime;
845 refreshGoodSlavesCount();
846 }
847
848 /* Master/slave cleanup Case 2:
849 * we lost the connection with the master. */
850 if (c->flags & CLIENT_MASTER) replicationHandleMasterDisconnection();
851
852 /* If this client was scheduled for async freeing we need to remove it
853 * from the queue. */
854 if (c->flags & CLIENT_CLOSE_ASAP) {
855 ln = listSearchKey(server.clients_to_close,c);
856 serverAssert(ln != NULL);
857 listDelNode(server.clients_to_close,ln);
858 }
859
860 /* Release other dynamically allocated client structure fields,
861 * and finally release the client structure itself. */
862 if (c->name) decrRefCount(c->name);
863 zfree(c->argv);
864 freeClientMultiState(c);
865 sdsfree(c->peerid);
866 zfree(c);
867}

890/* Write data in output buffers to client. Return C_OK if the client
891 * is still valid after the call, C_ERR if it was freed. */
892int writeToClient(int fd, client *c, int handler_installed) {
893 ssize_t nwritten = 0, totwritten = 0;
894 size_t objlen;
895 sds o;
896
897 while(clientHasPendingReplies(c)) {
898 if (c->bufpos > 0) {
899 nwritten = write(fd,c->buf+c->sentlen,c->bufpos-c->sentlen);
900 if (nwritten <= 0) break;
901 c->sentlen += nwritten;
902 totwritten += nwritten;
903
904 /* If the buffer was sent, set bufpos to zero to continue with
905 * the remainder of the reply. */
906 if ((int)c->sentlen == c->bufpos) {
907 c->bufpos = 0;
908 c->sentlen = 0;
909 }
910 } else {
911 o = listNodeValue(listFirst(c->reply));
912 objlen = sdslen(o);
913
914 if (objlen == 0) {
915 listDelNode(c->reply,listFirst(c->reply));
916 continue;
917 }
918
919 nwritten = write(fd, o + c->sentlen, objlen - c->sentlen);
920 if (nwritten <= 0) break;
921 c->sentlen += nwritten;
922 totwritten += nwritten;
923
924 /* If we fully sent the object on head go to the next one */
925 if (c->sentlen == objlen) {
926 listDelNode(c->reply,listFirst(c->reply));
927 c->sentlen = 0;
928 c->reply_bytes -= objlen;
929 /* If there are no longer objects in the list, we expect
930 * the count of reply bytes to be exactly zero. */
931 if (listLength(c->reply) == 0)
932 serverAssert(c->reply_bytes == 0);
933 }
934 }
935 /* Note that we avoid to send more than NET_MAX_WRITES_PER_EVENT
936 * bytes, in a single threaded server it's a good idea to serve
937 * other clients as well, even if a very large request comes from
938 * super fast link that is always able to accept data (in real world
939 * scenario think about 'KEYS *' against the loopback interface).
940 *
941 * However if we are over the maxmemory limit we ignore that and
942 * just deliver as much data as it is possible to deliver. */
943 if (totwritten > NET_MAX_WRITES_PER_EVENT &&
944 (server.maxmemory == 0 ||
945 zmalloc_used_memory() < server.maxmemory)) break;
946 }
947 server.stat_net_output_bytes += totwritten;
948 if (nwritten == -1) {
949 if (errno == EAGAIN) {
950 nwritten = 0;
951 } else {
952 serverLog(LL_VERBOSE,
953 "Error writing to client: %s", strerror(errno));
954 freeClient(c);
955 return C_ERR;
956 }
957 }
958 if (totwritten > 0) {
959 /* For clients representing masters we don't count sending data
960 * as an interaction, since we always send REPLCONF ACK commands
961 * that take some time to just fill the socket output buffer.
962 * We just rely on data / pings received for timeout detection. */
963 if (!(c->flags & CLIENT_MASTER)) c->lastinteraction = server.unixtime;
964 }
965 if (!clientHasPendingReplies(c)) {
966 c->sentlen = 0;
967 if (handler_installed) aeDeleteFileEvent(server.el,c->fd,AE_WRITABLE);
968
969 /* Close connection after entire reply has been sent. */
970 if (c->flags & CLIENT_CLOSE_AFTER_REPLY) {
971 freeClient(c);
972 return C_ERR;
973 }
974 }
975 return C_OK;
976}

1039/* Like processMultibulkBuffer(), but for the inline protocol instead of RESP,
1040 * this function consumes the client query buffer and creates a command ready
1041 * to be executed inside the client structure. Returns C_OK if the command
1042 * is ready to be executed, or C_ERR if there is still protocol to read to
1043 * have a well formed command. The function also returns C_ERR when there is
1044 * a protocol error: in such a case the client structure is setup to reply
1045 * with the error and close the connection. */
1046int processInlineBuffer(client *c) {
1047 char *newline;
1048 int argc, j;
1049 sds *argv, aux;
1050 size_t querylen;
1051
1052 /* Search for end of line */
1053 newline = strchr(c->querybuf,'\n');
1054
1055 /* Nothing to do without a \r\n */
1056 if (newline == NULL) {
1057 if (sdslen(c->querybuf) > PROTO_INLINE_MAX_SIZE) {
1058 addReplyError(c,"Protocol error: too big inline request");
1059 setProtocolError("too big inline request",c,0);
1060 }
1061 return C_ERR;
1062 }
1063
1064 /* Handle the \r\n case. */
1065 if (newline && newline != c->querybuf && *(newline-1) == '\r')
1066 newline--;
1067
1068 /* Split the input buffer up to the \r\n */
1069 querylen = newline-(c->querybuf);
1070 aux = sdsnewlen(c->querybuf,querylen);
1071 argv = sdssplitargs(aux,&argc);
1072 sdsfree(aux);
1073 if (argv == NULL) {
1074 addReplyError(c,"Protocol error: unbalanced quotes in request");
1075 setProtocolError("unbalanced quotes in inline request",c,0);
1076 return C_ERR;
1077 }
1078
1079 /* Newline from slaves can be used to refresh the last ACK time.
1080 * This is useful for a slave to ping back while loading a big
1081 * RDB file. */
1082 if (querylen == 0 && c->flags & CLIENT_SLAVE)
1083 c->repl_ack_time = server.unixtime;
1084
1085 /* Leave data after the first line of the query in the buffer */
1086 sdsrange(c->querybuf,querylen+2,-1);
1087
1088 /* Setup argv array on client structure */
1089 if (argc) {
1090 if (c->argv) zfree(c->argv);
1091 c->argv = zmalloc(sizeof(robj*)*argc);
1092 }
1093
1094 /* Create redis objects for all arguments. */
1095 for (c->argc = 0, j = 0; j < argc; j++) {
1096 if (sdslen(argv[j])) {
1097 c->argv[c->argc] = createObject(OBJ_STRING,argv[j]);
1098 c->argc++;
1099 } else {
1100 sdsfree(argv[j]);
1101 }
1102 }
1103 zfree(argv);
1104 return C_OK;
1105}

1107/* Helper function. Trims query buffer to make the function that processes
1108 * multi bulk requests idempotent. */
1109#define PROTO_DUMP_LEN 128
1110static void setProtocolError(const char *errstr, client *c, int pos) {
1111 if (server.verbosity <= LL_VERBOSE) {
1112 sds client = catClientInfoString(sdsempty(),c);
1113
1114 /* Sample some protocol to given an idea about what was inside. */
1115 char buf[256];
1116 if (sdslen(c->querybuf) < PROTO_DUMP_LEN) {
1117 snprintf(buf,sizeof(buf),"Query buffer during protocol error: '%s'", c->querybuf);
1118 } else {
1119 snprintf(buf,sizeof(buf),"Query buffer during protocol error: '%.*s' (... more %zu bytes ...) '%.*s'", PROTO_DUMP_LEN/2, c->querybuf, sdslen(c->querybuf)-PROTO_DUMP_LEN, PROTO_DUMP_LEN/2, c->querybuf+sdslen(c->querybuf)-PROTO_DUMP_LEN/2);
1120 }
1121
1122 /* Remove non printable chars. */
1123 char *p = buf;
1124 while (*p != '\0') {
1125 if (!isprint(*p)) *p = '.';
1126 p++;
1127 }
1128
1129 /* Log all the client and protocol info. */
1130 serverLog(LL_VERBOSE,
1131 "Protocol error (%s) from client: %s. %s", errstr, client, buf);
1132 sdsfree(client);
1133 }
1134 c->flags |= CLIENT_CLOSE_AFTER_REPLY;
1135 sdsrange(c->querybuf,pos,-1);
1136}

1138/* Process the query buffer for client 'c', setting up the client argument
1139 * vector for command execution. Returns C_OK if after running the function
1140 * the client has a well-formed ready to be processed command, otherwise
1141 * C_ERR if there is still to read more buffer to get the full command.
1142 * The function also returns C_ERR when there is a protocol error: in such a
1143 * case the client structure is setup to reply with the error and close
1144 * the connection.
1145 *
1146 * This function is called if processInputBuffer() detects that the next
1147 * command is in RESP format, so the first byte in the command is found
1148 * to be '*'. Otherwise for inline commands processInlineBuffer() is called. */
1149int processMultibulkBuffer(client *c) {
1150 char *newline = NULL;
1151 int pos = 0, ok;
1152 long long ll;
1153
1154 if (c->multibulklen == 0) {
1155 /* The client should have been reset */
1156 serverAssertWithInfo(c,NULL,c->argc == 0);
1157
1158 /* Multi bulk length cannot be read without a \r\n */
1159 newline = strchr(c->querybuf,'\r');
1160 if (newline == NULL) {
1161 if (sdslen(c->querybuf) > PROTO_INLINE_MAX_SIZE) {
1162 addReplyError(c,"Protocol error: too big mbulk count string");
1163 setProtocolError("too big mbulk count string",c,0);
1164 }
1165 return C_ERR;
1166 }
1167
1168 /* Buffer should also contain \n */
1169 if (newline-(c->querybuf) > ((signed)sdslen(c->querybuf)-2))
1170 return C_ERR;
1171
1172 /* We know for sure there is a whole line since newline != NULL,
1173 * so go ahead and find out the multi bulk length. */
1174 serverAssertWithInfo(c,NULL,c->querybuf[0] == '*');
1175 ok = string2ll(c->querybuf+1,newline-(c->querybuf+1),&ll);
1176 if (!ok || ll > 1024*1024) {
1177 addReplyError(c,"Protocol error: invalid multibulk length");
1178 setProtocolError("invalid mbulk count",c,pos);
1179 return C_ERR;
1180 }
1181
1182 pos = (newline-c->querybuf)+2;
1183 if (ll <= 0) {
1184 sdsrange(c->querybuf,pos,-1);
1185 return C_OK;
1186 }
1187
1188 c->multibulklen = ll;
1189
1190 /* Setup argv array on client structure */
1191 if (c->argv) zfree(c->argv);
1192 c->argv = zmalloc(sizeof(robj*)*c->multibulklen);
1193 }
1194
1195 serverAssertWithInfo(c,NULL,c->multibulklen > 0);
1196 while(c->multibulklen) {
1197 /* Read bulk length if unknown */
1198 if (c->bulklen == -1) {
1199 newline = strchr(c->querybuf+pos,'\r');
1200 if (newline == NULL) {
1201 if (sdslen(c->querybuf) > PROTO_INLINE_MAX_SIZE) {
1202 addReplyError(c,
1203 "Protocol error: too big bulk count string");
1204 setProtocolError("too big bulk count string",c,0);
1205 return C_ERR;
1206 }
1207 break;
1208 }
1209
1210 /* Buffer should also contain \n */
1211 if (newline-(c->querybuf) > ((signed)sdslen(c->querybuf)-2))
1212 break;
1213
1214 if (c->querybuf[pos] != '$') {
1215 addReplyErrorFormat(c,
1216 "Protocol error: expected '$', got '%c'",
1217 c->querybuf[pos]);
1218 setProtocolError("expected $ but got something else",c,pos);
1219 return C_ERR;
1220 }
1221
1222 ok = string2ll(c->querybuf+pos+1,newline-(c->querybuf+pos+1),&ll);
1223 if (!ok || ll < 0 || ll > 512*1024*1024) {
1224 addReplyError(c,"Protocol error: invalid bulk length");
1225 setProtocolError("invalid bulk length",c,pos);
1226 return C_ERR;
1227 }
1228
1229 pos += newline-(c->querybuf+pos)+2;
1230 if (ll >= PROTO_MBULK_BIG_ARG) {
1231 size_t qblen;
1232
1233 /* If we are going to read a large object from network
1234 * try to make it likely that it will start at c->querybuf
1235 * boundary so that we can optimize object creation
1236 * avoiding a large copy of data. */
1237 sdsrange(c->querybuf,pos,-1);
1238 pos = 0;
1239 qblen = sdslen(c->querybuf);
1240 /* Hint the sds library about the amount of bytes this string is
1241 * going to contain. */
1242 if (qblen < (size_t)ll+2)
1243 c->querybuf = sdsMakeRoomFor(c->querybuf,ll+2-qblen);
1244 }
1245 c->bulklen = ll;
1246 }
1247
1248 /* Read bulk argument */
1249 if (sdslen(c->querybuf)-pos < (unsigned)(c->bulklen+2)) {
1250 /* Not enough data (+2 == trailing \r\n) */
1251 break;
1252 } else {
1253 /* Optimization: if the buffer contains JUST our bulk element
1254 * instead of creating a new object by *copying* the sds we
1255 * just use the current sds string. */
1256 if (pos == 0 &&
1257 c->bulklen >= PROTO_MBULK_BIG_ARG &&
1258 (signed) sdslen(c->querybuf) == c->bulklen+2)
1259 {
1260 c->argv[c->argc++] = createObject(OBJ_STRING,c->querybuf);
1261 sdsIncrLen(c->querybuf,-2); /* remove CRLF */
1262 /* Assume that if we saw a fat argument we'll see another one
1263 * likely... */
1264 c->querybuf = sdsnewlen(NULL,c->bulklen+2);
1265 sdsclear(c->querybuf);
1266 pos = 0;
1267 } else {
1268 c->argv[c->argc++] =
1269 createStringObject(c->querybuf+pos,c->bulklen);
1270 pos += c->bulklen+2;
1271 }
1272 c->bulklen = -1;
1273 c->multibulklen--;
1274 }
1275 }
1276
1277 /* Trim to pos */
1278 if (pos) sdsrange(c->querybuf,pos,-1);
1279
1280 /* We're done when c->multibulk == 0 */
1281 if (c->multibulklen == 0) return C_OK;
1282
1283 /* Still not ready to process the command */
1284 return C_ERR;
1285}
1286
1287/* This function is called every time, in the client structure 'c', there is
1288 * more query buffer to process, because we read more data from the socket
1289 * or because a client was blocked and later reactivated, so there could be
1290 * pending query buffer, already representing a full command, to process. */
1291void processInputBuffer(client *c) {
1292 server.current_client = c;
1293 /* Keep processing while there is something in the input buffer */
1294 while(sdslen(c->querybuf)) {
1295 /* Return if clients are paused. */
1296 if (!(c->flags & CLIENT_SLAVE) && clientsArePaused()) break;
1297
1298 /* Immediately abort if the client is in the middle of something. */
1299 if (c->flags & CLIENT_BLOCKED) break;
1300
1301 /* CLIENT_CLOSE_AFTER_REPLY closes the connection once the reply is
1302 * written to the client. Make sure to not let the reply grow after
1303 * this flag has been set (i.e. don't process more commands).
1304 *
1305 * The same applies for clients we want to terminate ASAP. */
1306 if (c->flags & (CLIENT_CLOSE_AFTER_REPLY|CLIENT_CLOSE_ASAP)) break;
1307
1308 /* Determine request type when unknown. */
1309 if (!c->reqtype) {
1310 if (c->querybuf[0] == '*') {
1311 c->reqtype = PROTO_REQ_MULTIBULK;
1312 } else {
1313 c->reqtype = PROTO_REQ_INLINE;
1314 }
1315 }
1316
1317 if (c->reqtype == PROTO_REQ_INLINE) {
1318 if (processInlineBuffer(c) != C_OK) break;
1319 } else if (c->reqtype == PROTO_REQ_MULTIBULK) {
1320 if (processMultibulkBuffer(c) != C_OK) break;
1321 } else {
1322 serverPanic("Unknown request type");
1323 }
1324
1325 /* Multibulk processing could see a <= 0 length. */
1326 if (c->argc == 0) {
1327 resetClient(c);
1328 } else {
1329 /* Only reset the client when the command was executed. */
1330 if (processCommand(c) == C_OK) {
1331 if (c->flags & CLIENT_MASTER && !(c->flags & CLIENT_MULTI)) {
1332 /* Update the applied replication offset of our master. */
1333 c->reploff = c->read_reploff - sdslen(c->querybuf);
1334 }
1335
1336 /* Don't reset the client structure for clients blocked in a
1337 * module blocking command, so that the reply callback will
1338 * still be able to access the client argv and argc field.
1339 * The client will be reset in unblockClientFromModule(). */
1340 if (!(c->flags & CLIENT_BLOCKED) || c->btype != BLOCKED_MODULE)
1341 resetClient(c);
1342 }
1343 /* freeMemoryIfNeeded may flush slave output buffers. This may
1344 * result into a slave, that may be the active client, to be
1345 * freed. */
1346 if (server.current_client == NULL) break;
1347 }
1348 }
1349 server.current_client = NULL;
1350}

1352void readQueryFromClient(aeEventLoop *el, int fd, void *privdata, int mask) {
1353 client *c = (client*) privdata;
1354 int nread, readlen;
1355 size_t qblen;
1356 UNUSED(el);
1357 UNUSED(mask);
1358
1359 readlen = PROTO_IOBUF_LEN;
1360 /* If this is a multi bulk request, and we are processing a bulk reply
1361 * that is large enough, try to maximize the probability that the query
1362 * buffer contains exactly the SDS string representing the object, even
1363 * at the risk of requiring more read(2) calls. This way the function
1364 * processMultiBulkBuffer() can avoid copying buffers to create the
1365 * Redis Object representing the argument. */
1366 if (c->reqtype == PROTO_REQ_MULTIBULK && c->multibulklen && c->bulklen != -1
1367 && c->bulklen >= PROTO_MBULK_BIG_ARG)
1368 {
1369 int remaining = (unsigned)(c->bulklen+2)-sdslen(c->querybuf);
1370
1371 if (remaining < readlen) readlen = remaining;
1372 }
1373
1374 qblen = sdslen(c->querybuf);
1375 if (c->querybuf_peak < qblen) c->querybuf_peak = qblen;
1376 c->querybuf = sdsMakeRoomFor(c->querybuf, readlen);
1377 nread = read(fd, c->querybuf+qblen, readlen);
1378 if (nread == -1) {
1379 if (errno == EAGAIN) {
1380 return;
1381 } else {
1382 serverLog(LL_VERBOSE, "Reading from client: %s",strerror(errno));
1383 freeClient(c);
1384 return;
1385 }
1386 } else if (nread == 0) {
1387 serverLog(LL_VERBOSE, "Client closed connection");
1388 freeClient(c);
1389 return;
1390 } else if (c->flags & CLIENT_MASTER) {
1391 /* Append the query buffer to the pending (not applied) buffer
1392 * of the master. We'll use this buffer later in order to have a
1393 * copy of the string applied by the last command executed. */
1394 c->pending_querybuf = sdscatlen(c->pending_querybuf,
1395 c->querybuf+qblen,nread);
1396 }
1397
1398 sdsIncrLen(c->querybuf,nread);
1399 c->lastinteraction = server.unixtime;
1400 if (c->flags & CLIENT_MASTER) c->read_reploff += nread;
1401 server.stat_net_input_bytes += nread;
1402 if (sdslen(c->querybuf) > server.client_max_querybuf_len) {
1403 sds ci = catClientInfoString(sdsempty(),c), bytes = sdsempty();
1404
1405 bytes = sdscatrepr(bytes,c->querybuf,64);
1406 serverLog(LL_WARNING,"Closing client that reached max query buffer length: %s (qbuf initial bytes: %s)", ci, bytes);
1407 sdsfree(ci);
1408 sdsfree(bytes);
1409 freeClient(c);
1410 return;
1411 }
1412
1413 /* Time to process the buffer. If the client is a master we need to
1414 * compute the difference between the applied offset before and after
1415 * processing the buffer, to understand how much of the replication stream
1416 * was actually applied to the master state: this quantity, and its
1417 * corresponding part of the replication stream, will be propagated to
1418 * the sub-slaves and to the replication backlog. */
1419 if (!(c->flags & CLIENT_MASTER)) {
1420 processInputBuffer(c);
1421 } else {
1422 size_t prev_offset = c->reploff;
1423 processInputBuffer(c);
1424 size_t applied = c->reploff - prev_offset;
1425 if (applied) {
1426 replicationFeedSlavesFromMasterStream(server.slaves,
1427 c->pending_querybuf, applied);
1428 sdsrange(c->pending_querybuf,applied,-1);
1429 }
1430 }
1431}

1451/* A Redis "Peer ID" is a colon separated ip:port pair.
1452 * For IPv4 it's in the form x.y.z.k:port, example: "127.0.0.1:1234".
1453 * For IPv6 addresses we use [] around the IP part, like in "[::1]:1234".
1454 * For Unix sockets we use path:0, like in "/tmp/redis:0".
1455 *
1456 * A Peer ID always fits inside a buffer of NET_PEER_ID_LEN bytes, including
1457 * the null term.
1458 *
1459 * On failure the function still populates 'peerid' with the "?:0" string
1460 * in case you want to relax error checking or need to display something
1461 * anyway (see anetPeerToString implementation for more info). */
1462void genClientPeerId(client *client, char *peerid,
1463 size_t peerid_len) {
1464 if (client->flags & CLIENT_UNIX_SOCKET) {
1465 /* Unix socket client. */
1466 snprintf(peerid,peerid_len,"%s:0",server.unixsocket);
1467 } else {
1468 /* TCP client. */
1469 anetFormatPeer(client->fd,peerid,peerid_len);
1470 }
1471}

1474 * if client->peerid is NULL, otherwise returning the cached value.
1475 * The Peer ID never changes during the life of the client, however it
1476 * is expensive to compute. */

1478 char peerid[NET_PEER_ID_LEN];
1479
1480 if (c->peerid == NULL) {

1483 }
1484 return c->peerid;

1554void clientCommand(client *c) {
1555 listNode *ln;
1556 listIter li;
1557 client *client;
1558
1559 if (!strcasecmp(c->argv[1]->ptr,"list") && c->argc == 2) {
1560 /* CLIENT LIST */
1561 sds o = getAllClientsInfoString();
1562 addReplyBulkCBuffer(c,o,sdslen(o));
1563 sdsfree(o);
1564 } else if (!strcasecmp(c->argv[1]->ptr,"reply") && c->argc == 3) {
1565 /* CLIENT REPLY ON|OFF|SKIP */
1566 if (!strcasecmp(c->argv[2]->ptr,"on")) {
1567 c->flags &= ~(CLIENT_REPLY_SKIP|CLIENT_REPLY_OFF);
1568 addReply(c,shared.ok);
1569 } else if (!strcasecmp(c->argv[2]->ptr,"off")) {
1570 c->flags |= CLIENT_REPLY_OFF;
1571 } else if (!strcasecmp(c->argv[2]->ptr,"skip")) {
1572 if (!(c->flags & CLIENT_REPLY_OFF))
1573 c->flags |= CLIENT_REPLY_SKIP_NEXT;
1574 } else {
1575 addReply(c,shared.syntaxerr);
1576 return;
1577 }
1578 } else if (!strcasecmp(c->argv[1]->ptr,"kill")) {
1579 /* CLIENT KILL <ip:port>
1580 * CLIENT KILL <option> [value] ... <option> [value] */
1581 char *addr = NULL;
1582 int type = -1;
1583 uint64_t id = 0;
1584 int skipme = 1;
1585 int killed = 0, close_this_client = 0;
1586
1587 if (c->argc == 3) {
1588 /* Old style syntax: CLIENT KILL <addr> */
1589 addr = c->argv[2]->ptr;
1590 skipme = 0; /* With the old form, you can kill yourself. */
1591 } else if (c->argc > 3) {
1592 int i = 2; /* Next option index. */
1593
1594 /* New style syntax: parse options. */
1595 while(i < c->argc) {
1596 int moreargs = c->argc > i+1;
1597
1598 if (!strcasecmp(c->argv[i]->ptr,"id") && moreargs) {
1599 long long tmp;
1600
1601 if (getLongLongFromObjectOrReply(c,c->argv[i+1],&tmp,NULL)
1602 != C_OK) return;
1603 id = tmp;
1604 } else if (!strcasecmp(c->argv[i]->ptr,"type") && moreargs) {
1605 type = getClientTypeByName(c->argv[i+1]->ptr);
1606 if (type == -1) {
1607 addReplyErrorFormat(c,"Unknown client type '%s'",
1608 (char*) c->argv[i+1]->ptr);
1609 return;
1610 }
1611 } else if (!strcasecmp(c->argv[i]->ptr,"addr") && moreargs) {
1612 addr = c->argv[i+1]->ptr;
1613 } else if (!strcasecmp(c->argv[i]->ptr,"skipme") && moreargs) {
1614 if (!strcasecmp(c->argv[i+1]->ptr,"yes")) {
1615 skipme = 1;
1616 } else if (!strcasecmp(c->argv[i+1]->ptr,"no")) {
1617 skipme = 0;
1618 } else {
1619 addReply(c,shared.syntaxerr);
1620 return;
1621 }
1622 } else {
1623 addReply(c,shared.syntaxerr);
1624 return;
1625 }
1626 i += 2;
1627 }
1628 } else {
1629 addReply(c,shared.syntaxerr);
1630 return;
1631 }
1632
1633 /* Iterate clients killing all the matching clients. */
1634 listRewind(server.clients,&li);
1635 while ((ln = listNext(&li)) != NULL) {
1636 client = listNodeValue(ln);
1637 if (addr && strcmp(getClientPeerId(client),addr) != 0) continue;
1638 if (type != -1 && getClientType(client) != type) continue;
1639 if (id != 0 && client->id != id) continue;
1640 if (c == client && skipme) continue;
1641
1642 /* Kill it. */
1643 if (c == client) {
1644 close_this_client = 1;
1645 } else {
1646 freeClient(client);
1647 }
1648 killed++;
1649 }
1650
1651 /* Reply according to old/new format. */
1652 if (c->argc == 3) {
1653 if (killed == 0)
1654 addReplyError(c,"No such client");
1655 else
1656 addReply(c,shared.ok);
1657 } else {
1658 addReplyLongLong(c,killed);
1659 }
1660
1661 /* If this client has to be closed, flag it as CLOSE_AFTER_REPLY
1662 * only after we queued the reply to its output buffers. */
1663 if (close_this_client) c->flags |= CLIENT_CLOSE_AFTER_REPLY;
1664 } else if (!strcasecmp(c->argv[1]->ptr,"setname") && c->argc == 3) {
1665 int j, len = sdslen(c->argv[2]->ptr);
1666 char *p = c->argv[2]->ptr;
1667
1668 /* Setting the client name to an empty string actually removes
1669 * the current name. */
1670 if (len == 0) {
1671 if (c->name) decrRefCount(c->name);
1672 c->name = NULL;
1673 addReply(c,shared.ok);
1674 return;
1675 }
1676
1677 /* Otherwise check if the charset is ok. We need to do this otherwise
1678 * CLIENT LIST format will break. You should always be able to
1679 * split by space to get the different fields. */
1680 for (j = 0; j < len; j++) {
1681 if (p[j] < '!' || p[j] > '~') { /* ASCII is assumed. */
1682 addReplyError(c,
1683 "Client names cannot contain spaces, "
1684 "newlines or special characters.");
1685 return;
1686 }
1687 }
1688 if (c->name) decrRefCount(c->name);
1689 c->name = c->argv[2];
1690 incrRefCount(c->name);
1691 addReply(c,shared.ok);
1692 } else if (!strcasecmp(c->argv[1]->ptr,"getname") && c->argc == 2) {
1693 if (c->name)
1694 addReplyBulk(c,c->name);
1695 else
1696 addReply(c,shared.nullbulk);
1697 } else if (!strcasecmp(c->argv[1]->ptr,"pause") && c->argc == 3) {
1698 long long duration;
1699
1700 if (getTimeoutFromObjectOrReply(c,c->argv[2],&duration,UNIT_MILLISECONDS)
1701 != C_OK) return;
1702 pauseClients(duration);
1703 addReply(c,shared.ok);
1704 } else {
1705 addReplyError(c, "Syntax error, try CLIENT (LIST | KILL | GETNAME | SETNAME | PAUSE | REPLY)");
1706 }
1707}

1710 * really commands, but are used in security attacks in order to talk to

1724 logged_time = now;
1725 }

1729/* Rewrite the command vector of the client. All the new objects ref count
1730 * is incremented. The old command vector is freed, and the old objects
1731 * ref count is decremented. */
1732void rewriteClientCommandVector(client *c, int argc, ...) {
1733 va_list ap;
1734 int j;
1735 robj **argv; /* The new argument vector */
1736
1737 argv = zmalloc(sizeof(robj*)*argc);
1738 va_start(ap,argc);
1739 for (j = 0; j < argc; j++) {
1740 robj *a;
1741
1742 a = va_arg(ap, robj*);
1743 argv[j] = a;
1744 incrRefCount(a);
1745 }
1746 /* We free the objects in the original vector at the end, so we are
1747 * sure that if the same objects are reused in the new vector the
1748 * refcount gets incremented before it gets decremented. */
1749 for (j = 0; j < c->argc; j++) decrRefCount(c->argv[j]);
1750 zfree(c->argv);
1751 /* Replace argv and argc with our new versions. */
1752 c->argv = argv;
1753 c->argc = argc;
1754 c->cmd = lookupCommandOrOriginal(c->argv[0]->ptr);
1755 serverAssertWithInfo(c,NULL,c->cmd != NULL);
1756 va_end(ap);
1757}

1770 * The new val ref count is incremented, and the old decremented.
1771 *
1772 * It is possible to specify an argument over the current size of the
1773 * argument vector: in this case the array of objects gets reallocated

1775 *
1776 * 1. Make sure there are no "holes" and all the arguments are set.

1786 c->argv[i] = NULL;
1787 }
1788 oldval = c->argv[i];
1789 c->argv[i] = newval;
1790 incrRefCount(newval);
1791 if (oldval) decrRefCount(oldval);
1792
1793 /* If this is the command name make sure to fix c->cmd. */
1794 if (i == 0) {
1795 c->cmd = lookupCommandOrOriginal(c->argv[0]->ptr);
1796 serverAssertWithInfo(c,NULL,c->cmd != NULL);
1797 }

1800/* This function returns the number of bytes that Redis is virtually
1801 * using to store the reply still not read by the client.
1802 * It is "virtual" since the reply output list may contain objects that
1803 * are shared and are not really using additional memory.
1804 *
1805 * The function returns the total sum of the length of all the objects
1806 * stored in the output list, plus the memory used to allocate every
1807 * list node. The static reply buffer is not taken into account since it
1808 * is allocated anyway.
1809 *
1810 * Note: this function is very fast so can be called as many time as
1811 * the caller wishes. The main usage of this function currently is
1812 * enforcing the client output length limits. */
1813unsigned long getClientOutputBufferMemoryUsage(client *c) {
1814 unsigned long list_item_size = sizeof(listNode)+5;
1815 /* The +5 above means we assume an sds16 hdr, may not be true
1816 * but is not going to be a problem. */
1817
1818 return c->reply_bytes + (list_item_size*listLength(c->reply));
1819}

1856/* The function checks if the client reached output buffer soft or hard
1857 * limit, and also update the state needed to check the soft limit as
1858 * a side effect.
1859 *
1860 * Return value: non-zero if the client reached the soft or the hard limit.
1861 * Otherwise zero is returned. */
1862int checkClientOutputBufferLimits(client *c) {
1863 int soft = 0, hard = 0, class;
1864 unsigned long used_mem = getClientOutputBufferMemoryUsage(c);
1865
1866 class = getClientType(c);
1867 /* For the purpose of output buffer limiting, masters are handled
1868 * like normal clients. */
1869 if (class == CLIENT_TYPE_MASTER) class = CLIENT_TYPE_NORMAL;
1870
1871 if (server.client_obuf_limits[class].hard_limit_bytes &&
1872 used_mem >= server.client_obuf_limits[class].hard_limit_bytes)
1873 hard = 1;
1874 if (server.client_obuf_limits[class].soft_limit_bytes &&
1875 used_mem >= server.client_obuf_limits[class].soft_limit_bytes)
1876 soft = 1;
1877
1878 /* We need to check if the soft limit is reached continuously for the
1879 * specified amount of seconds. */
1880 if (soft) {
1881 if (c->obuf_soft_limit_reached_time == 0) {
1882 c->obuf_soft_limit_reached_time = server.unixtime;
1883 soft = 0; /* First time we see the soft limit reached */
1884 } else {
1885 time_t elapsed = server.unixtime - c->obuf_soft_limit_reached_time;
1886
1887 if (elapsed <=
1888 server.client_obuf_limits[class].soft_limit_seconds) {
1889 soft = 0; /* The client still did not reached the max number of
1890 seconds for the soft limit to be considered
1891 reached. */
1892 }
1893 }
1894 } else {
1895 c->obuf_soft_limit_reached_time = 0;
1896 }
1897 return soft || hard;
1898}

1900/* Asynchronously close a client if soft or hard limit is reached on the
1901 * output buffer size. The caller can check if the client will be closed
1902 * checking if the client CLIENT_CLOSE_ASAP flag is set.
1903 *
1904 * Note: we need to close the client asynchronously because this function is
1905 * called from contexts where the client can't be freed safely, i.e. from the
1906 * lower level functions pushing data inside the client output buffers. */
1907void asyncCloseClientOnOutputBufferLimitReached(client *c) {
1908 serverAssert(c->reply_bytes < SIZE_MAX-(1024*64));
1909 if (c->reply_bytes == 0 || c->flags & CLIENT_CLOSE_ASAP) return;
1910 if (checkClientOutputBufferLimits(c)) {
1911 sds client = catClientInfoString(sdsempty(),c);
1912
1913 freeClientAsync(c);
1914 serverLog(LL_WARNING,"Client %s scheduled to be closed ASAP for overcoming of output buffer limits.", client);
1915 sdsfree(client);
1916 }
1917}

1919/* Helper function used by freeMemoryIfNeeded() in order to flush slaves
1920 * output buffers without returning control to the event loop.
1921 * This is also called by SHUTDOWN for a best-effort attempt to send
1922 * slaves the latest writes. */
1923void flushSlavesOutputBuffers(void) {
1924 listIter li;
1925 listNode *ln;
1926
1927 listRewind(server.slaves,&li);
1928 while((ln = listNext(&li))) {
1929 client *slave = listNodeValue(ln);
1930 int events;
1931
1932 /* Note that the following will not flush output buffers of slaves
1933 * in STATE_ONLINE but having put_online_on_ack set to true: in this
1934 * case the writable event is never installed, since the purpose
1935 * of put_online_on_ack is to postpone the moment it is installed.
1936 * This is what we want since slaves in this state should not receive
1937 * writes before the first ACK. */
1938 events = aeGetFileEvents(server.el,slave->fd);
1939 if (events & AE_WRITABLE &&
1940 slave->replstate == SLAVE_STATE_ONLINE &&
1941 clientHasPendingReplies(slave))
1942 {
1943 writeToClient(slave->fd,slave,0);
1944 }
1945 }
1946}

1948/* Pause clients up to the specified unixtime (in ms). While clients
1949 * are paused no command is processed from clients, so the data set can't
1950 * change during that time.
1951 *
1952 * However while this function pauses normal and Pub/Sub clients, slaves are
1953 * still served, so this function can be used on server upgrades where it is
1954 * required that slaves process the latest bytes from the replication stream
1955 * before being turned to masters.
1956 *
1957 * This function is also internally used by Redis Cluster for the manual
1958 * failover procedure implemented by CLUSTER FAILOVER.
1959 *
1960 * The function always succeed, even if there is already a pause in progress.
1961 * In such a case, the pause is extended if the duration is more than the
1962 * time left for the previous duration. However if the duration is smaller
1963 * than the time left for the previous pause, no change is made to the
1964 * left duration. */
1965void pauseClients(mstime_t end) {
1966 if (!server.clients_paused || end > server.clients_pause_end_time)
1967 server.clients_pause_end_time = end;
1968 server.clients_paused = 1;
1969}

1971/* Return non-zero if clients are currently paused. As a side effect the
1972 * function checks if the pause time was reached and clear it. */
1973int clientsArePaused(void) {
1974 if (server.clients_paused &&
1975 server.clients_pause_end_time < server.mstime)
1976 {
1977 listNode *ln;
1978 listIter li;
1979 client *c;
1980
1981 server.clients_paused = 0;
1982
1983 /* Put all the clients in the unblocked clients queue in order to
1984 * force the re-processing of the input buffer if any. */
1985 listRewind(server.clients,&li);
1986 while ((ln = listNext(&li)) != NULL) {
1987 c = listNodeValue(ln);
1988
1989 /* Don't touch slaves and blocked clients. The latter pending
1990 * requests be processed when unblocked. */
1991 if (c->flags & (CLIENT_SLAVE|CLIENT_BLOCKED)) continue;
1992 c->flags |= CLIENT_UNBLOCKED;
1993 listAddNodeTail(server.unblocked_clients,c);
1994 }
1995 }
1996 return server.clients_paused;
1997}

1999/* This function is called by Redis in order to process a few events from
2000 * time to time while blocked into some not interruptible operation.
2001 * This allows to reply to clients with the -LOADING error while loading the
2002 * data set at startup or after a full resynchronization with the master
2003 * and so forth.
2004 *
2005 * It calls the event loop in order to process a few events. Specifically we
2006 * try to call the event loop 4 times as long as we receive acknowledge that
2007 * some event was processed, in order to go forward with the accept, read,
2008 * write, close sequence needed to serve a client.
2009 *
2010 * The function returns the total number of events processed. */
2011int processEventsWhileBlocked(void) {
2012 int iterations = 4; /* See the function top-comment. */
2013 int count = 0;
2014 while (iterations--) {
2015 int events = 0;
2016 events += aeProcessEvents(server.el, AE_FILE_EVENTS|AE_DONT_WAIT);
2017 events += handleClientsWithPendingWrites();
2018 if (!events) break;
2019 count += events;
2020 }
2021 return count;
