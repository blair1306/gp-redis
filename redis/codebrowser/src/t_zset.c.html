35/* ZSETs are ordered sets using two data structures to hold the same elements
36 * in order to get O(log(N)) INSERT and REMOVE operations into a sorted
37 * data structure.
38 *
39 * The elements are added to a hash table mapping Redis objects to scores.
40 * At the same time the elements are added to a skip list mapping scores
41 * to Redis objects (so objects are sorted by scores in this "view").
42 *
43 * Note that the SDS string representing the element is the same in both
44 * the hash table and skiplist in order to save memory. What we do in order
45 * to manage the shared SDS string more easily is to free the SDS string
46 * only in zslFreeNode(). The dictionary has no value free method set.
47 * So we should always remove an element from the dictionary, and later from
48 * the skiplist.
49 *
50 * This skiplist implementation is almost a C translation of the original
51 * algorithm described by William Pugh in "Skip Lists: A Probabilistic
52 * Alternative to Balanced Trees", modified in three ways:
53 * a) this implementation allows for repeated scores.
54 * b) the comparison is not just by key (our 'score') but by satellite data.
55 * c) there is a back pointer, so it's a doubly linked list with the back
56 * pointers being only at "level 1". This allows to traverse the list
57 * from tail to head, useful for ZREVRANGE. */

69/* Create a skiplist node with the specified number of levels.
70 * The SDS string 'ele' is referenced by the node after the call. */
71zskiplistNode *zslCreateNode(int level, double score, sds ele) {
72 zskiplistNode *zn =
73 zmalloc(sizeof(*zn)+level*sizeof(struct zskiplistLevel));
74 zn->score = score;
75 zn->ele = ele;
76 return zn;
77}

79/* Create a new skiplist. */
80zskiplist *zslCreate(void) {
81 int j;
82 zskiplist *zsl;
83
84 zsl = zmalloc(sizeof(*zsl));
85 zsl->level = 1;
86 zsl->length = 0;
87 zsl->header = zslCreateNode(ZSKIPLIST_MAXLEVEL,0,NULL);
88 for (j = 0; j < ZSKIPLIST_MAXLEVEL; j++) {
89 zsl->header->level[j].forward = NULL;
90 zsl->header->level[j].span = 0;
91 }
92 zsl->header->backward = NULL;
93 zsl->tail = NULL;
94 return zsl;
95}

97/* Free the specified skiplist node. The referenced SDS string representation
98 * of the element is freed too, unless node->ele is set to NULL before calling
99 * this function. */
100void zslFreeNode(zskiplistNode *node) {
101 sdsfree(node->ele);
102 zfree(node);
103}

105/* Free a whole skiplist. */
106void zslFree(zskiplist *zsl) {
107 zskiplistNode *node = zsl->header->level[0].forward, *next;
108
109 zfree(zsl->header);
110 while(node) {
111 next = node->level[0].forward;
112 zslFreeNode(node);
113 node = next;
114 }
115 zfree(zsl);
116}

118/* Returns a random level for the new skiplist node we are going to create.
119 * The return value of this function is between 1 and ZSKIPLIST_MAXLEVEL
120 * (both inclusive), with a powerlaw-alike distribution where higher
121 * levels are less likely to be returned. */
122int zslRandomLevel(void) {
123 int level = 1;
124 while ((random()&0xFFFF) < (ZSKIPLIST_P * 0xFFFF))
125 level += 1;
126 return (level<ZSKIPLIST_MAXLEVEL) ? level : ZSKIPLIST_MAXLEVEL;
127}

129/* Insert a new node in the skiplist. Assumes the element does not already
130 * exist (up to the caller to enforce that). The skiplist takes ownership
131 * of the passed SDS string 'ele'. */
132zskiplistNode *zslInsert(zskiplist *zsl, double score, sds ele) {
133 zskiplistNode *update[ZSKIPLIST_MAXLEVEL], *x;
134 unsigned int rank[ZSKIPLIST_MAXLEVEL];
135 int i, level;
136
137 serverAssert(!isnan(score));
138 x = zsl->header;
139 for (i = zsl->level-1; i >= 0; i--) {
140 /* store rank that is crossed to reach the insert position */
141 rank[i] = i == (zsl->level-1) ? 0 : rank[i+1];
142 while (x->level[i].forward &&
143 (x->level[i].forward->score < score ||
144 (x->level[i].forward->score == score &&
145 sdscmp(x->level[i].forward->ele,ele) < 0)))
146 {
147 rank[i] += x->level[i].span;
148 x = x->level[i].forward;
149 }
150 update[i] = x;
151 }
152 /* we assume the element is not already inside, since we allow duplicated
153 * scores, reinserting the same element should never happen since the
154 * caller of zslInsert() should test in the hash table if the element is
155 * already inside or not. */
156 level = zslRandomLevel();
157 if (level > zsl->level) {
158 for (i = zsl->level; i < level; i++) {
159 rank[i] = 0;
160 update[i] = zsl->header;
161 update[i]->level[i].span = zsl->length;
162 }
163 zsl->level = level;
164 }
165 x = zslCreateNode(level,score,ele);
166 for (i = 0; i < level; i++) {
167 x->level[i].forward = update[i]->level[i].forward;
168 update[i]->level[i].forward = x;
169
170 /* update span covered by update[i] as x is inserted here */
171 x->level[i].span = update[i]->level[i].span - (rank[0] - rank[i]);
172 update[i]->level[i].span = (rank[0] - rank[i]) + 1;
173 }
174
175 /* increment span for untouched levels */
176 for (i = level; i < zsl->level; i++) {
177 update[i]->level[i].span++;
178 }
179
180 x->backward = (update[0] == zsl->header) ? NULL : update[0];
181 if (x->level[0].forward)
182 x->level[0].forward->backward = x;
183 else
184 zsl->tail = x;
185 zsl->length++;
186 return x;
187}

189/* Internal function used by zslDelete, zslDeleteByScore and zslDeleteByRank */
190void zslDeleteNode(zskiplist *zsl, zskiplistNode *x, zskiplistNode **update) {
191 int i;
192 for (i = 0; i < zsl->level; i++) {
193 if (update[i]->level[i].forward == x) {
194 update[i]->level[i].span += x->level[i].span - 1;
195 update[i]->level[i].forward = x->level[i].forward;
196 } else {
197 update[i]->level[i].span -= 1;
198 }
199 }
200 if (x->level[0].forward) {
201 x->level[0].forward->backward = x->backward;
202 } else {
203 zsl->tail = x->backward;
204 }
205 while(zsl->level > 1 && zsl->header->level[zsl->level-1].forward == NULL)
206 zsl->level--;
207 zsl->length--;
208}

210/* Delete an element with matching score/element from the skiplist.
211 * The function returns 1 if the node was found and deleted, otherwise
212 * 0 is returned.
213 *
214 * If 'node' is NULL the deleted node is freed by zslFreeNode(), otherwise
215 * it is not freed (but just unlinked) and *node is set to the node pointer,
216 * so that it is possible for the caller to reuse the node (including the
217 * referenced SDS string at node->ele). */
218int zslDelete(zskiplist *zsl, double score, sds ele, zskiplistNode **node) {
219 zskiplistNode *update[ZSKIPLIST_MAXLEVEL], *x;
220 int i;
221
222 x = zsl->header;
223 for (i = zsl->level-1; i >= 0; i--) {
224 while (x->level[i].forward &&
225 (x->level[i].forward->score < score ||
226 (x->level[i].forward->score == score &&
227 sdscmp(x->level[i].forward->ele,ele) < 0)))
228 {
229 x = x->level[i].forward;
230 }
231 update[i] = x;
232 }
233 /* We may have multiple elements with the same score, what we need
234 * is to find the element with both the right score and object. */
235 x = x->level[0].forward;
236 if (x && score == x->score && sdscmp(x->ele,ele) == 0) {
237 zslDeleteNode(zsl, x, update);
238 if (!node)
239 zslFreeNode(x);
240 else
241 *node = x;
242 return 1;
243 }
244 return 0; /* not found */
245}

248 return spec->minex ? (value > spec->min) : (value >= spec->min);

255/* Returns if there is a part of the zset is in range. */
256int zslIsInRange(zskiplist *zsl, zrangespec *range) {
257 zskiplistNode *x;
258
259 /* Test for ranges that will always be empty. */
260 if (range->min > range->max ||
261 (range->min == range->max && (range->minex || range->maxex)))
262 return 0;
263 x = zsl->tail;
264 if (x == NULL || !zslValueGteMin(x->score,range))
265 return 0;
266 x = zsl->header->level[0].forward;
267 if (x == NULL || !zslValueLteMax(x->score,range))
268 return 0;
269 return 1;
270}

272/* Find the first node that is contained in the specified range.
273 * Returns NULL when no element is contained in the range. */
274zskiplistNode *zslFirstInRange(zskiplist *zsl, zrangespec *range) {
275 zskiplistNode *x;
276 int i;
277
278 /* If everything is out of range, return early. */
279 if (!zslIsInRange(zsl,range)) return NULL;
280
281 x = zsl->header;
282 for (i = zsl->level-1; i >= 0; i--) {
283 /* Go forward while *OUT* of range. */
284 while (x->level[i].forward &&
285 !zslValueGteMin(x->level[i].forward->score,range))
286 x = x->level[i].forward;
287 }
288
289 /* This is an inner range, so the next node cannot be NULL. */
290 x = x->level[0].forward;
291 serverAssert(x != NULL);
292
293 /* Check if score <= max. */
294 if (!zslValueLteMax(x->score,range)) return NULL;
295 return x;
296}

298/* Find the last node that is contained in the specified range.
299 * Returns NULL when no element is contained in the range. */
300zskiplistNode *zslLastInRange(zskiplist *zsl, zrangespec *range) {
301 zskiplistNode *x;
302 int i;
303
304 /* If everything is out of range, return early. */
305 if (!zslIsInRange(zsl,range)) return NULL;
306
307 x = zsl->header;
308 for (i = zsl->level-1; i >= 0; i--) {
309 /* Go forward while *IN* range. */
310 while (x->level[i].forward &&
311 zslValueLteMax(x->level[i].forward->score,range))
312 x = x->level[i].forward;
313 }
314
315 /* This is an inner range, so this node cannot be NULL. */
316 serverAssert(x != NULL);
317
318 /* Check if score >= min. */
319 if (!zslValueGteMin(x->score,range)) return NULL;
320 return x;
321}

323/* Delete all the elements with score between min and max from the skiplist.
324 * Min and max are inclusive, so a score >= min || score <= max is deleted.
325 * Note that this function takes the reference to the hash table view of the
326 * sorted set, in order to remove the elements from the hash table too. */
327unsigned long zslDeleteRangeByScore(zskiplist *zsl, zrangespec *range, dict *dict) {
328 zskiplistNode *update[ZSKIPLIST_MAXLEVEL], *x;
329 unsigned long removed = 0;
330 int i;
331
332 x = zsl->header;
333 for (i = zsl->level-1; i >= 0; i--) {
334 while (x->level[i].forward && (range->minex ?
335 x->level[i].forward->score <= range->min :
336 x->level[i].forward->score < range->min))
337 x = x->level[i].forward;
338 update[i] = x;
339 }
340
341 /* Current node is the last with score < or <= min. */
342 x = x->level[0].forward;
343
344 /* Delete nodes while in range. */
345 while (x &&
346 (range->maxex ? x->score < range->max : x->score <= range->max))
347 {
348 zskiplistNode *next = x->level[0].forward;
349 zslDeleteNode(zsl,x,update);
350 dictDelete(dict,x->ele);
351 zslFreeNode(x); /* Here is where x->ele is actually released. */
352 removed++;
353 x = next;
354 }
355 return removed;
356}

359 zskiplistNode *update[ZSKIPLIST_MAXLEVEL], *x;
360 unsigned long removed = 0;
361 int i;
362
363
364 x = zsl->header;
365 for (i = zsl->level-1; i >= 0; i--) {

376 while (x && zslLexValueLteMax(x->ele,range)) {
377 zskiplistNode *next = x->level[0].forward;
378 zslDeleteNode(zsl,x,update);

383 }

387/* Delete all the elements with rank between start and end from the skiplist.
388 * Start and end are inclusive. Note that start and end need to be 1-based */
389unsigned long zslDeleteRangeByRank(zskiplist *zsl, unsigned int start, unsigned int end, dict *dict) {
390 zskiplistNode *update[ZSKIPLIST_MAXLEVEL], *x;
391 unsigned long traversed = 0, removed = 0;
392 int i;
393
394 x = zsl->header;
395 for (i = zsl->level-1; i >= 0; i--) {
396 while (x->level[i].forward && (traversed + x->level[i].span) < start) {
397 traversed += x->level[i].span;
398 x = x->level[i].forward;
399 }
400 update[i] = x;
401 }
402
403 traversed++;
404 x = x->level[0].forward;
405 while (x && traversed <= end) {
406 zskiplistNode *next = x->level[0].forward;
407 zslDeleteNode(zsl,x,update);
408 dictDelete(dict,x->ele);
409 zslFreeNode(x);
410 removed++;
411 traversed++;
412 x = next;
413 }
414 return removed;
415}

417/* Find the rank for an element by both score and key.
418 * Returns 0 when the element cannot be found, rank otherwise.
419 * Note that the rank is 1-based due to the span of zsl->header to the
420 * first element. */
421unsigned long zslGetRank(zskiplist *zsl, double score, sds ele) {
422 zskiplistNode *x;
423 unsigned long rank = 0;
424 int i;
425
426 x = zsl->header;
427 for (i = zsl->level-1; i >= 0; i--) {
428 while (x->level[i].forward &&
429 (x->level[i].forward->score < score ||
430 (x->level[i].forward->score == score &&
431 sdscmp(x->level[i].forward->ele,ele) <= 0))) {
432 rank += x->level[i].span;
433 x = x->level[i].forward;
434 }
435
436 /* x might be equal to zsl->header, so test if obj is non-NULL */
437 if (x->ele && sdscmp(x->ele,ele) == 0) {
438 return rank;
439 }
440 }
441 return 0;
442}

444/* Finds an element by its rank. The rank argument needs to be 1-based. */
445zskiplistNode* zslGetElementByRank(zskiplist *zsl, unsigned long rank) {
446 zskiplistNode *x;
447 unsigned long traversed = 0;
448 int i;
449
450 x = zsl->header;
451 for (i = zsl->level-1; i >= 0; i--) {
452 while (x->level[i].forward && (traversed + x->level[i].span) <= rank)
453 {
454 traversed += x->level[i].span;
455 x = x->level[i].forward;
456 }
457 if (traversed == rank) {
458 return x;
459 }
460 }
461 return NULL;
462}

464/* Populate the rangespec according to the objects min and max. */
465static int zslParseRange(robj *min, robj *max, zrangespec *spec) {
466 char *eptr;
467 spec->minex = spec->maxex = 0;
468
469 /* Parse the min-max interval. If one of the values is prefixed
470 * by the "(" character, it's considered "open". For instance
471 * ZRANGEBYSCORE zset (1.5 (2.5 will match min < x < max
472 * ZRANGEBYSCORE zset 1.5 2.5 will instead match min <= x <= max */
473 if (min->encoding == OBJ_ENCODING_INT) {
474 spec->min = (long)min->ptr;
475 } else {
476 if (((char*)min->ptr)[0] == '(') {
477 spec->min = strtod((char*)min->ptr+1,&eptr);
478 if (eptr[0] != '\0' || isnan(spec->min)) return C_ERR;
479 spec->minex = 1;
480 } else {
481 spec->min = strtod((char*)min->ptr,&eptr);
482 if (eptr[0] != '\0' || isnan(spec->min)) return C_ERR;
483 }
484 }
485 if (max->encoding == OBJ_ENCODING_INT) {
486 spec->max = (long)max->ptr;
487 } else {
488 if (((char*)max->ptr)[0] == '(') {
489 spec->max = strtod((char*)max->ptr+1,&eptr);
490 if (eptr[0] != '\0' || isnan(spec->max)) return C_ERR;
491 spec->maxex = 1;
492 } else {
493 spec->max = strtod((char*)max->ptr,&eptr);
494 if (eptr[0] != '\0' || isnan(spec->max)) return C_ERR;
495 }
496 }
497
498 return C_OK;
499}

501/* ------------------------ Lexicographic ranges ---------------------------- */
502
503/* Parse max or min argument of ZRANGEBYLEX.
504 * (foo means foo (open interval)
505 * [foo means foo (closed interval)
506 * - means the min string possible
507 * + means the max string possible
508 *
509 * If the string is valid the *dest pointer is set to the redis object
510 * that will be used for the comparision, and ex will be set to 0 or 1
511 * respectively if the item is exclusive or inclusive. C_OK will be
512 * returned.
513 *
514 * If the string is not a valid range C_ERR is returned, and the value
515 * of *dest and *ex is undefined. */
516int zslParseLexRangeItem(robj *item, sds *dest, int *ex) {
517 char *c = item->ptr;
518
519 switch(c[0]) {
520 case '+':
521 if (c[1] != '\0') return C_ERR;
522 *ex = 0;
523 *dest = shared.maxstring;
524 return C_OK;
525 case '-':
526 if (c[1] != '\0') return C_ERR;
527 *ex = 0;
528 *dest = shared.minstring;
529 return C_OK;
530 case '(':
531 *ex = 1;
532 *dest = sdsnewlen(c+1,sdslen(c)-1);
533 return C_OK;
534 case '[':
535 *ex = 0;
536 *dest = sdsnewlen(c+1,sdslen(c)-1);
537 return C_OK;
538 default:
539 return C_ERR;
540 }
541}

543/* Free a lex range structure, must be called only after zelParseLexRange()
544 * populated the structure with success (C_OK returned). */
545void zslFreeLexRange(zlexrangespec *spec) {
546 if (spec->min != shared.minstring &&
547 spec->min != shared.maxstring) sdsfree(spec->min);
548 if (spec->max != shared.minstring &&
549 spec->max != shared.maxstring) sdsfree(spec->max);
550}
551
552/* Populate the lex rangespec according to the objects min and max.
553 *
554 * Return C_OK on success. On error C_ERR is returned.
555 * When OK is returned the structure must be freed with zslFreeLexRange(),
556 * otherwise no release is needed. */
557int zslParseLexRange(robj *min, robj *max, zlexrangespec *spec) {
558 /* The range can't be valid if objects are integer encoded.
559 * Every item must start with ( or [. */
560 if (min->encoding == OBJ_ENCODING_INT ||
561 max->encoding == OBJ_ENCODING_INT) return C_ERR;
562
563 spec->min = spec->max = NULL;
564 if (zslParseLexRangeItem(min, &spec->min, &spec->minex) == C_ERR ||
565 zslParseLexRangeItem(max, &spec->max, &spec->maxex) == C_ERR) {
566 zslFreeLexRange(spec);
567 return C_ERR;
568 } else {
569 return C_OK;
570 }
571}

573/* This is just a wrapper to sdscmp() that is able to
574 * handle shared.minstring and shared.maxstring as the equivalent of
575 * -inf and +inf for strings */
576int sdscmplex(sds a, sds b) {
577 if (a == b) return 0;
578 if (a == shared.minstring || b == shared.maxstring) return -1;
579 if (a == shared.maxstring || b == shared.minstring) return 1;
580 return sdscmp(a,b);
581}

583int zslLexValueGteMin(sds value, zlexrangespec *spec) {
584 return spec->minex ?
585 (sdscmplex(value,spec->min) > 0) :
586 (sdscmplex(value,spec->min) >= 0);
587}

589int zslLexValueLteMax(sds value, zlexrangespec *spec) {
590 return spec->maxex ?
591 (sdscmplex(value,spec->max) < 0) :
592 (sdscmplex(value,spec->max) <= 0);
593}

595/* Returns if there is a part of the zset is in the lex range. */
596int zslIsInLexRange(zskiplist *zsl, zlexrangespec *range) {
597 zskiplistNode *x;
598
599 /* Test for ranges that will always be empty. */
600 if (sdscmplex(range->min,range->max) > 1 ||
601 (sdscmp(range->min,range->max) == 0 &&
602 (range->minex || range->maxex)))
603 return 0;
604 x = zsl->tail;
605 if (x == NULL || !zslLexValueGteMin(x->ele,range))
606 return 0;
607 x = zsl->header->level[0].forward;
608 if (x == NULL || !zslLexValueLteMax(x->ele,range))
609 return 0;
610 return 1;
611}

639/* Find the last node that is contained in the specified range.
640 * Returns NULL when no element is contained in the range. */
641zskiplistNode *zslLastInLexRange(zskiplist *zsl, zlexrangespec *range) {

646 if (!zslIsInLexRange(zsl,range)) return NULL;
647
648 x = zsl->header;
649 for (i = zsl->level-1; i >= 0; i--) {
650 /* Go forward while *IN* range. */
651 while (x->level[i].forward &&
652 zslLexValueLteMax(x->level[i].forward->ele,range))
653 x = x->level[i].forward;
654 }
655
656 /* This is an inner range, so this node cannot be NULL. */
657 serverAssert(x != NULL);
658
659 /* Check if score >= min. */
660 if (!zslLexValueGteMin(x->ele,range)) return NULL;
661 return x;
662}

768/* Returns if there is a part of the zset is in range. Should only be used
769 * internally by zzlFirstInRange and zzlLastInRange. */
770int zzlIsInRange(unsigned char *zl, zrangespec *range) {
771 unsigned char *p;
772 double score;
773
774 /* Test for ranges that will always be empty. */
775 if (range->min > range->max ||
776 (range->min == range->max && (range->minex || range->maxex)))
777 return 0;
778
779 p = ziplistIndex(zl,-1); /* Last score. */
780 if (p == NULL) return 0; /* Empty sorted set */
781 score = zzlGetScore(p);
782 if (!zslValueGteMin(score,range))
783 return 0;
784
785 p = ziplistIndex(zl,1); /* First score. */
786 serverAssert(p != NULL);
787 score = zzlGetScore(p);
788 if (!zslValueLteMax(score,range))
789 return 0;
790
791 return 1;
792}

976unsigned char *zzlInsertAt(unsigned char *zl, unsigned char *eptr, sds ele, double score) {
977 unsigned char *sptr;
978 char scorebuf[128];

980 size_t offset;

983 if (eptr == NULL) {

985 zl = ziplistPush(zl,(unsigned char*)scorebuf,scorelen,ZIPLIST_TAIL);
986 } else {

988 offset = eptr-zl;

990 eptr = zl+offset;
991

994 zl = ziplistInsert(zl,sptr,(unsigned char*)scorebuf,scorelen);

999/* Insert (element,score) pair in ziplist. This function assumes the element is
1000 * not yet present in the list. */
1001unsigned char *zzlInsert(unsigned char *zl, sds ele, double score) {
1002 unsigned char *eptr = ziplistIndex(zl,0), *sptr;
1003 double s;
1004
1005 while (eptr != NULL) {
1006 sptr = ziplistNext(zl,eptr);
1007 serverAssert(sptr != NULL);
1008 s = zzlGetScore(sptr);
1009
1010 if (s > score) {
1011 /* First element with score larger than score for element to be
1012 * inserted. This means we should take its spot in the list to
1013 * maintain ordering. */
1014 zl = zzlInsertAt(zl,eptr,ele,score);
1015 break;
1016 } else if (s == score) {
1017 /* Ensure lexicographical ordering for elements. */
1018 if (zzlCompareElements(eptr,(unsigned char*)ele,sdslen(ele)) > 0) {
1019 zl = zzlInsertAt(zl,eptr,ele,score);
1020 break;
1021 }
1022 }
1023
1024 /* Move to next element. */
1025 eptr = ziplistNext(zl,sptr);
1026 }
1027
1028 /* Push on tail of list when it was not yet inserted. */
1029 if (eptr == NULL)
1030 zl = zzlInsertAt(zl,NULL,ele,score);
1031 return zl;
1032}

1115void zsetConvert(robj *zobj, int encoding) {
1116 zset *zs;
1117 zskiplistNode *node, *next;
1118 sds ele;
1119 double score;
1120
1121 if (zobj->encoding == encoding) return;
1122 if (zobj->encoding == OBJ_ENCODING_ZIPLIST) {
1123 unsigned char *zl = zobj->ptr;
1124 unsigned char *eptr, *sptr;
1125 unsigned char *vstr;
1126 unsigned int vlen;
1127 long long vlong;
1128
1129 if (encoding != OBJ_ENCODING_SKIPLIST)
1130 serverPanic("Unknown target encoding");
1131
1132 zs = zmalloc(sizeof(*zs));
1133 zs->dict = dictCreate(&zsetDictType,NULL);
1134 zs->zsl = zslCreate();
1135
1136 eptr = ziplistIndex(zl,0);
1137 serverAssertWithInfo(NULL,zobj,eptr != NULL);
1138 sptr = ziplistNext(zl,eptr);
1139 serverAssertWithInfo(NULL,zobj,sptr != NULL);
1140
1141 while (eptr != NULL) {
1142 score = zzlGetScore(sptr);
1143 serverAssertWithInfo(NULL,zobj,ziplistGet(eptr,&vstr,&vlen,&vlong));
1144 if (vstr == NULL)
1145 ele = sdsfromlonglong(vlong);
1146 else
1147 ele = sdsnewlen((char*)vstr,vlen);
1148
1149 node = zslInsert(zs->zsl,score,ele);
1150 serverAssert(dictAdd(zs->dict,ele,&node->score) == DICT_OK);
1151 zzlNext(zl,&eptr,&sptr);
1152 }
1153
1154 zfree(zobj->ptr);
1155 zobj->ptr = zs;
1156 zobj->encoding = OBJ_ENCODING_SKIPLIST;
1157 } else if (zobj->encoding == OBJ_ENCODING_SKIPLIST) {
1158 unsigned char *zl = ziplistNew();
1159
1160 if (encoding != OBJ_ENCODING_ZIPLIST)
1161 serverPanic("Unknown target encoding");
1162
1163 /* Approach similar to zslFree(), since we want to free the skiplist at
1164 * the same time as creating the ziplist. */
1165 zs = zobj->ptr;
1166 dictRelease(zs->dict);
1167 node = zs->zsl->header->level[0].forward;
1168 zfree(zs->zsl->header);
1169 zfree(zs->zsl);
1170
1171 while (node) {
1172 zl = zzlInsertAt(zl,NULL,node->ele,node->score);
1173 next = node->level[0].forward;
1174 zslFreeNode(node);
1175 node = next;
1176 }
1177
1178 zfree(zs);
1179 zobj->ptr = zl;
1180 zobj->encoding = OBJ_ENCODING_ZIPLIST;
1181 } else {
1182 serverPanic("Unknown sorted set encoding");
1183 }
1184}

1218/* Add a new element or update the score of an existing element in a sorted
1219 * set, regardless of its encoding.
1220 *
1221 * The set of flags change the command behavior. They are passed with an integer
1222 * pointer since the function will clear the flags and populate them with
1223 * other flags to indicate different conditions.
1224 *
1225 * The input flags are the following:
1226 *
1227 * ZADD_INCR: Increment the current element score by 'score' instead of updating
1228 * the current element score. If the element does not exist, we
1229 * assume 0 as previous score.
1230 * ZADD_NX: Perform the operation only if the element does not exist.
1231 * ZADD_XX: Perform the operation only if the element already exist.
1232 *
1233 * When ZADD_INCR is used, the new score of the element is stored in
1234 * '*newscore' if 'newscore' is not NULL.
1235 *
1236 * The returned flags are the following:
1237 *
1238 * ZADD_NAN: The resulting score is not a number.
1239 * ZADD_ADDED: The element was added (not present before the call).
1240 * ZADD_UPDATED: The element score was updated.
1241 * ZADD_NOP: No operation was performed because of NX or XX.
1242 *
1243 * Return value:
1244 *
1245 * The function returns 1 on success, and sets the appropriate flags
1246 * ADDED or UPDATED to signal what happened during the operation (note that
1247 * none could be set if we re-added an element using the same score it used
1248 * to have, or in the case a zero increment is used).
1249 *
1250 * The function returns 0 on erorr, currently only when the increment
1251 * produces a NAN condition, or when the 'score' value is NAN since the
1252 * start.
1253 *
1254 * The commad as a side effect of adding a new element may convert the sorted
1255 * set internal encoding from ziplist to hashtable+skiplist.
1256 *
1257 * Memory managemnet of 'ele':
1258 *
1259 * The function does not take ownership of the 'ele' SDS string, but copies
1260 * it if needed. */
1261int zsetAdd(robj *zobj, double score, sds ele, int *flags, double *newscore) {
1262 /* Turn options into simple to check vars. */
1263 int incr = (*flags & ZADD_INCR) != 0;
1264 int nx = (*flags & ZADD_NX) != 0;
1265 int xx = (*flags & ZADD_XX) != 0;
1266 *flags = 0; /* We'll return our response flags. */
1267 double curscore;
1268
1269 /* NaN as input is an error regardless of all the other parameters. */
1270 if (isnan(score)) {
1271 *flags = ZADD_NAN;
1272 return 0;
1273 }
1274
1275 /* Update the sorted set according to its encoding. */
1276 if (zobj->encoding == OBJ_ENCODING_ZIPLIST) {
1277 unsigned char *eptr;
1278
1279 if ((eptr = zzlFind(zobj->ptr,ele,&curscore)) != NULL) {
1280 /* NX? Return, same element already exists. */
1281 if (nx) {
1282 *flags |= ZADD_NOP;
1283 return 1;
1284 }
1285
1286 /* Prepare the score for the increment if needed. */
1287 if (incr) {
1288 score += curscore;
1289 if (isnan(score)) {
1290 *flags |= ZADD_NAN;
1291 return 0;
1292 }
1293 if (newscore) *newscore = score;
1294 }
1295
1296 /* Remove and re-insert when score changed. */
1297 if (score != curscore) {
1298 zobj->ptr = zzlDelete(zobj->ptr,eptr);
1299 zobj->ptr = zzlInsert(zobj->ptr,ele,score);
1300 *flags |= ZADD_UPDATED;
1301 }
1302 return 1;
1303 } else if (!xx) {
1304 /* Optimize: check if the element is too large or the list
1305 * becomes too long *before* executing zzlInsert. */
1306 zobj->ptr = zzlInsert(zobj->ptr,ele,score);
1307 if (zzlLength(zobj->ptr) > server.zset_max_ziplist_entries)
1308 zsetConvert(zobj,OBJ_ENCODING_SKIPLIST);
1309 if (sdslen(ele) > server.zset_max_ziplist_value)
1310 zsetConvert(zobj,OBJ_ENCODING_SKIPLIST);
1311 if (newscore) *newscore = score;
1312 *flags |= ZADD_ADDED;
1313 return 1;
1314 } else {
1315 *flags |= ZADD_NOP;
1316 return 1;
1317 }
1318 } else if (zobj->encoding == OBJ_ENCODING_SKIPLIST) {
1319 zset *zs = zobj->ptr;
1320 zskiplistNode *znode;
1321 dictEntry *de;
1322
1323 de = dictFind(zs->dict,ele);
1324 if (de != NULL) {
1325 /* NX? Return, same element already exists. */
1326 if (nx) {
1327 *flags |= ZADD_NOP;
1328 return 1;
1329 }
1330 curscore = *(double*)dictGetVal(de);
1331
1332 /* Prepare the score for the increment if needed. */
1333 if (incr) {
1334 score += curscore;
1335 if (isnan(score)) {
1336 *flags |= ZADD_NAN;
1337 return 0;
1338 }
1339 if (newscore) *newscore = score;
1340 }
1341
1342 /* Remove and re-insert when score changes. */
1343 if (score != curscore) {
1344 zskiplistNode *node;
1345 serverAssert(zslDelete(zs->zsl,curscore,ele,&node));
1346 znode = zslInsert(zs->zsl,score,node->ele);
1347 /* We reused the node->ele SDS string, free the node now
1348 * since zslInsert created a new one. */
1349 node->ele = NULL;
1350 zslFreeNode(node);
1351 /* Note that we did not removed the original element from
1352 * the hash table representing the sorted set, so we just
1353 * update the score. */
1354 dictGetVal(de) = &znode->score; /* Update score ptr. */
1355 *flags |= ZADD_UPDATED;
1356 }
1357 return 1;
1358 } else if (!xx) {
1359 ele = sdsdup(ele);
1360 znode = zslInsert(zs->zsl,score,ele);
1361 serverAssert(dictAdd(zs->dict,ele,&znode->score) == DICT_OK);
1362 *flags |= ZADD_ADDED;
1363 if (newscore) *newscore = score;
1364 return 1;
1365 } else {
1366 *flags |= ZADD_NOP;
1367 return 1;
1368 }
1369 } else {
1370 serverPanic("Unknown sorted set encoding");
1371 }
1372 return 0; /* Never reached. */
1373}

1375/* Delete the element 'ele' from the sorted set, returning 1 if the element
1376 * existed and was deleted, 0 otherwise (the element was not there). */
1377int zsetDel(robj *zobj, sds ele) {
1378 if (zobj->encoding == OBJ_ENCODING_ZIPLIST) {
1379 unsigned char *eptr;
1380
1381 if ((eptr = zzlFind(zobj->ptr,ele,NULL)) != NULL) {
1382 zobj->ptr = zzlDelete(zobj->ptr,eptr);
1383 return 1;
1384 }
1385 } else if (zobj->encoding == OBJ_ENCODING_SKIPLIST) {
1386 zset *zs = zobj->ptr;
1387 dictEntry *de;
1388 double score;
1389
1390 de = dictUnlink(zs->dict,ele);
1391 if (de != NULL) {
1392 /* Get the score in order to delete from the skiplist later. */
1393 score = *(double*)dictGetVal(de);
1394
1395 /* Delete from the hash table and later from the skiplist.
1396 * Note that the order is important: deleting from the skiplist
1397 * actually releases the SDS string representing the element,
1398 * which is shared between the skiplist and the hash table, so
1399 * we need to delete from the skiplist as the final step. */
1400 dictFreeUnlinkedEntry(zs->dict,de);
1401
1402 /* Delete from skiplist. */
1403 int retval = zslDelete(zs->zsl,score,ele,NULL);
1404 serverAssert(retval);
1405
1406 if (htNeedsResize(zs->dict)) dictResize(zs->dict);
1407 return 1;
1408 }
1409 } else {
1410 serverPanic("Unknown sorted set encoding");
1411 }
1412 return 0; /* No such element found. */
1413}

1415/* Given a sorted set object returns the 0-based rank of the object or
1416 * -1 if the object does not exist.
1417 *
1418 * For rank we mean the position of the element in the sorted collection
1419 * of elements. So the first element has rank 0, the second rank 1, and so
1420 * forth up to length-1 elements.
1421 *
1422 * If 'reverse' is false, the rank is returned considering as first element
1423 * the one with the lowest score. Otherwise if 'reverse' is non-zero
1424 * the rank is computed considering as element with rank 0 the one with
1425 * the highest score. */
1426long zsetRank(robj *zobj, sds ele, int reverse) {
1427 unsigned long llen;
1428 unsigned long rank;
1429
1430 llen = zsetLength(zobj);
1431
1432 if (zobj->encoding == OBJ_ENCODING_ZIPLIST) {
1433 unsigned char *zl = zobj->ptr;
1434 unsigned char *eptr, *sptr;
1435
1436 eptr = ziplistIndex(zl,0);
1437 serverAssert(eptr != NULL);
1438 sptr = ziplistNext(zl,eptr);
1439 serverAssert(sptr != NULL);
1440
1441 rank = 1;
1442 while(eptr != NULL) {
1443 if (ziplistCompare(eptr,(unsigned char*)ele,sdslen(ele)))
1444 break;
1445 rank++;
1446 zzlNext(zl,&eptr,&sptr);
1447 }
1448
1449 if (eptr != NULL) {
1450 if (reverse)
1451 return llen-rank;
1452 else
1453 return rank-1;
1454 } else {
1455 return -1;
1456 }
1457 } else if (zobj->encoding == OBJ_ENCODING_SKIPLIST) {
1458 zset *zs = zobj->ptr;
1459 zskiplist *zsl = zs->zsl;
1460 dictEntry *de;
1461 double score;
1462
1463 de = dictFind(zs->dict,ele);
1464 if (de != NULL) {
1465 score = *(double*)dictGetVal(de);
1466 rank = zslGetRank(zsl,score,ele);
1467 /* Existing elements always have a rank. */
1468 serverAssert(rank != 0);
1469 if (reverse)
1470 return llen-rank;
1471 else
1472 return rank-1;
1473 } else {
1474 return -1;
1475 }
1476 } else {
1477 serverPanic("Unknown sorted set encoding");
1478 }
1479}

1485/* This generic command implements both ZADD and ZINCRBY. */
1486void zaddGenericCommand(client *c, int flags) {
1487 static char *nanerr = "resulting score is not a number (NaN)";
1488 robj *key = c->argv[1];
1489 robj *zobj;
1490 sds ele;
1491 double score = 0, *scores = NULL;
1492 int j, elements;
1493 int scoreidx = 0;
1494 /* The following vars are used in order to track what the command actually
1495 * did during the execution, to reply to the client and to trigger the
1496 * notification of keyspace change. */
1497 int added = 0; /* Number of new elements added. */
1498 int updated = 0; /* Number of elements with updated score. */
1499 int processed = 0; /* Number of elements processed, may remain zero with
1500 options like XX. */
1501
1502 /* Parse options. At the end 'scoreidx' is set to the argument position
1503 * of the score of the first score-element pair. */
1504 scoreidx = 2;
1505 while(scoreidx < c->argc) {
1506 char *opt = c->argv[scoreidx]->ptr;
1507 if (!strcasecmp(opt,"nx")) flags |= ZADD_NX;
1508 else if (!strcasecmp(opt,"xx")) flags |= ZADD_XX;
1509 else if (!strcasecmp(opt,"ch")) flags |= ZADD_CH;
1510 else if (!strcasecmp(opt,"incr")) flags |= ZADD_INCR;
1511 else break;
1512 scoreidx++;
1513 }
1514
1515 /* Turn options into simple to check vars. */
1516 int incr = (flags & ZADD_INCR) != 0;
1517 int nx = (flags & ZADD_NX) != 0;
1518 int xx = (flags & ZADD_XX) != 0;
1519 int ch = (flags & ZADD_CH) != 0;
1520
1521 /* After the options, we expect to have an even number of args, since
1522 * we expect any number of score-element pairs. */
1523 elements = c->argc-scoreidx;
1524 if (elements % 2 || !elements) {
1525 addReply(c,shared.syntaxerr);
1526 return;
1527 }
1528 elements /= 2; /* Now this holds the number of score-element pairs. */
1529
1530 /* Check for incompatible options. */
1531 if (nx && xx) {
1532 addReplyError(c,
1533 "XX and NX options at the same time are not compatible");
1534 return;
1535 }
1536
1537 if (incr && elements > 1) {
1538 addReplyError(c,
1539 "INCR option supports a single increment-element pair");
1540 return;
1541 }
1542
1543 /* Start parsing all the scores, we need to emit any syntax error
1544 * before executing additions to the sorted set, as the command should
1545 * either execute fully or nothing at all. */
1546 scores = zmalloc(sizeof(double)*elements);
1547 for (j = 0; j < elements; j++) {
1548 if (getDoubleFromObjectOrReply(c,c->argv[scoreidx+j*2],&scores[j],NULL)
1549 != C_OK) goto cleanup;
1550 }
1551
1552 /* Lookup the key and create the sorted set if does not exist. */
1553 zobj = lookupKeyWrite(c->db,key);
1554 if (zobj == NULL) {
1555 if (xx) goto reply_to_client; /* No key + XX option: nothing to do. */
1556 if (server.zset_max_ziplist_entries == 0 ||
1557 server.zset_max_ziplist_value < sdslen(c->argv[scoreidx+1]->ptr))
1558 {
1559 zobj = createZsetObject();
1560 } else {
1561 zobj = createZsetZiplistObject();
1562 }
1563 dbAdd(c->db,key,zobj);
1564 } else {
1565 if (zobj->type != OBJ_ZSET) {
1566 addReply(c,shared.wrongtypeerr);
1567 goto cleanup;
1568 }
1569 }
1570
1571 for (j = 0; j < elements; j++) {
1572 double newscore;
1573 score = scores[j];
1574 int retflags = flags;
1575
1576 ele = c->argv[scoreidx+1+j*2]->ptr;
1577 int retval = zsetAdd(zobj, score, ele, &retflags, &newscore);
1578 if (retval == 0) {
1579 addReplyError(c,nanerr);
1580 goto cleanup;
1581 }
1582 if (retflags & ZADD_ADDED) added++;
1583 if (retflags & ZADD_UPDATED) updated++;
1584 if (!(retflags & ZADD_NOP)) processed++;
1585 score = newscore;
1586 }
1587 server.dirty += (added+updated);
1588
1589reply_to_client:
1590 if (incr) { /* ZINCRBY or INCR option. */
1591 if (processed)
1592 addReplyDouble(c,score);
1593 else
1594 addReply(c,shared.nullbulk);
1595 } else { /* ZADD. */
1596 addReplyLongLong(c,ch ? added+updated : added);
1597 }
1598
1599cleanup:
1600 zfree(scores);
1601 if (added || updated) {
1602 signalModifiedKey(c->db,key);
1603 notifyKeyspaceEvent(NOTIFY_ZSET,
1604 incr ? "zincr" : "zadd", key, c->db->id);
1605 }
1606}

1643/* Implements ZREMRANGEBYRANK, ZREMRANGEBYSCORE, ZREMRANGEBYLEX commands. */
1644#define ZRANGE_RANK 0
1645#define ZRANGE_SCORE 1
1646#define ZRANGE_LEX 2
1647void zremrangeGenericCommand(client *c, int rangetype) {
1648 robj *key = c->argv[1];
1649 robj *zobj;
1650 int keyremoved = 0;
1651 unsigned long deleted = 0;
1652 zrangespec range;
1653 zlexrangespec lexrange;
1654 long start, end, llen;
1655
1656 /* Step 1: Parse the range. */
1657 if (rangetype == ZRANGE_RANK) {
1658 if ((getLongFromObjectOrReply(c,c->argv[2],&start,NULL) != C_OK) ||
1659 (getLongFromObjectOrReply(c,c->argv[3],&end,NULL) != C_OK))
1660 return;
1661 } else if (rangetype == ZRANGE_SCORE) {
1662 if (zslParseRange(c->argv[2],c->argv[3],&range) != C_OK) {
1663 addReplyError(c,"min or max is not a float");
1664 return;
1665 }
1666 } else if (rangetype == ZRANGE_LEX) {
1667 if (zslParseLexRange(c->argv[2],c->argv[3],&lexrange) != C_OK) {
1668 addReplyError(c,"min or max not valid string range item");
1669 return;
1670 }
1671 }
1672
1673 /* Step 2: Lookup & range sanity checks if needed. */
1674 if ((zobj = lookupKeyWriteOrReply(c,key,shared.czero)) == NULL ||
1675 checkType(c,zobj,OBJ_ZSET)) goto cleanup;
1676
1677 if (rangetype == ZRANGE_RANK) {
1678 /* Sanitize indexes. */
1679 llen = zsetLength(zobj);
1680 if (start < 0) start = llen+start;
1681 if (end < 0) end = llen+end;
1682 if (start < 0) start = 0;
1683
1684 /* Invariant: start >= 0, so this test will be true when end < 0.
1685 * The range is empty when start > end or start >= length. */
1686 if (start > end || start >= llen) {
1687 addReply(c,shared.czero);
1688 goto cleanup;
1689 }
1690 if (end >= llen) end = llen-1;
1691 }
1692
1693 /* Step 3: Perform the range deletion operation. */
1694 if (zobj->encoding == OBJ_ENCODING_ZIPLIST) {
1695 switch(rangetype) {
1696 case ZRANGE_RANK:
1697 zobj->ptr = zzlDeleteRangeByRank(zobj->ptr,start+1,end+1,&deleted);
1698 break;
1699 case ZRANGE_SCORE:
1700 zobj->ptr = zzlDeleteRangeByScore(zobj->ptr,&range,&deleted);
1701 break;
1702 case ZRANGE_LEX:
1703 zobj->ptr = zzlDeleteRangeByLex(zobj->ptr,&lexrange,&deleted);
1704 break;
1705 }
1706 if (zzlLength(zobj->ptr) == 0) {
1707 dbDelete(c->db,key);
1708 keyremoved = 1;
1709 }
1710 } else if (zobj->encoding == OBJ_ENCODING_SKIPLIST) {
1711 zset *zs = zobj->ptr;
1712 switch(rangetype) {
1713 case ZRANGE_RANK:
1714 deleted = zslDeleteRangeByRank(zs->zsl,start+1,end+1,zs->dict);
1715 break;
1716 case ZRANGE_SCORE:
1717 deleted = zslDeleteRangeByScore(zs->zsl,&range,zs->dict);
1718 break;
1719 case ZRANGE_LEX:
1720 deleted = zslDeleteRangeByLex(zs->zsl,&lexrange,zs->dict);
1721 break;
1722 }
1723 if (htNeedsResize(zs->dict)) dictResize(zs->dict);
1724 if (dictSize(zs->dict) == 0) {
1725 dbDelete(c->db,key);
1726 keyremoved = 1;
1727 }
1728 } else {
1729 serverPanic("Unknown sorted set encoding");
1730 }
1731
1732 /* Step 4: Notifications and reply. */
1733 if (deleted) {
1734 char *event[3] = {"zremrangebyrank","zremrangebyscore","zremrangebylex"};
1735 signalModifiedKey(c->db,key);
1736 notifyKeyspaceEvent(NOTIFY_ZSET,event[rangetype],key,c->db->id);
1737 if (keyremoved)
1738 notifyKeyspaceEvent(NOTIFY_GENERIC,"del",key,c->db->id);
1739 }
1740 server.dirty += deleted;
1741 addReplyLongLong(c,deleted);
1742
1743cleanup:
1744 if (rangetype == ZRANGE_LEX) zslFreeLexRange(&lexrange);
1745}

1908/* Check if the current value is valid. If so, store it in the passed structure
1909 * and move to the next element. If not valid, this means we have reached the
1910 * end of the structure and can abort. */
1911int zuiNext(zsetopsrc *op, zsetopval *val) {
1912 if (op->subject == NULL)
1913 return 0;
1914
1915 if (val->flags & OPVAL_DIRTY_SDS)
1916 sdsfree(val->ele);
1917
1918 memset(val,0,sizeof(zsetopval));
1919
1920 if (op->type == OBJ_SET) {
1921 iterset *it = &op->iter.set;
1922 if (op->encoding == OBJ_ENCODING_INTSET) {
1923 int64_t ell;
1924
1925 if (!intsetGet(it->is.is,it->is.ii,&ell))
1926 return 0;
1927 val->ell = ell;
1928 val->score = 1.0;
1929
1930 /* Move to next element. */
1931 it->is.ii++;
1932 } else if (op->encoding == OBJ_ENCODING_HT) {
1933 if (it->ht.de == NULL)
1934 return 0;
1935 val->ele = dictGetKey(it->ht.de);
1936 val->score = 1.0;
1937
1938 /* Move to next element. */
1939 it->ht.de = dictNext(it->ht.di);
1940 } else {
1941 serverPanic("Unknown set encoding");
1942 }
1943 } else if (op->type == OBJ_ZSET) {
1944 iterzset *it = &op->iter.zset;
1945 if (op->encoding == OBJ_ENCODING_ZIPLIST) {
1946 /* No need to check both, but better be explicit. */
1947 if (it->zl.eptr == NULL || it->zl.sptr == NULL)
1948 return 0;
1949 serverAssert(ziplistGet(it->zl.eptr,&val->estr,&val->elen,&val->ell));
1950 val->score = zzlGetScore(it->zl.sptr);
1951
1952 /* Move to next element. */
1953 zzlNext(it->zl.zl,&it->zl.eptr,&it->zl.sptr);
1954 } else if (op->encoding == OBJ_ENCODING_SKIPLIST) {
1955 if (it->sl.node == NULL)
1956 return 0;
1957 val->ele = it->sl.node->ele;
1958 val->score = it->sl.node->score;
