54/* Compute the sha1 of string at 's' with 'len' bytes long.
55 * The SHA1 is then xored against the string pointed by digest.
56 * Since xor is commutative, this operation is used in order to
57 * "add" digests relative to unordered elements.
58 *
59 * So digest(a,b,c,d) will be the same of digest(b,a,c,d) */
60void xorDigest(unsigned char *digest, void *ptr, size_t len) {
61 SHA1_CTX ctx;
62 unsigned char hash[20], *s = ptr;
63 int j;
64
65 SHA1Init(&ctx);
66 SHA1Update(&ctx,s,len);
67 SHA1Final(hash,&ctx);
68
69 for (j = 0; j < 20; j++)
70 digest[j] ^= hash[j];
71}

73void xorObjectDigest(unsigned char *digest, robj *o) {
74 o = getDecodedObject(o);
75 xorDigest(digest,o->ptr,sdslen(o->ptr));
76 decrRefCount(o);
77}
78
79/* This function instead of just computing the SHA1 and xoring it
80 * against digest, also perform the digest of "digest" itself and
81 * replace the old value with the new one.
82 *
83 * So the final digest will be:
84 *
85 * digest = SHA1(digest xor SHA1(data))
86 *
87 * This function is used every time we want to preserve the order so
88 * that digest(a,b,c,d) will be different than digest(b,c,d,a)
89 *
90 * Also note that mixdigest("foo") followed by mixdigest("bar")
91 * will lead to a different digest compared to "fo", "obar".
92 */
93void mixDigest(unsigned char *digest, void *ptr, size_t len) {
94 SHA1_CTX ctx;
95 char *s = ptr;
96
97 xorDigest(digest,s,len);
98 SHA1Init(&ctx);
99 SHA1Update(&ctx,digest,20);
100 SHA1Final(digest,&ctx);
101}

103void mixObjectDigest(unsigned char *digest, robj *o) {
104 o = getDecodedObject(o);
105 mixDigest(digest,o->ptr,sdslen(o->ptr));
106 decrRefCount(o);
107}

109/* Compute the dataset digest. Since keys, sets elements, hashes elements
110 * are not ordered, we use a trick: every aggregate digest is the xor
111 * of the digests of their elements. This way the order will not change
112 * the result. For list instead we use a feedback entering the output digest
113 * as input in order to ensure that a different ordered list will result in
114 * a different digest. */
115void computeDatasetDigest(unsigned char *final) {
116 unsigned char digest[20];
117 char buf[128];
118 dictIterator *di = NULL;
119 dictEntry *de;
120 int j;
121 uint32_t aux;
122
123 memset(final,0,20); /* Start with a clean result */
124
125 for (j = 0; j < server.dbnum; j++) {
126 redisDb *db = server.db+j;
127
128 if (dictSize(db->dict) == 0) continue;
129 di = dictGetSafeIterator(db->dict);
130
131 /* hash the DB id, so the same dataset moved in a different
132 * DB will lead to a different digest */
133 aux = htonl(j);
134 mixDigest(final,&aux,sizeof(aux));
135
136 /* Iterate this DB writing every entry */
137 while((de = dictNext(di)) != NULL) {
138 sds key;
139 robj *keyobj, *o;
140 long long expiretime;
141
142 memset(digest,0,20); /* This key-val digest */
143 key = dictGetKey(de);
144 keyobj = createStringObject(key,sdslen(key));
145
146 mixDigest(digest,key,sdslen(key));
147
148 o = dictGetVal(de);
149
150 aux = htonl(o->type);
151 mixDigest(digest,&aux,sizeof(aux));
152 expiretime = getExpire(db,keyobj);
153
154 /* Save the key and associated value */
155 if (o->type == OBJ_STRING) {
156 mixObjectDigest(digest,o);
157 } else if (o->type == OBJ_LIST) {
158 listTypeIterator *li = listTypeInitIterator(o,0,LIST_TAIL);
159 listTypeEntry entry;
160 while(listTypeNext(li,&entry)) {
161 robj *eleobj = listTypeGet(&entry);
162 mixObjectDigest(digest,eleobj);
163 decrRefCount(eleobj);
164 }
165 listTypeReleaseIterator(li);
166 } else if (o->type == OBJ_SET) {
167 setTypeIterator *si = setTypeInitIterator(o);
168 sds sdsele;
169 while((sdsele = setTypeNextObject(si)) != NULL) {
170 xorDigest(digest,sdsele,sdslen(sdsele));
171 sdsfree(sdsele);
172 }
173 setTypeReleaseIterator(si);
174 } else if (o->type == OBJ_ZSET) {
175 unsigned char eledigest[20];
176
177 if (o->encoding == OBJ_ENCODING_ZIPLIST) {
178 unsigned char *zl = o->ptr;
179 unsigned char *eptr, *sptr;
180 unsigned char *vstr;
181 unsigned int vlen;
182 long long vll;
183 double score;
184
185 eptr = ziplistIndex(zl,0);
186 serverAssert(eptr != NULL);
187 sptr = ziplistNext(zl,eptr);
188 serverAssert(sptr != NULL);
189
190 while (eptr != NULL) {
191 serverAssert(ziplistGet(eptr,&vstr,&vlen,&vll));
192 score = zzlGetScore(sptr);
193
194 memset(eledigest,0,20);
195 if (vstr != NULL) {
196 mixDigest(eledigest,vstr,vlen);
197 } else {
198 ll2string(buf,sizeof(buf),vll);
199 mixDigest(eledigest,buf,strlen(buf));
200 }
201
202 snprintf(buf,sizeof(buf),"%.17g",score);
203 mixDigest(eledigest,buf,strlen(buf));
204 xorDigest(digest,eledigest,20);
205 zzlNext(zl,&eptr,&sptr);
206 }
207 } else if (o->encoding == OBJ_ENCODING_SKIPLIST) {
208 zset *zs = o->ptr;
209 dictIterator *di = dictGetIterator(zs->dict);
210 dictEntry *de;
211
212 while((de = dictNext(di)) != NULL) {
213 sds sdsele = dictGetKey(de);
214 double *score = dictGetVal(de);
215
216 snprintf(buf,sizeof(buf),"%.17g",*score);
217 memset(eledigest,0,20);
218 mixDigest(eledigest,sdsele,sdslen(sdsele));
219 mixDigest(eledigest,buf,strlen(buf));
220 xorDigest(digest,eledigest,20);
221 }
222 dictReleaseIterator(di);
223 } else {
224 serverPanic("Unknown sorted set encoding");
225 }
226 } else if (o->type == OBJ_HASH) {
227 hashTypeIterator *hi = hashTypeInitIterator(o);
228 while (hashTypeNext(hi) != C_ERR) {
229 unsigned char eledigest[20];
230 sds sdsele;
231
232 memset(eledigest,0,20);
233 sdsele = hashTypeCurrentObjectNewSds(hi,OBJ_HASH_KEY);
234 mixDigest(eledigest,sdsele,sdslen(sdsele));
235 sdsfree(sdsele);
236 sdsele = hashTypeCurrentObjectNewSds(hi,OBJ_HASH_VALUE);
237 mixDigest(eledigest,sdsele,sdslen(sdsele));
238 sdsfree(sdsele);
239 xorDigest(digest,eledigest,20);
240 }
241 hashTypeReleaseIterator(hi);
242 } else if (o->type == OBJ_MODULE) {
243 RedisModuleDigest md;
244 moduleValue *mv = o->ptr;
245 moduleType *mt = mv->type;
246 moduleInitDigestContext(md);
247 if (mt->digest) {
248 mt->digest(&md,mv->value);
249 xorDigest(digest,md.x,sizeof(md.x));
250 }
251 } else {
252 serverPanic("Unknown object type");
253 }
254 /* If the key has an expire, add it to the mix */
255 if (expiretime != -1) xorDigest(digest,"!!expire!!",10);
256 /* We can finally xor the key-val digest to the final digest */
257 xorDigest(final,digest,20);
258 decrRefCount(keyobj);
259 }
260 dictReleaseIterator(di);
261 }
262}

264void debugCommand(client *c) {
265 if (c->argc == 1) {
266 addReplyError(c,"You must specify a subcommand for DEBUG. Try DEBUG HELP for info.");
267 return;
268 }
269
270 if (!strcasecmp(c->argv[1]->ptr,"help")) {
271 void *blenp = addDeferredMultiBulkLength(c);
272 int blen = 0;
273 blen++; addReplyStatus(c,
274 "DEBUG <subcommand> arg arg ... arg. Subcommands:");
275 blen++; addReplyStatus(c,
276 "segfault -- Crash the server with sigsegv.");
277 blen++; addReplyStatus(c,
278 "panic -- Crash the server simulating a panic.");
279 blen++; addReplyStatus(c,
280 "restart -- Graceful restart: save config, db, restart.");
281 blen++; addReplyStatus(c,
282 "crash-and-recovery <milliseconds> -- Hard crash and restart after <milliseconds> delay.");
283 blen++; addReplyStatus(c,
284 "assert -- Crash by assertion failed.");
285 blen++; addReplyStatus(c,
286 "reload -- Save the RDB on disk and reload it back in memory.");
287 blen++; addReplyStatus(c,
288 "loadaof -- Flush the AOF buffers on disk and reload the AOF in memory.");
289 blen++; addReplyStatus(c,
290 "object <key> -- Show low level info about key and associated value.");
291 blen++; addReplyStatus(c,
292 "sdslen <key> -- Show low level SDS string info representing key and value.");
293 blen++; addReplyStatus(c,
294 "ziplist <key> -- Show low level info about the ziplist encoding.");
295 blen++; addReplyStatus(c,
296 "populate <count> [prefix] [size] -- Create <count> string keys named key:<num>. If a prefix is specified is used instead of the 'key' prefix.");
297 blen++; addReplyStatus(c,
298 "digest -- Outputs an hex signature representing the current DB content.");
299 blen++; addReplyStatus(c,
300 "sleep <seconds> -- Stop the server for <seconds>. Decimals allowed.");
301 blen++; addReplyStatus(c,
302 "set-active-expire (0|1) -- Setting it to 0 disables expiring keys in background when they are not accessed (otherwise the Redis behavior). Setting it to 1 reenables back the default.");
303 blen++; addReplyStatus(c,
304 "lua-always-replicate-commands (0|1) -- Setting it to 1 makes Lua replication defaulting to replicating single commands, without the script having to enable effects replication.");
305 blen++; addReplyStatus(c,
306 "error <string> -- Return a Redis protocol error with <string> as message. Useful for clients unit tests to simulate Redis errors.");
307 blen++; addReplyStatus(c,
308 "structsize -- Return the size of different Redis core C structures.");
309 blen++; addReplyStatus(c,
310 "htstats <dbid> -- Return hash table statistics of the specified Redis database.");
311 setDeferredMultiBulkLength(c,blenp,blen);
312 } else if (!strcasecmp(c->argv[1]->ptr,"segfault")) {
313 *((char*)-1) = 'x';
314 } else if (!strcasecmp(c->argv[1]->ptr,"panic")) {
315 serverPanic("DEBUG PANIC called at Unix time %ld", time(NULL));
316 } else if (!strcasecmp(c->argv[1]->ptr,"restart") ||
317 !strcasecmp(c->argv[1]->ptr,"crash-and-recover"))
318 {
319 long long delay = 0;
320 if (c->argc >= 3) {
321 if (getLongLongFromObjectOrReply(c, c->argv[2], &delay, NULL)
322 != C_OK) return;
323 if (delay < 0) delay = 0;
324 }
325 int flags = !strcasecmp(c->argv[1]->ptr,"restart") ?
326 (RESTART_SERVER_GRACEFULLY|RESTART_SERVER_CONFIG_REWRITE) :
327 RESTART_SERVER_NONE;
328 restartServer(flags,delay);
329 addReplyError(c,"failed to restart the server. Check server logs.");
330 } else if (!strcasecmp(c->argv[1]->ptr,"oom")) {
331 void *ptr = zmalloc(ULONG_MAX); /* Should trigger an out of memory. */
332 zfree(ptr);
333 addReply(c,shared.ok);
334 } else if (!strcasecmp(c->argv[1]->ptr,"assert")) {
335 if (c->argc >= 3) c->argv[2] = tryObjectEncoding(c->argv[2]);
336 serverAssertWithInfo(c,c->argv[0],1 == 2);
337 } else if (!strcasecmp(c->argv[1]->ptr,"reload")) {
338 rdbSaveInfo rsi, *rsiptr;
339 rsiptr = rdbPopulateSaveInfo(&rsi);
340 if (rdbSave(server.rdb_filename,rsiptr) != C_OK) {
341 addReply(c,shared.err);
342 return;
343 }
344 emptyDb(-1,EMPTYDB_NO_FLAGS,NULL);
345 if (rdbLoad(server.rdb_filename,NULL) != C_OK) {
346 addReplyError(c,"Error trying to load the RDB dump");
347 return;
348 }
349 serverLog(LL_WARNING,"DB reloaded by DEBUG RELOAD");
350 addReply(c,shared.ok);
351 } else if (!strcasecmp(c->argv[1]->ptr,"loadaof")) {
352 if (server.aof_state == AOF_ON) flushAppendOnlyFile(1);
353 emptyDb(-1,EMPTYDB_NO_FLAGS,NULL);
354 if (loadAppendOnlyFile(server.aof_filename) != C_OK) {
355 addReply(c,shared.err);
356 return;
357 }
358 server.dirty = 0; /* Prevent AOF / replication */
359 serverLog(LL_WARNING,"Append Only File loaded by DEBUG LOADAOF");
360 addReply(c,shared.ok);
361 } else if (!strcasecmp(c->argv[1]->ptr,"object") && c->argc == 3) {
362 dictEntry *de;
363 robj *val;
364 char *strenc;
365
366 if ((de = dictFind(c->db->dict,c->argv[2]->ptr)) == NULL) {
367 addReply(c,shared.nokeyerr);
368 return;
369 }
370 val = dictGetVal(de);
371 strenc = strEncoding(val->encoding);
372
373 char extra[128] = {0};
374 if (val->encoding == OBJ_ENCODING_QUICKLIST) {
375 char *nextra = extra;
376 int remaining = sizeof(extra);
377 quicklist *ql = val->ptr;
378 /* Add number of quicklist nodes */
379 int used = snprintf(nextra, remaining, " ql_nodes:%u", ql->len);
380 nextra += used;
381 remaining -= used;
382 /* Add average quicklist fill factor */
383 double avg = (double)ql->count/ql->len;
384 used = snprintf(nextra, remaining, " ql_avg_node:%.2f", avg);
385 nextra += used;
386 remaining -= used;
387 /* Add quicklist fill level / max ziplist size */
388 used = snprintf(nextra, remaining, " ql_ziplist_max:%d", ql->fill);
389 nextra += used;
390 remaining -= used;
391 /* Add isCompressed? */
392 int compressed = ql->compress != 0;
393 used = snprintf(nextra, remaining, " ql_compressed:%d", compressed);
394 nextra += used;
395 remaining -= used;
396 /* Add total uncompressed size */
397 unsigned long sz = 0;
398 for (quicklistNode *node = ql->head; node; node = node->next) {
399 sz += node->sz;
400 }
401 used = snprintf(nextra, remaining, " ql_uncompressed_size:%lu", sz);
402 nextra += used;
403 remaining -= used;
404 }
405
406 addReplyStatusFormat(c,
407 "Value at:%p refcount:%d "
408 "encoding:%s serializedlength:%zu "
409 "lru:%d lru_seconds_idle:%llu%s",
410 (void*)val, val->refcount,
411 strenc, rdbSavedObjectLen(val),
412 val->lru, estimateObjectIdleTime(val)/1000, extra);
413 } else if (!strcasecmp(c->argv[1]->ptr,"sdslen") && c->argc == 3) {
414 dictEntry *de;
415 robj *val;
416 sds key;
417
418 if ((de = dictFind(c->db->dict,c->argv[2]->ptr)) == NULL) {
419 addReply(c,shared.nokeyerr);
420 return;
421 }
422 val = dictGetVal(de);
423 key = dictGetKey(de);
424
425 if (val->type != OBJ_STRING || !sdsEncodedObject(val)) {
426 addReplyError(c,"Not an sds encoded string.");
427 } else {
428 addReplyStatusFormat(c,
429 "key_sds_len:%lld, key_sds_avail:%lld, key_zmalloc: %lld, "
430 "val_sds_len:%lld, val_sds_avail:%lld, val_zmalloc: %lld",
431 (long long) sdslen(key),
432 (long long) sdsavail(key),
433 (long long) sdsZmallocSize(key),
434 (long long) sdslen(val->ptr),
435 (long long) sdsavail(val->ptr),
436 (long long) getStringObjectSdsUsedMemory(val));
437 }
438 } else if (!strcasecmp(c->argv[1]->ptr,"ziplist") && c->argc == 3) {
439 robj *o;
440
441 if ((o = objectCommandLookupOrReply(c,c->argv[2],shared.nokeyerr))
442 == NULL) return;
443
444 if (o->encoding != OBJ_ENCODING_ZIPLIST) {
445 addReplyError(c,"Not an sds encoded string.");
446 } else {
447 ziplistRepr(o->ptr);
448 addReplyStatus(c,"Ziplist structure printed on stdout");
449 }
450 } else if (!strcasecmp(c->argv[1]->ptr,"populate") &&
451 c->argc >= 3 && c->argc <= 5) {
452 long keys, j;
453 robj *key, *val;
454 char buf[128];
455
456 if (getLongFromObjectOrReply(c, c->argv[2], &keys, NULL) != C_OK)
457 return;
458 dictExpand(c->db->dict,keys);
459 for (j = 0; j < keys; j++) {
460 long valsize = 0;
461 snprintf(buf,sizeof(buf),"%s:%lu",
462 (c->argc == 3) ? "key" : (char*)c->argv[3]->ptr, j);
463 key = createStringObject(buf,strlen(buf));
464 if (c->argc == 5)
465 if (getLongFromObjectOrReply(c, c->argv[4], &valsize, NULL) != C_OK)
466 return;
467 if (lookupKeyWrite(c->db,key) != NULL) {
468 decrRefCount(key);
469 continue;
470 }
471 snprintf(buf,sizeof(buf),"value:%lu",j);
472 if (valsize==0)
473 val = createStringObject(buf,strlen(buf));
474 else {
475 int buflen = strlen(buf);
476 val = createStringObject(NULL,valsize);
477 memcpy(val->ptr, buf, valsize<=buflen? valsize: buflen);
478 }
479 dbAdd(c->db,key,val);
480 signalModifiedKey(c->db,key);
481 decrRefCount(key);
482 }
483 addReply(c,shared.ok);
484 } else if (!strcasecmp(c->argv[1]->ptr,"digest") && c->argc == 2) {
485 unsigned char digest[20];
486 sds d = sdsempty();
487 int j;
488
489 computeDatasetDigest(digest);
490 for (j = 0; j < 20; j++)
491 d = sdscatprintf(d, "%02x",digest[j]);
492 addReplyStatus(c,d);
493 sdsfree(d);
494 } else if (!strcasecmp(c->argv[1]->ptr,"sleep") && c->argc == 3) {
495 double dtime = strtod(c->argv[2]->ptr,NULL);
496 long long utime = dtime*1000000;
497 struct timespec tv;
498
499 tv.tv_sec = utime / 1000000;
500 tv.tv_nsec = (utime % 1000000) * 1000;
501 nanosleep(&tv, NULL);
502 addReply(c,shared.ok);
503 } else if (!strcasecmp(c->argv[1]->ptr,"set-active-expire") &&
504 c->argc == 3)
505 {
506 server.active_expire_enabled = atoi(c->argv[2]->ptr);
507 addReply(c,shared.ok);
508 } else if (!strcasecmp(c->argv[1]->ptr,"lua-always-replicate-commands") &&
509 c->argc == 3)
510 {
511 server.lua_always_replicate_commands = atoi(c->argv[2]->ptr);
512 addReply(c,shared.ok);
513 } else if (!strcasecmp(c->argv[1]->ptr,"error") && c->argc == 3) {
514 sds errstr = sdsnewlen("-",1);
515
516 errstr = sdscatsds(errstr,c->argv[2]->ptr);
517 errstr = sdsmapchars(errstr,"\n\r"," ",2); /* no newlines in errors. */
518 errstr = sdscatlen(errstr,"\r\n",2);
519 addReplySds(c,errstr);
520 } else if (!strcasecmp(c->argv[1]->ptr,"structsize") && c->argc == 2) {
521 sds sizes = sdsempty();
522 sizes = sdscatprintf(sizes,"bits:%d ",(sizeof(void*) == 8)?64:32);
523 sizes = sdscatprintf(sizes,"robj:%d ",(int)sizeof(robj));
524 sizes = sdscatprintf(sizes,"dictentry:%d ",(int)sizeof(dictEntry));
525 sizes = sdscatprintf(sizes,"sdshdr5:%d ",(int)sizeof(struct sdshdr5));
526 sizes = sdscatprintf(sizes,"sdshdr8:%d ",(int)sizeof(struct sdshdr8));
527 sizes = sdscatprintf(sizes,"sdshdr16:%d ",(int)sizeof(struct sdshdr16));
528 sizes = sdscatprintf(sizes,"sdshdr32:%d ",(int)sizeof(struct sdshdr32));
529 sizes = sdscatprintf(sizes,"sdshdr64:%d ",(int)sizeof(struct sdshdr64));
530 addReplyBulkSds(c,sizes);
531 } else if (!strcasecmp(c->argv[1]->ptr,"htstats") && c->argc == 3) {
532 long dbid;
533 sds stats = sdsempty();
534 char buf[4096];
535
536 if (getLongFromObjectOrReply(c, c->argv[2], &dbid, NULL) != C_OK)
537 return;
538 if (dbid < 0 || dbid >= server.dbnum) {
539 addReplyError(c,"Out of range database");
540 return;
541 }
542
543 stats = sdscatprintf(stats,"[Dictionary HT]\n");
544 dictGetStats(buf,sizeof(buf),server.db[dbid].dict);
545 stats = sdscat(stats,buf);
546
547 stats = sdscatprintf(stats,"[Expires HT]\n");
548 dictGetStats(buf,sizeof(buf),server.db[dbid].expires);
549 stats = sdscat(stats,buf);
550
551 addReplyBulkSds(c,stats);
552 } else {
553 addReplyErrorFormat(c, "Unknown DEBUG subcommand or wrong number of arguments for '%s'",
554 (char*)c->argv[1]->ptr);
555 }

560void _serverAssert(const char *estr, const char *file, int line) {
561 bugReportStart();
562 serverLog(LL_WARNING,"=== ASSERTION FAILED ===");
563 serverLog(LL_WARNING,"==> %s:%d '%s' is not true",file,line,estr);
564#ifdef HAVE_BACKTRACE
565 server.assert_failed = estr;
566 server.assert_file = file;
567 server.assert_line = line;
568 serverLog(LL_WARNING,"(forcing SIGSEGV to print the bug report.)");
569#endif
570 *((char*)-1) = 'x';
571}

573void _serverAssertPrintClientInfo(const client *c) {
574 int j;
575
576 bugReportStart();
577 serverLog(LL_WARNING,"=== ASSERTION FAILED CLIENT CONTEXT ===");
578 serverLog(LL_WARNING,"client->flags = %d", c->flags);
579 serverLog(LL_WARNING,"client->fd = %d", c->fd);
580 serverLog(LL_WARNING,"client->argc = %d", c->argc);
581 for (j=0; j < c->argc; j++) {
582 char buf[128];
583 char *arg;
584
585 if (c->argv[j]->type == OBJ_STRING && sdsEncodedObject(c->argv[j])) {
586 arg = (char*) c->argv[j]->ptr;
587 } else {
588 snprintf(buf,sizeof(buf),"Object type: %u, encoding: %u",
589 c->argv[j]->type, c->argv[j]->encoding);
590 arg = buf;
591 }
592 serverLog(LL_WARNING,"client->argv[%d] = \"%s\" (refcount: %d)",
593 j, arg, c->argv[j]->refcount);
594 }
595}

597void serverLogObjectDebugInfo(const robj *o) {
598 serverLog(LL_WARNING,"Object type: %d", o->type);
599 serverLog(LL_WARNING,"Object encoding: %d", o->encoding);
600 serverLog(LL_WARNING,"Object refcount: %d", o->refcount);
601 if (o->type == OBJ_STRING && sdsEncodedObject(o)) {

835/* Return a file descriptor to write directly to the Redis log with the
836 * write(2) syscall, that can be used in critical sections of the code
837 * where the rest of Redis can't be trusted (for example during the memory
838 * test) or when an API call requires a raw fd.
839 *
840 * Close it with closeDirectLogFiledes(). */
841int openDirectLogFiledes(void) {
842 int log_to_stdout = server.logfile[0] == '\0';
843 int fd = log_to_stdout ?
844 STDOUT_FILENO :
845 open(server.logfile, O_APPEND|O_CREAT|O_WRONLY, 0644);
846 return fd;
847}

849/* Used to close what closeDirectLogFiledes() returns. */
850void closeDirectLogFiledes(int fd) {
851 int log_to_stdout = server.logfile[0] == '\0';
852 if (!log_to_stdout) close(fd);
853}

855/* Logs the stack trace using the backtrace() call. This function is designed
856 * to be called from signal handlers safely. */
857void logStackTrace(ucontext_t *uc) {
858 void *trace[101];
859 int trace_size = 0, fd = openDirectLogFiledes();
860
861 if (fd == -1) return; /* If we can't log there is anything to do. */
862
863 /* Generate the stack trace */
864 trace_size = backtrace(trace+1, 100);
865
866 if (getMcontextEip(uc) != NULL) {
867 char *msg1 = "EIP:\n";
868 char *msg2 = "\nBacktrace:\n";
869 if (write(fd,msg1,strlen(msg1)) == -1) {/* Avoid warning. */};
870 trace[0] = getMcontextEip(uc);
871 backtrace_symbols_fd(trace, 1, fd);
872 if (write(fd,msg2,strlen(msg2)) == -1) {/* Avoid warning. */};
873 }
874
875 /* Write symbols to log file */
876 backtrace_symbols_fd(trace+1, trace_size, fd);
877
878 /* Cleanup */
879 closeDirectLogFiledes(fd);
880}

925/* A non destructive memory test executed during segfauls. */
926int memtest_test_linux_anonymous_maps(void) {
927 FILE *fp;
928 char line[1024];
929 char logbuf[1024];
930 size_t start_addr, end_addr, size;
931 size_t start_vect[MEMTEST_MAX_REGIONS];
932 size_t size_vect[MEMTEST_MAX_REGIONS];
933 int regions = 0, j;
934
935 int fd = openDirectLogFiledes();
936 if (!fd) return 0;
937
938 fp = fopen("/proc/self/maps","r");
939 if (!fp) return 0;
940 while(fgets(line,sizeof(line),fp) != NULL) {
941 char *start, *end, *p = line;
942
943 start = p;
944 p = strchr(p,'-');
945 if (!p) continue;
946 *p++ = '\0';
947 end = p;
948 p = strchr(p,' ');
949 if (!p) continue;
950 *p++ = '\0';
951 if (strstr(p,"stack") ||
952 strstr(p,"vdso") ||
953 strstr(p,"vsyscall")) continue;
954 if (!strstr(p,"00:00")) continue;
955 if (!strstr(p,"rw")) continue;
956
957 start_addr = strtoul(start,NULL,16);
958 end_addr = strtoul(end,NULL,16);
959 size = end_addr-start_addr;
960
961 start_vect[regions] = start_addr;
962 size_vect[regions] = size;
963 snprintf(logbuf,sizeof(logbuf),
964 "*** Preparing to test memory region %lx (%lu bytes)\n",
965 (unsigned long) start_vect[regions],
966 (unsigned long) size_vect[regions]);
967 if (write(fd,logbuf,strlen(logbuf)) == -1) { /* Nothing to do. */ }
968 regions++;
969 }
970
971 int errors = 0;
972 for (j = 0; j < regions; j++) {
973 if (write(fd,".",1) == -1) { /* Nothing to do. */ }
974 errors += memtest_preserving_test((void*)start_vect[j],size_vect[j],1);
975 if (write(fd, errors ? "E" : "O",1) == -1) { /* Nothing to do. */ }
976 }
977 if (write(fd,"\n",1) == -1) { /* Nothing to do. */ }
978
979 /* NOTE: It is very important to close the file descriptor only now
980 * because closing it before may result into unmapping of some memory
981 * region that we are testing. */
982 fclose(fp);
983 closeDirectLogFiledes(fd);
984 return errors;
985}

988/* Scans the (assumed) x86 code starting at addr, for a max of `len`
989 * bytes, searching for E8 (callq) opcodes, and dumping the symbols
990 * and the call offset if they appear to be valid. */
991void dumpX86Calls(void *addr, size_t len) {
992 size_t j;
993 unsigned char *p = addr;
994 Dl_info info;
995 /* Hash table to best-effort avoid printing the same symbol
996 * multiple times. */
997 unsigned long ht[256] = {0};
998
999 if (len < 5) return;
1000 for (j = 0; j < len-4; j++) {
1001 if (p[j] != 0xE8) continue; /* Not an E8 CALL opcode. */
1002 unsigned long target = (unsigned long)addr+j+5;
1003 target += *((int32_t*)(p+j+1));
1004 if (dladdr((void*)target, &info) != 0 && info.dli_sname != NULL) {
1005 if (ht[target&0xff] != target) {
1006 printf("Function at 0x%lx is %s\n",target,info.dli_sname);
1007 ht[target&0xff] = target;
1008 }
1009 j += 4; /* Skip the 32 bit immediate. */
1010 }
1011 }
1012}

1014void sigsegvHandler(int sig, siginfo_t *info, void *secret) {
1015 ucontext_t *uc = (ucontext_t*) secret;
1016 void *eip = getMcontextEip(uc);
1017 sds infostring, clients;
1018 struct sigaction act;
1019 UNUSED(info);
1020
1021 bugReportStart();
1022 serverLog(LL_WARNING,
1023 "Redis %s crashed by signal: %d", REDIS_VERSION, sig);
1024 if (eip != NULL) {
1025 serverLog(LL_WARNING,
1026 "Crashed running the instuction at: %p", eip);
1027 }
1028 if (sig == SIGSEGV || sig == SIGBUS) {
1029 serverLog(LL_WARNING,
1030 "Accessing address: %p", (void*)info->si_addr);
1031 }
1032 serverLog(LL_WARNING,
1033 "Failed assertion: %s (%s:%d)", server.assert_failed,
1034 server.assert_file, server.assert_line);
1035
1036 /* Log the stack trace */
1037 serverLogRaw(LL_WARNING|LL_RAW, "\n------ STACK TRACE ------\n");
1038 logStackTrace(uc);
1039
1040 /* Log INFO and CLIENT LIST */
1041 serverLogRaw(LL_WARNING|LL_RAW, "\n------ INFO OUTPUT ------\n");
1042 infostring = genRedisInfoString("all");
1043 serverLogRaw(LL_WARNING|LL_RAW, infostring);
1044 serverLogRaw(LL_WARNING|LL_RAW, "\n------ CLIENT LIST OUTPUT ------\n");
1045 clients = getAllClientsInfoString();
1046 serverLogRaw(LL_WARNING|LL_RAW, clients);
1047 sdsfree(infostring);
1048 sdsfree(clients);
1049
1050 /* Log the current client */
1051 logCurrentClient();
1052
1053 /* Log dump of processor registers */
1054 logRegisters(uc);
1055
1056#if defined(HAVE_PROC_MAPS)
1057 /* Test memory */
1058 serverLogRaw(LL_WARNING|LL_RAW, "\n------ FAST MEMORY TEST ------\n");
1059 bioKillThreads();
1060 if (memtest_test_linux_anonymous_maps()) {
1061 serverLogRaw(LL_WARNING|LL_RAW,
1062 "!!! MEMORY ERROR DETECTED! Check your memory ASAP !!!\n");
1063 } else {
1064 serverLogRaw(LL_WARNING|LL_RAW,
1065 "Fast memory test PASSED, however your memory can still be broken. Please run a memory test for several hours if possible.\n");
1066 }
1067#endif
1068
1069 if (eip != NULL) {
1070 Dl_info info;
1071 if (dladdr(eip, &info) != 0) {
1072 serverLog(LL_WARNING|LL_RAW,
1073 "\n------ DUMPING CODE AROUND EIP ------\n"
1074 "Symbol: %s (base: %p)\n"
1075 "Module: %s (base %p)\n"
1076 "$ xxd -r -p /tmp/dump.hex /tmp/dump.bin\n"
1077 "$ objdump --adjust-vma=%p -D -b binary -m i386:x86-64 /tmp/dump.bin\n"
1078 "------\n",
1079 info.dli_sname, info.dli_saddr, info.dli_fname, info.dli_fbase,
1080 info.dli_saddr);
1081 size_t len = (long)eip - (long)info.dli_saddr;
1082 unsigned long sz = sysconf(_SC_PAGESIZE);
1083 if (len < 1<<13) { /* we don't have functions over 8k (verified) */
1084 /* Find the address of the next page, which is our "safety"
1085 * limit when dumping. Then try to dump just 128 bytes more
1086 * than EIP if there is room, or stop sooner. */
1087 unsigned long next = ((unsigned long)eip + sz) & ~(sz-1);
1088 unsigned long end = (unsigned long)eip + 128;
1089 if (end > next) end = next;
1090 len = end - (unsigned long)info.dli_saddr;
1091 serverLogHexDump(LL_WARNING, "dump of function",
1092 info.dli_saddr ,len);
1093 dumpX86Calls(info.dli_saddr,len);
1094 }
1095 }
1096 }
1097
1098 serverLogRaw(LL_WARNING|LL_RAW,
1099"\n=== REDIS BUG REPORT END. Make sure to include from START to END. ===\n\n"
1100" Please report the crash by opening an issue on github:\n\n"
1101" http://github.com/antirez/redis/issues\n\n"
1102" Suspect RAM error? Use redis-server --test-memory to verify it.\n\n"
1103);
1104
1105 /* free(messages); Don't call free() with possibly corrupted memory. */
1106 if (server.daemonize && server.supervised == 0) unlink(server.pidfile);
1107
1108 /* Make sure we exit with the right signal at the end. So for instance
1109 * the core will be dumped if enabled. */
1110 sigemptyset (&act.sa_mask);
1111 act.sa_flags = SA_NODEFER | SA_ONSTACK | SA_RESETHAND;
1112 act.sa_handler = SIG_DFL;
1113 sigaction (sig, &act, NULL);
1114 kill(getpid(),sig);
1115}

1135 serverLogRaw(level|LL_RAW,buf);
1136 b = buf;
1137 }
1138 }

1164void watchdogScheduleSignal(int period) {
1165 struct itimerval it;
1166
1167 /* Will stop the timer if period is 0. */
1168 it.it_value.tv_sec = period/1000;
1169 it.it_value.tv_usec = (period%1000)*1000;
1170 /* Don't automatically restart. */
1171 it.it_interval.tv_sec = 0;
1172 it.it_interval.tv_usec = 0;
1173 setitimer(ITIMER_REAL, &it, NULL);
1174}

1176/* Enable the software watchdog with the specified period in milliseconds. */
1177void enableWatchdog(int period) {
1178 int min_period;
1179
1180 if (server.watchdog_period == 0) {
1181 struct sigaction act;
1182
1183 /* Watchdog was actually disabled, so we have to setup the signal
1184 * handler. */
1185 sigemptyset(&act.sa_mask);
1186 act.sa_flags = SA_ONSTACK | SA_SIGINFO;
1187 act.sa_sigaction = watchdogSignalHandler;
1188 sigaction(SIGALRM, &act, NULL);
1189 }
1190 /* If the configured period is smaller than twice the timer period, it is
1191 * too short for the software watchdog to work reliably. Fix it now
1192 * if needed. */
1193 min_period = (1000/server.hz)*2;
1194 if (period < min_period) period = min_period;
1195 watchdogScheduleSignal(period); /* Adjust the current timer. */
1196 server.watchdog_period = period;
1197}

1199/* Disable the software watchdog. */
1200void disableWatchdog(void) {
1201 struct sigaction act;
1202 if (server.watchdog_period == 0) return; /* Already disabled. */
1203 watchdogScheduleSignal(0); /* Stop the current timer. */
1204
1205 /* Set the signal handler to SIG_IGN, this will also remove pending
1206 * signals from the queue. */
1207 sigemptyset(&act.sa_mask);
1208 act.sa_flags = 0;
1209 act.sa_handler = SIG_IGN;
1210 sigaction(SIGALRM, &act, NULL);
1211 server.watchdog_period = 0;
1212}
